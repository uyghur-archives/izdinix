ياخشى بوپتۇ ، مېنىڭچە ئەزالارنى كونا يېڭى دەپ ئايرىشنىڭ ھاجىتى يوق ،
بىلگىنىڭىزنى سۆزلەڭ ،بىلمىگەننى ئۆگىنىڭ دىگەن گەپ ياقىدۇ ماڭا ،
مەلۇم بىر ئەزانىڭ مەلۇم بىر مەسىلىدە پىكىرى ئورۇنسىز بولسا ئۇنىڭ تۈزۈك بىر نەرسە بىلمەيدىغانلىقىنى بىلدۈرمەيدۇ ، ھەر ئادەمنىڭ ئۆز ئەھۋالىغا يارىشا بىلىدىغان ۋە بىلمەيدىغانلىرى بولىدۇ ، ھەم مەلۇم تېمىغا بولغان قىزىقىشىمۇ ئوخشاش بولمايدۇ ، شۇڭا مەلۇم بىر تېمىغا ھەرخىل پىكىرلەرنىڭ چۈشىشى تەبىئىي ، مۇھىمى ئەڭ توغرا بولغان پىكىر ۋە مۇلاھىزە تەتىجىسىنى تەھلىل قىلىپ، ھەزىم قىلىپ ئۆز بىلىمىمىزغا ئايلاندۇرۇش ،
ئارتۇقچە سەكرەپ كېتىش ، مەسىلىنىڭ ماھىيىتىنى ، ياكى ھەقىقىي ئەھۋالنى بىلىپ تۇرۇپ ئىتراپ قىلماسلىق ياخشى ئەھۋال ئەمەس ،
يەنە بىر مەسىلە : مەلۇم تېمىغا نىسبەتەن كونا ئەزادىن بىرسى رەدىيە نوقتىسىدىن پىكىر قىلغان بولسا ئۇنىڭغا ئەگىشىپ قالغانلارمۇ رەدىيە پىكىرى بېرىش ياخشى ئەمەس ، تېما ياكى پىكىر يازغۇچى قانداق ئادەم بولىشىدىن قەتئىي نەزەر ئۇنىڭ ئاستىغا پىكىر يازغۇچى ئۆز مېڭىسىدە ئەكىس ئەتكەن تەھلىل نەتىجىسىنى ياكى قارىشىنى ئوتتۇرغا قويسا بولىدۇ ، خاتا سۆزلەپ قويۇپ باشقىلارنىڭ ئەيىپلىشىگە ئوچرارمەنمۇ دەپ ئەنسىرەشنىڭ ھاجىتى يوق ، مۇھىمى ئەگەر خاتا پىكىر قىلغان بولسا توغرا پىكىرنى دادىللىق بىلەن قۇبۇل قىلىش ،
ئېينىشتىيىن مۇ ئەڭ كىلەڭسىز ياسالغان لاي ئورۇندۇقنى سىنپتا دادىللىق بىلەن ئوقۇتقۇچى ۋە ساۋاقداشلىرىغا كۆرسىتەلىگەن ، چۇنكى ئۇ ھەقىقەتەن ئۆزى قول سېلىپ نەچچەتال ئورۇندۇق ياسىغان ،
پىكىرىم تۈگىدى ، بېشىنى يېزىپ ئاخىرىغا كەلگۈچە پىكىر چېچىلىپ كەتتىمۇ قانداق ، ئەيىپكە بويرۇماڭلار !
