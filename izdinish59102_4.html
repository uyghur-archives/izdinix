مەن دەيمە دوسلارھازىر ساقچىلار مۇقۇملىقىنىڭ ئالدىقى سپىدە جان تىكىپئشلەۋاتىدۇ.شۇڭا ساقچىلارنىڭ قۇلقىغا يۈتۈپ كەتتى، ئوغىرلىۋالدى دېگەن گەپلەرخوش ياقماي قالامىكى .
شۇڭا بۇ خەۋەرگە ئاپتونوم رەيونمىزدىكى يۇقرى دەرىجىلىك رەھپەرنىڭ يۈكسەك دەرىجىدە كۆڭۈل بۆلشىگە دېققەت قىلدۇرشىمىز كىرەك.مىلەتلەر ئتتپاقلىقى نوقتىسدىن چىقىپ تۇرۇپمەملىكەتلىك سىياسى كڭەش ئەزاسى،شىنجاڭ ھەربى رويوننىڭ باش قوماندانى، ئاپتونوم رايونلۇق پارتىكومنىڭ باش سىكىرىتارى جاڭ چۈنشىيەننىڭ قۇلقىغا يەتكۈزۈش كىرەك. بۇنىڭ ئەل كۆڭلىنى مايىل قىلىپ شنجاڭنىڭ ئەبەدىي ئەمىنلىكى ئۈچۈن بىسىلغان بىر ئۇلۇغ قەدەم ئىكەنلىكىنى بىلدۈرشىمىز كىرەك. ئازراق كاللا يۈگۈرتىسەكلە بۇ ئىزدەش تورنى پۈتكۈل مەملىكەت دائىرسىگە كىڭەيتىش لازىملىقى ھەمپۈتكۈل مەملىكەتكە كڭەيگەن ساقچىلار ئىزدەش تورىنى قۇرۇپ چىققىاندىلا ئىزدىركىنىڭ بولدىغانلىقىنى ھىس قىلىشقا بولىدۇ.   دوسلار ھەممىزىگە ئايان بولدۇكى چوقۇم ۋە چوقۇم ئاممنىڭ كۈچىىگە تاينىپ ئوماق سىڭلىمىزنى مەنچە تاپالمايمىز،چۈنكى يۈتۈپ كەتكىنىگە ئاز ۋاقىت بولمىدى.   شۇڭا بىرنجى قەدەمدە سۈرىيەنىڭ ئانىسى ئانلىق ھىسياتى بىلەنلا تولغان بىر پارچە خەتنى ئامال قىلىپ جاڭ باش سىكىرىتارنىڭ قولىغا يەتكۈزۈشى كىرەك.مۇشۇ خەت يەتكۈزۈلگەندىن كىينلا رەھپەرلىرىمىز قانداق قىلىش ۋە قانداق ئىلانبىرشنى ئۆزى ئۇقىدۇ.
ئاخىرىدا ئوماق سىڭلىمىزنىڭ تىزراق ئىزدىركىنىڭ بولشىنى ئۈمىد قىلمەن.

مەن دەيمە دوسلارھازىر ساقچىلار مۇقۇملىقىنىڭ ئالدىقى...

ناۋاتتەك گەپ بولۇپتۇ، بۈگۈنمۇ يەنە يۈتكەن ئۈچ بالىنىڭ خەۋىرى بىر ئاپتۇ دوستلار چەمبىرىكىنى، شۇڭا ھەممەيلەن ھەرىكەتكە كېلىپ پۇقرالارنىڭ بۇ قەلب ساداسىنى يۇقۇرغا يەتكۈزىشىمىز كېرەك، ئەسلى تۈپ يىلتىزىدىن يوقۇتىشقا تېەىشلىكلەر مۇشۇ ئادەم بىدىكلىرى زادى. مېنىڭچە ئىچكىرگە ماڭغان ھەر قانداق كىچىك بالىنىڭ نوپۇس، تۇغۇلغان ئىسپات قەغىزى قاتارلىق بىر قاتار ماتېرياللىرىنى تەكشۈرسىلا سەپەردىكى ئانا-بالا، ئاتا-بالىلىق مۇناسىۋەت ئېنىقلىنامدىكىن دەيمەن، بىزنىڭ زېرەك خەلق ساقچىلىرىمىز بۇنچىلىك ئىشنى چوقۇم قىلالايدۇ. بىزدە ياخشى پىكىر تەكلىپلەر بولسا يۇقۇرغا دېسەك زىيىنى يوق، ئاندىن بارلىق ئاتا-ئانىلار يۈكسەك دەرىجىدە ھوشۇيارلىقىمىزنى ئاشۇرۇپ، كوچا-كويلاردا، بازارلاردا، مەھەلىلەردە بالىلارنى تاشلاپ قويۇپ ئۆز ئىشىمىزنى قىلىپ، ئۈندىدار ئويناپ ئولتۇرماي بالىلىرىمىزغا ئىەە بولايلى، قارا قوللارنىڭ ھىيلىسى بەك يامان ھازىر.

تۈنگۈن كەچ ئۇيقۇم كەلمەي ئوماق سىڭلىمىزنىڭ ئشىنى ئويلاپ چىقىپتىمەن ،قاپ يۈرەكلىك بىلەنتۆۋەندىكىلەرنى تەسەۋۇر قىلىپ چىقىتىم. بۇ ئارقىلىق سۈرىيەنى ئىزدەش دائىرسىنى تارايتىشنى ماقسەت قىلغان .
1. سڭلىمىزنى ئېلىپ كەتكىنى بىر ئايال بولشى مۇمكىن، شۇندىلا كىشلەرنىڭ كۆپ گۇمانىنى قوزغىمايدۇ ،ئۇ ئاۋال سىڭلىمىزنى ھوشسزلاندۇرۇپئېلىپ ماڭغان .
2.ئۇ مەلۇم ۋاقىت سىڭلىمىز يۇتۇپ كەتكەن يەرلەردە كۆزتىش ئىلىپ بارغان ، بۇ شۇنى چۈشەندۈردىكى ئۇ ئايال شۇ يىقىن ئەتراپتىن مىھمانخانىلاردىن ياتاق ئىلىپ ياتقان بولشى كىرەك، سۈرىيەنى ئىلىپ ماڭغاندىن كىيىن چوقۇم ۋە چوقۇم ئالدىراشلا ياتاقنى قايتۇرماي ماڭغان . بۇ ئنتايىن مۇھىم يىپ ئۈچى بىز سۈرىيە يۈتۈپ كەتكەن كۈنى كىم سۈرىيە يۈتۈپ كىتىشتىن بۇرۇن ياتاقتىن چىقىپ شۈ چىقىپ كەتكىنىچە ياتاق قايتۇرمىدى، مۇشۇنى ئىنىقلاش بەك مۇھىم.
3.ئۇ ئايال چوقۇم ئايروپىلان بىلەن ماڭدۇ،شۇڭا سڭلىمىز تۇرۇشلۇق ۋىلايەتتىن مەملىكىتىمىزگە قايىى چوڭ شەھەرلەرىگە
ئاۋىئاتسىيە لىنىيىسى ئىچىلغان ،سۈريىە يۈتۈپ كەتكەندىن كىيىنكى ئەڭ بالدۇر ئۈچقان نۆۋەتچى ئايروپىلان قايسى بۇنى ئىنىقلاش ھەمدە ياتاقنى قايتۇرماي چىقىپ كەتكۈچنىڭ كىمىلىك نومۇرى بىلەن ئايروپىلانغا تىزملاتقۇچنىڭ كىملىك نومۇرى بارمۇ يوق بۇنى سۈرۈشتە قىلىشئەگەر بولساسۈرىيەنى ئېلىپ كەتكۈچنىڭ شۇ بولشى ئىنىق.
كۆپ نەرسىلەرنى تەسەۋۇر قىلغان بولساممۇ ئەمما مۇشۇلارنى ئوبىكتىپ ئەمەلىيەتكە ئەڭ ماس كىلدۇ دەپ قاراپ يازدىم.
