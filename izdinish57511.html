بىر ئىستۇدېنتنىڭ 40 مىڭ يۈەن پۇلى قىسقا ئۇچۇر ئارقىلىق ئالداپ كېتىل...

شۈ لۇلۇ

   شىنجاڭدىكى مەلۇم ئالىي مەكتەپنىڭ 4 – يىللىق ئوقۇغۇچىسى جېڭ يى (ئۆزگەرتىلدى) خۇنەن تېلېۋىزىيە ئىستانسىسى ‹‹مەن ناخشىچى›› پىروگراممىسىنىڭ سادىق كۆرۈرمىنى بولۇپ، ناخشىچىلارنىڭ ئاجايىپ ماھارىتىدىن تەسىرلىنىش بىلەن بىللە، ئېھتىياتسىزلىق قىلىپ ئالدامچىلار كۆڭۈل قويۇپ قۇرغان ‹‹مۇكاپات چىقىش ئالدامچىلىقى›› توزىقىغا چۈشۈپ قالغان. ئالداپ كېتىلگەن پۇلنى ئۇ ئۇنىۋېرسىتېتتىكى مەزگىلدە ئوقۇپ تۇرۇپ جاپالىق ئىشلەپ يىغقان ئىكەن. 6 – ئاينىڭ 20 – كۈنى جېڭ يى مۇخبىرغا ئالدانغانلىق جەريانىنى سۆزلەپ بېرىپ، تېخىمۇ كۆپ كىشىنىڭ ئاگاھ بولۇشىنى تەۋسىيە قىلدى.

   بۇ يىل 4 – ئاينىڭ 25 – كۈنى، جېڭ يى ناتونۇش نومۇردىن كەلگەن يان تېلېفون ئۇچۇرىنى تاپشۇرۇۋالىدۇ، ئۇچۇر مەزمۇنى ‹‹مەن ناخشىچى›› پىروگراممىسىدا ئۇنىڭغا 2 – دەرىجىلىك مۇكاپات چىققانلىقى، مۇكاپات بۇيۇمىنىڭ ‹‹ئالما›› ماركىلىق خاتىرە كومپيۇتېر بىلەن 168 مىڭ يۈەن مۇكاپات پۇلى ئىكەنلىكى توغرىسىدىكى ئۇقتۇرۇش ئىكەن. تۇيۇقسىز ئاسماندىن چۈشكەن بۇ زور مۇكاپات جېڭيىنى بەكمۇ شادلاندۇرغان. گەرچە ئۇ بىر ئاز ئىككىلەنگەن بولسىمۇ، ئەمما تەلەي سىناپ بېقىش ئويى بىلەن 400 دىن باشلانغان تېلېفون نومۇرىغا تېلېفون قىلغان. قارشى تەرەپتىن بىر ئايالنىڭ ئاۋازى ئاڭلانغان. جېڭ يى بۇنى باشقا خېرىدار مۇلازىمىتى تېلېفونلىرىغا ئوخشىتىپ، ھوشيارلىقىنى بوشاتقان. ئۇنىڭ ئۈستىگە ئۇ 400دىن باشلانغان تېلېفون نومۇرلار ئادەتتە مۇنتىزم ئورۇنلارنىڭ تېلېفون نومۇرى دەپ ئويلاپ، قىلچە گۇمانلانمىغان.

   مۇلازىمەت خادىمى ئۇنىڭغا ‹‹مەن ناخشىچى›› پىروگراممىسى ‹‹تەلەيلىك كۆرۈرمەن›› گۇرۇپپىسىنىڭ مۇئامىلىدارلار دىرېكتورى ئۇنىڭ مۇكاپاتىنى نەقلەشتۈرۈش ئىشلىرىغا مەسئۇل بولىدىغانلىقىنى ئېيتقان. مۇئامىلىدارلار دىرېكتورى ئۇنىڭغا: مۇكاپات بۇيۇمى ۋە مۇكاپات سوممىسىنى نەقلەشتۈرۈش ئۈچۈن ئالدىن 5900 يۈەن سۇغۇرتا ھەققى تاپشۇرۇش كېرەكلىكىنى، پۇلنى ياۋ جۈلەن نامىدىكى سودا – سانائەت بانكىسى كارتىسىغا ئۇرۇشنى ئېيتىدۇ. جېڭ يى كۆرسىتىلگەن ھېسابات نومۇرىغا پۇلنى ئۇرغاندىن كېيىن، مۇئامىلىدارلار دىرېكتورى يەنە %20 مۇكاپات تاپاۋىتى بېجى ئۈچۈن 33 مىڭ 800 يۈەن تاپشۇرۇش كېرەكلىكىنى ئېيتىدۇ. زور مۇكاپاتنى ئېلىشقا تاقىتى تاق بولغان جېڭ يى ئەقلىي پەرەز قىلىش ئىقتىدارىنى يوقىتىپ، مەكتەپ پۈتكۈزگەندىن كېيىن خىزمەت تېپىشقا ئىشلەتمەكچى بولغان پۇلىنى قارشى تەرەپ كۆرسىتىپ بەرگەن ھېسابات نومۇرىغا ئۇرىدۇ.

   ‹‹مۇكاپات نەقلەشتۈرۈش›› جەريانى بىر سائەتكىمۇ يەتمەيدۇ، جېڭ يى يانفونىنى تۇتقىنىچە ھاياجان ئىلكىدە نەقلەشتۈرۈلگەن مۇكاپات سوممىسىنىڭ ھېسابات نومۇرىغا كىرىشىنى كۈتىدۇ. ئىككى سائەت ۋاقىت ئۆتۈپ كېتىدۇ، ئەمما بانكا ھېساباتىغا مۇكاپات سوممىسى كىرگەنلىك ھەققىدە ھېچقانداق ئۇچۇر كەلمەيدۇ. جېڭ يى سەل جىددىيلىشىپ مۇئامىلىدارلار دىرېكتورىغا تېلېفون قىلىدۇ، ئەمما قارشى تەرەپنىڭ تېلېفونى ئۇلانمايدۇ، ئۇ قايتا – قايتا ئۇرۇپمۇ قارشى تەرەپنىڭ تېلېفونىنى ئالالمايدۇ. شۇ چاغدا جېڭ يى ئاندىن ئېسىگە كېلىپ، ئۆزىنىڭ ئالدانغانلىقىنى ھېس قىلىپ، دەرھال ساقچىغا دېلو مەلۇم قىلىدۇ.

   كېيىن جېڭ يى توردىن ئىزدەش ئارقىلىق خۇنەن تېلېۋىزىيە ئىستانسىسى ‹‹مەن ناخشىچى›› پىروگرامما گۇرۇپپىسىنىڭ 2 – ئايدىلا ئۇقتۇرۇش ئېلان قىلىپ، ‹‹تەلەيلىك كۆرۈرمەن›› لەرنىڭ مۇكاپاتقا ئېرىشىش ئۇچۇرى ھەم ئۇچۇردىكى تور پونكىتى ئادرېسىنىڭ ‹‹مەن ناخشىچى›› پىروگراممىسى گۇرۇپپىسى بىلەن ھېچقانداق ئالاقىسى يوقلۇقى، پىروگرامما گۇرۇپپىسىدىكىلەرنىڭ كۆرۈرمەنلەرگە ھېچقانداق لاتارېيە تارتىش پائالىيىتى ئۇيۇشتۇرمىغانلىقىنى جاكارلىغانلىقىنى بىلىدۇ.

   مۇخبىرنىڭ ئىگىلىشىچە، ئالدامچىلىق قىلمىشىدىكى 400دىن باشلانغان نومۇر تور تېلېفونىدا ئۇرۇلغان بولۇپ، ئۇ بىر خىل VOIP (كۆچمەنIP، ئىز قوغلاپ رازۋېدكا قىلىش تەس بولىدىغان) تېخنىكىسى ئىكەن. بۇ خىل يۇمشاق دېتال تېخنىكىسى ئارقىلىق ئالدامچىلار قارشى تەرەپ تېلېفونىغا ھەر قانداق نومۇرنى چىقىرالايدىكەن، جۈملىدىن 400، 110، 120 نومۇرلىرىنىمۇ نىقابلىيالايدىكەن.

   ھازىر، ئۈرۈمچى شەھەرلىك ج خ ئىدارىسى جىنايى ئىشلار رازۋېدكا تارماق ئەترىتى مەزكۇر تېلېفون ئالدامچىلىقى دېلوسىنى تەكشۈرۈۋېتىپتۇ.

ئالدىنقى ھەپتە  تىيەنجىن ئاۋىئاتسىيە شىركىتىنڭ تور بىتىدىن دادام ۋە ئۆزۈمگە بېلەت سېتىۋالغان، ئايروپىلان ئۇچۇشقا بىر كۈن قالغان ۋاقىتتا ئايروپىلاندا كاشىلا كۆرۈلگەنلىك سەۋەپلىك دادامنىڭ بېلىتىنىڭ قايتۇرۋىتىلگەنلىكى، تەپسىلى ئىشلارنى 400دىن باشلانغان بىر مۇلازىمەت تېلىفونىدىن سۈرۈشتۈرش لازىملىقى توغۇرلۇق بىر ئۇچۇر تاپشۇرۋالدىم، ئۇچۇر ئەۋەتلگەن 400 دىن باشلانغان مۇلازىمەت تېلېفونىغا تېلېفون قىلسام سىچۈن تەلەپپۇزىدا سۆزلەيدىغان بىر ئايال ئايروپىلاندىن كاشىلا كۆرۈلگەنلىك سەۋەبىدىن بېلەتنىڭ قايتۇرۇلغانلىقىنى، ئىككى سائەتتىن كېيىننكى ئايروپىلانغا ئۆزگەرتىشىم كېرەكلىكىنى، ۋە ئۆزگەرتىش ھەققى ئۈچۈن 120 سوم تاپشۇرۇشۇم كېرەك ئىكەنلىكىنى، پۇلنى بىر سودا-سانائەت بانكىسىنىڭ كارتىسىغا سېلىشمنى، ئالپەيدە تۆلەشكە بولمايدىغانلىقىنى ئېيتتى، نېمە سەۋەبتىن ئىككى بىېلەتتىن بىرنىڭ قايتۇرۇلغانلىقىنى سورىسام گېپىڭنى چۈشىنەلمىدىم دەيدۇ، تېلېفوننى قويۇۋىتىپ ئۇچۇرنى قايتا ئاچسام مەن بېلەت سېتىۋالغاندا ئۇچۇر ئەۋەتكەن نومۇردىن ئەمەس بەلكى 4006088 دىن باشلانغان بىر نومۇردىن كەپتۇ، توردىن شۇ نومۇرنى تەكشۈرسەم مەخسۇس يېشى 40 ياشتىن يۇقىرى كىشلەرنىڭ توردىكى بىلەت سېتىۋىلىش ئۇچۇرىغا ئېرىشىۋىلىپ شۇنداق يالغان ئۇقتۇرۇش ئارقىلىق ئالدامچىلىق قىلىدىغان نومۇر ئىكەنلىكىنى ئۇقتۇم، ئاندىن دەرھال تىيەنجىن ئاۋىئاتسىيە شىركىتىنىڭ تاۋباۋدىكى مۇلازىمەت سۇپىسىدىن سۈرۈشتۈرسەم بېلەتنىڭ نورمال ئېكەنلىكىنى، ئەتە ۋاقتىدا ئۆلگۈرۈپ بېرىشمنى ئىيتتى، 120 سوم ئۈچۈنمۇ نۇمۇس قىلماي ئالدامچىلىق قىلىۋاتقان يولداشلارغا قاتتىق ھەيران قالدىم

نەۋرە تازلار بۇ لاتارىيەنى مالىيە ئۇنۋىرستىتنىڭ ئالدىدىن تېپىۋېلىپ كا تېپىۋالدىم دەپ ئويناۋېتىپتىكەن،بۇ بىر تاللا لاتارىيە بېلىتى ئەمەس،بەلكى بىرمۇنچە لاتارىيە،تېخى ھىچقايسىسنىڭ مۇكاپاتنى كۆرىدىغان يېرىنى قىرماپتىكەن،مەن  ئىچىم پۇشۇپ ئولتۇرۇپلا بىر قانچسىنى قىرىپ باقسام بىرسىدىن ئىككىنچى دەرىجىلىك مۇكاپات چىقىپ قالدى،سوممىسى 280 مىڭكەن،باشتا يالغانغۇ بۇ دەپ  ئانچە ئېرەڭشىمەپتىمەن،كىيىن راس بوپ قاممىسۇن يەنە دەپ لاتارىيە بېلىتىنىڭ ئۈستىدىكى نۇمۇرغا تىلىفۇن قىلسام  پۇتۇڭخۇانى زورىغا، قىينىلىپراق  سۆزلەيدىغان بىر ئايال تىلىفۇننى ئېلىپ، مېنىڭ ھەقىقەتەن 280 مىڭ يۈەن مۇكاپاتقا ئېرىشكەنلىكىمنى، تىزدىن كىملىكىنى ۋە لاتارىيە بېلىتنى نۇسقىلاپ لاتارىيە بېلىتىدىكى فاكىس نۇمۇرىغا فاكىس قىلىپ بېرىشىمنى ئېيتتى،باشتا مۇشۇنداق ئۇسۇل بىلەن باشقىلارنىڭ سەخسى ئۇچۇرىنى يىغىدىغانلار ئوخشايدۇ دەپ ئويلاپتىمەن،كىيىن يۇرتقا يېنىپ كېلىپ  توردىن ئىزدەپ باقسام بۇ بىر ئالدامچىلىقكەن،ساقچى تەرەپنىڭ بۇ خىل ئالدامچىلىققا بەرگەن باھاسىمۇبار بولۇپ ،ساقچى تەرەپ بۇ خىل ئالدامچىلىق دىلولىرىنىڭ قىينىچىلىق دەرىجىسى يۇقۇرى،ئاسان پاش قىلىغىلى بولمايدۇ دەپ قارايدىكەن. يېنىپ كىلىپ  ھەپتىدىن كىيىن يېقىن ئۆتىدىغان بىر ئاكا ئاغىنىمىز يېنىمغا كېلىپ ھېلىقى گەپنى دىمەسمۇ.سېنىڭ تېپىۋالغنىڭ بىيجىڭ بىرنىمە پەن-تېخنىكا شىركىتى دەپ خەت بار لاتارىيە ئەمەستۇ دىسەم ،دەل شۇ دەيدۇ،ھەي ئاكا،مۇكاپاتقا ئېرىشسەڭ پۇلۇڭدىن ئازراق بەر دەپ  تۇرىۋالمايىتتىم ،ماندا بىر ئىش بارتى قانداق قىلىمەن دەپ بىر مەسلىھەت سېلىۋەتكەن بولساڭ نىمە بولاپ كىتە بولغىيتى دىسەم كۆزەمگە قاراپ تۇرىدۇ(لاتارىيەنى نەدىن تېپىۋالغان دىسەم تۇپراق بېشىغا چىققاندا تېپىۋالغان دەيدۇ،قارىسام بۇ لاتارىيە ھەممە يەردە بار ئوخشايدۇ).ساقچىغىمۇ مەلۇم قىلدى،بانكىغا بېرىپ كانىمۇ تەكشۈردى پۇل سالغان كارتا  پوچتا بانكىسنىڭ كارتىسى  بولۇپ  ،خېبىيلىق بىرسنىڭ نامىدىكەن ،ج خ دىكىلەر ساڭا ئىدارە نامدىن ئالاقە يېزىپ بىرەيلى،ئۆزەڭ بىر بېرىپ باققىن دەپتۇ.باش كۆزى يوق كېتىپ قالدىدە 80 مىڭ كوي.

چەنتۇ،ئۇيغۇر،ئۇرغۇي،ئۇيغۇرمۇ ياكى مۇسۇلمانمۇ.بۇلار ئۇيغۇرنىڭ يۈزيىللىق ئىدىئولوگىيەسىنى مەلۇم دەرىجىدە ئەكىس ئەتتۈرۈپ بىرىدىغان ناملار،چەنتۇ ئالاھىدە مىللى خاسلىققا قارىتىلغان نام ئەمەس،مەسىلەن ناخشا - ئۇسۇلغا ماھىر دىگەندەك،چەنتۇلۇقتا ئۆتكۈگەن كۈنلىرىمىزنى 60-70 يىل تىللاپ كەلدۇق،لېكىن ھەرقانچە قىلساقمۇ چەنتۇدىن ياخشى بولالمىدۇق.چەنتۇ دىگەن نامنىڭ ھاقارەت تۈسىنى ئالغىنى راس ،لېكىن چەنتۇلۇق ھاقارەت ھىس قىلغۇدەك ئىش ئەمەس، چەنتۇلۇقتىن ئۇيغۇرلۇققا كۆچۈپمۇ ئاسماندا ئۇچۇپ كەتمدۇق.بۇ ئاكا ئاغىنەمدە چەنتۇلۇقنىڭ بەزى ئالاھىدىكلىرى ساقلانغان،ئۇنىڭ 80 مىڭ كويغا قويۇلۇپ كەتكىنى ماختىغۇدەك ئىش ئەمەس،لېكىن نەپرەتلەنگۈدەكمۇ ئىش ئەمەس،بۇ ئاغىنەم ئەجىر قىلماي كەلگەن نىمەتنىڭ ھارام ئىكەنلىكىدىن ئىبارەت بۇ ئەقەللى ساۋاتنى بىلسىمۇ ئۇنىڭغا ئەمەل قىلمىغان،بۇ بىر تەرىپى ،يەنە بىر تەرىپى بولسا ئۇنىڭدا دۇنياغا مۇرەسسەسىز ھالدا ئىشلارنىڭ تەپەككۇرسىز ئۆز قىياپىتى ۋە ئۆز قەلبىدىكى ئۆزىنىڭ پورتىرىتنى ئۆلچەم قىلغان قەلىب لوگىكىسى بويىچە مۇئامىلە قىلىشتىن ئىبارەت ئوغۇزخاندىن باشلانغان دۇنيانى ئىدىيەسىز يالىڭاچ ھالەتتە ئويلاشتىن ئىبارەت بىر ئاساسلىق خاراكتىر ساقلانغان.بۇ ئۇيغۇرنىڭ تۈپكى خاراكتىرلىنىڭ بىرسى،يەنە بىر تەرىپى بولسا ئۆزنى خارلاش دەرىجىسگە بارغان شان-سۆھرەت تۇيغۇسى،شۇڭا 40 مىڭ كويغا قويۇلغنى باشقىلارغا دىيەلەيدۇ،80  مىڭغا قويۇلغنى باشقىلارغا دىيەلمەيدۇ.
ئويناپ سۆزلىسەكمۇ ئويلاپ سۆزلەيلى،باشقىلارنىڭ ئاچىسى بولسا سىزنىڭمۇ ئاچىڭىز ،سىڭلىڭىز بار،بۇلارمۇ يوق دىگەندە ئانىڭىزمۇ بولسا بار،سىز دىيەلىگەننى باشقىلارمۇ دىيەلەيدۇ.
