يۇمۇرلار

توشقان بىلەن تاشپاقىنىڭ مۇسابىقىسى باشلىنىپتۇ،مۇسابىقە باشلىنىشى بىلەن توشقان خېلىلا ئۇزاپ كېتىپتۇ،تاشپاقا ئۆز مېڭىشىدا مىسىلداپ كېتىۋاتسا ئېغىر قېپىنى يۈدۈپ ئاران كېتىۋاتقان قۇلۇلىنى كۆرۈپ ئىچى ئاغرىپ دۈمبىسىگە مىندۈرىۋاپتۇ،ماڭە - ماڭ بىريەرگە كەلسە بىر چۈمۈلە كېتىۋاتقان ئۇنىمۇ ئېلىۋاپتۇ.چۈمۈلە تاشپاقىنىڭ دۈمبىسىدە قۇلۇلىنى كۆرۈپ " ياخشىمۇ سىز " دىيىشىگە قۇلۇلە قوپۇپ "چىڭ ئولتا ،ما ئاداش يامان تېز ماڭىدىكەن " دىگۈدەك .
----------------------
بىر توڭ قىشتا يېرىم كېچىدە بىر ئۆيگە ئوت كېتىپتۇ،ئاتا - ئانا دەرھال  ئۆيدىن قېچىپ قارىسا ئوغلى يوق ،ئاپىسى شۇ ھامان دات - پەرياد كۆتۈرۈپ ۋاقىراپتۇ " قېنى سەن ئوغلۇم ........."ئۆيدىن ئوغلى ئالدىرىماي جاۋاپ بېرىپتۇ" كىيىملىرىمنى كىيىۋاتىمەن "
-- ۋاي بالام ما ئۆي كۈيۈپ كېتەي دەۋاتسا نىمە كىيىم كىيىش .چاققان چىق !-- دەپ ۋاقىراپتۇ ئاپىسى .بەش مىنۇت ئۆتۈپتۇ قارىسا ئوغلى يەنە چىقمىغان ،ئاپىسى جىلى بولۇپ يەنە ۋاقىراپتۇ "ۋاي بالام،ساقمۇ سەن ،نىمىشقا چىقمايسەن؟ "
كىيىم كەيمە دىگەنتىڭ،كىيىملىرىمنى سېلىۋاتىمەن -- دىگۈدەك ئوغلى .
-----------------------------------------------
دەپنىخانىغا ئۈچ جەسەتنى ئېلىپ كەپتۇ،دەپنە خادىمى قارىسا ئۈچىلىسىنىڭ چىرايىدا كۈلكە قېتىپ قالغان پېتى ئۆلگەن ،ھەيران بولۇپ ساقچىدىن سوراپتۇ " ئەپەندىم ،بۇلار قانداق ئۆلگەن ؟ "
-- ماۋۇ ئاداشقا تۈنۈگۈن ئون مىليون دوللار مۇكاپات چىقىپتىكەن،خوشاللىغىدا تولا كۈلۈپ ئۆلۈپ قاپتۇ ،ماۋۇ ئاداشنىڭ ئايالى ئۇدا 8 قىز تۇغۇپ تۈنۈگۈن بىر ئوغۇل توغۇپتىكەن ،شۇنىڭ خوشاللىغىدا ئوغلىنى تۇتۇپ ئولتۇرۇپ ھىجايغان پېتى ئۆلۈپ قاپتۇ -- دەپتۇ ساقچى .
-- ئەمىسە ماۋۇ بەكرەك ھىجىيىپ قالغان ئاداشچۇ ؟ دەپ سوراپتۇ دەپنە خادىمى ئاخىرقى جەسەتنى كۆرسۈتۈپ .
-- بۇ بىچارە تۈنۈگۈن بىر دەرەخكە يامىشىۋېتىپتىكەن ،دەل شۇ چاغدا چاقماق چېقىلسا بولىدۇ،بۇ ئاداش تېخى بىرسى ئۆزىنى رەسىمگە تارتىۋاتقان ئوخشايدۇ دەپ ...............

-----------------------------------------------
بىر ئادەم بالكونىنى تازىلاۋېتىپ بىر كونا جىن چىراق تېپىۋاپتۇ،چىراقنى ئەمدى سۈرتۈپ تۇرۇشىغا ئىچىدىن بىر دىۋە سەكرەپ چىقىپ"ئېي ئادەم،مېنى قۇتقۇزغانلىغىڭ ئۈچۈن بىر تەلىۋىڭنى ئورۇندايمەن،قېنى ئېيتقىن "دەپتۇ ،ھىلىقى ئادەم تۇرۇپ كېتىپ "چىراق ئىچىدىن چىققان باشقا جىنلار ئۈچ تەلەپنى ئورۇندايدىكەندۇق، سىز بىرنىلا دەيسىزغۇ " دىيىشىگە دىۋە قوپۇپ " مەن دىگەن ئاتۇشلۇق دىۋە ،بولە چاققان تەلىۋىڭنى دە "دەپتۇ،ھىلىقى ئادەم ئويلىنىۋېلىپ " مۇشۇ مۈشۈكنىڭ توققۇز جىنى بولىدىكەن،مېنىڭمۇ توققۇز جېنىم بولسا "دەپتۇ ، "تەلىۋىڭ ئىجاۋەت " دەپلا دىۋە يوقاپ كېتىپتۇ . كۈنلەر ئۆتۈپ ھىلىقى ئادەم زېرىكىپ بىر قېتىم ئۆلۈپ باقماقچى بولۇپتۇ،ئويان - ئويلاپ بۇيان ئويلاپ پويىز چاقىدا ئۆلۈپ باقماقچى بوپتۇدە رېلىستا يېتىۋاپتۇ ،ۋاي ئەپسۇس بۇ ئادەم شۇ پېتى بىراقلا ئۆلۈپ قاپتۇ  ،" نىمە ؟دىۋىنى يالغانچىكەن  دەمسىز ؟ ياق، ياق ،ئۇ كۈنكى پويىز نەق ئون ۋاگونلۇق ئىكەن "
.......................................................................................................
-- قاراڭ دوختۇر ، يېقىندىن بېرى كۆزۈم يىراقنى كۆرەلمەيدىغان بولۇپ قالدى -- دەپتۇ بىمار .
-- شۇنداقمۇ -- دەپتۇ دوختۇر ۋە بىمارنى باشلاپ دەرىزە ئالدىغا كېلىپ قۇياشنى كۆرسۈتۈپ تۇرۇپ " ئاۋۇ نىمە " دەپ سوراپتۇ .
-- قۇياش !
-- ئەمەسە دەڭچۇ قېنى ،ئۇنۇڭدىن يىراق يەنە نەنى كۆرمەكچىتىڭىز --- دىگۈدەك دوختۇر .
-----------------------------------------------
بىر ئادەم بېلىق تۇتماقچى بولۇپ باشتا سازاڭنى يەمچۈك قىلىپ بېقىپتۇ،نەچچە سائەتكىچە قارماققا ھىچنىمە ئېلىنماپتۇ،بولكىنى يەمچۈك قىلىپ بېقىپتۇ،ھىچنىمە يوق ،گۆشنىمۇ يەمچۈك قىلىپ بېقىپتۇ ،يەنە ھىچنىمە يوق ،ئاچچىغىدا 100 سوملۇق پۇلدىن بىرنى قارماققا ئىلىپلا سۇغا تاشلاپتىكەن سۇ ئاستىدىن بېلىقلار " ھېي ئوغرى ،سەي بازىرى كۆرۈپ قالدىڭمۇ بۇ يەرنى " دىگۈدەك
-----------------------------------------------
ھاياتىڭىزنى قەدىرلەي دىسىڭىز ،جوڭگو دۆلەت كوماندىسىدىن يىراق تۇرۇڭ !
-----------------------------------------------بىر كۈنى مەن ئىشىك - دەرىزىنى ھىم تاقاپ ،پەردىلەرنى چۈشۈرۈپ ئەمدىلا مەركىزىي تېلىۋىزىيە ئىستانىسىنىڭ 5 - قانىلىنى ئېچىپ جوڭگو دۆلەت كوماندىسىنىڭ مۇسابىقىسىنى كۆرۈپ تۇرغان ئىدىم،ئاغىنىلىرىم كېلىپ ئىشىكنى ئۇرغىلى تۇرۇشتى . مەن ھاپىلا - شاپىلا تېلىۋىزورنى ئۆچۈرۈپ ئىشىكنى ئاچتىم .ئۇلار مېنىڭ ئوڭايسىز ھالىمنى كۆرۈپ " ئەسلىدە سەن ئىشىكنى چىڭ تاقاپ سېرىق فىلىم كۆرىۋېتىپسەندە " دىدى ،مەن دەرھال ئۆزەمنى ئاقلاپ "ئىشىڭنى قىلىشە ماڭ،سېرىق فىلىمنى سەندەك ئىشەك گۇيلار كۆرىدۇ  " دىدىم ،ئاڭغىچە بىر .ئاغىنەم تېلىۋىزورنى ئېچىپ " ھە سەن ئەسلى جوڭگو دۆلەت كوماندىسىنىڭ مۇسابىقىسىنى كۆرۈۋاتقانما "دىدى،يۈزۈم  دەرھال ئوت ئېلىپ  قىزىرىپ كەتتى - دە  دەرھاللا " ياق ،ياق ،مەن ئەسلى سېرىق فىلىم كۆرۈۋاتقان " دىدىم .
-----------------------------------------------يەنە ھىلىقى دىۋە بار جىن چىراق بىر جوڭگولۇققا ئۇچراپ قاپتۇ،دىۋە "مېنى قۇتقۇزىۋالغىنىڭغا رەھمەت  ،مەن ما ئاتۇشلۇق دىۋە،بىرلا تەلىۋىڭنى ئورۇندايمەن،چاققان بولە " دەپتىكەن ھىلىقى جوڭگولۇق "مۇشۇ ياپۇنىيەنى يەر يۈزىدىن يوق قىلىۋەتسەڭ " دەپتۇ،دىۋە قوپۇپ " ئاداشە ۋا،بۇنداق تەس ئىشنى دىگۈچە باشقا تەلەپنى دىگىنە " دەپتىكەن جوڭگولۇ قوپۇپ " ئەمىسە جوڭگو دۆلەت  پۇتبول كوماندىسى بۇ قېتىم چىمپىيون بولسۇن " دەپتۇ ،دىۋە كۆتتىنى قاشلاپ تۇرۇپ كېتىپ "باشتىكى تەلىۋىڭ نىمىتى ؟ " دىگۈدەك
