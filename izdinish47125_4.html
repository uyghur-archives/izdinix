مۇشۇنداق چېگىش خىياللار ئىلكىدەكاللامغا تەسىر قىلغان ماۋۇ ئىشلارنى ئويلاپ قالدىم.ھويلا قورۇيىمىزنىڭ ئالدىدا ئاكا-ئۇكا ناۋاي، قاسساپلار بىرەر يۈز ئائىلىكنى گۆش، نان بىلەن تەمىنلەيدۇ. ماۋۇ يولداشلار ھور نىنى بىلەن كەسكەن چۆپىنى يىمەي، خاۋچىي دىيىشىپ، يولدىلا يەپ ماڭىدۇ بەزىلىرى.ناۋاي ئاكىمىزنىڭ ئايالى كىچچىكىنە كەلگەن دۇكاندا يولدىشىغا ياردەملىشىپ نان ياسلايدۇ، ناۋاي ئۇستام يوغان پىيالىگە تۇزنى قۇيىۋېلىپ ھورىنى پۇرقىرىتىپ، قىزىق تونۇرغا چاچقىلى تۇرىدۇ، ئەمدى مېڭىشنى ئۆگەنگەن قىزىدادىسىنىڭ ئەتراپىدا قولىدا بىر لاتا توپنى كۆتۈرۈپ ئويناپ يۈرىدۇ، دادىسى پات-پات قىزىنىڭ ئىسمىنى ئەكىلىتىپ، يولغا چىقىپ كەتمىسۇن دەپ توۋلاپ قويىدۇ، چۈشلىكى ئىشتىن چۈشۈپ كەلسەم يەنە شۇ كۆرۈنۈش، ناۋايخانا دەلئاھالىكلەر قورۇسىنىڭ نەق ئوتتۇرىسىنى كېسىپ ئۆتىدىغان يولنىڭ يېنىغا جايلاشقان، چۈشلىكى ناۋاي ئۇستامنىڭ ئايالى يول بۇيىغىلا ئەپچىلگىنە بىر قازاننى ئېسىپ يا سۇيقاشا چۆپ تاشلاۋاتقان، يا بولمىسا لەغمەننىڭ چۆپىنى سىلىۋاتقان كۆرۈنۈش چىلىقىدۇ، ئەپچىلگىنە كىچىككىنە دۇكانغا كىرىپ باقمىغان بولساممۇ، 10 كىۋادىراتقا يەتمەيدىغان قەلەي تاختاي قۇرۇلمىلىق ئۆينىڭ بۇلۇڭىدا بىر تېلېۋېزور، ۋ س دئاپپاراتى قۇيۇلغان بۇلۇپ، كەچلىكلىرى ئىشتىن كەش كېلىپ قالسام، شۇ ئۆيدىن ناخشا-مۇزىكا، ئىتوت-پىتوتنىڭ ئاۋازى بىلەن ئائىلە ئەزالىرىنىڭ بەخت كۈلكىلىرى ئاڭلىنىدۇ،شۇ چاغلاردا ھەۋەسلىنىپ ئۆتۈپ كېتەتتىم. بايا ئىشتىن كېلىۋېتىپ يەنە بىر ئىشقا ھەيران قالدىم، ناۋاي ئۇستاملارنىڭ سودىلىقى دۇكاندا بولغاچقا، ئۆزى، ئايالى، بىر ئىنىسى كۈندە دۇكاننى چۆرگىلەيدۇ، ناۋاي ئۇستامنىڭ ئايالى ياردەملەشكەچ، قىزىنىڭ كىر-قاتلىرىنى يۇيۇپ ھالىدىن خەۋەر ئالىدۇ. مېنى ھەيران قالدۇرغىنى بۈگۈنمۇ ناۋاينىڭ ئايالى قىزىنىڭ كىيىملىرىنى يۇغان چېغى، ناۋايخانىنىڭ ئۇدۇلىدىكى بىر تۈپ پاكار قۇيۇق يۇپۇرماقلىق قارلىغاچ دەرەخكە قىزىنىڭبىر مۇنچە كىيىملىرىنى كىيىم ئاسقۇدا يايمىدەكلا ئېسىۋېىتىپتۇ، قاراپ ئۆتۈپ كەتتىم، بەخت دىگەن بەلكىم شۇنچىلىك نەرسىدۇ، ئۇلارنىڭمۇ غەم-قايغۇلىرى باردۇ، ئەممامەندەك ئايال-بالىلىرىنى سېغىنمسىمۇ بۇلىدۇ، گەرچە يول تۈپىگىلابىر كونا چىلەكتىن ياسالغان ئوچاققا ئەپچىل قازان ئېسىپ ئاش قىلىپ يىسىمۇ، مەن سىرتتىكى ئاشپۇزۇللاردىن يىگەن تاماقلاردىن جىق پاكىز، قۇۋۋەتلىك، مېھىرلىك. ئۇلار شۇنداق ئۆتكەن كۈنىگە شۈكرى قىلىپ، كۆڭلى توق ئۆتىۋېرىدۇ... ئۇلارنىڭتۇرمۇش ئاددى بولسىمۇ، ئەمما بەخىتلىك. مەن يۇقارقى مىساللارنى دىيىش ئارقىلىق، ئۆزۈمنى بەخىتسىز ۋە ياكى باشقا گەپنى دىمەكچىمۇ ئەمەس، بىزمۇ بىر يەرگە كەلسەكلا شۇلاردەك خۇشال ئۆتىمىز. پەرزەنتىمىزنىڭ تاتلىق قىلىقلىرىدىن ھوزۇرلىنىمىز. ھازىرقى گەپ بىزدە مۇشۇ نەرسىلەر كەم بۇلىۋاتقاچقا ئىچىمگە پاتماي، ھەل قىلىش باش قېتىنچىلىقىدا بۇ يەردىن ئەھلى تورداشتىن مەسلىھەت سوراپ يۈردۈم. (ھېلىمۇ خۇدانىڭ بەرگەن كۈنلىرىگە مىڭ شۈكرى)، ئادەمنىڭ تاماسى مەڭگۈ قانمايدىغان چېغى.
