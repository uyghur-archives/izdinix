تۇرپاندىكى تۇيغۇلار

                                            تۇرپاندىكى تۇيغۇلار                    

                                                 1  
                                     يوچۇن بىر ئاۋاز مېنى يېمەكتە،
                                     قىزىرىپ پىشقان بۇ دۈشەنبىدە.

                                                  2
                                     ياشاۋاتىمەن چۈشلەردىكىدەك،
                                     ياكى ياشاشنى چۈشەۋاتىمەن.
                                     رىئاللىقىمدا بىر قورام تاش بار،
                                     ئەلھەمدۇلىللا، كۈچەۋاتىمەن.

                                                 3
                                     تۇرمۇش تونۇرى كۆيەر يالقۇنلاپ،
                                     سويۇپ ئالماق تەس ئىسسىق بىر ناننى.
                                     مەن ئاشۇ رايىش،سوزۇلغان جۈپ قول،
                                     تىلىدىم ئاخىر گۈزەل ئىماننى.

                                                 4
                                     تالاي ئادەملەر ئۆتتى زىمىندىن،
                                     مەڭگۈلۈك ۋەتەن قىلغىنى قەبرە.
                                     بىز قۇش ئەمەسمىز،ياكى گۈل ئەمەس،
                                     بەلكى سىناققا يارالغان بەندە.

                                                5
                                     ھەر ھەپتىدە بىر جۈمئەدىن چىقىپ،
                                     كاتتا دېڭىزغا كىرىمىز ئېقىپ.
                                     بىر تال ئۇچقۇنمىز،بىز ۋاقىتتىكى،
                                     ئۆچۈرەر بىر كۈن پەرىشتە كېلىپ.

                                                6
                                    يالماپ تۈگەتتۇق باھارنى،يازنى،
                                    قارلار سۆيمەكتە تۈندەك بۇ چاچنى.
                                    بىر قەدەھ كەلتۈر تىمتاس شەنبىدە،
                                    بىزگە كېرىكى ۋەھدەت شارابى.

                                                7
                                    ھەپتە ياشىدىم يۇرتۇم قوينىدا،
                                    مۇساپىر بولۇپ تۇيغۇلىرىمغا.
                                    يامغۇر چۈشمەكتە،يامغۇر چۈشمەكتە،
                                    دالدىسىز تۇرغان ئويغۇرلىرىمغا.


                     

                                          ئۆچۈرۈلۈش



                                     تېلفونىمىزنى
                                     خالىساقلا ئۆچۈرۋېتىمىز،
                                     يوق قىلىمىز ياكى ئۈنىنى.

                                     جۈپ كۆزلىرىم ئۆچۈرۋېتىلدى،
                                     سەھەر يەنە يېقىلدى خالاپ.
                                     ھايات ئۇچقۇن بىر لەھزىدىكى.

                                     يانفون،مەن ۋە دۇنيا ئۈچىمىز،
                                     ئوچۇق تۇرۇپ،يوپيورۇق تۇرۇپ،
                                     بىر خالاشتا تامان ئۆچىمىز.

                                     ئالەمدىكى ھەممە شەيئىنىڭ،
                                     بىر ئۆچۈشتە ئوخشاش تەقدىرى.
                                     نۇر تارقىتىش، مەڭگۈ ئۆچمەسلىك،
                                     ئاللاھقىلا خاستۇر ئاخىرى.


                                                  مۇساپە     
                             
                                 
                                       يەتتە كۈن ياشىدىم يۇرتۇم قوينىدا،
                                       مۇساپىر بولۇپ مەن ئۆز يۈرۈگۈمگە.

                                       گۈل- ئېگىسى گۈل ئەمەس،
                                       ئاي نۇرى ئاينىڭ ئەمەس.
                                       يوپۇرماقمۇ شاختىكى،
                                       سەرگەرداندۇر بولدى بەس.

                                       پارلىغان قۇياش،
                                       كىرپىكتىكى ياش،
                                       يومۇلاق ياشاش،
                                       ھەممىگە ئوخشاش،
                                       باردۇر بىر توختاش.

                                       مەن زىمىندا كېتىۋېتىپ،
                                       باقتىم جىمجىت ئاسمانغا.
                                       سەجدە بىلەن يەتمىسەك،
                                       ئەقىل يەتمەس سۇبھانغا.

  

                                           تەۋەككۈل

                                        دوستلىرىم
                                        ئالقانلىرىدا چوغ،يۈزلىرى يورۇق،
                                        مەن تېخىچە نۇرسىز،قۇپقۇرۇق.....

                                        شاماللارغا ئايلىنالايتتىم،
                                        ئەركە قۇشتەك ياشىيالايتتىم.
                                        مەن نەفسىمنى يۈك قىلىپ ئاخىر،
                                        بۇ دۇنياغا سۇ بولۇپ ئاقتىم.

                                        ھەقىقەت
                                        تاشتۇر گويا سۈكۈناتتىكى،
                                        يا تاشتىكى سۈكۈت ئېھتىمال.
                                        پات-پات ئۇندىن قاچماق بولىمىز،
                                        تۇرۇپ تولۇق قىلىنغان قامال.

                                        بىز گۈل ۋاقىتتىكى،
                                        بىر ئېچىلىپ بولارمىز غازان.
                                        بىز قۇل مەڭگۈلۈككە،
                                        ئاگاھ بولغىن،ئاگاھ بول ئىنسان.

                                        ئابدۇۋەلى چوغ تۇت بىر دەقىق،
                                        دىلىڭغا نۇر بەرگەي ئەرراھمان.


                                                يارا

                                       كۈتىۋاتسام سۈبھى پەيتىنى،
                                       ئەڭ ئاخىرقى بىر تال شام بىلەن.
                                       تۈن باغرىدا پورەكلىگىنى،
                                       مېنىڭ گۈلدەك دىل يارام ئىكەن.

                                       سۇ يارىسى-دەريادا،
                                       قۇش يارىسى-سامادا.
                                       ئېگىمىزنى قانداق تونۇيتتۇق،
                                       گۈزەل ھەسرەتكە بولماي مۇپتىلا.

                                       ساڭا سېغىنسام ئەي ئاللاھ
                                       تاڭ ئېتىپ كەتتى بىردىنلا.

                                                ئەمما ئىچمىدى

                                      ئىچ-باغرىنى سۇغا چىلىدى

                                         تەشنا بولغان بىر ئادەم
                                سۇنى ئويلىمىدى ياكى ئىچىمىدى
                                               ئۇ سۇدا ئىدى.

                                                         يارغول

                                               
                                                 خۇددى نۇردەك بىر ئادەم


                                          [ ئابدۇۋەلى مىرزايوف قارىيغا ئاتاپ]


                                              ئۆتكەن ئەسىردە
                                              ئەنجاندىكى نامسىز قەسىردە،
                                              كۈندەك گۈزەل ياندى بىر چىراغ.

                                              يارالغاننى كۈيلەۋاتاتتىم،
                                              بىر ھارارەت چۈشتى توساتتىن،
                                              ئەڭ غايىۋىي خىيالغا پاتتىم.

                                              ئۇنىڭ
                                              ئورنى مېھراپ
                                              ئەمرى ھىجاپ.
                                              ياشاۋاتقىنى
                                              ئاسماندىن چۈشكەن بىر كىتاب.

                                              ئۇ كۆيدۈرمەيدۇ،ئۆگىتىدۇ،
                                              زۇلمەتلەرنى يورۇتىدۇ.
                                              ئۆرنەك ئۈچۈن ئەسھابلارنى،
                                              ئۆزبېك تىلىدا ئويغىتىدۇ.

                                              ئۇ سۆزلەۋاتقاندا
                                              ئۇچۇپ كىرەر مەدرىسكە قۇشلار.
                                              قەلىبلەرنىڭ پىنھانلىرىدىن،
                                              كۆك قەھرىگە ئېشىك ئېچىلار.

                                              ئۇشتۇمتۇت
                                              بىر يىگىت تەكبىر توۋلايدۇ،
                                              ئەتراپقا بىردىن،
                                              تەۋھىد ھىدلىرى تارايدۇ...

                                              بىر نۇر
                                              ئۆچۈرۋېتىلدى.
                                              بىر ئىنسان
                                             كۆچۈرۋېتىلدى،
                                              ئۆزى سۆيگەن زاتنىڭ ئالدىغا.   

                                             ئۇ ياشىماس كۆرۈمىدە،
                                             مەڭگۈ ياشار ئۆلۈمىدە.
                                             كىمنى كۈيلەش كېرەكلىكىنى،
                                             تۇنۇپ يەتتىم بەرگەن ئىلمىدە.




                                              تۈز سىزىق


                                             ھاياتىڭنىڭ يولىدا
                                             ئۆلگۈچىنىڭ پۇتى بىلەن ماڭ.
                                             نامازشامدەك قىسقا پەيىتسەن،
                                             ئەزىنىڭمۇ توۋلىنىپ بولغان.

                                             بىراۋنى كۈتىۋاتقان بولساڭ،ئۇ سەن،
                                             بار مەنانى ئوراپ تۇرغىنى يوق تەن.
                                             يوللار سەندە كېتىۋاتقاندا،
                                             نىشانىڭدىن ئازغان بولىسەن.

                                             بۇرادەر شۇنى ساڭا دەپ قوياي..

                                             يىلدا بىر تۇغۇلساڭ ئىزدىمە مېنى،
                                             بىر قېتىم ئۆلسەڭلا دوستۇم سەن چوقۇم.
                                             كۈندە بىر ئۆلسەڭمۇ يىراق يۈر مەندىن،
                                             بىر دېگەن نەقەدەر چوڭقۇر بىر ئۇقۇم.




                                                          چېنىقىش

         
                                                يىغلاش باشقا بىرخىل كۈلۈشتۇر،
                                                ياشاش ئۆلۈم تامان يۈرۈشتۇر.

                                               ئايلىنىۋاتقان ئىستىرىلكىدەك،
                                               قايتىپ كېلىمىز ئۆز-ئۆزىمىزگە.
                                               سۈزۈك نۇرنى كۆرەتتۇق بەلكىم،
                                               باقالىساق تۈننىڭ كۆزىگە.

                                               بىز سۈكۈتكە پېتىپ باقىمىدۇق،
                                               قەلبىمىزگە ئېقىپ باقمىدۇق.
                                               ئادەملىكنىڭ ئېغىرلىقىنى،
                                               تېخىچىلا تېتىپ باقىمىدۇق.

                                              قېچىش بىر خىل يېقىنلىشىشتۇر،
                                              كېچىش بەلكىم چىن ئېرىشىشتۇر.

                                              بىر ھەممىمىز بىلىپ بولۇپتۇق،
                                                ياشىماقنىڭ ئاسانلىقىنى.
                                                      قىيامەتنىڭ
                                                      قاچانلىقىنى..

                                               مەن چېنىقىۋاتىمەن
                                               ياشارتىدىغان مېۋىلەرنى يەۋاتىمەن،
                                               ناھايىتى تېز قېچىۋاتىمەن....


                                              ھۇجرامدىكى سائەتنىڭ
                                              ئايلانماقتا ئىستىرىلكىسى.

بۈگۈن ۋە باشقىلار


بۈگۈنمۇ سۇدەكلا ئېقىپ تۈگىدى،
بىر دەريا ئاقماقتا جېنىمنى بويلاپ.
ھاياتنىڭ ئۆركىشى سىڭمەكتە مەيۈس.
ئۆلۈمنىڭ بىپايان چۆلىگە قاراپ.

بىر چۈشنى ئەسلىتەر ئۆتكەن كۈنلىرىم،
ئىچىمدە چاراسلاپ گۈلخان يانىدۇ.
نە قالۇر ۋاقىتنىڭ سېخىي باغرىدىن،
چەكسىز بىر يېگانە ئىنسان قالىدۇ.

قۇشلارنىڭ ئۈنى يوق ئەتەمدە بەلكىم،
كۈي- ناخشا ئېيىتمىدىم كېلەچىكىمگە.
يۈزلەندىم ئەڭ رايىش بەندىلەرگە خاس،
      بۈگۈن ۋە ئۇندىكى ھەممە ھەممىگە.
