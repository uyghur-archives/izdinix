ئەسسالامۇ ئەلەيكۇم قېرىنداشلار!ئاللاھنىڭ ئىلتىپاتى بىلەن ئايالىمنىڭ بويىدا قېلىپ،پېقىرنى چەكسىز خۇرسەنچىلىككە چۆمدۈردى،ئاللاھدىن نامازلىرىمدا سالىھ بىر پەرزەنت سوراپ تۇرۇپتىمەن.لېكىن ئايالىم ئۆزى تاماق ئەتكىلى قازان بېشىغا كەلسە ھۆ ......بولۇپ ھېچ ھالى قالمايۋاتىدۇ،تاماقنىغۇ ئاساسەن ئاپام ئېتىۋاتىدۇ،ئايالىم قارىشىپ بېرىدۇ.بىراق ئۇ  نېىمىگە سېزىك بولغېنىنى بىلەلمەي بەك جاپادا قالدى،ھېلى پېتر مانتا.ھېلى كاۋاپ دېگەندەك گۆش تۈرىدىكى نەرسىلەرگە ھېرىس بوپ يەيدۇ لېكىن يەپ بولۇپ ئۇزۇن ئۆتمەي ياندۇرۇدۇ.بۇ ئەھۋالنى بىر دورا ساتىدىغان ئاغىنىگە دىسەم.ئۇ ۋىتامىن ۋە كالىتىسى يىتىشمىگەنلىكنىڭ ئىشى دەيدۇ،ۋە ماڭا ئۆزى سېتىۋاتقان كالتىسى ۋە ۋىتامىن ،كۆپ خىل بىرىكمە ۋىتامىن،ئاقىسىل دېگەندەك ئوقەتلەرنى ئېلىپ،مۇشۇ ھامىلدارلىق مەزگىلىدە يىگۈزسەم بالىنىڭ قوساقتىكى ئۆسۈپ -يىتىلىشگە بەك پايدىلىق دەپ چۈشەندۈردى.لېكىن ئايالىم قورقۇپ تۇرۇدۇ ئەپ بېرەي دىسەم.يەنە بىر تېۋىپ بۇرادىرىم بولسا مىڭ سۇمغا ئۇنداق دورا ئەپ بەرمەي،نوقۇت تالقىنى[ھەر خىل مېغىزلار ۋە نوقۇي قوشۇپ ئۆيدە ياساپ]بېرىشىمنى تەۋسىيە قىلدى.بېشىم قېتىپ قالدى،زادى مۇشۇ ئاياللار ھامىلدار مەزگىلىدە شۇ خىل ۋىتامىن ،كالتسى دېگەندەك نەرسىلەرنى يىسە بولامدۇ يۇق؟ ئەگەر بولمىسا  قانداق ئوزۇقلىنىش كېرەك؟ئاڭلىسام بالا قوساقتىكى چاغدا كالتىسى تولۇقلاپ بەرمىسە،راخىت بوپ قالىمىش.شۇڭا ئىزدىنىشتىكى دوختۇر-تېۋىپ،ھەكىملەرنىڭ مۇشۇ ئىشتا تەپسلىرەك چۈشەنچە بېرىشىنى ئۈمىت قىلاتتىم

مەنمۇ بېشمدىن ئۆتكەننى ، كۈزۈم كۆرگەننى جىجىلاپ باقاي .

ئايالىم ھامىلدار مەزگىلدە ، زىيالى ئويناپ يۈرگەن يەرگە يېقىن يەردە ئىدۇق بىزمۇ .

سىزىك سەت تۇتتى كاساپەتنى ، نىمە يىسە شۇ يانىدۇ ، سۇ ئىچسە سۇ .كوڭلۈمنى ئىلەشتۈرۈپ تۇكۇرۈپلا ساراڭ قىلۋەتتى .

دوختۇرغا باردۇق . پەن تېخنىكا دەھشەت تەرەققى قىلغان مۇشۇ دۆلەتنىڭ دوختۇرلىرى جېنىدا سىزىككە ئامال يوق دەپ كۈزىنى پارقىرتىپ تۇرسا مەن قانداق قىلىپ بىرەلەيتىم . قۇلۇمدىن كىلىشىچە ھالىدىن خەۋەر ئالدىم شۇ .  

دوختۇر دىدىلەركى :

ھامىلە قۇساقتا پەيدا بولغاندىن باشلاپ ، بالياتقۇ بالىنى بېقىش ئۈچۈن نۇرمال ۋاقىتتا يوق بولغان ھورمۇن ئاجىرتىپ چىقىردۇ. بەدەن شۇ ھورمۇنغا قارىشى ھەرخىل ئىنكاس قايتۇردۇ . كۆڭلى ئېلىشىش ، ياندۇرۇش .......دىگەنلەر شۇنىڭ سەۋەبى . بەدەن ئوخشىمىغاندىن كىيىن ، قايتۇردىغان ئىنكاسمۇ ئادەمگە بېقىپ ئوخشىمايدۇ ، شۇڭا ئامال يوق.

بىرەر ئايلىق بولغاندا پەقەت بۇلالمىدى ، يەنە دوختۇرغا ئاپاردىم . يەنە شۇ كونا گەپ ، ئامال يوق. لىكىن بىرنەرسە يىيەلمەي ، ھەتتا سۇمۇ ئىچەلمىگەندىكىن ، بەدەن بەك سۇسىزلىنىپ كىتىدۇ دەپ ، تۇمۇردىن بىر قۇتا سۇيۇقلۇق ئېسىپ قويدى . تەركىبى نىمىدۇ دەپ قارىسام سۇكەن شۇ .

14 ھەپتىدىن كىيىن ياخشى بۇلۇپ كىتىدۇ ، چىداڭلار دىدى ، ماقۇل دىمەي ئامال قانچە ؟ 14ھەپتىدىن كىيىن راست قۇيۇپ بەردى سىزىك .

ھامىلە 4 ئايلىق بولغىچە ئانىنىڭ بەدىندىن ئۇزۇقلۇق قۇبۇل قىلمايدۇ، شۇڭا بۇمەزگىلدە ئۇ، بۇ نەرسىلەرنى تۇلۇقلاپ ئاۋارە بولماڭلار دىدى.

4ئايدىن كىيىن تەكشۇرۇپ ، ھامىلە نۇرمال ئىكەن ، بىر نەرسە تۇلۇقلايمىز دەپ ئاۋارە بولماڭلار،ئانىسى نۇرمال ئۇزۇقلىنپ يۈرۈۋەرسۇن دىدى ، ماقۇل دەپ شۇ بۇيچە قىلدۇق . ئاخىردا بالا قەرەلىدە ، تەبىئى تۇغۇلغان بولدى . ئاللاغا مىڭ شۈكرى ، ھازىرمۇ ساغلام چوڭ بۇلۋاتىدۇ .

يەنە ، مىكىرو دولقۇنلۇق ئۇچاق ، قول تېلفۇن .....دىگەندەك ئەتىراپىمىزدىكى نەرسىلەردىن قورقۇپ يۇرۇپتىمىز ، ئۇمۇ ئىلىم-پەن ئىسپاتلىمىغان ئىش ئىكەن ، بۇرۇن قانداق ياشغان بولساڭ شۇنداق ياشاۋەرگىن دەپ يولغا سالدى .
