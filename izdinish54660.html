ئىزدىنىش مۇنبىرى › مەشھۇر شەخىسلەر › غېنى سەمەت: رىۋايەتلىك سەنئەتكار دېڭ لىجۈن

Muyunchur يوللانغان ۋاقتى 2013-8-28 14:26:20

غېنى سەمەت: رىۋايەتلىك سەنئەتكار دېڭ لىجۈن

                     رىۋايەتلىك سەنئەتكار دېڭ لىجۈن



                                  غېنى سەمەت



‹‹ جۇڭگولۇق بار يەردە ، دېڭ لىجۈننىڭ ناخشىسى بار ›› . بۇتەرىپ بەلكىم ئۇنىڭ دۇنيادىكى نوپۇسى ئەڭ كۆپ خەلق -خەنزۇلارنىڭ ئارىسىدىكى غايەت زور شۆھرىتىنى سۈپەتلەشكە كۇپايە قىلار. دېڭ لىجۈن خەنزۇ قەۋمىنىڭيېرىم ئەسىردىن بۇيانقى ناخشا-مۇزىكا سەنئىتىگە ئەڭ زور، ئۈزۈل-كىسىل تەسىر كۆرسەتكەن ، بىر-ئىككى دەۋر ئەۋلادلارنىڭ ناخشا-مۇزىكىدىن زوقلىنىش تۇيغۇسىدا، ئىستېتېكىسىدا ئىنقىلاپ خاراكتېرلىك ئۆزگىرىش پەيدا قىلغان كاتتا سەنئەتكار. جۇڭگو جانبازلىقىنى دۇنياغا تونۇتۇشتىكى ئەزىمەت- لى شىياۋلۇڭ دېيىلسە، جۇڭخۇا ئاممىباب ناخشىچىلىقىنى دۇنياغا تونۇتۇشتىكى گېگانت چولپان دەل دېڭ لىجۈندۇر. شۇڭىلاشقىمۇ، ئامېرىكىدا چىقىدىغان نوپۇزلۇق ‹‹ دەۋر ›› ژۇرنىلى 20-ئەسىردىكىئەڭ زور تەسىرگە ئىگە سەنئەتكارلارنى باھالىغاندا ، جۇڭخۇا قەۋمىدىن ئىككى ئادەم كىرگەن : بىرى، لى شياۋلۇڭ، يەنە بىرى دېڭ لىجۈن. 50 يىل مابەينىدە ھېچقانداق سەنئەتكار جۇڭخۇا قەۋمىگە بۇنچىلىك چۇڭقۇر تەسىر كۆرسىتەلىگەن ئەمەس. ئۇنىڭ ئوتلۇق ئاۋازى، لاتاپەتلىك ھۆسىن – جامالى، جەلىپكار سەھنە ئوبرازى، مول ۋە نادىر ناخشىلىرى، قىسمەتلەرگە باي تۇرمۇشى، سىرلىق ئۆلۈمى ...ئۆز دەۋرىدىكى بىر ئەۋلاد كىشىلەرنى مەست، ھەيرانۇ-ھەس قىلىپلا قالماي، زامانىمىزدىمۇ يەنىلا سالماقلىق تەسىر كۈچىنى، مەپتۇنكارلىقىنى ساقلاپ كەلمەكتە.

تۇرمۇشى ۋە سەنئەت ھاياتى رىۋايەتلەرگە ئايلىنىپ كەتكەن مەشھۇر ناخشىچى دېڭ لىجۈن 1953-يىلى 29-ماي تەيۋەننىڭ يۈنلىن ناھىيەسىدە دۇنياغا كەلگەن.دېڭ لىجۈننىڭ دادىسى خېبىيلىكبولۇپ، خۇاڭپۇ ھەربىي مەكتىپىنىڭ 14-قارار ئۇقۇغۇچىسى ، ئانىسى شەندۇڭلۇق بولۇپ، ئائىلە ئايالى ئىدى. دېڭ ئائىلىسى 1949-يىلى گومىنداڭغا ئەگىشىپ ، تەيۋەن يۈنلىندا ماكان تۇتقان. دېڭ لىجۈن ئەنە شۇ ئائىلىنىڭ تۆتىنچى بالىسى ئىدى. مەسئۇلىيەتچان ئاتا-ئانىسىنىڭ بىر قوللۇق كۆيۈنىشى بىلەن ساغلام تەربىيەلىنىۋاتقان دېڭ لىجۈن 1958-يىلى پىڭدۇڭ شەھىرى كۆھىقاپ ( شىيەن گۇڭ ) تىياتىرخانىسىدا بىرەر يىلبالېت ئۆگىنىدۇ .1959-يىلى تەيبېي ناھىيە لۈجوۋ باشلانغۇچ مەكتىپىگە كىرىدۇ. 1960-يىلى تۇنجى سەنئەت ئۇستازى لى چېڭچىڭدىن ناخشا ئۆگىنىشكە كىرىشىدۇ. 1962-يىلى ‹‹ سەنكاڭ مۇزىكا ئەترىتى ›› گە ئەگىشىپ، ھال سۇراش ئويۇنىغا قاتنىشىدۇ. 1963-يىلى جۇڭخۇا رادىيو ئىستانسىسى ئۇيۇشتۇرغانناخشا مۇسابىقىسىگە قاتنىشىپ 10 يېشىدىلا چمپىيونلۇقنىڭ تەمىنى تېتىيدۇ. ئارىلىقتا بىر قانچە قېتىملىق دېكلىماتسىيە مۇسابىقىلىرىدە ياخشى نەتىجىلەرگە ئېرىشىدۇ. 1966-يىلى ھەقىقەت ئاۋازى رادىيو ئىستانسىسى ئۇيۇشتۇرغان ناخشا كۇرسىغا قاتنىشىپ، كۆرۈنەرلىك نەتىجىلىرى بىلەن ئۇستازلارنىڭ ئالقىشىغا ئېرىشىدۇ ھەمدە ‹‹ ئالتۇن ئات ›› پىلاستىنكا شىركىتى ناخشا مۇسابىقىسىدە چىمپىيۇن بولىدۇ. بۇ جەرياندا نۇرغۇن ئورۇنلاردىن ناخشا ئېيتىش تەكلىپىنى تاپشۇرىۋالىدىغان نامدار كىچىك ناخشىچىغا ئايلىنىدۇ. 1967-يىلى ئالەم پىلاستىنكا شىركىتىگە ئەزا بولۇپ ، ناخشىلىرىنى ئۈنگە ئېلىشقا كىرىشىدۇ. شۇ يىلى 9-ئايدا تۇنجى پىلاستىنكىسى ـ ‹‹ گۈل دۇمباق›› نەشردىن چىقىدۇ.14 يېشىدا نەشردىن چىققان بۇ پىلاستىىنكا دېڭ لىجۈننىڭ كەسپىي سەنئەت ھاياتى باشلانغانلىقىنىڭ بەلگىسى بولۇپ قالىدۇ. بۇ پىلاستىنكا نەشر قىلىنغان مەزگىلدە تەيۋەن ناخشا –مۇزىكا ساھەسىدە ئاممىباب ناخشىلار دەۋر سۈرىۋاتقان بولۇپ، لېنتىدىكى ساددا، يېقىملىق، ئاددىي رېتىمدىكى خەلق ناخشىلىرى ۋە ئاممىباب ناخشىلار تېزلا ئالقىشقا سازاۋەر بولۇپ، مودا ناخشىغا ئايلىنىپ كېتىدۇ. 1968-يىلى 15 يېشىدا تەيۋەن تېلېۋىزىيە شىركىتى ئىشلىگەن ‹‹چولپانلار بىرلەشمىسى›› ناملىق پىروگراممىدا نومۇر ئورۇنلاشقا تەكلىپ قىلىنىپ، ئۆزىنىڭ ئېكران ھاياتىنىڭ تۇنجى قەدىمىنى باسىدۇ.1969-يىلى ‹‹ باش دىرېكتور، رەھمەت سىزگە ›› ناملىق فىلىمدە ئايال باش پېرسۇناژنىڭ رولىنى ئالىدۇ. شۇ يىلى ئۇ يەنە تەيۋەندىكى تۇنجى تېلېۋىزىيە تىياتىرلىرىدىنبولغان ‹‹كىرىستال›› نىڭ ئاساسلىق ناخشىسىنى ئېيتىدۇ. تەيۋەن تېلېۋىزىيە ئىستانسىسىنىڭ ‹‹كۈندە بىر چولپان›› ناملىق پىروگراممىسىغا قاتنىشىپ ‹‹ تالانتلىق ئايال ناخشىچى ›› دېگەن شەرەپكە ئېرىشىدۇ. شۇ يىلى يەنە سىنگاپور 1-خانىمىنىڭتەكلىپىگە ئاساسەن ، سىنگاپورغا چىقىپ خەير-ســـاخــاۋەت ئويۇنىغا قاتنىشىدۇ.

   1973-يىلى دېڭ لىجۈن تەيبېي شىلىن ئامېرىكا مەكتىپىگە ئوقۇشقا كىرىپ، مۇنتىزىم ئىنگلىز تىلى ئۆگىنىشكە كىرىشىدۇ.   1973-يىلى 3-ئايدا دېڭ لىجۈن ياپونىيەگە بارىدۇ. 1977-يىلىغىچە بولغان تۆت يىلدا، ئۇنىڭ ياپونىيەدە سەككىز دانە بىرلەشمە ۋە يالغۇز كىشىلىك ناخشا پىلاستىنكىسى نەشر قىلىنىدۇ، ھەربىر پىلاستىنكىسىنىڭ سېتىلىش مىقدارى ئالدىنقى 30 بازارلىق پىلاستىنكىنىڭ ئىچىگە كىرىپ، قاتتىق بازار تاپىدۇ. دېڭ لىجۈننىڭ سەنئەت ھاياتىدا شىياڭگاڭمۇ مۇھىم ئورۇن تۇتىدۇ. ئۇ شياڭگاڭغا بۇرۇنلا يۈزلەنگەن. 1970-يىلى شياڭگاڭدا ‹‹ ئاق ئاپتاپپەرەس يېغى ســـاخــاۋەت خانىشى ›› نامىغا ئېرىشىدۇ ۋە شۇ دەۋردىكىيېشى ئەڭ كىچىك ســـاخــاۋەتچى چولپانبولۇپ قالىدۇ. 10-ئايدا جاڭ جۇڭ بىلەن ھەمكارلىشىپ، ئۆزىنىڭيەنە بىر يېڭى فىلىمى - ‹‹ ناخشا مەستانىسى خېنىم ›› نى ئىشلەيدۇ.1971-يىلى دېڭ لىجۈن شياڭگاڭ، سىنگاپۇر، مالايسىيا، فىلىپپىن ، تايلاند، ۋېيتنام قاتارلىق دۆلەتلەرنى ئايلىنىپ ئويۇن قويىدۇ .بۇ قېتىملىق سەپەر دېڭ لىجۈننىڭ شەرقىي جەنۇبىي ئاسىيا ئەللىرىدىكى كېيىنكى شۆھرىتىگە ئاساس سالىدۇ. 1972-يىلى دېڭ لىجۈن ناخشا مەستانىلىرى كولۇبى _ << ‹ گۈزەل ياشلىق › دوستلۇق ئۇيۇشمىسى >> شياڭگاڭدا دەسلەپكى قەدەمدە قۇرۇلىدۇ. 1976-يىلى رەسمىي يۇسۇندا ۋۇجۇتقا كېلىدۇ.شۇ يىلى يەنە لىفېڭ پىلاستىنكا شىركىتى بىلەن توختاملىىشىدۇ . ئۆمرىدىكى تۈنجى سۆيگۈنى لىن جېنفا بىلەن تونىشىدۇ ، شۇنداقلا ‹‹ ئاق ئاپتاپپەرەس يېغى ســـاخــاۋەت خانىشى ›› نامىغا يەنە بىر قېتىم مۇيەسسەر بولىدۇ.

   تەيۋەن ، شياڭگاڭ ، شەرقىي جەنۇبىي ئاسىيا ئەللىرى ۋە ياپونىيەدە كۈنسېرى ئالقىشلىنىۋاتقان دېڭ لىجۈن شۇ يىللاردەبۇ دىيارلاردا كەينى-كەينىدىن ناخشا كېچىلىكلىرىنى ئۆتكۈزىدۇ، بۇ ئايدا سىنگاپۇردا ناخشا كېچىلىكى ئۆتكۈزسە، يەنە بىر ئايدا ياپۇنىيە سەھنىلىرىدە جەۋلان قىلىدۇ. ھەر قېتىمدا دېگۈدەك تىياتىرخانىلار ئادەم دېڭىزىغا ئايلىنىدۇ. پىلاستىنكىلىرى تەيۋەن، شياڭگاڭ، ياپۇنىيە ، شەرقىي جەنۇبىي ئاسىيا ئەللىرىدە نەشر قىلىنىپ،ئاجايىپ زور بازار تاپىدۇ. شۇ ئەللەردىكى تېلېۋىزىيە ئىستانسىسىلىرىدا داۋاملىق دېگۈدەك زىيارەت قىلىنىپ، تەسىر كۈچى شىددەت بىلەن ئاشىدۇ. پىلاستىنكىلىرى ھەر يىلى دېگۈدەك ‹‹ ئالتۇن پىلاستىنكا ›› مۇكاپاتلىرىغا، دېڭ لىجۈنمۇ ئەڭ يۇقىرى مۇزىكا مۇكاپاتلىرىغاۋە شان –شەرەپلەرگە كەينى – كەينىدىن نائىل بولۇپ، پۈتۈنشەرقىي ئاسىيا مىقياسىدا مىسلىسىز دەرىجىدىكى دېڭ لىجۈن قىزغىنلىقى كۆتۈرىلىدۇ.

ياپۇنىيەدە بىر قانچە پىلاستىنكىلىرى شىددەت بىلەن بازار تېپىۋاتقان، ئاۋمىنچە پىلاستىنكىسىمۇ زور تەسىر قوزغاۋاتقان دېڭ لىجۈن 1979-يىللىرى ئامېرىكىغا بېرىپ، ئىنگلىز تىلىنى پۇختىلايدۇ، ياپۇن ئەدەبىياتى، بىئولوگىيە ۋە ماتېماتىكىدىن بىلىم ئاشۇرىدۇ. 4-ئايدا كانادادا يالغۇز كىشىلىك ناخشا كېچىلىكى ئۆتكۈزىدۇ. 1980-يىلى ئامېرىكىنىڭ نيورك لىنكولىن مەركىزى، لوس-ئانژىلىس مۇزىكا مەركىزىدە ناخشا كېچىلىكى ئۆتكۈزۈپ، بۇ ھاكاۋۇر مەيدانلاردا تۇنجى قېتىم ناخشا كېچىلىكى ئۆتكۈزگەن جۇڭخۇالىقلارنىڭ بىرى بولۇش سۈپىتىدە ، جۇڭخۇا كۈيلىرىنى پۈتۈن دۇنياغا ياڭرىتىدۇ. 1983-يىلى 30 ياشقا تولغانلىقى، كەسپىي ھاياتىنىڭ 15 يىللىقى مۇناسىۋىتى بىلەن ‹‹ خىرە، سىرلىق تۇيغۇ ›› ناملىق كىلاسسىك پىلاستىنكىسىنى بازارغا سالىدۇ. بۇ پىلاستىنكا پۈتۈن ئاسىيانى قاپلاپ كېتىدۇ. شۇ يىلى يەنە ئويۇن قۇيۇش سەپىرىنى باشلاپ، بارلىق شەرقىي جەنۇبىي ئاسىيا ئەللىرىنى قايتىدىن زىلزىلىگە سېلىۋىتىدۇ. ئىككىنچى رەت ئامېرىكىغا چىقىپ ، قىمار شەھىرى لاسۋېگاستىكى ‹‹ قەيسەر قەسىرى ›› دە داغدۇغىلىق كېچىلىك ئۆتكۈزىدۇ. شۇ يىلىدىكى ئىستاستىكىغا ئاساسلانغاندا، شۇ ۋاقىتقا قەدەر ، دېڭ لىجۈننىڭ مەخسۇس پىلاستىنكىلىرى 109 قا يېتىدۇ ( ياپۇنچە پىلاستىنكىلىرى بۇنىڭ سىرتىدا )، پىلاستىنكىلىرىنىڭ شياڭگاڭ بازىرىدىكى سېتلىشىلا بەش مىليون پارچىدىن ئېشىپ ، خەنزۇ تىللىق پىلاستىنكىلار تارىخىدىكى ئەڭ يۇقىرى رېكوردنى يارىتىدۇ. 12-ئايدا شياڭگاڭ قىزىل قىرغاق تەنتەربىيە مەيدانىدا ئالتە كۈندە يەتتە مەيدان كېچىلىك ئۆتكۈزىدۇ، بېلەت كىرىمى تارىختىكى ئەڭ يۇقىرى چەككە چىقىدۇ، مەيداندىكى تاماشبىنلارنىڭ سانى 100 مىڭدىن ئاشىدۇ،بۇمۇ جۇڭگو تارىخىدىكى يېڭى رېكورد بوپقالىدۇ. شۇ يىلى ‹‹ شياڭگاڭدىكى 10 مۇنەۋۋەر ياش ئايال ›› نىڭ بىرىبولۇپ باھالىنىدۇ. 1984-يىلى ئەنگىلىيەگە ئوقۇشقا چىققان بولسىمۇ ، ئۆزىنىڭ كەسپىي پائالىيەتلىرىنى توختاتمايدۇ. 1-ئايدا تەيبېيدا ‹‹ بىر مىليارت چاۋاك ›› دېڭ لىجۈن ناخشا كېچىلىكىنى ئۆتكۈزىدۇ . بۇ كېچىلىك تۇنجى قېتىم چوڭ قۇرۇقلۇققا بىۋاستە ئاڭلىتىلىدۇ. ياپۇنىيەدە ‹‹ گۇناھىنى يۇيۇش ›› ناملىق پىلاستىنكىسى تارقىلىپ،زور تەسىر پەيدا قىلىدۇ. 1984-يىلى سىنگاپوردا ‹‹ بېيجىڭ ياشلار گېزىتى ›› نىڭ تېلېفون زىيارىتىنى قوبۇل قىلىدۇ. بۇ ئۇنىڭ تۈنجى قېتىم چوڭ قۇرۇقلۇق تارتقۇلىرى بىلەن ئۇچرىشىشى ئىدى. بۇ جەرياندا ، يەنە نۇرغۇن پىلاستىنكىلىرى تەيۋەن ، ئاۋمىندا نەشر قىلىنغاندىن سىرىت ، ‹‹ يارىم ›› ناملىق ياپۇنچە پىلاستىنكىسى ياپونىيەدە نەشر قىلىنىپ تېخىمۇ زور زىلزىلە قوزغايدۇ. ئۇندىن باشقاياپونىيە NHKزالىدا يالغۇز كىشىلىك ناخشا كېچىلىكى ئۆتكۈزۈپ، بۇنداق شان-شەرەپكە ئېرىشكەن تۇنجى جۇڭگولۇق بولۇپ قالىدۇ. ئامېرىكا ‹‹ دەۋر ›› ژۇرنىلى تەرىپىدىن ‹‹ دۇنيادىكى ئەڭ مەشھۇر يەتتە ئايال ناخشا چولپىنى ›› ، ‹‹ دۇنيادىكى ئەڭ ئالقىشلىق ئون ئايال ناخشا چولپىنى ›› بولۇپ باھالىنىپ، قوش شەرەپكە ئېرىشكەن تۇنجى ئاسىيالىق ناخشىچى تارىخىنى يارىتىدۇ. 1987-يىلى يەنە ئويۇن قۇيۇشنى داۋاملاشتۇرۇپ، تەيۋەن، شياڭگاڭ، ئامېرىكا ، فىرانسىيە قاتارلىق ئەللەرنى چاڭ كەلتۈرىۋېتىدۇ. ‹‹ دېڭ لىجۈن تاللانما ناخشىلىرىدىن 225 پارچە ›› تۇنجى قېتىم چوڭ قۇرۇقلۇقتا – بېيجىڭدا تارقىتىلىدۇ. ئۇ تاكى 1990-يىلىغا قەدەر يەنە دۇنيانىڭ ھەر قايسى جايلىرىدا كېچىلىك ئۆتكۈزۈش، لېنتا ئىشلەش، تۈرلۈك پائالىيەتلەرگە قاتنىشىش بىلەن ئالدىراش يۈرۈيدۇ. بۇ جەرياندا نۇرغۇن مۇكاپات ۋە شان-شەرەپلەرگە ئېرىشىدۇ. نۇرغۇن تاراتقۇلارنىڭ زىيارىتىنى قوبۇل قىلىدۇ. ئۇنىڭغا ئاتالغان مەخسۇس پىروگراممىلار تېخىمۇ كۆپ ئىشلىنىدۇ.

   1990-يىلى كەسپىي پائالىيەتلىرىنى ئازايتىپ، تۇرمۇشىنى يېڭىدىن باشلاش مەقسىتىدە ، فىرانسىيەگە بېرىپ ئولتۇراقلىشىدۇ. شۇ يىلى دادىسى ۋاپات بولىدۇ، بۇرۇندىن خەير –ســـاخــاۋەت ئىشلىرىغا كۆڭۈل بۆلۈپ كېلىۋاتقان دېڭ لىجۈن شۇندىن كېيىن ، خەير-ســـاخــاۋەتكە تېخىمۇ بېرىلىپ كېتىدۇ. 1991-يىلى تاراتقۇلارغا تۇنجى قېتىم فىرانسىيەلىك ئوغۇل دوستى پائۇلنى ئاشكارىلايدۇ.بۇ يىللاردا ئۇ خەير-ســـاخــاۋەت كېچىلىكلىرى ۋە پائالىيەتلىرىدىن باشقا سورۇنلاردا ئاساسەن دېگۈدەك كۆرۈنمەيدۇ. 1994-يىلى 6-ئايداتەيۋەندە ‹‹ مەڭگۈلۈك خۇاڭپۇ ›› ھال سۇراش كېچىلىكىگە قاتنىشىدۇ. بۇ ئۇنىڭ تەيۋەندىكى ئەڭ ئاخىرقى قېتىملىق ئاشكارا ناخشا ئېيتىشى بولۇپ قالىدۇ. 10-ئايدا ياپونىيەدىكى ئەڭ ئاخىرقى قېتىملىق ئويۇن قويىشى ھېسابلىنىدىغان ‹‹ خەلق ناخشىلىرى كېچىلىكى ›› گە قاتنىشىدۇ. 1995-يىلى 3-ئايدا ئائىلىسىدىكىلەر بىلەن جەم بولۇپ، شياڭگاڭدىكى تاراتقۇلارنىڭ زىيارىتىنى قوبۇل قىلىدۇ. 3-ئايدا شياڭگاڭ تېلېۋىزىيە ئىستانسىسىنىڭ پائالىيىتىگە قاتنىشىدۇ. بۇ ئۇنىڭ ئەڭ ئاخىرقى قېتىم ئاممىۋېي پائالىيەتلەرگە قاتنىشىشى ۋە ئەل-ئاۋامغا كۆرۈنىشى بولۇپ قالىدۇ. 1995-يىلى 8-ماي تايلاندتىكى بىر مېھمانخانىدازىققا كېسىلى ئۇشتۇمتۇت قوزغىلىپ 42 يېشىداۋاپات بولىدۇ( ئۆلۈمى سىرلىق دەپ قارىلىپ، ھازىرغىچە تالاش-تارتىش ئۈزۈلگىنى يوق ). كېيىن تەيبېيدىكى ‹‹ جۈنيۈەن ›› ( بامبۇكزار ) غا يۆتكەپ كېلىنىپ دەپن قىلىنىدۇ.

            

جۇڭخۇا ئاممىباب، مودا ناخشىچىلىقىنىڭمۇنەۋۋەر ۋەكىلى، بىر دەۋر ئاممىباب ناخشا ئېقىمىنىڭ پېشىۋاسى، جۇڭخۇا ناخشا – مۇزىكا ساھەسىدىكى مىسلىىسىز سەنئەتكار دېڭ لىجۈن ئۆز ھاياتىدا خەنزۇ ئەدەبىي تىلى ، فۇجيەن شىۋېسى، ئاۋمېن شىۋېسى، ياپۇن تىلى ۋە ئىنگلىز تىللىرىدا 3000 ئەتىراپىدا ناخشا ئېيتىقان. بۇ ناخشىلار دەۋرنىڭ يالدامىسى سۈپىتىدە جۇڭخۇا ناخشا –مۇزىكا خەزىنىسىدىكى بىباھا بايلىق ھېسابلىنىپلا قالماي، ھېلىھەممۇ تەسىر كۈچىنى يۇقاتقىنى يوق. ئۇنىڭ خەنزۇ تىلى ۋە چەت ئەل تىللىرىدا دۇنيادىكى ھەر قايسى ئەللەردە دېگۈدەك نەشر قىلىنغان، تارقىتىلغان پىلاستىنكىلىرىنىڭ ئېنىق سانىنى بېكىتىش قىيىن. ئۇنىڭ ۋەكىللىك ناخشىلىرىدىن ‹‹ تاتلىقىم ›› ، << يۈرىكىم مىسلى ئايدۇر ›› ، ‹‹ ھايات يولىدا سەيلى قىلىش ›› ، ‹‹ كۆڭلۈم پەقەت سەندىدۇر ›› ، ‹‹ سەھراغا مۇھەببەت ›› ، ‹‹شىمالىي جۇڭگو باھارى ›› ، ‹‹ سىم-سىم يامغۇر ›› ، ‹‹ كىچىك شەھەر قىسسىسى ›› ، ‹‹ ئۇنۇت ئۇنى ›› ، ‹‹ ئۆتمۈشنى ئەسلەش ›› ، ‹‹ جانانىم ›› ، ‹‹ ئۇزۇن ئۆمۈر تىلىكى ›› ، ‹‹ دېڭىزنىڭ مۇڭى ››، ‹‹ مېيخۇا گۈلى ›› ، ‹‹ كېچىدىكى خۇشپۇراق ›› ، ‹‹ خۇش جانان ›› ، ‹‹ مەڭگۈلۈك سۆيگۈ ›› ... قاتارلىقلار خەنزۇ ئاممىباب ، مودا ناخشا سەنئىتىدىكىئابىدىلەردۇر.

دېڭ لىجۈن ئاممىباب ناخشا سەنئىتىدە ئاجايىپ ئۇتۇق قازانغان سەنئەتكاردۇر. ئۇ 20 يىل ئىچىدىلا ئۆزى يالغۇز دېگۈدەك دۇنيادىكى نوپۇسى ئەڭ كۆپ قەۋمنىڭ ئاممىباب، مودا ناخشا تارىخىنىڭ ئالتۇن قۇرلىرىنى يازدى. بۇ سۈپىتىنىڭ يۇقىرىلىقى ، سانىنىڭ كۆپلىكى ، تەسىرىنىڭ ئاجايىپ كۈچلۈكلىكى بىلەن نەچچە ئەۋلاد مۇنەۋۋەر ياش سەنئەتكارلار ئاران ۋۇجۇتقا كەلتۈرەلىگىدەك دەرىجىدىكى ئۇتۇق ھېسابلىناتتى. شۇڭا، ئۇ تارىخنى قىسقارتقان گېگانت سەنئەتكار دەپ تەرىپلىنىدۇ.دېڭ لىجۈننىڭ ئاۋازىدىن بىر خىل نەپىسلىك، ئېسىلزادىلىك، ئازادىلىك چىقىپ تۇرىدۇ.ئۇنىڭ كۈيلىرى كەم-كۇتىسىز بىر خىل گۈزەللىكنى ھېس قىلدۇرىدۇ. بۇ ئۇنىڭ لاتاپەتلىك سىياقى، دولقۇنلىنىپ تۇرغان قەلبى، يېقىملىق تەبەسسۇمى، جەلىپكار سەھنە ئوبرازى بىلەن قۇشۇلۇپ، كىشىدە مەڭگۈ ئۇنتۇلماس تەسىر پەيدا قىلىدۇ.   ئۆتكەن ئەسىرنىڭ 70-، 80-، 90- يىللىرىدا ياشلىق دەۋرىنى ئۆتكۈزگەن ھەر قانداق بىر خەنزۇ پەرزەنتىنىڭ ئىچىدە ئۇنىڭ ناخشىلىرىغا مەست بولمىغانلار يوق دېيەرلىك. ھازىر ئۇلار ئوتتۇرا ياش ۋە ياشانغانلىق دەۋرىدە تۇرسىمۇ ، دېڭ لىجۈننىڭ ناخشىلىرىنى قايسىدۇر بىر كوچىدا، نەدىدۇر بىر خىلۋەت چايخانىدا ئاڭلاپ قالسىلا قەدىمىنى توختىتىپ، شۇ دەۋردىكى قايناق ھاياتى، تۇنجى مۇھەببىتىنى سۈيۈنۈش ئىچىدە ئەسكە ئېلىشىدۇ.

خەنزۇ ئاممىباب ناخشىچىلىقىنى دۇنياغا ئۇلاش ۋە تونۇتۇشتىكى ئەڭ مۇنەۋۋەر ۋە ئاۋانگارت تۆھپىكارمۇ شۆبھىسىزكى دېڭ لىجۈن ھېسابلانسا كېرەك. دۇنيا خەنزۇ ئاممىباب كۈيلىرىنى ئۇنىڭ سېھىرلىك ئاۋازىدىن بىلدى، دېيىش ئارتۇقلۇق قىلماس بەلكىم.

   دېڭ لىجۈننىڭ شۇ دەرىجىدە مۇۋەپپەقىيەت قازىنىشى ئالدى بىلەن ئۇنىڭ ناخشا ئېيتىشتىكى خاسلىقى بىلەن مۇناسىۋەتلىك .ئاۋاز تۈسى جەھەتتىن، دېڭ لىجۈننىڭ تەبىئىي ئاۋازى تاتلىق، سىلىق، نەپس بولۇپ، 70-80-يىللاردىكى ئاياللارنەپىس ئاۋاز تۈسىنىڭ ئالاھىدىلىكىنى تولۇق مۇجەسسەملىگەن . تىل ( تەلەپپۇز ) جەھەتتىن قارىغاندا ، دېڭ لىجۈن شىماللىق جۇڭگولۇقنىڭ پەرزەنتى بولۇش سۈپىتىدە، ئەدەبىي تىلنى ئىنتايىن ئۆلچەملىك تەلەپپۇز قىلالايدۇ. ئۆلچەملىك تەلەپپۇز ئۇنى خەلق ناخشىلىرى ۋە جۇڭگوچە پۇراقتىكى ئاممىباب ناخشىلارنى ئۆلچەملىك ئېيتىش ئەۋزەللىكىگە ئىگە قىلغان. ئۇنىڭ ئۈستىگە بىر قانچە تىپىك خەنزۇ شېۋىسىگە پىششىقلىقىۋە تۆت-بەش خىل چەت ئەل تىلىنى بىلىشىئۇنىڭ ئوخشىمىغان شېۋە ، تىل ، قاتلام رايونلىرىدا يىلتىز تارتىشىغا ئاساس ھازىرلىغان. ناخشا ئېيتىش ماھارىتى جەھەتتە، ئۇنىڭ ئاۋاز دائىرىسى ئادەتتىكىچە بولۇپ، كەڭ ھېسابلانمايدۇ. لېكىن ، ئۇ راست-يالغان ئاۋازلارنى ماسلاشتۇرۇپ ناخشا ئېيتىشقا ئىنتايىن ماھىر. ئۇنىڭ ھەقىقي ئاۋازى يېقىملىق، تەبىئىي بولۇپ، ساختا ئاۋاز بىلەن پەردازلاپ، شۇنچىلىك گۈزەل بىر ئارىلاشما ئاۋازغا ئايلاندۇرىۋېتىدۇ. بۇ ئارىلاشما ئاۋاز كىشىگە نەپس، سىلىق، سەنئەتكە تۇيۇنغاندەك تۇيغۇ بېرىدۇ. ناخشا ئېيتقاندا ئاۋازى يەڭگىل، نەپىسى سىلىق مېڭىپ، كىشىگە تولىمۇ ئەركىن-ئازادە كەيپىيات بەخش ئېتىدۇ. ئۇنىڭ دەرتلىك ئاۋاز چىقىرىش ئۇسسۇلى دولقۇنلۇق ئاۋازنى ئەپلىك بىر تەرەپ قىلىش ماھارىتى بىلەن بىرلىشىپ، دېڭ لىجۈننىڭ ئۆزىگە خاس ئاۋاز ماھارىتىنى تېخىمۇ ۋايىغا يەتكۈزگەن . ئۇندىن باشقا ، دېڭ لىجۈننىڭ مۇزىكا سەۋىيە ئۆلچىمى ۋە ئۇدار سېزىمى ئىنتايىن يۇقىرى. ئۇ ساز بولمىغان شارائىتتىمۇ ناخشىنىڭ ھۆددىسىدىن تامامەن چىقالايدۇ. ئۇ ئىنچىكە مۇزىكا پەدىلىرىنى جايىدا بىر تەرەپ قىلغاندىن باشقا يەنە ئىشلەتكەن تېكىستلەرنىمۇ شۇنچىلىك راۋان چۈشەندۈرۈپ بېرەلەيدۇ. ئۇندىن باشقا ، ئۇ ئۆز ناخشىلىرىدا جۇڭگو ئەلنەغمىلىرىدىكى سەكرەتمە ۋە سېيرىلما ئاۋازلاردىن ئۇستىلىق بىلەن پايدىلانغاچقا، كۈي گىرۋەكلىرى رۇشەن، جانلىق ۋە ياڭراق بولۇپ، كىشىلەرنىڭ قەلبىگە تەبىئىي سىڭىپ كېتىدۇ.

   ئۇنىڭ ناخشىلىرى ئەنە شۇنداق تىپىك ئالاھىدىلىككە تۇيۇنغاچقا ، ھۇزۇلانغۇچىلار قاتلىمىدىكى ياش –قورام پەرقىنى بۇزۇپ تاشلاپ، تېزلا تارقىلىپ خەلق ئىچىگە چۇڭقۇر سىڭىپ كېتەلىگەن. خۇددى تەيۋەن تېلېۋىزىيە شىركىتىنىڭ ئاتاقلىق رىياسەتچىسى تيەن ۋېنجۇڭ ئەپەندىم ئېيتقاندەك: ئۇنىڭ كۈيلىرى ھەممە ئادەمگە باب كېلىدۇ، ئىككى ياشلىق بۇۋاقتىن تارتىپ، 200 ياشلىق بۇۋايلارغىچە ھەممىسى ياقتۇرۇپ ئاڭلايدۇ...

بۈگۈنكى جۇڭگو ناخشا ساھەسىدىكى نۇرغۇن چولپانلار دېڭ لىجۈننىڭ تەسىرىدە يېتىلگەن بولۇپ، داڭلىق ئايال ناخشا چولپىنى ۋاڭ فېيمۇ ئۇنى مەبۇد ۋە ئۇستاز سۈپىتىدە كۆرىدىغانلىقىنى ئىپتىىار بىلەن تىلغا ئالىدۇ . 2008-يىلى جۇڭگونىڭ‹‹ خاسىيەتلىك كىمە 7-نومۇر ››ى ئالەمگە چىققاندا، دېڭ لىجۈننىڭ ‹‹ ئۇزۇن ئۆمۈر تىلىكى ›› ناملىق ناخشىسى ئالەم بوشلۇقىدا ياڭرىتىلىدۇ. ئۇ ۋاپات بولۇپ 14 يىل ئۆتكەندىن كېيىنكى- 2009-يىلىدىكى توردا بېلەت تاشلاش ئارقىلىق باھالىنىدىغان ‹‹ جۇمھۇرىيەت قۇرۇلغان 60 يىلدىن بۇيانقى ئەڭ تەسىركار مەدەنىيەت ئەربابلىرىنى تاللاش ›› پائالىيىتىدە ، سەككىز مىليون 500 مىڭدىن كۆپرەك بېلەتكە ئېرىشىپ، لاۋشې، جىن يۇڭ، ۋاڭ فېي، جۇۋ جېلۈن قاتارلىقلارنى مۇتلەق ئۈستۈنلۈك بىلەن يېڭىپ، ئەڭ ئالدىنقى قاتارغا تىزىلىدۇ. بۇ دېڭ لىجۈننىڭ بۈگۈنكى ئەۋلادلار ئارىسىدىمۇ زور تەسىرگە ئىگە ئىكەنلىكىنىڭ ئاددىي بىر دەلىلى.

شۇنداق، يەن-خۇاڭ ئەۋلادلىرىنىڭ قەلبىدە ‹‹ مەڭگۈ يىقىلماس ئابىدە ›› تىكلىگەن بۇسەنئەتكار بۇ دۇنيادىنۋاقىتسىز خوشلاشقان بولسىمۇ ، لېكىن، ئۇنىڭ ئەسەرلىرى يەنىلا مىليارد چاۋاكلارنىڭ ئارىسىدا.
