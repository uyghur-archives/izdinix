ئەسسالامۇئەلەيكۇم دوستلار ،مېنىڭ قىزىم بۇ يىل بەش ياشقا كىردى ،ئۆتكەن يىل يىل  ئاخىردىن باشلاپ ئۇخلىغاندا ئاغىزىنى ئېچىپ خورەكتەك ئاۋاز چىقىرىپ ئۇخلايدىغان بۇلۇپ قالدى، بۇرنىدا بىر نەرسە تۇسۇلۇپ قالغاندەك خوقىراپ ئۇخلايدۇ .شۇنىڭ بىلەن يېڭى يىلدا ئۇرۇمچىگە ئاپىرىپ 2-دوختۇرخانىنڭ ‹ئېغىز-بۇرۇن -كاناي بۇلىمى›نىڭ مۇتەخسسس نۇمۇرىنى ئېلىپ كۆرسەتتۇق،تەكشۇرۇش نەتىجىسىدە  بادامسىمان بەز يوغىناش(腺样体肥大)،ئانگىناسى يوغىناش (扁桃体肥大)دەپ دىئاگىنوز قويۇپتۇ ،دوختۇر چوقۇم ئوپراتسىيە قىلىسىلەر دەيدۇ. بىز دورا –ئۇكۇل بىلەن داۋالىغى بولمامدۇ دەپ سورىساق ،‹قەتئىي  بولمايدۇ ،چوقۇم  ئوپراتسسيە قىلىمىز ›دەيدۇ.بالا تېخى ئەمدى بەش ياشقا كىردى ،بىزنىڭ بۇ كىچىك بالىمىزنى ئوپراتسسيە قىلدۇرغىمىز يوق .
تور بىلەن ئىزدىنىپ كۆرسەم ،بۇ كىسەل ئاساسلىقى 3ياشتىن 10ياشقىچە بالىلار ئىچىدە كۆپ كۆرۇلدىكەن ،12ياشتىن كېيىن بۇرنىدىكى  بۇ بەز ئۆزلىكىدىن كىچىكلەپ ،يىگىلەپ يوقاپ كىتىدىكەن .بۇ بەز يوغىناپ كەتكەن بالىلارنىڭ  دەل ۋاقتىدا داۋاسىنى قىلمىغاندا بىر قاتار ئالامەتلار يۇز بېرىدىكەن ،مەسلەن مېڭىسىگە ئوكسىگىن يىتشمەسلىك سەۋەپلىك ھۇشىدىن كېتىش ،مېڭىسىنىڭ تەرەقىياتى توسقۇنلۇققا ئۇچىراش ،ئاغىزىنى ئېچىپ ئۇخلاۋەرگەنلىك بىلەن چىشى ۋە چىرايى ئۆزگىرىپ كىتىش ،،،،دىگەندەك .

مېنىڭ مەسلىھەتىم، ئوپراتسىيە قىلدۇرماڭ. مېنىڭ قىزىمنىڭمۇ بادامسىمان بەز يوغان ئىدى، ئىككى-ئۈچ ياش ۋاقىتتىن باشلاپ خورەك تارتىپ ئۇخلايتتى. نۇرغۇن دوختۇرغا كۆرسىتىپ كەتتۇق، كۆپىنچىسى ئوپراتسىيە قىلدۇرۇش مەسىلىھەتىنى بەردى. ئەمما بىز قىلدۇرمىدۇق. يەتتە ياشقا كىرگەن يىلى ياۋروپاغا ساياھەتكە چىقىپ قالدۇق، ياۋروپادا دوختۇرغا كۆرسەتسەم، ئۇلار «ھېچقىسى يوق، كىچىك ئىشكەن، بالا چوڭ بولغانسىرى يوقۇلۇپ كېتىدۇ» دېدى. قانداق ئىشلاردا دىققەت قىلىدىغانلىقىمىزنى سورىساق، ئامال بار زۇكام دېگەندەك ئىممۇنت كۈچىنى تۆۋەنلىتىۋىتىدىغان كېسەللەردىن ساقلانسۇن، كۆپرەك دىققەت قىلىڭلار. ئۆينىڭ نەملىكىنى ساقلاڭلار دېدى. شۇنىڭ بىلەن خاتىرجەم قايتىپ كېلىپ، ئويدە نەملىك پەيدا قىلغۇچنى كۆپ ئىشلەتتۇق، زۇكام تېگىشتىن ساقلىنىشقا تىرىشتۇق، بۇرۇننى نەملىتىدىغان بىر پۈركۈيدىغان دورا باركەن، ئانچە-مۇنچە ئۇنى پۈركۈدۇق. شۇنداق قىلىپ، بۇ يىل ئون ياشقا كىردى. بادامسىمان بەز كۆرۈنەرلىك كىچىكلىدى. ئاغرىپ قالغان ۋاقىتلار بولمىسا ئانچە خورەك تارتىپمۇ كەتمەيدۇ. شۇڭا دەيمەن، بۇ كېسەل ئوپرااتسىيە قىلدۇرىدىغان كېسەل ئەمەس، ئوپراتسىيە قىلدۇرماڭ، ئەمما بالىنىڭ ئىممۇنېت كۈچىنى ئاشۇرۇشقا دىققەت قىلىڭ، ئۆينىڭ نەملىكىنى ساقلاڭ.

مېنىڭ ئوغلۇمنىڭمۇ بادامسىمان بېزى ئىششىپ كېتىپ ئىككى قېتىم دوختۇرخانىغا ئاپارساق ئوخشاشلا ئوپېراتسىيە قىلىشنى تەۋسىيە قىلغان . ئاخىرى ئۇيغۇر تېبابىتىدىن بىر داۋاسى چىقىپ قالارمۇ دېگەن ئۈمىد بىلەن ئاپتۇنۇم رايونلۇق ئۇيغۇر تېبابەت دوختۇرخانىسىغا ئاپىرىپ كۆرسەتتۇق . ئۇ دوختۇرخانىدىكى دوختۇر ھازىر بالىلاردا بۇ كېسەلنىڭ بەك كۆپ ئىكەنلىگىنى ئېيتتى ھەم تەكشۈرۈپ كۆرۈپ بىر تەرىپىنىڭ ئىككىنچى ، يەنى بىر تەرىپىنىڭ ئۈچىنجى دەرىجىلىك ئىكەنلىگىنى ئېيتتى ھەم ھازىرچە دورا بىلەن داۋالىنىپ بېقىشنى  تەۋسىيە قىلىپ دورا يېزىپ بەردى ،ھەم خېلى ئۇزۇندا ئاندىن ساقىيىدىغانلىقىنى ئېيتتى. ئوغلۇم ھازىر شۇ دورىلارنى(ئەفىفىي چوڭ ، مەجۈنى چۆپچىن ئەلا) ئىچىپ تۇرىۋاتىدۇ .كېسەللىك ئەھۋالى كۆرۈنەرلىك ياخشىلىنىپ كەتمىگەن بولسىمۇ لېكىن ئېغىرلىشىپ كەتمەي بىر خىل ھالەتتە تۇرىۋاتىدۇ . لېكىن يېمەك-ئىچمەك جەھەتتە قاتتىق پەرھىز تۇتۇشى كېرەككەن . مەسىلەن قوي گۆشىدىن باشقا گۆش يېمەسلىكى ، بەرەڭگە ، مۇچ ، كاۋا ، پىياز قاتارلىقلارنى ھەم بانان ۋە نەشپۈتتىن باشقا مېۋىلەرنى يېمەسلىكى،سوغۇق ئىچىملىككە يېقىن كەلمەسلىگى  كېرەككەن .  ئامال بار قورۇمىغان تاماقلارنى يېيىشى ھەم تاماققا تەم خۇرۇچلارنى سالماسلىقى كېرەككەن . شۇندقلار كۆپرەك ھەركەت قىلىشى كېرەككەن .
ئەلبەتتە كېسەللىك ئەھۋالى بەك ئېغىر بولسا ئوپېراتسىيە قىلىشى لازىمكەن ، ھەم دوستلىرىمىنىڭ ئىچىدىمۇ بالىلىرى مۇشۇ كېسەل بىلەن ئاغرىپ ئوپېراتسىيە قىلىپ ساقايتقانلار بار.
