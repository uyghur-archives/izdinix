مەن ۋە دەرەخ


                                                           نويابىر

                                                           يارغول

                                          توساتتىن كىمنىدۇر تىللىغۇم كېلەر،
                                          ئويلايمەن كىمنىدۇر مىجىۋېتىشنى.
                                          قەبىھ بىر تۇيغۇدا تۇرسام تەپچىرەپ،
                                          ئىسسىق چاي سۇنىدۇ كىمدۇر بىركىشى.

                                          ئەڭ يارقىن ئازابتۇر چىرايلىق قىزلار،
                                          ئازابتۇر ئۆزگىگە تەۋەلىكى ھەم.
                                          ئىسىتقان تاماقتەك قايتا ۋە قايتا،
                                          بۇ سۇلغۇن كۈنلەردە قالمىدى بىر تەم.

                                          ئاسماننىڭ كەينىنى قىلدىم تەسەۋۋۇر،
                                          يامغۇردەك تۆكۈلگۈم كەلدى شۇ دەمدە.
                                          كەڭ كەتكەن ئادەملەر ئورمانلىقىدا،
                                          بىر مەيۈس ئادەممەن چەكسىز يېگانە.

                                          بۇ قارسىز پەسىل ھەم سارغۇچ رېئاللىق،
                                          غازاندەك روھىمنى يەلپۈتەر رەت-رەت.
                                          كىمنىدۇر تىللاشنى ئويلاپ ئولتۇرسام،
                                          نويابىر كېلىپتۇ مۇھەببەت ئىزلەپ.

                                                                              

                                                                                مەن ۋە دەرەخ

                                                                                    يارغول

                                                               ئەڭ مەيۈس كۈنلەرنى ياشاۋاتىمەن،
                                                               ھىجراننىڭ رەڭگىدە ئېچىلدى دۇنيا.
                                                               نەدىندۇر كەلمەكتە شۇملۇقنىڭ ھىدى،
                                                               ئىچىمگە مۆكۈندۈم چەكسىز ھەم تەنھا.
   
                                                               باغرىمدا يەنىلا شۇ كونا ئاغرىق،
                                                               مەي باغلاپ يېتىلگەن بىر ساپاق يارا.
                                                               مەن ئۇنى بىر تالدىن تېتىيمەن ئاخىر،
                                                               چىلىشىپ سارغايغان رېئاللىقىمغا.

                                                               ئۆلگۈدەك قورقىمەن يالغۇز قېلىشتىن،
                                                               قورقىمەن بىراۋنى ئىزدەشتىن يەنە.
                                                               شاھ مەشرەپ ياشىغان بولغۇيدى قانداق،
                                                               شۇنچىلىك مەردانە،شۇنچە يېگانە.

                                                               نالىلىك ياپراقلار ئاياقلىرىمغا،
                                                               تۆكۈلۈپ بارماقتا بۇ كەچكۈز ئارا.
                                                               رەڭگىنى تۈگەتتى ئۆكسۈپ بىر دەرەخ،
                                                               شۇ قەدەر ئوخشايىتتىم مەن ھەم ئۇنىڭغا.

ئون كۈن بولدى

يارغول


شېئىرسىز يۈرگىلى ئون كۈنچە بوپتۇ،
تاشقىنلاپ توختىغان دەريادەك گويا.
چاك بولغان پايانسىز قىرغاقلىرىمدا،
گۈل بولۇپ ئېچىلدى گۈزەل بىر يارا.

بىر ناخشا بار ئىدى ئۇنتۇيالمىدىم،
ھەم قانداق توۋلاشنى بىلمىدىم ئەسلا.
دۇنيادا ھىجراندىن باشقىسى ئاخىر،
بېرەلمەس ھاياتقا ھەقىقىي مەنا.

تاغلارنىڭ سۈكۈتى ئەڭ چىن رىۋايەت،
تىمتاسلىق ئاداققى ئەڭ تاتلىق كۈيدۇر.
ھېچيەردە ئۆي يوقتۇر سۆيگۈدىن باشقا،
مۇھەببەت لاۋۇلداپ ئوت كەتكەن ئۆيدۇر.

يۈرگىنىم يەنىلا شۇ توپىلىق يول،
ئۈستۈمدە قۇياش ھەم سۈكۈترەڭ ئاسمان.
بىر مىسرا يازمىدىم ئون كۈنچە بولدى،
ئەمما ھەر تىنىقىم ئۆزۈڭگە ئايان.



ھېكايىمىز


ئاداققى تۆكۈلۈش پەسلىدە تۇردۇم،
قەلبىمدە يەنە شۇ ئاغرىق، ھاياجان.
كەل سېنى باغرىمغا چىڭ بېسىۋالاي،
چۈنكى بىز ئايرىلىپ كېتىمىز ھامان.

غازانلار سۆزلىگەي ھېكايىمىزنى،
كەچكۈزگە، ئېرىققا، سەرسان قۇشلارغا،
مۇزسىمان ئادەملەر شۇنچە ئالدىراش،
قىززىقماس مۇھەببەت دېگەن ئىشلارغا.

ھېلىقى ناخشىنى ئۇنتۇيالمىدىم،
ئۇنتۇشلا بەخىتتىن بېرىدۇ دېرەك.
سەن مېنى ھېچقاچان ئۇنتىيالمايسەن،
مەن مەڭگۈ بەخىتلىك بولمىسام كېرەك.

زىمىستان ھەققىدە سۆزلەپ دىلىمغا،
ئاستا ۋە ئاستا سىڭمەكتە قۇياش.
بەك ئۇزاق ياشىدۇق بىزلەر ئۆز ئارا،
بولۇشۇپ سۆيۈملۈك يالغان باغرى تاش.

نېمىنى ئويلايسەن گۈلۈم شۇ تاپتا،
كۆكسۈڭدە قىزىرىپ يانامدۇ گۈلخان.
بۇ ھىجران چۆلىنىڭ چەكسىزلىكىدە،
يۈرمەكتە تەمتىرەپ بىز سالغان كارۋان.

كىمنىڭ لىق بىر دۇنيا يېشى يوق دەيسەن،
مەن ئەمما يوللارغا چېچىپ باقمىدىم.
كەل سېنى جىمجىت بىر باغاشلىۋالاي،
بىز ھېلى ئايرىلىپ كېتىمىز جېنىم.

   زىددىيەت

يارغول


                                                  ياشايمەن دېسەملا ھايات قالمايمەن،
                                                  دائىم ساق قالىمەن ئۆلمەكچى بولۇپ.
                                                  ئادەملەر ئىچىدە ئۆزۈم سىرتىدا،
                                                  ھۆركىرەپ يىغلايمەن ھىجىيىپ تۇرۇپ.

                                                  سۈكۈتكە ئايلاندىم سۆزۈم جىق بولغاچ،
                                                  كۆزۈمنى يۇمدۇم ھەم ئېنىق كۆرەي دەپ.
                                                  بۇ قەلبىم بەخىتكە مۇھتاج بولغاچقا،
                                                  يېڭى بىر ئازابنى يۈرۈيمەن ئىزدەپ.

                                                  مۇھەببەت ئەشەددىي زوراۋانلىقتۇر،
                                                  مېھىر بار قان تامغان قېلىچ بىسىدا.
                                                  ئۇچۇشنى ئۆگەندىم چۈشۈشۈم ئۈچۈن،
                                                  يوقۇلۇش ئۈچۈنلا مەۋجۇتتۇر دۇنيا.
                                                  
                                                 سۆيۈلگەن بولغاچقا نەپرەتكە قالدىم،
                                                 ساق ياشاش ئۈچۈن مەن قالدىم ئېلىشىپ.
                                                 بۇ دۇنيا لىق تولغان مۇھەببەت بىلەن،
                                                 شۇڭا مەن ھىجرانغا قالدىم ئېرىشىپ.
      

129 يوللانغان ۋاقتى 4 كۈن ئالدىدا

                                                          يارغول
           
                                                      مەندىكى يارا

                                          ئېغىرلاپ بارماقتا ئىچىمدىكى غەم،
                                          تېشىمدا بىر دۇنيا ئازابنىڭ رەڭگى.
                                          بىر گۈزەل ھەسرەتكە چىلىنىپ تۇرۇپ،
                                          مۇھەببەت ھەققىدە ئېيتىمەن ناخشى.

                                          چوڭقۇرلاپ بارماقتا مەندىكى يارا،
                                          ساقايماس لوقماننى چىللىساممۇ ھەم.
                                          ئوتلۇق بىر ناز بىلەن تۇتىسەن زەھەر،
                                          يارامنى داۋالاش ئۈچۈن ئەي سەنەم.

                                          ئېگىزلەپ بارماقتا دىلدىكى ھەسرەت،
                                          ئېگىلىپ بارماقتا قەددىم شۇ قەدەر.
                                          شۇ ئېگىز چوققىغا چىقىپ قارىسام،
                                          بۇ ھايات يەنىلا گۈزەل بىلىنەر.

                                           ئۇلغىيىپ بارماقتا ئىسسق ياشلىرىم،
                                          بولسىمۇ بۇ دۇنيا مۇدھىش ھەم سوغۇق.
                                          ئەگەر كىم يىغلاشنى ئۆگۈنەلىسە،
                                          بىر پۈتۈن بەخىتكە ئېرىشەر تولۇق.
