فۇتۇ باتارىيە كونتۇرول توك يوللېرى

  ئادەتتە فۇتۇ باتارىيە زەرەتلەش تەمىنلەش توك يوللېرى تۇغرىلىق ماتىرياللار لازىم بولسا بەيدۇدىن ئىزدىسە بەك كونا تىپتىكىلېرىنىلا تاپقىلى بۇلىدىكەن، شۇڭا مۇشۇ ئىشنى قىلىۋاتقانلارنىڭ 2000-يىللاردىن ئىشلىتىلىۋاتقان توك يوللېرىدىن ئازراق چۈشەنچە ھاسىل قىلىشى ئۈچۈن بۇ تىمىنى يازدىم.
ئەلۋەتتە فۇتۇ باتارىيەدىن چىققان توكنى ئادەتتىكى بىسىم مۇقىملاشتۇرۇش توك يوللېرى ئارقىلىق مۇقىملاشتۇرۇپلا ئاكۇمۇلياتورغا زەرەتلىسىمۇ بۇلىدۇ، ئەمما فۇتۇ ئىلىكتىر مەنبەسى قىممەترەك بولغاچقا ئادەتتىكى كىرىستال ئۈچ قۇتۇپلۇق لامپىلار زەرەتلەۋاتقاندا توك ئىقىمى چوڭ بولغانچە بىر قىسىم ئىنىرگىيە ئىسراپچىلىقى بۇلىدىغانلىقتىن(ئادەتتىكى لامپىدا 5 ئامپىر توك ئۈتكەندە كۈپىنچە يىغقۇچى قۇتۇپ بىلەن تارقاتقۇچى قۇتۇپ ئارسىدىكى بىسىم 0.6-1 ۋولىتقىچە بۇلىدۇ، بۇ 3-5ۋاتقىچە توكنىڭ ئورۇنسىز يۇقاپ كەتكىنىگە ھىساپ، يەنە يۈككىمۇ 0.4-1 ۋولىتقىچە توك بىسىمى كەم بارسا نورمال خىزمەت قىلمايدۇ)كۈپىنچە موس لامپا ئارقىلىق زەرەتلەش ۋە يۈككە توك تەمىنلەش ئەمەلگە ئاشۇرۇلىدۇ.

  بۇ خىل توك يۇلىدا ئاساسەن مۇسبەت قۇتۇپ توك سىملېرى بىر تۇتاش ئۇلىنىپ مەنفى بىسىمدىن زەرەتلەش ۋە يۈك تەمىنلەش ئىشقا ئاشۇرۇلىدۇ.تېخنىكىسى يۇقىرىلېرىدا ئاكۇمۇلياتور بىلەن فۇتۇ باتارىيەنىڭ توك بىسىمىنى سىلىشتۇرۇپ فۇتۇ باتارىيە توك بىسىمى ئاكۇمۇلياتور توك بىسىمىدىن 0.01 ۋولىت ئاشقان ھامان Q6Q7لەر ئۈتكۈزۈش ھالىتىگە كىرىپ  زەرەتلەش ھالىتىگە كىرىدۇ، نورمال زەرەتلەش توك بىسىمى 14.3V بۇلۇپ، شۇ بىسىمغا يەتكەندە زەرەتلەش كونتۇرۇوللىغۇچى Q6Q7 لەر ئۈزۈلىدۇ، ئاكۇمۇلياتور زەرەتلىنىپ بولمىغان ۋاقىتتا زەرەتلەش تۇختىسا سىكونىتقا بارمايلا توك بىسىمى تۈۋەنلەيدۇ، تىخى 14.2V قا چۈشۈپ بولماي تۇرۇپلا يەنە قايتا زەرەتلەش ھالىتىگە كىرىدۇ. بۇ جەريان شۇنداق قايتىلىنىپ باشقا يۈك ئۇلانمىغان ھالەتتە ئاكۇمۇلياتور زەرەتلىنىپ بولغاندا نەچچە سىكونىتتا بىر سىكونىتقا يەتمىگەن ۋاقىت زەرەتلەپ تۇرىدۇ. ناۋادا زەرەتلەش كونتۇرولى مەلوم سەۋەپ بىلەن تۇتاش ھالەتتە بۇلۇپ قالسا توك بىسىمى V15-14.3V قىچە يۇقىرلاپ قالسا Q8-Q7 موس لامپىلار ئاپتۇماتىك ئۈزۈلۈپ، زەرەتلەشنى ۋە يۈككە توك تەمىنلەشنى تۇختىتىدۇ.
  بەزى كىچىك شىركەتلەر بۇلارنى كونتۇرول قىلىشتا كۈپىنچە  LM324，LM393 قاتارلىق سىلىشتۇرغۇچلاردىن پايدىلىنىپلا ئەمەلگە ئاشۇرغان بۇلۇپ، قۇغدىنىش ئىقتىدارلېرى ياخشى ئەمەس.
  باشلامچى شىركەتلەر بولسا يۇقارقىدەك كونتۇرول ئىقتىدارلېرىنى ئاردىنو ۋە C تىلى ئارقىلىق كود يىزىپ باشقۇرىدىغان يەككە ئۈزەكلەردە ئەمەلگە ئاشۇرغان ھەم تېنىچ ھالەتتىكى ئۆزىگە سەرىپ قىلىدىغان توك مىقتارىنى زور دەرىجىدە كىچىكلىتىش ئۈچۈن يەككە ئۈزەك ۋە LED ھالەت ۋە توك مىقتار كۆرسەتكۈچ چىراقلېرىغا تەمىنلەيدىغان 5ۋولىت توكنىمۇ ياخشى تەڭشەلگەن DC_DC توك بىسىم تۈۋەنلەتكۈچ توك يوللېرى ئارقىلىق تەمىنلەيدۇ.
  ئۆلچەملىك فۇتۇ توك مەنبەسى
  يۇقىرى بىسىمدىن قۇغدىنىش ۋە ئەسلىگە كىلىش ، ئاكۇمۇلياتور تۈۋەن بىسىمدىن قۇغدىنىش ۋە ئەسلىگە كىلىش ، يۈككە تەمىنلىنىدىغان توكنىڭ يۇقىرى ئىقىملىق ياكى قىسقا يول ھالىتىدىن قۇغدىنىش ۋە ئەسلىگە كىلىش ئىقتىدارلېرى يۇقىرى بۇلۇپ، قايتا-قايتا قىسقا يول بولسىمۇ ئۆزلىكىدىن مەلوم ۋاقىتتىن كىيىنلا ئەسلىگە كەلتۈرىدۇ.
  ئاردىنو ۋە C تىلى تۇغرىلىق ئىزدىنىپ باقمىدىم، ھەم ئۇ تۇغرىلىق كۆپ بىلمەيمەن. 太阳能用单片机 دەپ يىزىپ ئىزدىسە خېلى ماتىرياللار بار ئىكەن.
ardino كودىمۇ باركەن.
   بۇرۇن تاغلىق بەزى كىشىلەرنىڭ ماشنىنىڭ رىگۇلياتورىنى زەرەتلەش كونتۇروللىغۇچى قىلىپ ئىشلەتكىنىنى كۆرگەن ئىدىم. ھىچقانداق كونتۇروللىغۇچ بولمىغان ئەھۋالدا ئادەم قاراپ تۇرۇپ زەرەتلىگىنى ياخشى، توك يۇلىنى پۇختا بىلىدىغانلار بولسا شۇنى چۈشەنگىلى بۇلىدۇكى ماشنا رىگۇلياتورى ئارقىلىق زەرەتلىگەندە ئاكۇمۇلياتور بىسىمى بىر قىتىم 14 ۋولىتقا چىققاندىن كىيىنلا ئۈزۈلگەندىن كىيىن قايتا زەرەتلەش ھالىتىگە كىرەلمەيدۇ. بۇنىڭ بىلەن ئاكۇمۇلياتور كۆپ بولسا %50 زەرەتلەنگەن بۇلىدۇ. ناۋادا موس لامپىلىق رىگولياتور بولسا %70كە چىقىشىمۇ مومكىن. شۇڭا يەنىلا ئۆزىنىڭ زەرەتلەش كونتۇروللىغۇچىسىنى ئىشلەتكەن ياخشى.

ئىسل تېمىلار جىقكەن سىلىدە، داۋاملاشقاي

ArislanMemet يوللانغان ۋاقتى 2015-3-3 22:41:50

ئالدىنقى قېتىم ئېلان قىلغان تېمىدا، بۇرۇن تىلۋىزۇردىن كۆرگەن كۈن تۆك تاختىسى توغرۇلۇق ئازراق ئىنكاس يازغان ئىدىم.
توردىن ئىزلەپ يۈرۈپ، تۆۋەندىكى مەزمۇننى تاپتىم. لىكىن بۇ خەۋەر بەك كونا ئىكەن، 2009-يىلنىڭ ئىكەن. قارىغاندا ئالدىنقى قېىتم ئىنكاس يازغاندا، ۋاقىتنى خاتا ئەستە تۇتۇۋالغان بولۇشۇم كېرەك. بۇ ماقالىنىڭ مەزمۇنى تۆۋەندىكىچە:
قوياش ئىنىرگىيەسىدىن ئۈنۈملۈك پايدىلىنىش، ھازىرغىچە ھەر قايسى دۆلەت پەن-تېخنىكا خادىملىرىنىڭ ئالدىغا قويۇلغان، زۆرۈر ھەل قىلىشقا تېگىشلىك مەسىلە.
ئاۋىستىرالىيە  University of New South Wales يېڭى جەنۇبى ۋېلىس ئۇنۋېرسىتى يۇقۇرى ئۈنۈملۈك قوياش ئېنىرگىيە تاختىسىنى ياساپ چىقتى. بۇ كۈن توك تاختىسىنىڭ قۇياش ئىنىرگىيەسىنى ئايلاندۇرۇش نىسپىتى 43%، بولۇپ بۇرۇنقى 42.7% لىك رىكوردنى بۇزدى. تەتقىقاتچىلار بىرىكمە ماتىرياللىق ئېنىرگىيە يىققۇچنىڭ ئايلاندۇرۇش نىسپىتى تاق ماتىرياللىق ئىنىرگىيە يىققۇچتىن كۆپ يۇقۇرى بولىدىغانلىقىنى بايقىغان.
ئەنئەنىۋى سېلتىسىلىق(خەنچە گۈي دىگەننى سىلتسىي دەمدۇق يا؟) تاختاينىڭ ئىنىرگىيە يىغىش نىسپىتى 7%~ 13% گىچە بولۇپ، بۇ بىرىكمە ماتىرياللىق تاختايدىن كۆپ تۆۋەن.
بۇ تاختىغا ئىشلىتىلگەن بىرىكمە ماتىرياللار بىر قەدەر قىممەت بولغاچقا، تاۋار شەكلىدە ئىشلەپچىقىرش ھازىرچە مۇمكىن ئەمەس، لىكىن بۇ تېخنىكا كەلگۈسىدىكى ئىنسانلارغا نىسپەتەن پايدىلىق ئۇچۇر بىلەن تەمىنلىشى مۇمكىن.

يۇقۇردىكىسى ماقالىنىڭ ئاساسلىق مەزمۇنى، ئىچىدىكى ئىلمىنتلارنىڭ خەنچە ئۇيغۇرچە ناملىرىنى تازا بىلىپ كەتمىگەچ يازمىدىم. بۇندىن باشقا يەنە قوياش ئىنىرگىيەسى توغرۇلۇق ماتىريال ئىزلىسەم، تۆۋەندىكى ئۇچۇرغا ئىرىشتىم. قوياشنىڭ ھەر سىكونتا يەر شارىنى تەمىنلىگەن ئىنىرگىيەسى 5 مىليون توننا كۆمۈرنى كۆيدۈرگەندىكى ئىنىرگىيەگە تەڭ.
