مۇھەببەتلىشىۋاتقان مەزگىللەردە قىزلار ھەمىشە :_ بىز ئايرىلىپ كېتەيلى_ دەيدۇ .
ئوغۇللار بولسا ئۆزىنىڭ تەبىئىتى بويىچە قاتتىق غەزەپلىنىدۇ ، " چوقۇم باشقا بىرىگە كۆڭلى چۈشۈپ قېلىپ مەندىن ئايرىلماقچى بولۇۋاتىدۇ " دەپ گۇمان قىلىدۇ ، ھەتتا ۋارقىراپ - جارقىراپ قىزنىڭ ۋىجدانسىزلىقىنى ، تاشيۈرەكلىكىنى ئەيىبلەيدۇ ، ئەمما قىز راستتىنلا ئارقىغا ئۆرۈلۈپ ماڭغان ئاشۇ دەقىقتە ئازاب ۋە غەزەب ئىچىدە ئارقىسىدىن قاراپ تۇرماقتىن سىرت ، " كەتمەڭ " دېگەن بىر ئېغىز سۆزنىمۇ قىلالمايدۇ .

قىز يولبويى ئوغۇلنىڭ ئارقىسىدىن يۈگۈرۈپ كېلىپ قولىدىن تارتىپ ئۆزىنى ئېلىپ قېلىشىنى ، قەدىرلىكىم ،سىزنى سۆيىمەن ، ئايرىلمايلى ، دېيىشىنى كۈتىدۇ ، ئەمما كۆز ياشلىرى ئېقىپ تۈگىگەندىمۇ ھېچقانداق سادا بولمايدۇ ، مۇھەببەت مانا مۇشۇنداق ئاخىرلىشىدۇ ....

ئوغۇللار قىزلارنى قانچىلىك چۈشىنىدۇ ؟ ئەمىلىيەتتە قىزلارنىڭ " ئايرىلايلى" دېگەن سۆزى ئوغۇللارنىڭ " ياق ، كەتمەڭ " دېگەن سۆزىگە بولغان ئىنتىزارلىقىدىن باشقا نەرسە ئەمەس!

ھەرقېتىم " ئايرىلايلى " دېگەندە قىزلار قورقۇنچ ئىچىدە تۇرغان بولىدۇ ، سىلەرنىڭ راستلا ئايرىلىپ كېتىشىڭلاردىن قورقىدۇ ، ھەر قېتىم " ئايرىلايلى" دېگەندە قىز ئىنتىزارلىق ئىچىدە تۇرغان بولىدۇ ، سىلەرنىڭ " كەتمەڭ " دېيىشىڭلارغا ، ئۇنىڭغا نەقەدەر قىيمايدىغانلىقىڭلارنى بىلدۈرۈشۈڭلارغا ئىنتىزار بولغان بولىدۇ . ھەرقېتىم " ئايرىلايلى " دېگەندە قىز ئامالسىزلىق ئىچىدە قالغان بولىدۇ . چۈنكى ئۇ سىزدىكى ھەربىر كىچىك ئۆزگىرىشلەر تۈپەيلىدىن مۇھەببىتىڭىزنىڭ يەنىلا بۇرۇنقىدەك ئىكەنلىگە ئىشەنچ قىلالمايدۇ ، شۇڭا ئامالسىزلىقتىن " ئايرىلىش " ئارقىلىق مۇھەببەتنى دوغا تىكمەكچى بولىدۇ .ئەگەر ئۇتتۇرۇپ قويسا سىز ئۇنى يېتەرلىك سۆيمىگەن بولسىز! ھەرقېتىم راستلا ئايرىلىپ كەتكەندە قىزلارنىڭ يۈرىكى لەختە - لەختە بولۇپ كېتىدۇ ! ئوغۇللار نېمىشقا قىزلارنىڭ قەلبىنى چۈشەنمەيدىغاندۇ ؟

قىزنىڭ " ئايرىلايلى " دېيىشى سىزنى ھەقىقىي سۆيگەنلىكى ئۈچۈن ! سىزنى بەكلا مۇھىم ئورۇنغا قويۇۋالغانلىقى ئۈچۈن ! شۇڭا سىزدىكى ھەربىر كىچىك ئۆزگىرىش ئۇنى ساراسىمىگە سالىدۇ ، ئەندىشىگە چۆككۈزىدۇ . ئۇ پەقەت سىزنىڭ ئۇنى يەنە داۋاملىق سۆيىدىغان سۆيمەيدىغانلىقىڭىزنى ئايدىڭلاشتۇرۋالماقچى خالاس !

قىزلار مۇھەببەتتىكى تېڭىرقاش ئوغۇللارنىڭ ئەڭ ئازابلىق جاۋابىنى ئاڭلاشقا يېتەرلىك كۈچ قۇۋۋەت بېرەلەيدۇ دەپ ئويلايدۇ . قىزلار مۇھەببەت بىر ۋېكلىيۇچاتىلغا ئوخشاش تاق قىلىپ ئۇلىنىپ ، تاق قىلىپ ئۈزىلىدۇ دەپ ئويلايدۇ . قىزلار توك مەنبەسىنى دەرھال ئۈزۈۋەتسە خېيىم - خەتەر كېلىپ چىقمايدۇ دەپ ئويلايدۇ . قىزلار ئايرىلىش بىلەنلا بارلىق تېڭىرقاشلارنى ، ئازابلارنى ، ئەندىشىلەرنى يوقاتقىلى بولىدۇ دەپ ئويلايدۇ ،قىزلار ھەرقانداق يارا ئېغىزى ۋاقىتنىڭ ئۆتىشى بىلەن ئالدىرماي پۈتىدۇ دەپ ئويلايدۇ . قىزلار " ئايرىلايلى" دېگەندە سىلەرنى چوقۇم ئۆزىنى تۇتۇپ قالىدۇ دەپ ئويلايدۇ !

بىراق بۇلارنىڭ ھەممىسى قىزلارنى ئۆزىنىڭلا خىياللىرى . قانچىلىگەن ئوغۇللار مانا مۇشۇنداق ۋاقىتتا ئازاب ئىچىدە ئۆزى سۆيگەن قىزنىڭ كېتىشىگە قاراپ تۇرغان - يۇ ، توسۇپ قالمىغان ؟ نىمىشقا ئوغۇللار قىزلارنىڭ قەلبىنى چۈشىنەلمەيدۇ ؟
