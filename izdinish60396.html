ئىزدىنىش مۇنبىرى › ئۇيغۇر تىبابىتى › ئادەم تىنى مۇھىم ئەزالىرىنىڭ ئىقتىدارى ۋە مۇھىم ئورنى

ئادەم تىنى مۇھىم ئەزالىرىنىڭ ئىقتىدارى ۋە مۇھىم ئورنى

بۇ يازمىنى ئاخىرىدا   hiyalqi تەھرىرلىگەن. ۋاقتى2014-10-26 13:47

كۆپ ساندىكى ئادەم تىنىنىڭفىزىئۇلوگىيىلىك ئىقتىدارىنىئۈگۈنۈپ باقمىغان كىتاپخانلارغا قۇلايلىق يارىتىش ئۈچۈن ، تۆۋەندە ئادەم تىنىدىكى ئەڭ مۇھىم بولغان بىر قىسىم ئۇرۇن ۋە ئەزالارنىڭ ئىقتىدارى ۋە جايلىشىشىنى چۈشەندۈرۈپ بىرىمىز .
I يۈرەك
يۈرەكنىڭ رۇلى بولسا قاننىڭ ھەركەتلىنىشىگە تۈرتكە بۇلۇش ،ئەزا ،تۇقۇلمىلارنى يىتەرلىك قان بىلەن تەمىنلەش ،تۆۋەندىكى بەدەننى ئوكسىگىن ۋە ھەر خىل ئۇزۇقلۇق ماددىلار بىلەن تەمىنلەش ، مىتابولىزىممەھسۇلاتلىرىنى بىللە ئېلىپ يۈرۈش(مەسىلەن :كاربۇن تۆت ئوكسىد، ئورىيا، سۈيدۈك كىسلاتاسى قاتارلىقلار )، ھۈجەيرىنىڭ نورمال ماددا ئالمىشىشى ۋە ئىقتىدارىنى ساقلاپ قېلىشتۇر . بەدەننىڭ ئىچكى قىسىمىدىن ئاجرىلىپ چىققان ھەر خىل ھورمۇن ۋە باشقا بەدەن سۇيۇقلىقى ئىلمىنىتلىرىمۇقان ئايلىنىش سېستىمىسى ئارقىلىق ھەر قايسى ھۈجەيرىلەرگە قۇشۇلۇپ ،ئورگانىزىمنىڭ سۇيۇقلۇق تەڭشىشىنى ئەمەلگە ئاشۇرىدۇ . بەدەن ئىچكى مۇھىتىنىڭ نىسپى مۇقىملىقىنى ساقلايدۇ .ئۇنىڭدىن باشقا قاننى قوغداش فونكىسيىسىنى ئەمەلگە ئاشۇرۇش شۇنداقلا بەدەن تىمپۇراتۇرىسىنىڭ نىسبى تۇراقلىق بۇلۇشىنى تەڭشەشمۇ ، قاننىڭ قان تۇمۇرىدا توختىماستىن ئايلىنىشقا تايىندۇ . قاننىڭ ئايلىنىشى بولسا يۈرەك « پومپىسى » نىڭ رۇلى ئارقىلىق ئەمەلگە ئاشىدۇ .
رەڭدار پەردىدە يۈرەك قىسىمىدا كۈپۈنكى ھالەت كۈرۈلسە ،دائىم مۇزدەك سۇ ئىچىدىغانلىقىنى ، ئاسان ھاياجانلىندىغانلىقىنى بىلدۈرىدۇ . سەل پېتنقى ھالەت كۈرۈلسە ،يۈرەك ئىقتىدارى چىكىنپ كەتكەنلىگىنى ، بەدەن سوغوقتىن قورقىدىغانلىقىنى ، خاراكتىرى بىر قەدەر ئىچكى مىجەز ئىكەنلىگىنى بىلدۈرىدۇ . رەڭدار پەردە يۈرەك قىسىمدا ئويمان كۈرۈلسە ، دائىم 脑闷气短         مىڭىگە قان يىتىشمەسلىك يۈز بىرىدۇ .

IIجىگەر
جىگەر بولسا ئادەم تىنىدىكى ئەڭ چوڭ ھەزىم قىلىش بىزى ،شۇنداقلا ئادەم تىنىدىكى مىتابولىزىم ( يېڭى ماددا ئالماشتۇرۇش ) نىڭ مەركىزى . ئادەم تىنى ئېھتىياجلىق بولغان ئىنىرگىيە بىز يىگەن ئۇزۇقلۇقلاردىن كىلىدۇ . ئۇلار كاربۇن سۇ بىرىكمىلىرى ، ئاقسىل ۋە مايلاردىن ئىبارەت . بۇ ئۇزۇقلۇق ماددىلارنىڭ ماددا ئالمىشىش جەريانى ۋە ئۆز- ئارا ئالمىشىشى ، ئاساسلىقى جىگەردە ئېلىپ بېرىلىدۇ . جىگەر ئادەم تىنىدىكى مۇھىم زەھەرنى تازىلايدىغان ئەزا ، ئۇ ئورگانىزىمنى قوغداپ ، زىيانغا ئۇچرىشىدىن ساقلايدۇ . رەڭدار پەردىدە جىگەر ئورنىدا داغ ۋە يېرىق كۈرۈلسە ، بۇ بەدەننىڭ زەھەر قايتۇرۇش ئىقتىدارىنىڭ تۆۋەنلەپ كەتكەنلىگىنى ئىپادىلەيدۇ . بۇ چاغدا ئاسان چارچايدۇ ، يۈزى قارىداپ كىتىدۇ ياكى ئاللىبۇرۇن جىگەرنى ماي قاپلاشكۈرۈلگەن بۇلىدۇ . بۇ ۋاقىتتا جەزمەن كۆپرەك ئارام ئېلىش ، ئەخلەت يىمەكلىكلەرنى ئازراق يىيىش كېرەك .

III ئاشقازان
ئاشقازان بىر خىل ئۆتكۈنچى شەكىللىك ئوتتۇرا بەدەن بوشلۇقى ئەزاسى ، ئۇ ھەزىم قىلىش يۇلىنىڭئەڭ كىڭەيگەن قىسىمى ، قورساق بوشلۇقىنىڭ سول يۇقۇرى قىسىمىغا جايلاشقان . يىمەكلىكلەر ۋاقىتلىق توختايدىغان ۋە ھەزىم قىلىندىغان سورۇن، ئاشقازاننىڭ كىرىش ئېغىزى 宽门دەپ ئاتىلىدۇ . قىزىل ئۆڭگەچ بىلەن تۇتىشىپ تۇرىدۇ . چىقىش ئېغىزى 幽门 دەپ ئاتىلىدۇ ،12 بارماق ئۈچەي بىلەن تۇتىشىپ تۇرىدۇ . ئاشقازان شىللىق پەردىلىرى ئىچىدە نۇرغۇن ئاجراتما بەزلەر بۇلۇپ ، شىلىمشىق سۇيۇقلۇق ، ئاشقازان تېمى (پوستى)مۇسكۇللىرى يىغىلغان چاغدا يىمەكلىكلەرنىڭ كىچىك ئۈچەيگە كىرىشىكە ھەيدەكچىلىك قىلىپلا قالماستىن يەنە يىمەكلىكلەرنى ئارىلاشتۇرۇپ ئىزىش رۇلى بار .
رەڭدار پەردىدە ئاشقازان ھالقىسى يۇملاق بولمىسا ياكى قېنىق كۈلەڭگە كۈرۈلسە ،بۇ دائىم بەدەننىڭ ھەزىم قىلىش ، سۈمۈرۈش ئىقتىدارى چىكىنىپ كەتكەنلىگىنى ، ئۇزۇقلۇق ماددىلارنى سۈمۈرۈشى يىتەرلىك ئەمەسلىگىنى ، ناھايىتى ئاسانلا ئاشقازان كۈپىدىغانلىقىنى ، ئاشقازان ئاغرىيدىغانلىقىنى بىلدۈرىدۇ . شۇڭلاشقا ئەڭ مۇھىم بولغىنى تۇرمۇش تەرتىۋىگە ئەھمىيەت بىرىش ۋاقتىدا تاماق يىيىش ، ئاسان قۇبۇل قىلىنىدىغان يىمەكلىكلەرنى كۆپرەك يىيىشۋە يىمەك-ئىچمەكنى مۇۋاپىق ماسلاشتۇرۇشكېرەك .

IV ئاشقازان ئاستى بىزى
ئاشقازان ئاستى بىزى يۇقۇرقى قورساق قىسىمىغا جايلاشقان ئۇ ئادەم تىنىدىكى 2- چوڭ ھەزىم قىلىش بىزى بۇلۇپ ، جىگەر ،ئۆت ، ئاشقازان ، 12 بارماق ئۈچەي ۋە تال قاتارلىقلار بىلەن قوشنا . ئاشقازان ئاستى بىزى گەرچە كىچىك بولسىمۇ ،ئۇنىڭ رۇلى ناھايىتى چوڭ . شۇڭا ئۇنى ، ئادەم تىنىدىكى ئەڭ مۇھىم ئەزالارنى بىرى دەپ ئېيتىشقا بۇلىدۇ . ئاشقازان ئاستى بىزى ئادەم تىنىدىكى ئەڭ مۇھىم ئەزالارنىڭ بىرى دىيىشىمىزدىكى سەۋەپ ئۇنىڭ ئىچكى ۋە سىرتقى چىقىرىش ئىقتىدارىغا ئىگە بەز بولغانلىغى ،ئۇنىڭ فىزىئۇلوگىيلىك رۇلى ، پاتولوگىيلىك ئۆزگىرىشلىرىنىڭ ھەممىسىنى ھاياتلىق ئۇچۇرى بىلەن ئۆز- ئارا مۇناسىۋەتلىك بولغانلىقىدا . ئاشقازان ئاستى بىزىئاجرىتىپ چىقارغان بەز سۇيۇقلۇقىدىكى بىر نەچچە خىل ھەزىم قىلىش ئىنزىملىرى يىمەكلىكنىڭ ھەزىم قىلىنىشى جەريانىدا « باش رول » نى ئۆتەيدۇ . بۇلۇپمۇ ياغلارنىڭ ھەزىم قىلىنىشىغا نىسبەتەن چۇقۇم شۇنداق بۇلىدۇ . ئاجرىتىپ چىقىرىش جەھەتتە ، گەرچە ئاشقازان ئاستى بىزىنىڭ ھەجىمى كىچىك بولسىمۇ ، بىراق ئۇ كۆپ خىل ئىقتىدارغا ئىگە ئىچكى ئاجىرتىپ چىقىرىش ھۈجەيرىلىرىگە ئىگە بولغان بۇلىدۇ . مەسىلەن: 胰高血糖素 ،胰岛素، 胃必素قاتارلىقلارنى ئاجرىتىپ چىقىرىدۇ . بۇ ھۈجەيرىلەر ھورمۇن ئاجىرتىپ چىقىرىشتا ماددىلارنى ھەزىم قىلىش سۈمۈرۈشكە قاتنىشىپلا قالماستىن بەلكى پۈتۈن بەدەننىڭ فىزىئولوگىيلىك ئىقتىدارىنى تەڭشەشكە مەسئۇل بۇلىدۇ . ئەگەر بۇ ھۈجەيرىلەردە كىسەللىك ئۆزگۈرىشى بولسا سۈمۈرگەن ماددىلار ئېشىپ قالىدۇ ياكى يىتىشمەيدۇ ، ئوخشاشلا كىسەللىك كىلىپ چىقىدۇ .

V قالقانسىمان بەز
قالقانسىمان بەز – قالقانسىمان بەز ھورمۇنىنى ئاجرىتىپ چىقىرىدۇ ،ئاساسلىق رۇلى ماددا ۋە ئىنىرگىيە ئالماشتۇرۇشنى ئىلگىرى سۈرۈش ، ھەمدە ئادەم تىنىنىڭ ئۈسۈش ۋە يىتىلىش جەريانىنى كونتىرول قىلىش ؛قالقانسىمان بەز ھورمۇنى مۇتلەق كۆپ قىسىم تۇقۇلمىلارنىڭ ئوكسىگىننى سەرپ قىلىش نىسبىتىنى يۇقىرى كۈتىرىدۇ ،قەنتلەرنىڭ سۇ سۈمۈرۈشى ۋە جىگەردىكى قەنت ئىلمىنتىنىڭ پارچىلىنىشىنى ئىلگىرى سۈرىدۇ ،خولىستىرىننىڭ چۈشۈشىنى كۈچەيتىدۇ ،ئاقسىل ۋە ئىنزىملارنىڭ شەكىللىنىشىنى ئىلگىرى سۈرىدۇ . يەنە قاندىكى كالتىسىي مىقدارىنىڭ تەڭشىلىشىنى كونتىرول قىلىشقا ياردەم بىرىدۇ . قالقانسىمان بەز ئىقتىدارى   تۆۋەنلەپ كەتسە ماددا ئالماشتۇرۇش ئىقتىدارىنى تۆۋەنلىتىۋېتىدۇ ،شۇنىڭ بىلەن بەدەندىكى ھەرقايسى ئەزالارغا تەسىر كۆرسىتىدۇ ،روھى چۈشكۈنلۈك ،غەزەپلىنىش ،ھەسەت قىلىش ،ئايرىلىپ كىتىش ( يەنى ئاتالمىشمۇھەببەتتە ) قاتارلىقلارنىڭ ھەممىسى قالقانسىمان بەز ھورمۇنىنىڭ ئاجرىلىپ چىقىشىنى ئازايتىدۇ .
ئەگەر تەكشۈرۈلگۈچىنىڭ رەڭدار پەردىسى ئۈستىدە قالقانسىمان بەزنىڭ مۇناسىپ ئورنىدا ئويمان كۈرۈنسە، ئۈزىنىڭقالقانسىمان بىزىدە مەسىلە بارلىقىغا دوختۇرخانىدا دىئاگنۇز قۇيۇلۇپ باقمىغان بولسا ئۇنىڭدا زىيادە ئۇرۇقلۇق ،ئاسان ھۇدۇقۇش ۋە جىددىيلىشىش قاتارلىق ئەھۋاللارنىڭ بار –يوقلىقىغا دىققەت قىلىش كىرەك ،ئەگەر يۇقارقىدەك ئەھۋاللار كۈرۈلسە دوختۇرخانىغا بېرىپ بىئولۇگىيىلىك لابۇراتىرىيىلىك تەكشۈرتۈش ئېلىپ بېرىپ كىسەللىكنى بىكىتىش ھەمدە بالدۇرراقكەيپىياتنى تەڭشەش ياكى كىسەلنىداۋالاش كېرەك .
VI بۆرەك ئۈستى بېزى
ئۇ بۆرەكنىڭ ئۈستىقىسمىدىكى ئىچكى ئاجراتما بېزى ،ئۇنىڭ ئىقتىدراى ،قان بېسىمى ،يۈرەك سۇقۇش رىتىمى ۋە قاندىكى ناترىي كالىينىڭ تەڭپۇڭلىقى قاتارلىقلارنى تەڭشەشكە ياردەم بىرىش .بۆرەك ئۈستى بېزى چىكىدىن ئېشىپ كەتسە بۆرەك ئۈستى بىزى ھورمۇنلىرى سەرپ بۇلۇپ قان بېسىمى تۆۋەنلەش ،بۆرەك ،تىرە ۋە ئۆپكىنىڭ ئاجرىتىپ چىقىرىش ئىقتىدارىنى ئاجىزلاشتۇرۇشقا ئېلىپ بارىدۇ .ئەگەر قاندىكى توكسىنلار زىيادە كۆپ بولسا بۆرەك ئۈستى بېزىنىڭ ئىقتىدارىنى تۆۋەنلىتىۋېتىدۇ .

VIIبۆرەك
سۈيدۈك ئىشلەپچىقىرىدۇ ،ئادەم تىنىدىكى سۇ مىقدارىنىڭ تەڭپۇڭلۇقىنى ساقلايدۇ ،ئادەم تىنىدىكى ماددا ئالماشتۇرۇش مەھسۇلاتلىرى ۋە زەھەرلىك ماددىلارنى ( مەسىلەن :سۈيدۈك ، سۈيدۈك كىسلاتاسى ،مۇسكۇل ئانگىدرىت 肌酐) قاتارلىق تەركىۋىدە ئوكسىگىن بولغان ماددىلارنى چىقىرىپ تاشلاپ ،ئادەم تىنىنىڭ نورمال فىزىئولوگىيىلىك پائالىيىتىنى قوغدايدۇ ،شۇنىڭ بىلەن بىر ۋاقىتتا ماددا ئالماشتۇرۇش جەريانىداشەكىللەنگەن كىسلاتالىق ماددىلارنى سۈيدۈك ئارقىلىق بەدەن سىرتىغا چىقىرىپ تاشلاپ ،كىسلاتا –ئىشقار تەڭپۇڭلۇقىنى ساقلايدۇ ،ئۇنىڭدىن باشقا بىر قىسىم ماددىلارنى ئاجرىتىپ چىقىرىدۇ ،بىر قىسىم ماددىلارنى بىرىكتۈرۈپ ئادەم تىنىنىڭ فىزىئولوگىيىلىك ئىقتىدارىنى تەڭشەيدۇ ،مەسىلەن ،قان بېسىمىغامۇناسىۋەتلىك بولغان بەز، مەزى بىزى ھورمۇنى قاتارلىقلارنى ئاجرىتىپ چىقىرىدۇ ،تەڭشەيدۇ ،قىزىل ھۈجەيرە شەكىللەندۈرۈش ھورمۇنىنى ئاجرىتىپ چىقىرىدۇ ( كەم قانلىقنى كەلتۈرۈپ چىقىرىدىغان ئامىلنى ئازايتىدۇ ).
بۆرەك رايۇنىنىڭ ھەرخىل ھالەتلىرىئويمان ،يېرىق ۋە داغ قاتارلىقلارنى ئۆز ئىچىگە ئالىدۇ ،بۇلارنىڭ ھەممىسى رەڭدار پەردىدە دائىم كۈرۈلىدىغان ئەھۋاللاردۇر .چۈنكى بۆرەك جوڭگۇ تىبابىتىدە ئىرسىيەت ئاساسى   دەپ ئاتىلىدۇ ،مەنىسى «تۇغما ئاساس» «تەبىئىي ئاساس » دىگەن مەنىلەردە ،بىز يىگەن نورغۇن ئانتسبىئوتىك (Antsbiotic ) دورىلارنىڭ بەدەن سىرتىغا چىقىرىلىدىغان يۇلى جىگەر ئەمەس بەلكى بۆرەك. شۇڭلاشقا بەدەننىڭ ئۆزگىرىشى كۆپىنچە بۆرەكنىڭ زىيانغا ئۇچرىشىدىن باشلىنىدۇ ،رەڭدار پەردە ئۈستىدە بىزنىڭ كۆرگىنىمىز بۇرۇندىن قېلىپ قالغان مەسىلىمۇ ياكى يېڭىدىن پەيدا بولغان مەسىلىمۇ ؟بۇنى رەڭدار پەردىدە ئوخشاش ۋاقىتتا كۈرۈلگەن باشقا ھالەتلەرگە ئاساسەن بىرلەشتۈرۈپ قايتىدىن ئۇنىۋىرسال مۇلاھىزە قىلىش كېرەك .

VIII ئۆپكە ۋە كاناي
ئۆپكە- ئۆپكە ئوڭ قانىتى ۋە ئۆپكە سول قانىتى دەپ بۈلىنىدۇ ، ئۇ ئۇچلۇق سىممىتىرىكشەكىللىك ئەزا ،ئۇ كۆكرەك بوشلۇقىنىڭ كۆپ قىسمنى ئىگەللەيدۇ ،كاناي بىلەن ئۆپكە شارچىسىدىن تۈزۈلىدۇ . ئۆپكىنىڭ ئىقتىدارى قاننى ئوكسىگىن بىلەن بىرىكتۈرىدۇ ھەمدەكاربۇن (IV-)ئوكسىدنى چىقىرىۋېتىدۇ ،بەدەن تىمپۇراتۇرىسىنى ،كىسلاتا – ئىشقار تەڭپۇڭلۇقىنى كونترول قىلىدۇ ،ئۆپكە –كاناي ئىقتىدارىدا توسالغۇ بولسا ئۈچەي يۇلى ،بۆرەك ،تىرە ،لىمفا سېستىمىسى ،يۈرەك ،مۇستەقىل نېرۋا ۋە بەدەندىكى باشقا ئەزا تۇقۇلمىلارغا تەسىر كۆرسىتىپ ،قاندىكى كاربۇنات كىسلاتا تەڭپۇڭلۇقىنى بۇزۇۋېتىدۇ ،ئوكسىگىننىڭ بىرىكىشىنى ئاجىزلاشتۇرىدۇ ،باشقا ياللۇغلارنىڭ پەيدا بۇلۇشى مۇمكىنچىلىگىنى ھەمدە قاننىڭ كىسلاتالىق دەرىجىسنى يۇقىرى كۆتىرىۋېتىدۇ. ئۆپكە ئىقتىدارى نورمال بولمىسا زىيادە سىزىمچانلىق ،يۆتەل ،لىمفا ئايلىنىشى توسقۇنلۇققا ئۇچراش ،چارچاش ،پۈتۈن بەدەن سىرقىراپ ئاغرىش ، ماددا ئالمىشىش تۆۋەنلەش قاتارلىقلارنى كەلتۈرۈپ چىقىرىشى مۈمكىن .شۇڭلاشقا بىز رەڭدا پەردە ئۈسىتىدە ئۆپكە قىسمىدا ئويمان ياكى يېرىق كۈرۈلگەن چاغدا كىسەلنىڭ كىسەل تارىخى ،تۇرمۇش ئادىتى قاتارلىقلارنى تەپسىلى سوراپ ،قارشى تەرەپكە تاماكا چەكمەسلىك ،ئەخلەت يىمەكلىكلەرنى يىمەسلىك ھەققىدە تەكلىپ بىرىشىمىز لازىم .

IX لىمفا سېستىمىسى
لىمفا سېستىمىسى - بادام بەز ،سازاڭسىمان ئۈسۈكچە ،تال ،سۈت بېزى ،كۆكرەك بېزى قاتارلىقلارنى ئۆز ئىچىگە ئالىدۇ ،لىمفا سېستىمىسىنىڭ رۇلى – ئادەم بەدىنىنىڭ ھەرقايسى جايىدا پەيدا بولغان ئەخلەتلەرنى تۇشۇپ يىغىشتىن ئىبارەت. لىمفا سېستىمىسىنىڭ ئىقتىدارى توسالغۇغا ئۇچرىسا بادامبەز ،سۈت بىزى ،سازاڭسىمان ئۈسۈكچە ، بۇغۇم ، مۇستەقىل نېرۋا سېستىمىسى ۋە ئاجرىتىپ چىقىرىش ئەزالىرى تەسىرگە ئۇچرايدۇ .


X ئۈچەي يۇلى
ئۈچەي يۇلى ئادەم تىنىدىكى مۇھىم بولغان ئاجرىتىپ چىقىرىش ئەزاسى . كىچىك ئۈچەي ئاساسلىقى سۈمۈرىدۇ ، چوڭ ئۈچەي ئاساسلىقى ئاجرىتىپ چىقىرىدۇ ،ئۈچەي يۇلى ياخشى بولمىسا باشقا ئاجرىتىپ چىقىرىش ئەزالىرىنىڭ مەسئۇلىيىتىگە تەسىر يەتكۈزىدۇ ،يەنە قان ۋە لىمفادىكى توكسىنلارنى كۆپەيتىپ ئۇلارنى پۈتۈن بەدەندە تىندۇرۇپ قۇيىدۇ ،بۇلۇپمۇ ئىرسىيەت خاراكتىرلىك ئاجىزراق بولغان ئەزاغا بەكرەك تەسىرى بۇلىدۇ ،يەنى يۈرەك ،مىڭە ،ئۆپكە ،كاناي ،بۆرەك ، كۆپىيىش سېستىمىسى ،جىگەر ، ئۆت ،لىمفا سېستىمىسى ۋە تىرە قاتارلىقلارغا تەسىر كۆرسىتىدۇ .ئۈچەي يۇلى نورمال بولسا سالامەتلىكنى ئىلگىرى سۈرىدۇ ،ئەكسىچە بولسا پۈتۈن بەدەندىكى ئەزالارغا تەسىر كۆرسىتىدۇ .

XIتىرە
تىرە ئادەم تىنىدىكى يۈزى ئەڭ چوڭ بولغان ئەزا ،ئۇ – قوغداش ، ئاجرىتىپ چىقىرىش ،بەدەن تىمپۇراتۇرىسىنى تەڭشەش ، سىرتقى مۇھىتنىڭ غىدىقلىشىنى قۇبۇل قىلىش قاتارلىق ئىقتىدارلارغا ئىگە .تىرە مەسلىسىنىڭ ئاساسلىق مەنبەسى –ئۈچەي يۇلىدىكى توكسىندىن كىلىدۇ ،ئۈچەي يۇلىدا توكسىن زىيادە كۆپ بولسا تىرە قورغاقلىشىدۇ ، زىيادە يىرىكلىشىدۇ ، 长疱                   ئەسۋە ( 丘疹 ) ،دانىخورەك ( 痤疮) ،干藓قاتارلىقلارنى پەيدا قىلىدۇ ،باشقا ئەزالارغىمۇ تەسىر كۆرسىتىدۇ ،شۇنداقلا ئىقتىدارى نورمال بولمىغان ئۆپكە – بۆرەك،لىمفا سېستىمىسى ، ھەزىم قىلىش سېستىمىسى قاتارلىقلارنىڭ تەسىرىگىمۇ ئۇچرايدۇ .
تىرە ماددا ئالمىشىش مىتابولىزىم ھالقىسى قارا رەڭلىك بولسا تىرىنىڭ ماددا ئالمىشىشىنىڭ ياخشى ئەمەسلىگىنى شۇنداقلا بەدەن ئىچىدىكى توكسىىننى ھەيدەشنىڭ يىتەرلىك ئەمەسلىگىىنى چۈشەندۈرىدۇ ، بۇلارغا قارىتا كۆپ ھەرىكەت قىلىپ كۆپ تەرلەش ،تىرىنىڭ پاكىزلىقىنى ساقلاش ۋە ئىچكى ئەزالاردىكى توكسىننى ھەيدەش قاتارلىقلارغا تەكلىپ بىرىش كېرەك .
