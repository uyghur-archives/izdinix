يىقىن بىر ئاغىنام بۇلدىغان كىلىشەلمەسلىك سەۋەبى بىلەن ئايالىدىن ئاجىرشىپ كەتكەن،كىيىن بىرئاغىنىسىنىڭ تويىدا بىر قىز بىلەن تۇنۇشۇپ قاپتۇ، بىر مەزگىل ئارلاشقاندىن كىيىن توي قىلماقچى بۇپتۇ،ئارلىشىش جەريانىدا ئۇ قىزدا ھىچقانداق غەيرى ئالامەتلەر كۆرۈنمەپتۇ، توي كۈنى  ئەتتىگەندە بىرنەچچە يىقىن ئاغىنلەر نىكاھغا باردۇق،نىكاھ ئوقۇلدى تويمۇ تۈگىدى كەچتە يىگىت بىلەن قىزنى ھوجۇرغا ئەكىرىپ قۇيۇپ قايتىپ كەلدۇق،ئەتىسى ئاغىنام نىكاھتىن ئاجىراشتى،سەۋەبىنى دەمسىلەر ئالدىرماڭلار ھازىر دەپ بىرىمەن،ئاغىنەمنىڭ دەپ بىرىشچە بىز قايتىپ كەتكەندىن كىيىن يىگىت بىلەن قىز بىردەم پاراڭلىشىپ ئولتۇرۇپتۇ،ئارلىقتا قىز چىقىپ گىرىملىرىنى يۇيۋىتىپ كىرىپتۇ، ئاغىنەم ئەمدى ئارام ئالايلى دەپ ئايالىنى چاقىرىپتۇ، شۇ ئەسنادا توۋە ئۇ قىزنىڭ كۆزىگە قان تۇلۇپ كۆزى يوغىناپ چىرايى شۇنداق سەتلىشىپ كىتىپتۇ دە ئاغىنەمنىڭ گىلىدىن بوغۇپتۇ، ئاغىنەم ئاران تەستە گىلدىن ئۇ قىزنىڭ قۇلىنى ئاجىرتىپتۇ،قارىسا ئۇ قىز ئاغىنەمگە شۇنداق سەت خىرىس قىلىپ تۇرغۇدەك ئاغىنەم ئازىراق ھەركەت قىلسىلا ئۇ قىز ھازىرلا ئىتىلدىغاندەك سەت بىر خىل ئاۋاز چىقىرىپ تۇرۇپتۇ،ئاغىنەمنىڭ دەپ بىرىشچە مۇشۇ كەمگىچە ئۇنداق قورقۇنۇشلۇق ئىشنى كۆرۈپ باقماپتىكەن،ئاغىنام بەك قوقاپ كەتكەشكە يەردىن بىشىنى كۆتىرەلمەپتۇ،بىر چاغدا ئۇ قىز ئاستا مىڭىپ مىھمانخانا ئۆيگە چىقىپ كىتىپتۇ،ئاغىنام ياتاق ئۆينىڭ ئىشكىدىن ئاستا مىھمانخانا ئۆيگە قارىسا ھىلىقى قىز ئاغىنامغا تىكىلىپ بىر خىل سەت ئاۋازدا خىرىس قىلىپ تۇرغۇدەك،ئاغىنام مىھمانخانا ئۆيگە قاراشقىمۇ پىتىنالماي نەچچە سائەت ياتاق ئۆيدە قوقاپ ئولتۇرۇپتۇ،بىر چاغ بولغاندا ئۇ قىز ئەسلىگە كەپتۇ ئاغىنامدىن نىمىش بولغانلىقىنى سوراپتۇ،ئاغىنام بۇلۇپ ئۆتكەن ئىشلارى دەپ بىرىپتۇ،ئاغىنام ئۇ قىزنى ئەمدى ئوڭشالغان ئوخشايدۇ دەپ قۇلىنى تۇتۇپ تۇرۇشىغا ئۇ قىز يەنە شۇنداق قورقۇنۇشلۇق ھالىتىگە ئۆزگۈرۈپ قاپتۇ،ئاغىنام توي كىچىسىنى شۇنداق قورقۇنۇشلۇق ئىشلار بىلەن ئۆتكۈزۈپتۇ،بامداتقا ئەزان چىققان چاغدا ئۇ قىز يەنە ئوڭشۇلۇپ قاپتۇ، ئۇ قىز ئاغىنامنىڭ زورلىشى بىلەن ئۆزىنىڭ جىنى بارلىقىنى ئەرلەر چىقىلماقچى بولسا ئاغىنام كۆرگەندەك سەتلىشىپ كىتىدىغانلىقىنى دەپ بىرىپتۇ،ئاغىنام قوقاپ كەتكەشكە ئەتىسى سۆزىنى بىرىۋىتىپتىكەن.بەزەن يەرلىرىنى قىسقارتىۋەتتىم ،قايسىڭلار شۇنداق ئىشنى ئاڭلاپ باققان.

70 يىللاردا تاغام دادۇينىڭ 55 تىپلىق تىراكتۇرنى ھەيدەيدىكەنتۇق،تولاراق يەكەندىن قەشقەرگە ماي تۇشۇيدىكەنتۇق.بىر كۈنى باشقا بىر شوپۇر ئاغرىپ قالغانلىقتىن، تاغام ئۇ شوپۇرنىڭ ئورنىدا قەشقەگە چۈشتىن كىيىن، يەنە ناماز شام ۋاختى بىلەن يولغا چىقىپتۇ.تىراكتۇرغا ئىككى كۇزۇپ چىتىلغان بۇلۇپ ھەر بىر كۇزۇپنىڭ بىشىغا كۇزۇپنى يۇرۇتۇپ تۇردىغان چوڭ چىراغ ئورنۇتۇغلۇقكەن،تىزلا قاراڭغۇ چۈشۈپ كەتكەنلىكتىن تاغام كۇزۇپلارنىڭ چىرىغىنى ياندۇرۇپ قۇيۇپتۇ،يەكەندىن قەشقەگە بارغىچە ئۇ چاغدا ئىككى ياكى ئۈچ يەردە تىراكتورغا سۇ قاچىلايدىكەنتۇق،چۈنكى تىراكتوردىكى سۇ بەك قىززىپ كەتكەشكە توختاپ سۇ قاچىلاپ ماڭىدىكەنتۇق،تاغام پات پات كەينىگە يەنە كۇزۇپلارغا قاراپ قۇيۇپ مىڭىپتۇ،بىرەر ئەھۋال يوق،دائىم سۇ قاچىلايدىغان يەرگە كەلگەندە سۇ قاچىلاش ئۈچۈن يەرگە چۈشۈپ ماتۇرنىڭ سۈيىنى ئالماشتۇرۇپتۇ،ئازراق ماڭغاندىن كىيىن كەينىگە قارىسا ئەڭ كەينىدىكى كۇزۇپتا بىر ئايال پۇتىنى كۇزۇپنىڭ ئالدى رىشاتكىسىدىن ساڭگىلتىپ ئولتۇرغۇدەك تاغام بۇنى كۆرۈپ بايا سۇ قاچىلغان چاغدا مەندىن خىجىل بۇلۇپ چىقۋالغان ئوخشايدۇ دەپ كارى بولماي ئازىراق ماڭغاندىن كىيىن، باشقىلارنىڭ بۇ يولدا كىچىسى ماڭغانلارغا جىن كۆرنەمىش چىشىغا تەگسە بۇغۇپ قۇيامىش دىگەندەك قىلىشقان گەپلىرى ئىسىگە كىلىپ، بەدىنى توك سوققاندەك تىتىرەپ كىتىپتۇ، بىسمىللا دەپ كەينىگە ئاستا قارىسا ئىككىنىجى كۇزۇپتا ئولتۇرغان ئايال يەنە ئىككى ئايال بىلەن بىرىنجى كۇزۇپتا پۇتلىرىنى ساڭگىلتىپ ئولتۇرغۇدەك تاغامنىڭ پۈتۈن بەدىنى چىلىق چىلىق تەرلەپ كىتىپتۇ تەرەكتۇرنى تاس قاپتۇ ئاپىرىپلا دەرەخكە سوققىلى ،تاغام بىلدىغان ئايەتلىرىنى بىر باشتىن ئوقۇغۇلى تۇرۇپتۇ بىردەم ماڭغاندىن كىيىن قارىسا ئۇ بايامقى ئۈچ ئايال ئىككى بوپ قاپتۇ يەنە ئىككىنجى كۇزۇپقا چىقىۋاپتۇ،ئەشۇنداق مىڭىپ يەنە بىر سۇ قاچىلايدىغان يەرگە كەپتۇ، قارىسا تەلىيىگە باشقا بىر تەرەكتۇرغا سۇ قاچىلغىلى تۇرغانمىش تاغاممۇ دەرھال چۈشۈپ سۇنى قاچىلۋاپتۇدە بىسمىللا دەپ تەرەكتۇرنىڭ ئىشىكلىرىنى چىڭ تاقاپ شۇ ماڭغانچە باردىغان يىرىگە بىرىۋاپتۇ ۋە ئەتىسى كۈندۈزدە قايتىپ كەپتۇ ئۇ كىچىسى تەلىيىگە تاغامغا چىقىلماپتۇ.شۇ ئىش بوپ نەچچە كۈندىن كىيىن ئۈچ تەرەكتۇردا خوتەنگە ماي تۇشۇيدىغان بوپ قىلىپ يولدا كىتىۋاتسا ئارىسىدىكى بىر تەرەكتۇر يىرىم يولدا بۇزلۇپ قاپتۇ،بۇزلۇپ قالغان تەرەكتورنى سۆرەپ مىڭىپ يېرىم كىچە بوپ كىتىپتۇ تەرەكتۇرنىڭ ماتورى بەك قىززىپ كەتكەچكە تاغاملار مەسلەتلىشىپ شۇ يەردە ئۇخلاپ ئەتىسى ماڭماقچى بۇلۇپ ئازراق ئوت قالاپ جۇۋۇلىرىنى سىلىپ پاراڭلىشىپ يىتىپ ئۇخلاپ قاپتۇ،بىر چاغ بولغاندا بىرسىنىڭ تاغامنى چاقىرۋاتقانلىقىنى ئاڭلاپ ئويغىنىپ كىتىپتۇ،قارىسا بىر ھەمراھى يىراقتا تۇرۇپ تاغامنى چاقىرۋاتقانمىش تاغاممۇ نىمە بولغاندۇ دەپ ئالدىراش ئورنىدىن تۇرۇپ ھەمراھى چاقىرغان تەرەپكە مىڭىپتۇ شۇنچە تىز ماڭسىمۇ ئۇ ئادەمگە پەقەت يىتىشەلمەپتۇ.ئۇ ئادەم تىز بوغىن تىز بوغىن دەپ ئالدىدا مىڭىپتۇ تاغام ھەي توختاپ تۇر دەپ شۇنچە چاقىرسىمۇ كەينىگە قاراپمۇ قويماي كىتىپ بارمىش شۇ چاغدا تاغام مىڭىشتىن توختاپ يەنە بىر ھەمراھىنىمۇ بىللە ئىلۋالمىغىنىغا پۇشايمان قىپ كەينىگە ياناي دەپ تۇرسا ئۇ ئادەم توختاپ ھە كەلدۇق تىز بۇياققا كەل دەپ چاقىرىپتۇ،تاغام سەل ئىككىلىنىپراق قىشىغا بارسا يەرگە قارا دەپتۇ يەرگە قارىسا گىردىنىڭ چوڭلىقىدا بىر تۈشۈك تۇرغۇدەك تاغام ئولتۇرۇپ تۈشۈكتىن قارىسا شۇنداق چوڭ بىر زال زالنىڭ ئىچىدە ھەرخىل ھەرياڭزا چىراقلار يىنىقلىق ھەر خىل كىشىلەر ئولتۇرۇپ زىياپەت قىلۋىتىپتۇ،ھىلىقى ئادەم تاغامغا قانداق چۈشەمدۇق زىياپەتكە قاتنىشىپ چىقامدۇق دەپتۇ تاغام نىمە دىيىشىنى بىلەلمەي تۇرغاندا تۇيۇقسىز كەينىدىن ھىلىقى ھەمرالىرى چاقىرىپ كەپتۇ ،كەينىگە قارىسا ئىككى ھەمراسى تۇرغىدەك ھىلىقى ئۆزىنى باشلاپ كەلگەن ھەمراسى بىردەمدە يوق ھىلىقى يەردىكى تۈشۈكمۇ يوق ئۇلار تەرەكتۇرنىڭ قىشىغا قايتىپ كىلىپ تاغام بولغان ئىشنى ھەمرالىرىغا دەپ بىرىپتۇ،كىچىدە بىرسىنىڭ قۇسقى ئاغىرىپ كىتىپ تەرەكتۇرنىڭ يىنىغىراق بىرىپ تەرەت قىلىپ ئولتاسا تاغام ئورنىدىن چاچىراپ قۇپۇپ كىتىپ بارمىش ئۇ ئادەم ئورنىدىن قۇپۇپ يەنە بىرسىنىمۇ ئوغۇتۇپ  تاغامنىڭ كەينىدىن مىڭىپتۇ تاغام مىڭىپ ھىلىقى يەرگە كەلگەندە ئولتۇرۇپ يەرگە ئىڭىشىپ بىر نىمىلەرنى قىلغىلى تۇرۇپتۇ ئۇلار سەل قوقاپراق تاغامنى چاقىرىپتىكەن تاغام شۇنىڭ بىلەن ئەسلىگە كەپتۇ ،ئۇلار كىچىچە ئۇخلىماي تاڭنى ئاتقۇزۇپ تەرەكتۇرنى يەنە بىر ئوت ئالدۇرۇپتىكەن تەرەكتۇر ئوت ئاپتۇ شۇنىڭ بىلەن ئۇلار يولىغا راۋان بوپتۇ تاغام شۇ قايتىپ كەلگەندىن كىيىن تەرەكتۇر ھەيدىمەپتۇ ،نىمىشقا دىسەڭلار تەرەكتۇر ھەيدەپ يۈرگەن چاغلاردا ئاشنىڭغا ئوخشاش ئىشلارغا بەك كۆپ ئۇچىراپ كەتكەشكە قورقۇپ قاپتۇ.

مەنمۇ ئۆز بېشىمدىن ئۆتكەننى سۆزلەپ بىرەي راس ۋەقە  2014 يىلى 12 ئاينىڭ 31 كۈنى كەچ سائەت 9  چوڭ ئۆيدە ھەممىمىز دوختۇرخانىدىكى ئشلانى سۆزلەپ ئولتۇرىمىز مەن دوختۇرخانىدىن چىققىلى ئككى كۈن بولغان ۋاقىت ئىدى  بىردەم پاراڭ سوقۇپ  ئەمدى ياتايلى دەپ ياتماق بولدۇق  شۇنىڭ بىلەن مەنمۇ ئۆيۈمگە ماڭماق بۇلۇپ ئورنۇمدىن تۇردۇم  تۇيۇقسىز ئانامنىڭ ئېغىزى ئېچىلىپ قالدى   بالام كىسەلچان ئادەم يالغۇز ياتسا بولمايدۇ  جىن ئالىشىۋالىدۇ  بەش ئالتە كۇن يېنىمىزدا ياتقىن دىدى  مەن  ۋاي نىمدەيسە ئانا مەن كىچىك بالا بولمىسام  قوقمايمە دەپ ئانامنىڭ ياليالۋۇرشىغا قارىماي باغقا سېلىپ بەرگەن ئۆيۈمگە قاراپ ماڭدىم ئانام ۋە  سىڭىللىرىم كەينىمدىن چىقىپ ئۆيۈمگە ئوت قالاپ جايى سېلىپ چىقىپ كەتتى ئانام چىقىپ كىتىش ئالدىدا بېشىمغا قۇرئەننى ئەكىپ قۇيۇپ  بىر نىملەنى ئوقۇپ سۇپ سۇپ دەپ قۇيۇپ چىقىپ كەتتى  كەچ سائەت 10 ئون بولدى دەرھال توكنى ئۆچۈرۈپ تېلپۇننى قۇلۇمغا ئالدىم دە يوتقاننىڭ ئچىگە كاللامنى تىقىپ تورلارنى چۆرگىلەپ نەۋەت ئىزدىنىش  مۇنبىرگە كەپ قالدى مۇنبەرگە كىردىم قارىسام  ماۋۇ جىن توغۇرلۇق تىمىغا كۆزۈم چۈشۈپ قالدى شۇنىڭ بىلەن تىما ۋە ئنكاسلارنى ئوقۇشقا باشلىدىم ئوقۇغانسىرى قورقۇپ يۈرۈگۈم گۈپۈلدەشكە باشلىدى  ئانناڭنى قورقۇپ تىتىرەشكە باشلىدىم ھېچنىمىدىن ھېچنىمە يوق  ئۆي قۇپ قاراڭغۇ  تېخىمۇ سۈرلۈك بوپ كەتتى  ئادەم قورقۇنۇشلۇق ھىكايا رڭمانلارنى مۇشۇ كىچىدە قاپ قاراڭغىدا ئوقىسا ئۈنىمى باشقىچە بولىدىكەن  قورققۇلتۇغانسىرى قاپ قاراڭغۇ ئۆينىڭ بۇلۇڭ پۇچقاقلىرىدىن كۆزۈمگە غەلىتە بىرنەرسىلە كۆرۈنگىلى تۇردى ئاستا ئورنۇمدىن تۇرۇپ توكنى ياندۇردۋالدىم ھېچ ئۇخلايدىغاندەك تۇرمە بىر ۋاقىتتا سىيگۈم كەپ قالدى لىكىن سىرتقا چىقىشتىن قورقۇپ جىملا ياتتىم كاللامدا ھەرخىل قورقۇنۇچلۇق ججىنلانىڭ خىياللىرى قانچە ئويلىغانسىرى شۇنچە قورقۇۋاتىمە  «لاپ»  تۇيۇقسىز توك ئۆچۈپ قالدى يۈرۈگۈم «قارتلا» قىپقىپ سالدى قاراڭغۇدا قورقۇنۇچۇم تېخىمۇ كۈچەيدى  ئچىمدە نەچچىنى تىللىۋەتتىم توك ئدارسىدىكىلەرنى  سىيگۇم تېخى كىلىدۇ ئىندا   بىرچاغلاردا تاشقىرقى ئۆينىڭ ئشكى «دارققىدە» قىلمامدۇ ئاھ خۇدا  شۇ ۋاقىتتىكى قورقۇنۇشنى ئۆزۈملا بىلىمەن بەدىنىم تىتىرە ۋاتىدۇ  ئۆي تىپ تىنىچ پۈتۈن دىققىتىم بىلەم تاشقىرقى ئۆيدىكى شەپىگە دىققەت قىلىۋاتىمەن.........

تاشقىرقى ئۆيدە بىرنەرسىنىڭ شىپىرلىغان يەردە ماڭغان ئاۋازى ناھايتى بوش ئاڭلىنىۋاتىدۇ  ھەيي  ئانامنىڭ گىپىنى ئاڭلىمىغىنمغا مىڭ  پۇشايمان قىلىدىم  بىر ۋاقىتتا مەن ياتقان ئۆينىڭ ئشكى غىچچىچچچچچچچر قىلپ ئشىك ئېچىلغاندەك قىلدى تىلىم ئاغزىمغا تىقىلدى تۈرۈگۈم سوقۇۋاتامدۇ يا توختىدىمۇ ئۇقمايمە  بىر نەرسە شىپىرلاپ كىلىپلا دوككىدە قىلىپ سۆكىگە چىقتى  غال غال تىتىرەپ كەتتىم قۇتقۇزۇڭلا دەپ توۋلۇغۇم كەلدى لىكىن نمبولدىكىن توۋلىيالمىدىم  ئاستا ئاستا  يەنە ئاستا ماڭغان ئاۋاز ماڭا يېقىنلاپ كىلىۋاتىدۇ   ئاستا ياستۇققا چىققىاندەك قىلدى تۇيۇقسىز يۈزۈمگە ناھايتى يۇڭلۇق بىر نەرسە سۈركەلدى  ئاھ خۇدا ئەمدى مەن ئۆلدۈم  يۈزۈمنى بىر نەرسە يالىغاندەك قىلدى ئاندىن تېىخىمۇ قاتتىق سۈرۈلۈپ غىدىقىمنى كەلتۈرۈپ قوينۇمغا كىرمەكچى بولدى دەل شۇ ۋاقىتتا توك ياندى  ھەي  مىنىڭ ئاسلىنىم  ئىكەنغۇ  توۋا مىنى ئۇنتۇپ قالماپتۇ دە نەچچە كۇندىن بۇيان دوختۇرخانىدە ياتسام  پۇتۇن بەدىنىم چىپ چىپ تەرلەپ كىتىپتۇ مۈشۈكۈم تۈرتىلىپ كىرىپ قوچىقىمغا كەپلىشىپ  قۇساقلىرىمنى ئۇ يە بۇ يەنى يالاپ قۇيۇپ خورەك تارتىشقا باشلىدى شۇنداق قىلىپ بىر قىتىملىق جىن چۈشى ئاخىرلاشتى  شۈكرە خۇدايىم دەپ ئاستا يانغا ئۆرۈلدىم  بىراق كۆرپە ئەمدى ئۆلدەك تۇيۇلدى مۇندا سىلىسام ......ھەي سىيۋېتىپتىمە ئەمەسمۇ
