<script>
var limit="0:300"
if (document.images){
var parselimit=limit.split(":")
parselimit=parselimit*60+parselimit*1}
function beginrefresh(){
if (!document.images)
return
if (parselimit==1)
window.location.reload()
else{
parselimit-=1
curmin=Math.floor(parselimit/60)
cursec=parselimit%60
if (curmin!=0)
curtime=curmin+"M"+cursec+"S"
else
curtime=cursec+"New"
window.status=curtime
setTimeout("beginrefresh()",1000)
}}
window.onload=beginrefresh
</script>

_____________________

var limit="0:300"

بۇنىڭدىكى 300 سېكونتنى بىلدۇرىدۇ.
300 سېكونت توشقاندىن كىيىن بەت ئۆزلىكىدىن يېڭىلىنىدۇ.

curtime=curmin+"M"+cursec+"S"
else
curtime=cursec+"New"بۇنىڭدىكى S  سېكونت، M مۇنتنى ، New يىڭىلاشنى بىدۇرىدۇ.

ھالەت سىتونىدا 4m 59 S  دپ كورىنىدۇ. بۇ دىگەنلىك 4 مىنۇت 59 سېكونتتا يىڭىلىنىدۇ دىگەن مەنىنى بىلدۇرىدۇ.
