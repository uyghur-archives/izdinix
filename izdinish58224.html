ئىزدىنىش مۇنبىرى › مەملىكەت خەۋەرلىرى › جۇڭگو كۈچلۈك دۆلەتكە ئايلىنىش ئۈچۈن چوقۇم قىلىدىغان بەش مەيدان جەڭ

osmanuyghur يوللانغان ۋاقتى 2014-5-2 21:10:52

جۇڭگو كۈچلۈك دۆلەتكە ئايلىنىش ئۈچۈن چوقۇم قىلىدىغان بەش مەيدان جەڭ

جۇڭگو كۈچلۈك دۆلەتكە ئايلىنىش ئۈچۈن چوقۇم قىلىدىغان بەش مەيدان جەڭ

جۇڭگو ئۈزلۈكىسز تۈردە، تىنىچلىق بىلەن تەرەققىي قىلىشنى مەقسەت قىلىپ، ئۇرۇشنىڭ خەتىرىدىن ساقلانماقتا، ئەمما خەلقئارا ۋەزىيەتنىڭ ئۆزگىرىشى ۋە جۇڭگو ئەتراپى ۋەزىيتىنىڭ مۇقىم بولماسلىقىغا ئەگىشىپ، بولۇپمۇ يېقىندىن بۇيان ئامېرىكا داۋملىق جۇڭگو ئەتراپىدىكى دۆلەتلەرنى قۇتىرتىپ، جۇڭگو بىلەن سۈركىش ھاسىل قىلىشىغا سەۋەپ بولىشى شۇنداقلا پاكىستاننىڭ تېرورچىلىق تەھدىتىگە ئۇچۇرشى قاتارلىق بىر قاتار ئامىللار جۇڭگونىڭ بۇندىن كېيىن ئۈزلۈكىسز تىنىچ تەرەققىي قىلىشىنىڭ مۇمكىن بولمايدىغانلىقنى ئىسپاتلايدۇ.
ئۇنداقتا، جۇڭگونىڭ ئۇرۇش ئارقىلىق تەرەققىياتىنى ئىلىگىرى سۈرۈشى زۈرۈر بولسا، تۆۋەندىكى بەش مەيدان ئۇرۇشنى چوقۇم قىلىشى كېرەك. بۇ ئارقىلىق ئامېرىكنىڭ سۈيقەستىنى بىتچىت قىلغىلى ۋە بۇ بەش مەيدا ئۇرۇش ئارقىلىق جۇڭگونىڭ دۇنياۋى كۈچلۈك دۆلەتكە ئايلىنىش ئارزۇسىنى رىئاللىقا ئايلاندۇرغىلى بولىدۇ.

1. جەنۇبىي تاقىم ئاراللىرىنى قايتۇرۋىلىش جىڭى


جەنۇبىي دېڭىز جۇڭگونىڭ تەرەققىي قىلىشىغا ئۈزلۈكسىز توسالغۇ بولۇپ كەلگەن مۇھىم بىر ئامىل، جەنۇبىي دېڭىزنىڭ ئىشغالىيەتچىلەر قولىدا بىر كۈن تۇرۇشى جۇڭگوغا ئېلىپ كەلگەن ھاقارەت بولۇپلا قالماستىن بەلكى ئەڭ مۇھىمى جۇڭگونىڭ توشۇش لىنىيسى بىخەتەرلىكى توسالغۇغا ئۇچرايدۇ.

جەنۇبىي دېڭىز تاقىم ئاراللىرىدا مول بولغان بايلىق بولۇپلا قالماستىن يەنە جۇغراپىيەلىك ئۇرنىمۇ ئىنتايىن مۇھىم. جەنۇبىي دېڭىز ئاراللىرنى قايتۇرۋىلىش ياكى قايتۇرۋالالماسلىق جۇڭگونىڭ سىجىل تەرەققىيات قىلىش -قىلالماسلقى ۋە كۈچلۈك دۆلەت بولۇش - بولالماسلىقىنىڭ مۇھىم ئامىلى.

جۇڭگو جەنۇبىي دېڭىز تاقىم ئاراللىرىنى قايتۇرۋالالىسا، ئۇنداقتا جەنۇبى قۇڭگو دېڭىزى ھەقىقى مەنىدە جۇڭگونىڭ ئىچكى دېڭىزىغا ئايلىنىدۇ، ۋاقتى كەلگەندە جەنۇبى دېڭىزدىكى ئىگىلىك ھوقۇققا كاپالەتلىك قىلىپ قالماستىن يەنە جەنۇبىي دېڭىزغا ئەركىن چقىپ - كېرىش ئۈچۈن شەرت - شارائىت ھازىرلانغان بولىدۇ.
شۇڭلاشقا جۇڭگو يېڭى دۇنيادا كۈچلۈك دۆلەتكە ئايلىنىش ئۈچۈن بىرىنچى مەيدان ئۇرۇش جەنۇبىي دېڭىز ئۇرۇشدا چوقۇم غەلبە قىلىشى كېرەك.


2. تەيۋەننى ئازات قىلىش


كۆپچىلككە مەلۇم تەيۋەن جۇڭگۇنىڭ ئايرىلماس بىر قىسمى، گەرچە ھۆكۈمەت داۋاملىق تىنىچ يول بىلەن ھەل قىلىشنى تەكىتلەپ كىلۋاتقان بولسىمۇ، ئەمما تەيۋەن ئىچىدىكى <تەيۋەن مۇستەقىللىقى> كۈچلىرى نىڭ ئۈزلۈكسىز تەرەققىي قىلىشى بىلەن تىنىچ يول ئارقىلىق ھەل قىلىش ئۈمىدى كۆپۈككە ئايلانماقتا. ھازىرقى تەيۋەن جۇڭگونىڭ بىر پارچە يۈەك كېسىلى بولۇپلا قالماستىن يەنە ئامېرىكىنىڭ جۇڭگونىڭ بىخەرلىكىگە تەھدىد سىلىشتىكى بىر پىچكىسى. تەيۋەننى ئازاد قىلىش دۆلەتنىڭ زىمىن پۈتۈنلىكى قوغداپ قالماي قالماسىتىن يەنە جۇڭگونىڭ ھەقىقىي تۈردە كۈچلۈك دۆلەتكە ئايلىنىشى ۋە دۆلەت ئىچىدىكى خەلقنىڭ قەلبىنى بىر يەرگە يغقىلى بولىدۇ.


3. شىزاڭنىڭ جەنۇبىنى قايتۇرۋىلىش ئۇرۇشى


ھىندىستان يىقىنقى ئون نەچچە يىلدىن بۇيان ھەممە يەردە جۇڭگو بىلەن قارشىلىپ، جۇڭگونى ئۆزىنىڭ ئەڭ چوڭ دۈشمىنى دەپ قاراپ كەلدى. ھەمدە جۇڭگونىڭ ئىچكى قىسمىدىكى بۆلگۈنچى كۈچلەرنى قوللاپ، جۇڭگونىڭ ئچىدە تىرورلۇق پائالىيەتلەرنىڭ پارتلىشىغا سەۋەبچى بولدى. يەنە كېلىپ شىزاڭنىڭ جەنۇبىنى ئىشغال قىلىپ تۇرماقتا، مۇنداقچە ئيتقاندا ھىندىستان، جۇڭگونىڭ ئامېرىكىدىن قالسىلا ئەڭ چوڭ سىرتقى تەھدىتى ھىسابلىنىدۇ.

يىقىندىن بۇيانقى ھىندستاننىڭ دىپلوماتىيە سىياسىتىدىن شۇنى كۆرۋىلىشقا بولىدۇكى، ھىندىستان تەرەپ ئۈزلۈكسىز ھالدا ھەربىي تەرەققىياتنى ئىزلەپ ھەمدە ئامېرىكا بىلەن جىننىڭ بارىچە جۇڭگونى چەكلەشكە ئۇرۇنۋاتىدۇ. يەنە كېلىپ ئامېرىكىمۇ ھىندىستاننىڭ جۇڭگونىڭ تەرەققىياتىنى چەكلەشتە مۇھىم قورال بوپ بەرگەنلىكىدىن خۇشال بولۋاتىدۇ.
ھىندىستاننىڭ ھەربىي كۈچىنىڭ يۇقىرى كۆتۈرلىشىگە ئەگىشىپ، جۇڭگونىڭ غەربىي جەنۇب چىگىرا رايونلىرى تەھدىتكە ئۇچراۋاتىدۇ. شۇڭا شىزاڭنىڭ جەنۇبىنى قايتۇرۋىلىش جۇڭگونىڭ زىمىن پۈتۈنلىكىنى ئەسلىگە قايتۇرۇپلا قالماستىن يەنە ھىندىستانغا ئەجەللىك زەربە بەرگىلى، ھىندىستاننىڭ جۇڭگونىڭ غەربىي جەنۇب چىگرا رايونلىرىغا تەھدىد ئىلىپ كىلىشىدىن ساقلانغىلى بولىدۇ.

4.ھىندى ئوكيان دېڭىز ئۇرۇشى


جۇڭگو ھىندىستان بىلەن قۇرۇقلۇقتا ئۇرۇش قىلىغان ۋاقىتتا، يەنە ھىندىستان چوقۇم ھىندى ئوكياننى قامال قىلىدۇ، شۇنداقلا جۇڭگو دېڭىز ئارمىيسى بىلەن دۇئىلغا چۈشۈش پۇرسىتى ئىزدەپ، جۇڭگو دېڭىز ئارمىيسىنى يوقىتىشقا تىرىشىدۇ. ئوخشاشلا جۇڭگو دېڭىز ئارمىيسى جۇڭگونىڭ دېڭىز يولى بىخەتەرلىكى ئۈچۈن چوقۇم ھىندىستان دېڭىز ئارمىيسىنى مەغلۇپ قىلىدۇ. مەزكۇر ئۇرۇشتا ئەگەر جۇڭگو يەڭسە، ئۇنداقتا پۈتۈنلەي ھالدا ئامېرىكنىڭ قورشاپ يۇقىتشىدىن قۇتۇلۇپ، ئامېرىكىغا بىۋاستە تاقابىل تۇرۇشقا كۈچ ھازىرلايدۇ.


5. شەرقىي دېڭىز ئۇرۇشى


ياپونىيە جۇڭگونىڭ ئۇدا ئەشەددى دۈشمىنى بولۇپ كەلدى. ياپونىيە ئامېرىكا بىلەن ھەريەردە بىر سەپتە تۇرۇپ جۇڭگونى بسىپ كەلدى، يەنە كېلىپ دىياۋيۈ ئارىلى ئىگىلىك مەسىلىسى ئوتتۇرسىدا تالاش تارتىش مەۋجۇت. ياپونىيە - ئامېرىكا ئارمىيسى ئىتتىپاقىنىڭ كۈندىن كۈنگە كۈچىيشى ۋە ياپونىيە ئارمىيە كۈچىنىڭ ئۈزلۈكسىز يۇقىرى كۆتۈرلىشى سەۋەبىدىن جۇڭگونىڭ شەرقىي دېڭىز رايونلىرى كۈنسىرى تەھىدىدكە ئۇچراپ كەلدى.

جۇڭگونىڭ شەرقىي دېڭىزدا ئۇرۇش قىلىشى دىياۋيۈ ئارىلىنىڭ ئىگىلىك ھوقۇقىنى قايتۇرۋىلىپلا قالماسىتىن يەنە ريۇكۇ تاقىم ئاراللىرنى ئۆزىنى ئۆزى باشقۇرۇش پۇرسىتىگە ئىگە قىلىدۇ. ريۇكۇ تاقىم ئاراللىرى تارىختىن بۇيان جۇڭگوغا تەۋە جايلار بولسىمۇ ئەمما جۇڭگونىڭ قوينىغا قايتىپ كىلىشى رىئاللىققا ئايلانمايدۇ. ئەمما ريۇكۇ ئوخشاشلا ياپونغا تەۋە ئەمەس، ريۇكۇنىڭ ئاپتونۇمنى تەلەپ قىلىشىغا قارىتا خەلقئارا جەمئىيەت قارشى تۇرۇپ كەتمەيدۇ، يەنە كېلىپ جۇڭگونىڭ پايدا - مەنپەئىتىگە ماس كېلىدۇ؛ ريۇكۇ جۇڭگودىن ئاپتونۇم بولۇپ چىقىپ كەتسە، جۇڭگو ريۇكۇ دېڭىز رايونى ئارقىلىق غەربى تىنىچ ئوكيانغا ئەركىن - ئازادە چىقىپ كىرەلەيدۇ.

ئەلۋەتتە، جۇڭگونىڭ ئەڭ ئاخىرقى مەقسىتىمۇ ياپوننىڭ دېڭىز - ھاۋا ئارمىيە كۈچىنى تەلتۈكۈس بەربات قىلىش ئارقىلىق ئامېرىكنى غەربى تىنىچ ئوكياندىن چقىپ كىتىشكە قىستاش، بۇ ئارقىلىق جۇڭگو غەربىي تىنىچ ئوكياندىكى دېڭىز يوللىرىنى پۈتۈنلەي كونتىرول قىلالايدۇ.
