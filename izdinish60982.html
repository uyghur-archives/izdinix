بۇ يازمىنى ئاخىرىدا   tadubeg تەھرىرلىگەن. ۋاقتى2014-11-30 15:08


شىپاخانىمىزنىڭ دورىلىرى
زەفىران ئەرلەر قۇۋۋەت جەۋھىرى
بۇ مەھسۇلات ئەنئەنىۋى تىبابەتنىڭيۇمران بۇغا مۈڭگۈزى،ئامرىكا ئادەمگىياسى ،سۆئلەپ مىسرى، سەقەنقۇر، يىڭنە بېلىق، ئىپار، زەپەر قاتارلىق قىممەتلىك دورا ماتىرىياللىرىدنئۇيغۇر تېبابىتى دورىگەرلىك نەزەرىيەسى بويىچە تەييارلانغان بولۇپ ،ئېمۇنت كۈچىنى يۇقىرى كۆتۈرۈش، قاننى جانلاندۇرۇش،روھى كەيپىياتنى ياخشىلاش ،يۈرەك،مېڭە،جىىگەر،بۆرەك قاتارلىق ئەزالارنى قۇۋۋەتلەش،ھەر خىل سەۋەبلەردىن بولغان جىنسى ئاجىزلىقنى ياخشىلاش،مەنى ۋە ئۇنىڭ تەركىبىدىكى ئۇرۇقنى كۆپەيتىش تەسىرلىرى بولۇپ، ئەرلەردىكى ھەر خىل سەۋەبلەردىن بولغان جىنسى ئاجىزلىق، مەنى كەملىك،بالدۇر كېتىش،پەرزەنتلىك بولالماسلىق،ماغدۇرسىزلىق قاتارلىقلارغا ئىشلىتىلىدۇ.
ئالى نۇسخىلىق لوبوبى كەبىر ياكى جاۋارىش زەرئۇنى ئەنبىرى بىلەن بىرگە ئىشلەتسە ئۈنۈمى تېخىمۇيۇقىرى بولىدۇ.

مىقدارى 400 گرام، مۇقىم باھاسى 600 يۈئەن.

زەفىران ئاياللار قۇۋۋەت جەۋھىرى
بۇ جىگدە چېچىكى، نارگۈل ئۇرۇقى ، رىگماھى، پەنىرمايە،مەرۋايىت،زەپەر،ئەنبەر قاتارلىقلاردىن تەييارلانغان بولۇپ، ئىمۇنت كۈچىنى يۇقىرى كۆتۈرۈش، بالىياتقۇنى قۇۋۋەتلەش،ياللۇغ قايتۇرۇپ،ئۆسمىگە قارشى تۇرۇش،قاننى جانلاندۇرۇش تەسىرلىرى بولۇپ، بالىياتقۇنى قىزىتىپ، سوغۇق يەلنى تارقىتىدۇ، پەي-مۇسكۇللارنى چىڭىتىپ، ھۆللۈكنى قۇرۇتىدۇ، ئاق خۇننى توختىتىدۇ، ئادەت قالايمىقان بولۇش، ئاق خۇن كۆپ كېلىش، تۇغماسلىق ۋە ئاياللاردىكى جىنسى ئاجىزلىققا ئىشلىتىلىدۇ.
ئاياللار گۆھىرى بىلەن ئىشلەتسە ئۈنۈمى يۇقىرى بولىدۇ.
مىقدارى 400 گرام، باھاسى 600 يۈئەن.

مېڭە قۇۋۋەتلەش جەۋھىرى
بۇ ئامرىكا ئادەمگىياسى، چىلغوزا مېغىزى،سۆئلەپ مىسرى، مەرۋايىت،ئالتۇن ۋەرەق، كۈمۈش ۋەرەق،مەرۋايىت،زەھرە مۆھرە،ياقۇت،ئەنبەر،ياڭاق مېغىزى قاتارلىقلاردىنبولۇپ، مەركىزى نېرۋا سېستىمىسىنى قۇۋۋەتلەيدۇ،ھېسسى قۇۋۋەتنى ئاشۇرىدۇ، ئىچكى تاشقى سەزگۈلەرنى كۈچەيتىدۇ، يۈرەك بىئارام بولۇش، ئىچى سىقىلىش، چۈشكۈنلۈك، ماغدۇرسىزلىق،ئۇيقۇسىزلىق،ئۇنۇتقاقلىق، تىرىككەكلىك،زىرىككەكلىك، ئىمۇنت كۈچى ئاجىزلىق،رەئىس ئەزالار ئاجىزلىقىغا ياخشى مەنپەئەت قىلىدۇ.
داۋائىلمىشكى ياكى ئەۋرىشىم شەربىتى بىلەن ئىچسە ئۈنۈمى تېخىمۇ يۇقىرى بولىدۇ.
مىقدارى 400 گرام، باھاسى 600 يۈئەن.

ئالى نۇسخىلىق لوبوبى كەبىر
بۇ سېمىزلەر ، بەدىنىنى ھۆللۈك ۋە سۇغۇقلۇق قاپلاپ كەتكەنلەر ئۈچۈنئاقىرقەرھا ھىندى، جەدۋار ھىندى،سەقەنقۇر، يۇمران بۇغا مۈڭگۈزى ،بۇقا چۈيىسى،قۇشقاچ مېڭىسى،زەپەر، كۈشتە مۇرەككەپ قاتارلىقلاردىن مەخسۇس تەييارلانغانبولۇپ، بەدەندىكى ئارتۇقچە ھۆللۈك ۋە سۇغۇقلۇقنى يوقىتىدۇ، يۈرەك، مېڭە، جىگەر، بۆرەك ئاجىزلىقلىرى ۋە شۇ سەۋەبلەردىن بولغان جىنسى ئاجىزلىقنىڭ ھەر قايسى تۈرلىرىگە كۆپ ئىشلىتىلىدۇ. ئاجىزلىقنى يوقىتىپ، ئۇمۇمى بەدەننىڭ قۇۋۋىتىنى يۇقىرى كۆتۈرىدۇ. مەنى كۆپەيتىدۇ ۋەئۇرۇق سۈپىتىنى يۇقىرى كۆتۈرىدۇ. ئاياللارغا ئىشلىتىلسە بالىياتقۇ خىزمىتىنى ياخشىلاپ ھامىلدارلىقنى ئىلگىرى سۈرىدۇ، ئاق خۇننى تىزگىنلەيدۇ، قان ئازلىق ، بەل – دوۋسۇن ساھىسى يېقىمسىزلىنىش، ھەۋەس سۇس بولۇش قاتارلىقلارغا مەنپەئەت قىلىدۇ.
تۇغماسلىققا قارىتا زەپىران قۇۋۋەت جەۋھىرى ، جاۋارىش زەرئۇنى ئەنبىرى قاتارلىقلار بىلەنبىرگە بېرىلىدۇ.
مىقدارى 400 گرام، باھاسى 400 يۈئەن.

جاۋارىش زەرئۇنى ئەنبىرى
بۇ سۆئلەپ مىسرى، بۇقا چۈيىسى، ئوغرىتىكەن،كەرەپشە ئۇرۇقى، چىلغوزا، ئاقىرقەرھا ھىندى، زەپەر، ئەنبەر ئەشھەب قاتارلىقلاردىن تەييارلانغانبولۇپ، بۆرەكنى قۇۋۋەتلەش ۋە ئۇنىڭ سۈزۈش خىزمىتىنى ياخشىلاش، بۆرەك زەئىپلىكى،سۇغۇق ئۈتۈشسەۋەبلىرىدىن بولغان بەل ئاغرىقلىرىنى يوقىتىش، باھ زەئىپلىكىگە مەنپەئەت قىلىپ، شەھۋەتنى كۈچلەندۈرۈش، مەنىينىڭ سۇۋۇپ ۋە سۇيۇلۇپ كېتىشىنى ياخشىلاپ مەنىينى قىزىتىش، قۇيالدۇرۇش، كۈپەيتىش،تىز كېتىشىنى تىزگىنلەش، بەدەن ھۆللۈكلىرىنى قۇرۇتۇش، ئاشقازاننى قىزىتىپ ، يەل تارقىتىش قاتارلىق تەسىرلىرى بولۇپ، بۆرەكنى قۇۋۋەتلەش مەقسىتىدە كۆپ ئىشلىتىلىدۇ.
مىقدارى 400 گرام. باھاسى 400 يۈئەن

مادەتۇلھايات كەبىر
بۇ قارىمۇچ، ئاقىرقەرھا ھىندى، زىراۋەندە،شاترەنجى،چىلغوزا مېغىزى، سۆئلەپ مىسرى، زەپەر قاتارلىقلاردىن تەييارلانغان بولۇپ، مېڭە، بۆرەك ، ئاشقازاننى قۇۋۋەتلەش، قاننى جانلاندۇرۇش، زىيادە ھۆللۈكلەرنى قۇرۇتۇش، مەي-مۇسكۇللارنى چىڭىتىش، جىنسى قۇۋۋەتنى ئاشۇرۇش، ئۇمۇمى بەدەننى قۇۋۋەتلەش خۇسۇسىيتىگە ئىگە بولۇپ، مېڭە، باھ، بۆرەك ۋە ئۇمۇمى بەدەن ئاجىزلىقى، بەل، پۇت-قوللار ئاغرىش، بۆرەك سۇۋۇپ كېتىش، مەنى ۋە سۈيدۈك تۇتالماسلىق، دېئابىت، پالەچ، رېماتىزىم قاتارلىقلارغا ئىشلىتىلىدۇ.
مىقدارى 250 گرام، باھاسى 100 يۈئەن.

لوبوبى سەغىر
بۇ شاقاقۇل مىسرى، ئاقلانغان بادام مېغىزى، چىلغوزا مېغىزى، چىگىت مېغىزى، توشقان زەدىكى، سەۋزە ئۇرۇقى، خەشخەش ئۇرۇقى، قارا كۈنجۈت ،ئۈزۈم شىرىسى قاتارلىقلاردىن تەييارلانغانبولۇپ، سۈت ۋە مەنى كۆپەيتىدۇ، بۆرەك ۋە بەدەننى سەمرىتىدۇ، مېڭىنى قۇۋۋەتلەيدۇ.بەدەندىكى قۇرۇقلۇقنى يوقىتىدۇ، ئاساسلىقى ئەر-ئاياللاردىكىئۇرۇقلۇق،بۆرەك زەئىپلىكى سەۋەبىدىن بولغان جىنسى ئاجىزلىق، مەنى ئازلىق،قان ئازلىق، سۈت كەملىك، مېڭىگە قۇرۇقلۇق يېتىش قاتارلىقلارغاياخشى مەنپەئەت قىلىدۇ.
مىقدارى 400 گرام، باھاسى 120 يۈئەن.

مەجۈنى ئەقرەب
بۇ كاكىنەچ، جىنتىيانا، چايان، قۇندۇزقەھرى، جۇۋىنە قاتارلىقلاردىن تەييارلانغان بولۇپ، بۆرەك، دوۋساق، سۈيدۈك يوللىرىدىكى تاشنى ئېرىتىش، چۈشۈرۈش، ياللۇغ قايتۇرۇش،سۈيدۈك ھەيدەش، زەھەر قايتۇرۇش قاتارلىق تەسىرلەرگە ئىگە بولۇپ، بۆرەك ۋە دوۋساققا تاش چۈشۈش، سۈيدۈك يولى ياللۇغى، مەزى بېزى ياللۇغى ۋە مەزى بېزى يوغىناش، سۈيدۈك تۇتۇلۇش قاتارلىقلارغا مەنپەئەت قىلىدۇ.
جاۋارىش زەرئۇنى، ياللۇغ قايتۇرۇش چايلىقى قاتارلىقلار بىلەن ئىشلىتىلسە ئۈنۈمى تېخىمۇ يۇقىرى بولىدۇ.
مىقدارى 135گرام، باھاسى 100 يۈئەن.

جاۋارىش مەرۋايىت

بۇ مەرۋايىت، بەسباسە،زەنجىبىل، دارچىن، سېرىقماي قاتارلىقلاردىن تەييارلانغان بولۇپ، تالنى قۇۋۋەتلەيدۇ، ئاشقازاننى قىزىتىدۇ، سوغۇقتىن بولغان ئاشقازان ئاغرىقى، ئىچى سۈرۈش، ئىشتىھاسىزلىق، ئاشقازان-ئون ئىككى بارماق ئۈچەي يارىسى،ئادەتلەنگەن بويىدىن ئاجراشقاتارلىقلارغا ياخشى مەنپەئەت قىلىدۇ.
مىقدارى 400 گرام، باھاسى 200 يۈئەن.

داۋائىلمىشكى
بۇ كۈشتە مۇرەككەپ، دۇرنەج ئەقرەب، زۆرەنباد، بەھمەن، ئەۋرىشىم ، سازەج ھىندى، ئىپار قاتارلىقلاردىن تەييارلانغان بولۇپ، يۈرەك، مېڭە ۋە باھنى قۇۋۋەتلەش، روھنى كۈچەيتىپ كۆڭۈلگە خۇشلۇق كەلتۈرۈش، قاننى جانلاندۇرۇش، تەبىئى ھارارەتنى كۈچەيتىش، قان تومۇرلاردىكى توسالغۇلارنى ئېچىش، ھۆللۈكلەرنى قۇرۇتۇش تەسىرى بولۇپ، يۈرەك مىزاجىنىڭ سۇغۇقتىن بۇزۇلۇشى، نورمالسىز سەۋدا ، بەلغەم خىلىتلىرىنىڭ تەسىرىدىن كېلىپ چىققان يۈرەك مېڭە قان تۇمۇرلىرى تۇسۇلۇش، يۈرەك زەئىپلىكى، يۈرەك چالاڭغۇراش، يۈرەك – مېڭىگە قان يېتىشمەسلىك، قان بېسىمى تۆۋەنلەش، باش ئاغرىقى، روھى خۇنۇكلىشىش ۋە يۇقارقى سەۋەبلەردىن بولغانباھ ئاجىزلىقىغا ئىشلىتىلىدۇ.
ئەۋرىشىم شەربىتى بىلەن ئىچىلسە ئۈنۈمى يۇقىرى بولىدۇ.
مىقدارى 135 گرام، باھاسى 150 يۈئەن.

ئاياللار گۆھىرى
بۇ نارگۈل، جىگدە چېچىكى، رىگماھى، بۇقا چۈيىسى، يۇمران بۇغا مۈڭگۈزى، ئاقىرقەرھا ھىندى، زەپەر ، ئادراسمان ئۇرۇقى قاتارلىقلاردىن تەييارلانغان بولۇپ، قاننى جانلاندۇرۇپ، بالىياتقۇنى قىزىتىش، سۇغۇق يەلنى تارقىتىش، پەي-مۇسكۇللارنى چىڭىتىپ، ھۆللۈكنى قۇرۇتۇش تەسىرلىرى بولۇپ، ئاياللاردىكى جىنسى ئاجىزلىققامەخسۇس ئىشلىتىلىدۇ.زەفىران ئاياللار قۇۋۋەت جەۋھىرى بىلەنبىرگە ئىشلىتىلسە ئۈنۈمى يۇقىرى.
بىر قۇتىسى 250تال(كۇمىلاچ) باھاسى 100 يۈئەن.
