مۇزىكا بوشلۇقى بىلەن ھەقسىز تەمىنلىگەن سۇبات تورىغا رەھمەت ئېيتىمەن

تۇغۇلغاندا يالغۇز تۇغۇلغان، يەرشارىدا ماڭا ئوخشاشتىن يەنە بىرى يوق. رەببىم مىنى ياراتقاندا مىنىڭ ۋۇجۇدۇمغا سىلەر ياقتۇردىغان نۇرغۇن خىسلەتلەرنى، سىلەر ياقتۇرمايدىغان نۇرغۇن ئىللەتلەرنى قۇشۇۋېتىپتىكەن. ياشاش جەريانىدا يەنە نۇرغۇن ئۆزگىرىشلەرنى قىلدىم.
نۇرغۇن ئېسىل ئەخلاقلارنىڭ ساھىبى بولدۇم، نۇرغۇن قاملاشمىغان قىلىقلارنى ئۆگەندىم.
مىنىڭ ۋۇجۇدۇمدىن خۇشاللىق، قايغۇ ،  ۋاپا، يۈزسىزلىك، سەمىمىيلىك، راستچىللىق، قورقۇنچاقلىق، باتۇرلۇق، مۇھەببەت، نەپرەت، تىرىشچانلىق، ھورۇنلۇق ئەقىللىقلىق، دىۋەڭلىك ھەممىسى تېپىلىدۇ. ئۆزۈم خۇشال بولالايمەن، ئۆزۈم يىغلىيالايمەن. يەنە يېشىمنى سۈرتەلەيمەن ياكى سۈرۈتمەيلا كۈلەلەيمەن.
ھايات، تۇرمۇش، جەمىيەت مىنى ھەرخىل سورۇنلاردا، ھەرخىل روللارغا چىقىرىپ تىياتىر قۇيىدۇ. ھەرخىل ھارۋىلارنى قېتىپ، ھەرخىل داۋانلارغا ئىتتىرىدۇ.
بەزىدە چىرايىمدىن تەبەسسۇم كۆرسىڭىز بەزىدە قايغۇ كۆرىسىز.
بەزى ياراملىق بولغىنىمنى ھېس قىلسىڭىز ،بەزىدە لەقۋالىقىمنى ھېس قىلىپ قالىسىز.
سىز بىلەن مەلۇم ئوخشاشلىق، مەلۇم مۇددىئا، مەلۇم ئىستەك سەۋەبىدىن ھاياتتا ھەمكارلىشىپ قېلىشىم مۇمكىن، ھاياتتىكى ھەرقانداق مۇناسىۋەت. ھەمكارلىقتىن باشقا نەرسە ئەمەس.  خۇددى «ئېلىپ، بە» دىگەندەك.
ماڭا سۆيگۈبەرسىڭىز، مەن سىزگە ئۆزۈمنى بېغىشلايمەن. سىزگە ھەمدەمدە بولغان بولسام، سىز ھېچ بولمىغاندا مەن ئۈچۈن دۇئا قىلىسىز.
مەن سىزنىڭ ھاياتتىكى ھەرقانداق نەرسىنىڭ بىر بەدىلى بولىدىغىننى بىلسىكەن دەيمەن.
بەزى بەدىلىنى ئاۋۋال تۈلەپ ئاندىن ئېرىشىسىز، بەزى ئاۋۋال ئېرىشىپ ئاندىن بەدىلىنى تۈلەيسىز. ئوخشىمايدىغىنى ئاۋۋال ئېرىشىپ ئاندىن تۈلىگەن بەدەل ئېغىرراق بۇلىدۇ.
شۇنىڭ ئۈچۈن ھاياتتىكى بىكارلىق نەرسىلەرگە يۈگۈرمەڭ. مەندىن پايدىنىمەن دەپمۇ ئاۋارە بولماڭ ! مەندىن قانچىلىك پايدىلانسىڭىز ئۇنىڭ ھەققىنى تۈلەيسىز، يا ماڭا، يا باشقىلارغا . . .
ئۆزۈمنىڭ يىغىسىنى يىغلاپ يۈرسەم ساراڭكەن دىمەڭ. چۈنكى مىنى ياراتقاندا يىغىنىڭ خۇرۇچلىرىنى ماڭا سېلىۋەتكەن. ھاياتتا مەلۇم ئۆلچەمدىكى يىغلاش ۋەزىپەم بار. يەنە كۈلۈپ يۈرگىنىمنى كۆرسىڭىزئېلىشىپ قاپتۇ دىمەڭ. ئوخشاشالا ھاياتتا كۈلۈش ۋەزىپەم بار
سىز بىلەن توي قىلىپ، ئۆي تۇتۇپ، بالا-چاقلىق بۇلىشىم مۇمكىن. بۇ دىگەن ئىككەيلەن ھەمكارلىشىپ بىر كېمىنى ھەيدەش دەپ ئاتىلىدۇ. ھەرگىزمۇ مەن كېمىدە ياتسام بۇ ياق سۆرەپ ئاپىرىدۇ دەپ ئويلاپ قالماڭ! سىزنىڭمۇ ئۆزىڭىز ئادا قىلىشقا تىگىشلىك مەجبۇرىيەتلىرىڭىز بار. بۇلارنى ئۆزىڭىز تۇنىسىڭىز، ۋە ئادا قىلسىڭىز مەن تۇلىمۇ ياخشى ئەر بۇلالايمەن. ھەم مىنىڭمۇ ئۆزۈمگە لايىق مەجبۇرىيەتلىرىم بۇلىدۇ.
ئەلۋەتتە بەزەن كەيپىيات، كىسەل، ئالدىراشچىلىق سەۋەبىدىن مەجبۇرىيەتلىرىڭىزنى ئادا قىلىپ بولالمىسىڭىز تەشەببۇسكارلىق بىلەن ياردەملىشىمەن ۋە توغرا چۈشىنىمەن.
بۇنداق ئەھۋاللار مەندە يۈزبەرسىمۇ، سىزنى شۇنداق قىلالايدۇ دەپ ئويلايمەن.
كەيپىياتىڭىز ناچار بولسا، ئاچچىقىڭىز كىلىپ قالغان بولسا، ئامال بار ماڭا كۆرسەتمەڭ.
ئۆزىڭىزنىڭ ئاچچىقى، كەيپىياتى، ھاياجىنى سەۋەپلىك ماڭا ئازار بەرمەڭ، ئۆزىڭىزنىڭ مەنىۋى داۋالغۇشلىرىنى ئۆزىڭىز بىر تەرەپ قىلىشنى ئۈگۈنىۋېلىڭ. ناھەق بىرىلگەن ئازەر ئۇزۇنغىچە قەلبنى مۇجۇيدۇ، گەرچە مەن سۈكۈت قىلغان تەقدىردىمۇ.
مەن بىرقېتىممۇ مەيلىلا دەۋەتمەيمەن، چۈنكى بىرىنچى قېتىمكەن، مەيلى دىسە كەينىدىكىسىنىڭ تېخىمۇ ئېغىر بۇلىدىغىننى بېشىمدىن ئۆتكۈزۈپ باققانمەن.

يىغىسىنى يىغلاپ، ئاچچىقنى بېسىپ تۈگۈتىۋېتىڭ. چۈنكى، مەن ماسلىشالماي قالسام ئوت ئۈستىگە ياغ چېچىلىپ كىتىشى مۇمكىن.
تۇرمۇشتا بارچە جاپا- مۇشەققەتلەرنى كۈتۈرۈپ كەتكىلى بولسىمۇ. ئەمما، يېنىڭىزدىكىلەرنىڭ كەيپىيات تۇراقسىزلىقىغا ماسلاشماق مۈشكۈلدۇر، بۇلۇپمۇ بۇ ماسلىشىشنى داۋاملاشتۇرۇش . . .
مىنىڭ كەمچىلىكلىرىمنى كۆرگەندە ئەيىپلەپ، پىخىلداپ كۈلمەڭ، ئارتۇقچىلىقىمنى كۆرگەندە ھاياجانلىنىپ ياش تۈكۈپمۇ يۈرمەڭ. بۇنداق ئەھۋاللار مىنىڭ ۋۇجۇدۇمنىڭ ھەرخىل شەكىلدە زاھىر بۇلىشىدۇر.
ئىنساننىڭ جىسمىدا ئادىمىيلىك قىممەت، پەرىشتىلىك خىسلەت، ھايۋانلىق ئىللەتلەر بۇلىدۇ. ئادەم ماكان، زامان تەرپىدىن ھەرخىل شەكىلدە زاھىر بۇلىشى مۇمكىن. شۇنداقلا ئادەم ھاياتىدا  ئادەممۇ، پەرىشتىمۇ، ھايۋانمۇ بۇلۇپ ياشاپ باقىدۇ. ئۈستىدە دىگەندەك ياراتقاندا ئۇنىڭ مەۋجۇتلىقىغا قېتىۋىتىلگەن ئىللەتلەر ھامان بىركۈنى پۇرسەت تاپقاندا مۆڭرەپ چىقىدۇ.
مىنىڭ پالانى ئۈلىما، پۇكۇنى ھاجىم، قارىم، مۇئەللىم، ئاكام، ئۇكىللىرىمغا سېلىشتۇرماڭ، ئۇلارنىڭ خۇرۇچى باشقا، خۇيى باشقا، ئامراق، ئۆچ نەرسىللىرى باشقا، باشقا ۋاقىتتا، باشقا بىر ھالەتتە تۇغۇلغان. ئۇلارنىڭ ماڭا مەلۇم دەرىجدە بىردەكلىكى مىنىڭ ئۇلارنىڭ قېلىپىغا چۈشىشىمنى ياكى ئۇلارنىڭ مىنىڭ قېلىپىمغا چۈشىشىنى بىلدۈرمەيدۇ. بىز ھەممىمىز ئايرىم، ئايرىم بىر دۇنيادۇرمىز. بىزنى بىر نىشانغا يىتەكلىيەلىشىڭىز مۇمكىن. ئەمما، بىر خىل ھېسياتقا، بىرخىل مىجەزكە ئەكىلەلمەيسىز.

بىر تەرەپتىن ئېيتقاندا مەن قىزىرىپ پىشقان چىلانغا ئوخشايمەن. يەنە مىنىڭ تىكىنىممۇ          يىتىمەن. ئەمما، چىلاننى ئۈزەلمەي تىكىنىمگە سانچىلىۋېلىپ قايغۇرۇپ يۈرسىڭىز بۇ مەندىكى گۇناھ ئەمەس. ئەتىراپىڭىزدىكىلەرنى مىۋىسىنى قانداق ئۈچۈپ يىيىشنى بىلىشمۇ بىر پاراسەت.
ئۆزىىڭىزنىڭ كۈلكىسىدە كۈلۈڭ، ئۆزىڭىزنىڭ يىغىسىدا يىغلاڭ. باشقىلارنىڭ كۈلكىسىگە ئۆزىڭىزنىڭ يىغىسىنى سېلىشتۇرماڭ. شۇنداقلا باشقىلارنىڭ يىغىسىغا ئۆزىڭىزنىڭ كۈلكىسىنىمۇ سېلىشتۇرماڭ. ئىككىلىڭلارنىڭ يىغىسى ۋە كۈلكىسى ئوخشاش. پەقەت سىز كۆرگەن ۋاقتى ئوخشاش ئەمەس. سىز ئۇنىڭ يا يىغىسىنى، يا كۈلكىسىنى كۆرەلىدىڭىز خالاس.
دېڭىز ساھىلىدا ئولتۇرۇپ ئىسسىنغان قۇياش نۇرى بىلەن ئۆگزىدە ئولتۇرۇپ ئىسسىنغان قۇياش نۇرى ئوخشاش. دېڭىز ساھىلىغا بارغۇدەك مادارىڭىز بولمىسا ئۆگزىگە چىقىڭ.
دېڭىز سالىھىلىغا بېرىش ئۈچۈن، ئۆزىڭىزگە تەۋە بولمىغان، باشقىلارنىڭ مادارىغا چېقىلىپ يۈرمەڭ. بولمىسا ئاۋۋال سىز كۈلۈپ باشقىلار يىغلايدۇ. ئاندىن، باشقىلار كۈلۈپ سىز يىغلايسىز.
كۈلۈش ئۈچۈن چۇقۇم بىر نەرسىلەرگە ئېرىشىش كەتمەيدۇ. كۈلكىنىڭ خۇرۇچى ئۈزىڭىزدە بار. قولتۇقىڭىزنى قىچىقلاپ كۈلسىڭىزمۇ، نەچچە مىليۇن خەجلەپ كۈلگەن ئادەمنىڭ كۈلكىسىگە  ئوخشاش كۈلگە چىقىدۇ. ئوخشىمايدىغان يىرى سىز داۋاملىق قىچىقلاپ كۈلىۋەرسىڭىزمۇ بۇلىدۇ. ئەمما، نەچچە مىليۇن بىلەن كۈلگەن ئادەم دەرھال بۇ كۈلكىنى توختىتىپ  ئايىغىنى يىغىستۇرۇشنىڭ پىلانىنى قىلمىسا بولمايدۇ. داۋاملىك كۈلىمەن دىسە مەڭگۈ كۈلەلمەس بۇلۇپ قالىدۇ.
مەن بىلەن دوست بولسىڭىز، خوشنا بولسىڭىز، شىرىك بولسىڭىز بۇلىدۇ. ھايات چىغىرلىرىدا بىللە ماڭسىڭىز بۇلىدۇ. ئەمما، مىنىڭ يەنە سىز ياقتۇرمايدىغان نۇرغۇن تەرەپلىرىمنىڭ يارالمىشىمدا بىللە بارلىقىنى ئۇنتۇپ قالسىڭىز بولمايدۇ. مەن مەندۇرمەن. مەڭگۈ سىزنىڭ ئۆلچىمىڭىزدىكى مەن بولالمايمەن. ئۆزۈمنىڭ ئۆلچىمىدىكى مەندۇرمەن.
ئەگەر بۇ نوقتىنى ھېس قىلسىڭىز، مىنىڭ مەۋجۇتلىقىمغا بىر پۈتۈن ھۆرمەت قىلالىىسڭىز مەندىن نۇرغۇن پايدىلارغا ئېرىشىسىز. چۈنكى، يارالمىشىمدا ئەتىراپىمدىكىلەرگە مەنپەئەت يەتكۈزۈش ئىدىيە مەندە بىللە بار. ھاياتىمدا بىرقىسىم مەنپەئەتلەرنى يەتكۈزۈش ۋەزىپەم بار. بۇنى چۇقۇم قىلىمەن، يا سىزگە، يا باشقىلارغا.
رەنجىسەم قاپاق تۈرىيمەن، ئىچىم سىقىلسا كەلسە، كەلسمەس ۋارقىرايمەن. ۋارقىرىسام سىزگە ۋارقىرمايمەن. ئەمما، يېنىمدا سىز بولغاندىكىن سىز ئاڭلايسىز شۇ. . . بۇنداق ۋاقىتتا ماڭا ۋارقىردى، دىمەستىن مەن يېقىن بولغاندىكىن ئاڭلاپ قالدىم دەپ چۈشىنىڭ.
بەزىدە ئويلانمايلا سۆزلەيمەن. بەزىدە يولسىز كۈندەشلىك قىلىمەن.
بەزى ئىشلارنى بىكارغا ئاستىن-ئۈستۈن قىلىۋىتىمەن. بەزىدە ناھايىتى بىمەنە خىياللارنى قىلىمەن. بەزىدە جىنات ئۆتكۈزۈشنى ئارزۇ قىلىپ قالىمەن. ھەربىر ئادەمنىڭ كاللىسىدا بىر جىنايەتچىنىڭ باغلاقلىق ئىكەنلىكىنى ئۇنۇتماڭ. ياخشى كىشىلەر ئۇنى مەڭگۈ باغلاق ئىچىدە ساقلايدۇ. ئەمما، بەزى كىشىلەر قۇيۋېتىپ سالىدۇ.
غەزەپلىنىش، بوتناش، ھاياجانلىنىش، كەچ يېتىپ، ۋاقچە قۇپۇش، قالايمىقان پۇل خەجلەش،
ھورۇنلۇق قىلىش قاتارلىق غەيرى نۇرمال ئادەتلىرىمگە يۇلۇقۇپ قالىسىز. مانا مۇشۇ ئىللەتلەر مىنى پەرىشتىن ئايرىپ، ئادەملىكىمنى نامايەن قىلىپ تۇرىدۇ.
سۆيۈشكە تىگىشلىك بولمىغان لەۋلەرنى سۈيۈپ قالغانمەن. سۈيۈشكە تگىشلىكلىرنى قۇرۇق قالدۇرغانمەن. قىلىشقا تىگىشلىك ئىشلاردىن ۋاز كىچىپ، قىلىشقا تىگىشلىك بولمىغان ئىشلارنى قىلىپ قالغانمەن. شۇڭا پۇشايمانلىرىم، نادامەتلىرىم، غەم-قايغۇلىرىم بار.
ھاياتنىڭ بۇ مەنزىللىرىگە كەلگەندە ھە دىسە ئەتىراپتىكىلەردىن خۇشاللىق ئىزدەپ يۈرۈشنى قۇيۇپ ئىچىمدىكى خۇشاللىقتىن بەھىرمەن بۇلۇشنى ئۈگۈنىۋاتىمەن. شۇڭا مەن بىلەن بىللە بۇلۇشنى ئويلىغان ۋاقىتلىرىڭىزدا مەن يالغۇز كىتىپ قېلىشىم مۇمكىن.
ھەممە ئادەم قايغۇرۋاتسا مەن كۈلۈپ كىتىشىم مۇمكىن.
بۇ، مەندرۇمەن. يارتىلىشتا بارلىق خۇرۇچلار سېلىنىپ مۇكەممەل يارىتىلغان ئادەمدۇرمەن.
مىنى ۋە ئەتىراپتىكى بارلىق ئادەملەرنى مۇشۇنداق ئىدىيە بىلەن كۆرەلىسىڭىز نۇرغۇن زىددىيەتلەر ئازىيىشى مۇمكىن.
