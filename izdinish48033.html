ئەسسىلامۇ -ئەلەيكۇم ، شىنجاڭ تېلىۋىزىيە ئىستانسىدىكى رەھبەرلەر ، ئاكا - ھەدىلەر، تېچلىقمۇ؟

      ئەمدى ھەرقايسىڭلار بىلەن باداشقان قورۇپ ئولتۇرۇش پۇرسىتى بولمىغاندىكىن ئازراق كۆڭۈل سۆزلىرىم ۋە سېسىق گەپلىرىم بار ئىدى، مۇشۇ يەرگىلا تۆكەي دەپ ئويلىدىم .

      ھەرقايسىلىرى كۆپ جاپا -تارتىپ مەركەزنىڭ ۋە مەملىكەت ئىچى سىرتىدىكى خەۋەرلەرنى بىزگە تارقىتىپ تۇرىشۋاتىلا، كۆپ تەشەككۈر .

   ئاندىن بىزنىڭ سەنئەتخۇمار خەلق ئىكەنلىكىمىزنى كۆزدە تۇتۇپ سەنئەت پىروگراممىسى بەكلا تولا، مەشرەپ دەمدۇ، ناۋا دەمدۇ، ئاندىن سورايدىغان سوئاللىرىنىڭسەۋىيسىنىڭ بەك تۆۋەنلىكىكى، قىممىتىنىڭ يوقلىقى بىلەن بىر قانچە يىلدىن ئىزچىل قويىلىۋاتقان تەلەي يۇلتۇزى دەمدۇ، ئىشقىلىپ مۇشۇنداق نەرسە ساماندەك ... شۇڭلاشقىمۇ رىياسەتىچمۇ جىق مۇشۇ XJTV دا ! ئەمدى ئۇنىمۇ -دەيمىز بۇنى دەيمىز، يوقتىن ياخشى ...


    ھەرقايسىلىرىنىڭ چوڭ شىركەتنىڭ مەبلەغ سېلىشى بىلەن شىنجاڭدا پوتبول كۇلۇبى قۇرۇلۇپ مەملىكەت ب دەرىجىلىك مۇسابىقىگە چۈشىۋاتقانلىقىدىن خەۋەرلىرى بارمىكىن . بەلكىم خەۋەردار بولىشقانلا، ما نۆۋەت تېلىۋىزىيە ئىستانسىنىڭ خېلى كۆپ ئاتامانلىرى خىزمەتلىرىنى تاشلاپ مۇسابىقە كۆرگىلى بېرىپتۇ. شۇلار ئويلىمىغانمىدۇ دەيمەن، شىنجاڭدا 16 شەھەر بار، نۇرغۇن ئادەم مۇسابىقە كۆرەلمەيدۇ. مانى بىر سۈرەتكە ئېلىپ نەق مەيداندىن تارقىتايلى دىگەننى . ...............

ئەگەر بۇنىڭغا شارائىت ياكى ئىقتىساد يار بەرمايدۇ دەپ قالسىلازە تۈف قىلىپ تۈكۈرۈپ قويغىسى كېلىدۇ ئادەمنىڭ ...ياكى ماۋۇ مۇسابىقىنىڭ دەرىجىسى تۆۋەن دىيىشىسىلىمۇ بولمايدۇ، مەيلى قانچىلىك تۆۋەن بولسۇن شىنجاڭلىقلار ئۈچۈن بۇ دۇنيا لوڭقىسىدىنمۇ بەكرەك مۇھىم .....

      ئەگەر زادى پۇل يوقتى .، ئېلاندىن كىرگەن كىرىمنى بۆلىشىپ خەجلەپ بولۇپتىمىز دەپ قېلىشسىلا كېينكى نۆۋەت مۇسابىقە مەيدانىغا XJTV يېزىپ يوغان - يوغان ساندۇقتىن 30 نى قويۇپ قويىشالا، 40 مىڭ پوتبول مەستانىسى 5 يۈەندىن بېرەيلى. 200 مىڭ يۈەن دىگەن گەپ. لېكىن ماۋۇپۇلنىڭ 100 مىڭنى بۆلىشىپ خەجلىۋېلىشلىما قالغان 100 مىڭغا ما ئىشنى قىلغىلى بولىدۇ، تامامەن بولىدۇ .

ئۇنداق بولسا خەلقنىڭ ساداسىنى ئاڭلاپ ما ئىشنى ئەستايىدىل ئويلىشىپ بېقىشىلا.................

    تورداشلار توغرا ئېيتىدۇ، <<شىنجاڭ خەلق تېلېزىيە ئىستانسسى>> خەلقىمىزنىڭ كۆز يېشىغىغۇ مەلھەم بولالمىدى. بۇنىسىنى تېلېۋىزۇرنى قاچاندىن بۇيان كۆرگەن بولسام ھازىرغىچە خەلق كۆز يېشى ئۈچۈن ئىشلەنگەن پروگرامما كۆرمىدىم. كۆرگەنلىرىم ساختا كۈلكىلەر ئىچىدىكى بىچچارە ئوبرازلار خالاس!

   ئەمدى بۈگۈنكى تېمىدا ئېيتىلىۋاتقان پۇتبۇل مەيدانىدىكى بۇ ھاياجانلىق مىنۇتلاردا تېلېۋىزىيە ئىستانسسىنىڭ ئاپپاراتلىرى خىجىل بولۇپ قالغىنى...


    شىنجاڭ خەلق تېلېۋزىيە ئىستانسىسى خەلقىمىزنىڭ ئۇچۇر مەركىزى ئىدى، ئەپسۇسكى خەلقىمىز مەنىۋىيىتى ۋە ساغلاملىقىغا زور ئىلھام بولىدىغان40مىڭ كىشلىك بۇنداق كاتتا يىغىلىش توغرىسىدىكى ئۇچۇرلار بۇ ئاتالمىش تېلېۋزىيە ئىستانسسى ئۈچۈن <<قىممەتسىز>>بولۇپ قېلىشى، خەلق تېلېۋىزىيىسىنىڭ نەپىسى خەلق بىلەن تەڭ سوقماي قالدىمۇ قانداق؟! دېگەن سۇئالنى ھەر كىشىنىڭ ئويىغا سېلىشى تۇرغانلار گەپ.

    ئەمدى ئەمەلىي ئەھۋالغا قارايدىغان بولساق، تېلېۋىزىيە ئىستانسىسىنىڭ ئۇچۇر مەركىزىلىك ئورنىنى باشقا تارتاتقۇلارغا تارتقۇزۇپ قويامدۇ قانداق؟! مەنغۇ تېلېۋىزۇر كۆرمىگىلىك ساق 12يىل بولدى. چۈنكى تېلېۋىزىيەنىڭ بۈگۈنكى رولى كۆپىنچىسى بىزنىڭ مەنىۋىيىتىمىزنى بۇلغايدىغان مەنىۋىيەت ئەخلەتلىرى... ئاز بىر قىسمى بولسا يەنە شۇ ئورۇندىكى ئاز ساندىكى جاپاكەشلەرنىڭ تىرىشچانلىقى بىلەن ئەمەلگە ئېشىۋاتقان، مەنىۋىيەت ئوزۇقلىرى... ئەپسۇسكى، بېرىلگەن مەنىۋىيەت ئوزۇقىدىن ئەخلەتلەر كۆپ. بۇ خىل ئەمەلىيەت بىزدەك ئىسلام ئېتىقادى ئاساسىدا ئۆزتۇرمۇشىنىڭ ھەر بىر ئىنچىكە تەرەپلىرىگىچە قائىدە تۈزۈپ، ئۆرپ-ئادەتكە ئايلاندۇرغان مەدەنىي مىللەتنى ، ئۆز ئەسلىدىن ئاستا-ئاستا يىراقلاشتۇرۇپ، ھازىرقى ھەممىسى بىر-بىرىگە ئاغرىنىش ئىچىدە مىللەتنى ئىللەتكە ئايلاندۇرۋېتىدىغان ئەبجەش توپقا ئايلاندۇرۇپ قويدى.يوقارقىلار پەقەت مېنىڭ شەخسى قاراشلىرىم، شۇڭا ئون يىلدىن بۇيان تېلېۋىزۇر كۆرۈپ كەتمەيمەن...
