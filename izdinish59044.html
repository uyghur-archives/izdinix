دونۋالىد شىنجاڭ كوماندىسىغا باش تېرىنىر بۇلۇشنى ئۈمۈت قىلدى

   ئۇيغۇر ۋاسكېتبول تورى 7-ئاينىڭ 5-كۈنى خەۋىرى: دۆلەت ئەرلەر ۋاسكېتبول كوماندىسىنىڭ سابىق باش تېرىنىرى، شىنجاڭ گۇاڭخۇي ئۇچار يۇلۋاس كوماندىسىنىڭ سابىق باش تېرىنىرى ئامېرىكالىق دونۋالىد ئۆزىنىڭ شەخسىي مىكرو بىلوگىدا كىلەر مۇسابىقە پەسىلىدە شىنجاڭ گۇاڭخۇي كوماندىسىغا باش تېرىنىر بۇلۇشنى خالايدىغانلىقىنى بىلدۈرگەن.  ئەگەر بۇ ئىش رۇياپقا چىقسا ، دونۋالىد ئىككىنچى قىتىم شىنجاڭ كوماندىسىغا باش تىرىنىر بۇلۇپ قالىدۇ. تورداشلار بىلەن ئۆز-ئارا پىكىر ئالماشتۇرۇش جەريانىدا، دونۋالىد شىنجاڭ كوماندىسىنى CBA چىمپىيۇنلۇققا ئىرىشتۇرۇش ئىشەنچىسىنىڭ بارلىقىنى بىلدۇرگەن.  دونۋالىد شىنجاڭ گۇاڭخۇي كوماندىسىغا باش تېرىنىر بۇلۇشنى خالايدىغانلىقىنى بىلدۈرگەن بۇلسىمۇ ئەمما ھىچقانداق تەكلىپ تاپشۇرۇپ ئالمىغان.
  دونۋالىد مىكرو بىلوگىغا ؛ « مەن شىنجاڭ كوماندىسىغا قايتىپ بىرىشنى بەكمۇ خالايمەن، كوماندا خۇجايىنى چىمپىيۇن بۇلۇشنى ئۇيلايدۇ، مەن ئۇنى ياقتۇرىمەن. شىنجاڭ كوماندىسىنىڭ ھازىرقى سەپ تۈزۈلىشى ھەقىقەتەن ياخشى، توپ مەستانىلىرىمۇ بەك قېزغىن، مىنىڭ ئىشەنچىم بار، مەن شىنجاڭ كوماندىسىنى يىتەكلەپ چىمپىيۇنلۇقنى قولغا كەلتۈرەلەيمەن » دەپ يازغان. ئۇ ئارىدىن ئۇزۇن ئۆتمەي دەرھال  تورداشلار بىلەن ئۆز-ئارا پىكىر ئالماشتۇرۇش ئىلىپ بىرىپ مۇنۇلارنى بىلدۇرگەن؛ « ليۇۋېي بىلەن مەن يەنە تاڭ جىڭدۇڭ ۋە سۇۋىي بىرلىكتە ئاسىيا تەنھەركەت يىغىنىنىڭ چىمپىيۇنلۇقىنى قولغا كەلتۈرگەن. شىرئەلىجان، ليۇۋېي ۋە مەن بىرلىكتە قول تۇتۇشۇپ ئاسىيا ۋاسكېتبول لەۋھە تالىشىش مۇسابىقىسىنىڭ ئالتۇن مىدالىنى ئالغان. ھازىر مەن ئۇلار بىلەن بىرلىكتە CBA بىرلەشمىسىنىڭ چىمپىيۇنلۇقىنى ئېلىشنى خالايمەن ».  
   دونۋالىد يەنە توپ مەستانىلىرىگە تەشەككۈر ئېيتىشنى ئۇنتۇپ قالمىغان، « مەنمۇ شىنجاڭ توپ مەستانىلىرىنى ئەسلەيمەن، بۇ كوماندا ھەقىقەتەن بەك قالتىس سەپ شەكلىگە ئىگە ».  دونۋالىد ئۆزىنىڭ شىنجاڭ كوماندىسىغا باش تىرىنىر بۇلۇش ئارزۇسىنى ئىپادىلەش ئارقىلىق يىقىندا ئېقىپ يۈرگەن بىر سۆز-چۆچەكنى ئىسپاتلاپ قۇيغان،  ئۇ بۇلسىمۇ  شىنجاڭ گۇاڭخۇي ئۇچار يۇلۋاس كوماندىسىنىڭ باش تېرىنىرى سۈي ۋەنجۈننىڭ ئاللىقاچان ۋەزىپىسىدىن ئايرىلىشىدۇر.   
  دونۋالىد جۇڭگۇ ۋاسكېتبول مەستانىلىرىگە نىسپەتەن ناتۇنۇش ئەمەس، ئۇ يىقىنقى نەچچە يىلدا جۇڭگۇنىڭ ۋاسكېتبول ساھەسىىدە ئەڭ بەس-مۇنازىرە قۇزغىغان چەتئەللىك تىرىنىرلارنىڭ بىرى.   2009-يىلى يازدا  ياۋمىڭ شاڭخەي كوماندىسىنى سىتىۋالغاندىن كىيىن، دونۋالىد شاڭخەي كوماندىسىنىڭ تارىختىكى تۈنجى چەتئەللىك باش تىرىنىرى بۇلۇپ قالىدۇ. ئۇ ئىككى مۇسابىقە پەسلى شاڭخەي كوماندىسىغا باش تىرىنىر بۇلۇپ، 2009-2010 مۇسابىقە پەسلىدە،  ياۋمىڭ NBA بىرلەشىمسىگە كەتكەندىن كىيىنكى ئەڭ ياخشى نەتىجە يارىتىپ، يىرىم  ھەل قىلغۇچ باسقۇچىغا بۈسۈپ كىرىدۇ، ئەپسۇسكى يىرىم ھەل قىلغۇچ باسقۇچىدا 1-3 نەتىجە بىلەن گۇاڭدۇڭ كوماندىسىغا ئۇتتۇرۇپ قۇيىدۇ.
   2010-يىلى4-ئايدا،  دونۋالىد دۆلەت ئەرلەر ۋاسكېتبول كوماندىسىنىڭ باش تېرىنىرلىقىغا تەيىنلىنىپ، دۆلەت ئەرلەر ۋاسكېتبول كوماندىسى تارىخىدىكى ئۈچۈنچى چەتئەللىك باش تىرىنىرى بۇلۇپ قالىدۇ. شۇ يىلى دۇنيا ئەرلەر ۋاسكىتبول لەۋھە تالىشىش مۇسابىقىسىدە جۇڭگۇ كوماندىسىنى 16 كۈچلۈك ئىچىگە ئېلىپ كىرىدۇ. 2011-يىلى يازدا، شاڭخەي كوماندىسى دونۋالىد بىلەن بولغان توختامىنىڭ ئاياقلاشقانلىقىنى ئىلان قىلىدۇ. ئارىدىن ئۇزۇن ئۆتمەي دونۋالىد شىنجاڭ گۇاڭخۇي ئۇچار يۇلۋاس كوماندىسىنىڭ باش تىرىنىرى بۇلىدۇ.
  2011-يىلى 11-ئايدا، دونۋالىد باش تىرىنىرلىقىدىكى دۆلەت ئەرلەر ۋاسكېتبول كوماندىسى گۇاڭجۇدا ئۆتكۈزۈلگەن ئاسىيا تەنھەركەت يىغىنىدا چىمپىيۇن بۇلىدۇ.  ئارىدىن ئۇزۇن ئۆتمەيلا جىددى ھالدا 2011-2012 پەسىللىك CBA بىرلەشمىسى مۇسابىقىسىگە كىرىشىپ كىتىدۇ، بىراق شىنجاڭ كوماندىسىنىڭ نەتىجىسى كىشىنى رازى قىلالمايدۇ. 2011-يىلى 12-ئاينىڭ ئوتتۇرلىرىدا شىنجاڭ كوماندىسى ئىچكى قىسىىمىدا دونۋالىدنىڭ باش تىرىنىرلىقتىن چۈشۈرۇپ، جاڭ شىڭچۇەننىڭ باش تىرىنىرلىققا تەيىنلەنگەنلىگىنى ئىلان قىلىدۇ.  دونۋالىد شىنجاڭ كوماندىسىنىڭ باش تىرىنىرلىقىدىن ئايرىلغاندىن كىيىن، دۆلەت ئەرلەر ۋاسكېتبول كوماندىسىغا يىتەكچىلىك قىلىپ 2012 -يىللىق لوندۇن ئولىمپىك تەنھەركەت يىغىنىغا قاتنىشىدۇ، ئەمما نەتىجىسى ئانچە ياخشى بۇلمايدۇ.  
    دونۋالىد  شىنجاڭ گۇاڭخۇي ئۇچار يۇلۋاس كوماندىسىنىڭ باش تىرىنىرلىقىغا تەكلىپ قىلىنىپ ئۇزۇن ئۆتمەي شىرئەلىجاننى مۇداپىيەدە  ئەۋزەللىككە ئىگە دوپ قاراپ، ئاسىيا ۋاسكېتبول لەۋھە تالىشىش مۇسابىقىسىگە قاتنىشىدىغان ئاخىرىقى 12 كىشلىك ئەزالار سەپ تىزىلىگىگە كىرگۈزىدۇ، بىراق جۇڭگۇ ۋاسكېتبول ساھەسىدە شەخسىي مۇناسىۋەتكە تاياندى دەپ مۇنازىرە قۇزغىلىدۇ. شۇ قىتىملىق مۇسابىقىدە چىپىيۇنلۇقنى قولغا كەلتۈرۈپ 2012 -يىللىق لوندۇن ئولىمپىك تەنھەركەت يىغىنىغا قاتنىشىش سالاھىيىتىگە ئىگە بۇلىدۇ.   لوندۇن ئولىمپىك تەنھەركەت يىغىنىغا تەييارلىق قىلىش يىغىلىپ مەشىق قىلىش باشلانغاندىن كىيىن ئۇزۇن ئۆتمەي شىرئەلىجاننى 1-نۇمۇرلۇق ئورۇنغا ماس كەلمەيدۇ، 2-نۇمۇرلۇق ئورۇنغا بۇي ئىگىزلىكى ئەۋزەللىككە ئىگە ئەمەس دىگەننى باھانە قىلىپ دۆلەت كوماندىسىدىن بۇرۇنلا شاللاپ، ئۇيغۇر ۋاسكېتبول چۇلپىنىمىزنىڭ ئەڭ ئالى غايىسى بولغان ئولىمپىك تەنھەركەت يىغىنىغا قاتنىشىش ئارزۇسىنى بەرپات قىلىدۇ. ۋاسكېتبول ساھەسى شىرئەلىجاننىڭ شاللىنىشىنى دونۋالىدنىڭ  شىنجاڭ كوماندىسىدىن شەخسىي ئۈچ ئىلىشى بىلەن مۇناسىۋەتلىك بۇلىشى مۇمكىن دەپ تەھلىل قىلىشىدۇ.
   لوندۇن ئولىمپىك تەنھەركەت يىغىنى ئاخىرلاشقاندىن كىيىن، دونۋالىدنىڭ جۇڭگۇ ۋاسكېتبولچىلار جەمئىيىتى بىلەن بولغان توختامى ئاخىرلىشىدۇ. ئۇ جۇڭگۇدىن ئايرىلغاندىن كىيىن ئىزچىل ئىنتىرنىت تورى ئارقىلىق جۇڭگۇ ۋاسكېتبولچىلىقى ۋە CBA بىرلەشمىسىنىڭ تەرەقىياتىغا كۈڭۈل بۈلۈپ، ھامان بىر كۈنى جۇڭگۇغا قايتىپ ، ئۆزىنى يەنە بىر قىتىم ئىسپاتلاش پۈرسىتىنى كۈتۈپ كەلگەن.
  دونۋالىدنىڭ شىنجاڭ گۇاڭخۇي كوماندىسىغا باش تېرىنىر بۇلۇشنى خالايدىغانلىقىنى تۇغرىسىدىكى خەۋەر تارقالغاندىن كىيىن، شىنجاڭ گۇاڭخۇي ئۇچار يۇلۋاس ۋاسكېتبول كۇلۇبىنىڭ باش لىدىرى گوجىيەن مۇخبىرلارنىڭ زىيارىتىنى قۇبۇل قىلىپ،  دونۋالىدنىڭ شىنجاڭ گۇاڭخۇي ئۇچار يۇلۋاس ۋاسكېتبول كوماندىسىغا باش تىرىنىر بۇلۇشىنىڭ مۇمكىن ئەمەسلىگىنى بىلدۇرگەن.
   يىقىنقى كۈنلەردىن بۇيان نۇپۇزلۇق تەنتەربىيە تور بىكىەتلىرىدە  شىنجاڭ گۇاڭخۇي ئۇچار يۇلۋاس كوماندىسىنىڭ باش تېرىنىرى سۈي ۋەنجۈننىڭ كوماندا بىلەن بولغان توختامىنىڭ ئۈزۈلگەنلىكى توغرىسىدا خەۋەر تارقالدى. بۇ خەۋەرگە نىسپەتەن شىنجاڭ گۇاڭخۇي ئۇچار يۇلۋاس ۋاسكېتبول كۇلۇبى 6-ئاينىڭ 26-كۈنى مەخسۇس بايانات ئىلان قىلىپ «  كۇلۇپنىڭ  سۈي ۋەنجۈن بىلەن بولغان توختامىنى ئۈزمىگەنلىكىنى،  توختام مۇددىتىنىڭ بەش ئىكەنلىگىنى ئەمدى ئىككى يىل توشقانلىقىنى يەنە ئۈچ يىل توختام مۇددىتى بارلىقىنى، نۆۋەتتە  سۈي ۋەنجۈن كىسەل سەۋەبى بىلەن رۇخسەت سۇراپ لىياۋنىڭدا داۋالىنىۋاتقانلىقى، سۈي ۋەنجۈن نەتىجىسىنىڭ شىنجاڭدىكى ھەر قايسى ساھەنىڭ ۋە كۇلۇپنىڭ ئىتراپ قىلىشىغا ئىرىشكەنلىگىنى » ئىلگىرى سۈرگەن.  بىراق سۈن ۋەنجۈن ھازىرغىچە كوماندىدىن ئايرىلىپ تۇرغىنى ئۈچۈن بۇ خەۋەر يەنىلا ئوخشىىغان شەكىلدە بەس مۇنازىرەگە سەۋەپ بولماقتا...
