ئەسسلامۇ ئەلەيكۇم ئىزدىنشى ئەھلى پىقىر كوۋرۇكتىن سىلەرگە سالام ،،

--بوگۇن تۇرغا كىرىپ ‹ئارقا تېرىكىم بار ›ناملىق ئەزانىڭ مەن ھەققىدە يېزىلغان تىمىنى كورۇپ ھەم خوش بۇلدۇم ھەم كوڭلوم يىرىم بۇلدى ،بۇ تىما خۇددى دادىسى ئولوپ قالىنىڭ بالىنىڭ دادىسىنى كومۇپ بۇلوپ تۇپراق بشىدىكى جامائەتتىن مىنىڭ دادام قانداق ئادەم ئىدى دەپ سۇئال سۇرىغاندەكلا بولوپ قاپتۇ .ئادەمنىڭ قانداق ئىكەنلىگىنى ئىنسان قانداقمۇ باھا بەرسۇن ...

-ئۇنداق بۇلسا ئوزەمنى تۇنۇشتۇرۇپ قۇياي ،مەن غۇلجىنىڭ بىر يېزىسدىن لىكىن شەھەرگە بەك يىراق ئەمەس قاتناش ئەپلىك 7سۇملۇق يۇل ،راس گەپ قىلسام مەن كومپىيۇتۇر دىگەن نەرسىدىن پەقەتلا خەۋرىم يۇق ئىدى ،بىز كۇنى بىزنىڭ مەھەللىدىكى يۇنۇس دىگەن تۇنگان ئاغىنەم بىلەن شەھەرگە كىرىپ ئۇ مىنى ۋاڭباغا باشلاپ كىردى ۋە ماڭا بىر كومپىيۇتۇرنى ئىچىپ  ، ئايرۇپىلان ئاتدىغان ئۇيۇنچۇقنى ئىچىپ بەردى مەن بىر سائەتتە مائۇس تۇتۇشنى ئىككى سائەتتە باشقۇرشنى ئوگۇنىۋالدىم ۋە شۇندىن كىيىن كومپيۇتۇرغا باشچىلاپ كىرىپ كەتتىم يۇنۇس تۇنگاننىڭ ئاغىنىلىرى بىلەن ‹بەنتىياۋمىڭ ›دىگەن ئويۇننى تازا ئوينىدۇق ،ak47ماركىلىق قۇرال ۋە بۇمبا ئىشىلىتشى دىگەننى ئۇبدانلا ئوگۇنۇپ يۇنۇس تۇنگاننىڭ ئاغىنىلىرىنى دائىم يىڭىۋىلىپ يىرىم توخۇ ئالغۇزۇپ خىمىر دىگەنىنى نەچچە قىتىم سالغۇزۇپ قۇساق تۇيغۇزۇپ يىزىغا چىقىپ كىتتۇق ،لىكىن مەن بۇنىڭ بىلەن بۇ ئۇيۇنغا خۇمار بولوپ قالدىم ئۇڭۇمدىمۇ چۇشۇمدىمۇ شۇ ئۇيۇننى كورىمەن ئىتىشىمەن ،،،

بىر كونى چۇشۇمدە يۇنۇس تۇنگاننىڭ ئاغىنىلىرى بىلەن ئىتىشىپ تات-تات-تات ،،دەپ كوسار بىلەن ئورنۇمدىن تۇرۇپ ھۇيلىغا چىقىۋاتىمەن ۋە يۇنۇس ئوگۇزگە قارا -ئوگۇزدە بار دىققەت قىل دەپ ۋاقىراپ كىتىپتىمەن ،بۇنى ئاڭلاپ دادام بىلەن ئاكام مالغا ئوغرى كىرگەن ئوخشايدۇ دەپ دادام پالتىنى ئاكام كالا پۇقى ئالدىغان ئارىنى كوتۇرۇپ ھۇيلىدا يۇگرەپ يۇرۇپتۇ كەينىمدىن كىيىن ئولار مىنىڭ ئوخلاۋىتىپ يۇگرەپ يۇرگەنلىكىمنى بىلىپ ئاستا ئويگە ئەكىرىپ ياتقۇزۇپ قۇيۇپتۇ .ئەتىسى ئۇيغۇنسام ئاپام بىلەن  دادام يىغلاپ ئولتۇرىدۇ بىشىمدا نىمە بۇلدى دىسەم گەپ قىلمايدۇ ،ئاخشام نىمە چۇش كوردۇڭ دەيدۇ ،مەن ئولارغا مەن دائىم بەنتىياۋمىڭنى كورىمەن دىسەم كىمۇ ئۇ قايسى تۇنگاننىڭ قىزى دەيدۇ ،،(بىزنىڭ مەھەللىدە تۇنگان جىق بۇلغاچقا ئولار مىنى بىرەر تۇنگاننىڭ قىزنى ياخشى كورۇپ قالدى دىدىمىكا)

شۇنىڭ بىلەن ئاپام بىلەن دادام بۇ ئىشىنى مۇمامغا دەپتىكەن  ماڭا جىن چاپلىشىپ قاپتۇ ئوقۇتۇپ كىلەيلى دەپ شەھەردىكى بىر جىكەش خوتۇننىڭ ئويىگە ئاپاردى ،جىنكەس خۇتۇن دىسە قانداق خۇتۇنكى دىسەم كوزى ئالغاي ئادەتتىكلا خۇتۇنكەن ،مىنى يالغۇز ئىلىپ كۇزىنى يۇمۇۋىلىپ ئايەت دىسە ئايەت ئەمەس گەپ دىسە گەپ ئەمەس بىر نەرسىلەرنى ئوقۇۋىتىپ ئوزىنى يەرگە تاشلاپ خىخراپ بىر نىمىلەرنى دەپ تارتىشىپ قالدى ،مەن قورقۇپ كىتىپ ھەدە سىزگە نىمە بۇلدى 120گە تىلفۇن قىلايلى دەپ كەتتىم ۋە تىلفۇننى باسماقچى بۇلسام ئو خۇتۇن كۇزىنى قىيا ئىچىپ بۇلدى ئورما ھازىر ئۇڭشىلىپ قالىمەن دىدى ..ۋە ئۇرنىدىن تۇرۇپ كەتتى ئاندىن ئاپامنى چاقىرىپ ئاپامنىڭ قۇلىقىغا بىر نەرسىلەرنى دىدى ،ئاپام ئۇ خۇتۇنغا 50سۇم بىرىپ مىنى ئىلىپ كەتتى ...

شۇندىن كىيىن تورخانىغا خىلى ۋاقىت كىرمىدىم ،،بۇ يىل يىڭى يىلدا شەھەردىكى تۇققانلارنىڭ ئۇيىگە كىرىپ مۇنبەر دىگەن نەرسىنى كورۇپ قالدىم ۋە ئولار مىنى تىزىملىتىپ قۇيدى ئۇزەم كىتاپ ئوقۇشقا قىزىققاچقا مۇنبەرگە باشچىلاپلا كىرىپ كەتتىم .

--ھە ئاكام تۇغۇرلوق پاراڭ قىلىپ بىرىپ ھە ،،ئاكامنىڭ بىر ئايالى بىر بالىسى بار ئاكام ئايالى بىلەن ئايدا بىر قىتىم ئۇرۇشىدۇ .ھەر قىتىم ئاكامنىڭ ئايالى ئاكامنىڭ يۇزىنى تاتىلاپ قىيما چىيما قىلىۋىتىدۇ ،،ئوتكەندە ئاكام گۇش تۇغراش ماشىنىسى ئالىمەن دىگەن مەن نىمە قىلىسەن گۇش تۇغراش ماشىنسى ئىلىپ ئايالىڭ گۇش تۇغراشقا ئۇستىغۇ يۇزۇڭنى تاتلىغاچ گۇشنىمۇ تۇغرىۋەتسە بۇلىدىغۇ دەپ چاخچاق قىلسام دادامنىڭ دائىم ئاڭلايغان رادىيۇسىنى ماڭا قارىتىپ بىر ئاتقان مەن چاققانلىق قىلىپ چەتكە ئوتىۋالسام رادىيۇ تامغا تىگىپ دادام دائىم ئاڭلايدىغان مەركىزى خەلىق رادىيۇ ئىستانسىنىڭ چىقمايدىغان  بۇلوپ قالغان داادم غۇلجىنىڭ رادىيۇسىنى ئاڭلاپ يۇرىدۇ ،دادامغا دىمدىم دىسەم ئاكامنى چۇقۇم ئورىدۇ .

ئاكام كۇندە 10غىچە ئوخلايدۇ دادام تىللىغاندىن كىيىن ئورنىدىن تۇرىدۇ ،ئوتكەندە ئەتىگەنلىك چاي ئىچىپ ئولتۇرۇپ دادام ئاكامغا سەنمۇ چوڭ بۇلدۇڭ مەسچتىكە  نامازغا چىق مەن ئولوپ قالسام تۇپراق بىشىمغا چىقىدىغان ئادەم كىرەك دىگەن ئاكام چىيىنى سۇمۇرگەچ بامدىتىڭغا چىقالمايمەن جۇمە نامزىڭغا چىقىپ بىرەي دىگەن دادام ئاكامنى بەك سەت تىللىۋەتتى ..

ئەمدى كوۋرۇك دىگەن نامغا كەلسەك يۇگۇسلاۋيەنىڭ كوۋرۇك دىگەن كىنۇسىنى بەك ياخشى كورىمەن بولوپمۇ پارتلاپ كەتتىڭ پاتتىنا دىگەن يىرىنى كورسەم يىغلاپ كىتىمەن ئاندىن چۇڭ ئەيسانىڭ ئىپۇزىتىنى كۇرۇپ كۆلىۋالىمەن ،،بەزىدە زىرىكىپ كەتسەم بىردەم كۆۋرۇك دىگەن كىنۇنى بىر دەم چۇڭ ئەيسانىڭ ئىپىزۇتىنى كۆرۇپ   ئۇزەمنى بىردەم يىغلىتىپ بىردەم كۆلدۇرۇپ ئۆينايمەن ..ئۇندىن باشقا ئىلى دەريا كۆۋرىگە مىھرىم بار چۇنكى ھازىر يۇرۋاتقان قىز دۇستۇم بىلەن تۇنجى قىتىم كوۋرۇكتا ئۇچراشقان ۋە شۇ يەردە ماڭا ماقۇل بۇلغان ،،نۇرغۇن قىزلار مىنى ياراتمىدى ،بەزىلىرى مىنى نىرۋىسىدا چاتاق باركەن ،دىسە بەزىلىرى مىڭىشى بەك سەتكەن ،بەزىلىرى بەك قارۋاي كىلەڭسىزكەن ،دەپ ياراتمىدى ،،ھازىر قىز دۇستۇم مىنى كوڭلى تۇز ياخشى بالا دەپتۇ ئاپىسىغا ..بۇ قىز بىلەن كۈزدە توي قىلىمەن خۇدايىم بۇيرىسا ،،
