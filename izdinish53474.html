   جوڭگۇ پەنلەر
ئاكادېمىيىسى پىخسولوگىيە تەتقىقات ئىنستىتوتىدىكى ئاسپىرانتلىق ئوقۇشۇمنى
تاماملاپ،تۇغۇلۇپ ئۆسكەن يۇرتۇم ئۈرۈمچىگە قايتىپ كەلگىنىمگىمۇ بىر نەچچە ئاي
بوپقالدى.بۇ جەرياندا ساۋاقداش دوست-يارەنلەر،ئۇرۇق تۇغقان،قوشنا-قۇلۇملار بىلەن
كەڭ ئۇچراشتىم،ھال-مۇڭ بولدۇم.

قىزغىن،قايناق ھايات يەنە شۇ ئۆزگىچىلىكى بىلەن
داۋام قىلماقتا ئىدى.نەچچە يىل باشقا مەدەنىيەت توپلىرى بىلەن كۆپرەك ئۇچراشقان ھەم
ئوقۇش -تەتقىقات ئېلىپ كەلگەن ئۇچۇر-بىلىم ئىمكانى ۋە مەجبۇرىيەتلىرى بىلەن تولىمۇ
ئالدىراش يۈرگەچكىمۇ،بۇ بىر نەچچە ئايلىق ئۆز مەدەنىيەت تەۋەلىكىگە قايتىش مەندە
خېلى كۆپ ئويلىنىش پەيدا قىلدى.نۇرغۇن ئادەت،پىكىر-قاراش،مەسىلە-خاھىشقا باشقىدىن
قاراشقا،يېڭىدىن باھا بېرىشكە،ھەتتا بەزىلىرى ھەققىدە ئەستايىدىل ئويلىنىشقا مەجبۇر
بولدۇم.ئېچىۋىتىلگەن ئاڭ ئەسلىدىمۇ ئەقىل سۈزگۈچنىڭ بوش،ئاراچلىرىنىڭ كەڭ بۇلۇشىدا
ئەمەس،بەلكى ھىم،پۇختىلىقىدا ئىدى.ۋە ھالەنكى ئەقىل بىلەن باققانسىرى
ئارىمىزدا،بۇلۇپمۇ مەن بىلەن تەڭ دېمەتلىكلەر ئارىسىدا ئەۋج ئېلىپ كېتىۋاتقان
مۇنداق بىر قانچە مەسىلەبەكرەك دېققىتىمنى تارتتى.بۇلار جەمىيتىمىزدە ياشاۋاتقۇچى
ھەربىر شەخسكە مۇناسىۋەتلىك مەسىللەردىن ئىدى.يەنە كېلىپ دەلىل بىلەن تۇرغۇزۇلىشى
كېرەك بولغان بۇ قاراش -ئادەتلەر ساپلا ئاساسسىز غەلەت پەتىۋالىرىنىڭ دۆۋىسىدە
قالغان،بەزىلەر خاتا يۆنىلىشتە كېتىۋاتقانىدى.ئەلۋەتتە،ئادەمنىڭ تەقدىرىگە، جېنىغا
بېرىپ تاقىلىدىغان بۇ مەسىللەرگە سەل قاراشقا بولمايتى.شۇڭا،بۇ ھەقتە ئاز-تولا
ئىزدىنىپ،ئىزدەنگەنلىرىمنى ئوقۇرمەنلەر بىلەن ئورتاقلىشىشنى ئۆز بۇرچۇم دەپ
بىلدىم.

   2.ئوپېراتسىيە ئارقىلىق
ئېلىنغان بالا ئەقىللىق،ساغلام بولارمىش.ئۇلارچە،قاتتىق تولغاق ئەزىز بالىنى قىيناپ

ئاساسى يوق جۆيلۈشتىن باشقا نەرسە ئەمەس،ئەلۋەتتە.چۈنكى،توققۇز ئاي قورساقتا،سۇنىڭ
ئىچىدە ياتقان بالا تەبىئىي تۇغۇلۇش جەريانىدىلا رىتىملىق بېسىم،تەبىي كۆنۈش
جەريانىنى بېشىدىن ئۆتكۈزۈپ،چوڭ مېڭىسى،ئۆپكىسى،يۈرىكى تېخىمۇ تەرەقىي قىلالايدۇ ۋە

تولغاق بالىياتقۇدا ئۇزۇن مەزگىل ياتقان بالىغا نىسبەتەن زۆرۈر سىناق بۇلۇپ،مۇشۇ
جەريان ئارقىلىقلا بالىنىڭ ئۆپكىسى،يۈرىكى،قول-پۇت مۇسكۇللىرى سىرتقى مۇھىتتا

   5.تۇرمۇشىمىزنىڭ ھاللىنىشىغا
ئەگىشىپ،ھامىلىمۇ تېز يېتىلىدىغان،بوۋاقلارمۇ بۇرۇنقىدىن چوڭ،ئېغىر تۇغۇلىدىغان
بوپكەتتى.بىر قانچە دوستۇمنىڭ بوۋىقى تۆت كىلۇگىرامدىن ئېشىپ
تۇغۇلدى.مۇشۇنداق‹‹بودەك››بوۋاقلارمۇ ئانلارنىڭ قەيسەر ئوپېراتسىيىسىنى تاللىشىغا
سەۋەب بولماقتا.شۇڭا ئاياللار ھامىلدار ۋاقىتتا كۆپرەك ھەرىكەت قىلىپ،ئاز ئۇخلىشى
كېرەك،ئەلۋەتتە،تۇغۇتلۇق ئاياللارنىڭ داس سۆڭىكى ئالاھىدە كىچىك ياكى بىنورمال
بولسا،بالىياتقۇ يېرىلىپ كەتسە،تۇغۇت يولى بىنورمال ياكى ئۆسمە بولسا،ھامىلىنىڭ
ئورنى نورمال بولمىسا،ھامىلىنىڭ كىندىكى يۆگىشىپ قالسا،بالا ھەمرىھى چۈشۈپ
كەتسە،ھامىلە تۇنجۇقۇپ قالسا،تۇغۇت بەك قىيىن بوپكەتسە قەيسەر ئوپېراتسىيىسى ئەڭ
ياخشى تاللاشتۇر.قەيسەر ئوپېراتسىيىسى نۇرغۇن ئانا ۋە پەرزەنتنى بەزىدە ئەجەللىك
قىيىنچىلىقتىن قۇتۇلدۇرۇپ كەلدى.لېكىن بۇنداق بىنورمال بۇلۇش نىسبىتى %10ئەتراپىدا
بۇلۇپ،يېرىمىدىن كۆپرەك بولغۇسى ئانىنىڭ خەتەرگە تەۋەككۈل قىلىپ قەيسەر
ئوپېراتسىيىنى تاللىشى ئاقىلانە ئىش ئەمەس.تۇغۇت جەريانىدا كۆرۈلىدىغان نۇرغۇن
تەسلىك،قىيىنچىلىقنى تۇغۇشتىن بۇرۇنلا بايقاش،ئالدىنى ئېلىش تامامەن
مۇمكىن.ئاياللار ھامىلدار ۋاقتىدا لايىقىدا ئۇزۇقلىنىپ،قەرەللىك تەكشۈرتۈپ،مۇۋاپىق
ھەرىكەت قىلىپ،ئۆزى تۇغۇشقا تەييارلىق كۆرگىنى ياخشى.

كەچۈرگەن ئانا بىر ئۆلۈپ تىرىلگەندەك بولىدۇ.ئايالنىڭ تۇغۇت ئازابىنى كۆرگەن ئەرمۇ
دەل شۇنداق.مانا بۇ ھاياتتىكى بىر سىناق.ئىرادىمىزنى تېخىمۇ چىڭىتىدىغان،جاپالاردىن
قورقماس قىلىدىغان بىر سىناق.ھايات -ئۆلۈمھەققىدىكى ئۈزۈل-كېسىل قايتا ئويلىنىش

    تېخى تۈنۈگۈنكىدەكلا ئېسىمدە،شىئەن قاتناش ئۇنىۋېرسىتېتى
ئىنستىتۇتىنىڭ كلىنىكىلىك داۋالاش كەسپىدەئوقۇۋاتقان ۋاقىتلىرىمدا تۆتىنچى
يىللىقتىن باشلاپ بىر دوختۇرخانىغا پراكتىكىغا چۈشتۇق.مېنىڭ تۇنجى بۆلۈنگىنىم تۇغۇت
بۆلۈمى بولدى.تۇنجى كىرگەن ئوپېراتسىيەم تۇغۇت ئوپېراتسىيەسى بولدى.تۇغۇت كارۋىتىدا
بىر ئايال -بولغۇسى ئانا ياتاتتى.مەن پەقەت دوختۇرلارغا ياردەملىشىپ،ئايالغا
ئوكسىگىن پۇرىتىپ،ئۇنىڭ قولىنى تۇتۇپ تەسەللى بېرىپ،رىغبەتلەندۈرەتتىم.باشتا ئانچە
جىددىيلەشمىگەندەك قىلغانىدىم.ئەستايىدىللىق بىلەن ھەممىنى كۆزىتىپ
تۇراتتىم.ئايالنىڭ تولغىقى كۈچەيگەنسىرى ۋە ۋاقىت ئۇزارغانسىرى جىددىيلەشكىنىمدىن
پۇت-قۇلۇمدا جان قالمىغىلى تۇردى.ئايال چىدىغۇسىز تولغاق ئازابىدا نەچچە سائەت
قەيسەرلەرچە كۈچىدى.ماڭا ئۇ ئايال بىر بالا ئەمەس،بىر پۈتۈن دۇنيانى تۇغۇۋاتقاندەك
تۇيۇلۇپ كەتتى.جىددىيلىشىشكە باشلىدىم.بالىنىڭ بېشى،ياق يەر شارىنىڭ بىر ئۇچى
كۆرۈندى.باش پاتماي چىقىش ئېغىزى ناركۇزسىز كېسىلدى.ئايال تۇغۇتنىڭ ئازابىدىن بۇ
كېسىلىشىنى سەزمىگەن ئىدى.جىرتىلداپ قان ئېتىلىپ چىقتى.مەن قان توختىتىش ئۈچۈن
قولۇمغا ئالغان داكىنى تۇتقان پېتى ئايلىنىپ كېتىپتىمەن.ئاپئاقلىق ئىچىدە،كۆزنى
چاققۇدەك يوپيۇرۇق بىر يەرنى كۆردۈم.قۇلىقىم چىڭقىلىپ چېچنىمىنى
ئاڭلىيالمىدىم،يىراقتىن تولىمۇ غۇۋا گۇدۇڭ-گۇدۇڭ كېلەتتى...خېلىدىن كېيىن سەل-پەل
ھۇشۇمغا كېلىپ،ئاستا-ئاستا كۆزۈمنى ئاچىتىم.دوختۇرلار سۆزلىشىۋاتاتى،بالىمۇ،ئانىمۇ
يىغلاۋاتاتتى،يېنىمدا سېستىرالار.بىرەرسى ماڭا سۇ ئىچكۈزىد.يەنە بىرسى:‹‹ئەتىگەن
ناشتا قىلمىغانمۇ؟ئوپېراتسىيەخانىدا ھاۋا يېتىشمەي قاپتۇ ساڭا،قاندىن
قوركىدىكەنسەن...››دېدى،ئۇلار مەن ھوشۇمدىن كەتكەندىن كېيىن مېنى ياندىكى ئۆيگە
ئەكىرىپ ياتقۇزۇپ قويغانىكەن.

مەڭگۈ ئېسىمدىن چىقمايدىغان ئوپېراتسىيە ئىدى.ئۇنىڭ بىخۇد روھىيىتىمگە ئېلىپ كەلگەن

تەبىئەتنىڭ ئىلاھىي قانۇنىيىتىنىڭ ئۇلۇغ ھېكمىتىنى قايتىدىن تېخىمۇ ئېنىق

تاللاپ بەرگەن ھەم شۇ ئىسىم قۇيۇلغان بالىلارمۇ نەچچە ئوندىن ئاشتى.‹‹مال ئېگىسىنى
دورىمىسا ھارام››دېگەندەك،ھازىر دوستلىرىممۇ مەندىن ئىسىم سورايدىغان
بۇلۇشتى.بېيجىڭدىكى چېغىمدا بىر دوستۇم يېڭى تۇغۇلغان قىزىغا ئىسىم تېپىشىپ بەرسەڭ
دەپ تېلىفۇن قىپتۇ.مەن ئىنتېر تورىدىن ئىزدەپ،ئۇنىڭ ئىسمىغا يېقىن قىلىپ
ئۇيغۇرچە،تۈركچە،پارسچە،ئەرەبچە نۇرغۇن گۈزەل ئىسىمنى تېپىپ ئىزاھاتى بىلەن
ئەۋەتتىم.بىزنىڭ تىلىمىزدا كىرمە سۆزلەر بىر قەدەر كۆپ بولغاچقا،بۇ ئىسىملارنىڭ
ھەممىسىنى دېگۈدەك بىز چۈشىنەتتۇق.ئارىدىن يەنە ئۇزۇن ئۆتمەي يەنە بىر دوستۇمنىڭ
سىڭلىسى يەڭگىپتۇ.ئۇلارنىڭ ‹‹قۇرئان كەرىم››دىكى ئىسىملاردىن تاللاپ قويغۇسى بار
ئىكەن.‹‹قۇرئان كەرىم››دە ئىسمى زىكىر قىلىنغان ئاياللاردىن پەقەت ھەۋۋا
ئانىمىز،ئىبراھىم ئەلەيھىسالامنىڭ ئايالى سارە،پىرئوننىڭ ئايالى ئاسىيە،شەباھ
پادىشاھلىقىنىڭ خانىشى بىلقىس،ئېيسا ئەلەيھىسالامنىڭ ئانىسى بۈۋىمەريەمدىن ئىبارەت
بەش ئايالنىڭ ئىسمى بار ئىكەن،قالغان ئاياللار تىلغا ئېلىنغان بىلەن ئىسمى
ئېيتىلمىغانىدى.ئۇنىڭ ئۈستىگە چوقۇم‹‹قورئان كەرىم››دە گۈزەل ئاڭلىنىدىغان ئىسىمگە
ئوخشاش سۆزلەر نۇرغۇن.لېكىن ئۇلار ئىسىم ئەمەس پېئىل،ئالماش،سۈپەت بۇلۇشى
مۇمكىن.مەنىسىنى بىلمەي تۇرۇپ پەرزەنتىگە ئىسىم قۇيۇش چوڭ مەسئۇلىيەتسىزلىك
جۈملىسىدىن ئىدى.دوستۇم سىڭلىسىنىڭ قىزىغا ‹‹قورئان كەرىم››دىن تاللىغان ئىسىمنىڭ
مەنىسىنى مەندىن سوراپ تېلىفۇن قىپتۇ.ئۇلار سۈرە فاتىھەدىكى ‹‹لەزىنە››دېگەن سۆزنى
چىرايلىق ھېس قىلىپ ئىسىم قويماقچى ئىكەن.مەن ئىزدەپ كۆرسەم،بۇ سۆز
ئالماش،مەنىسى‹‹ئۇلار››ئىكەن،دىنىمىز مۇسۇلمانلار ئىسمىنىڭ گۈزەل بۇلۇشى ۋە
بالىلارغىمۇ گۈزەل ئىسىملارنى قۇيۇشنى تەكىتلەيدۇ.ئالللاھتىن باشقىغا قۇلچىلىقىنى
ئىپادىلەيدىغان ئىسىم،پەقەت ئاللاھقىلا خاس بولغان ئىسىم،كۇفرىلارغا خاس   
ئىسىم،شەيتان-ئىبلىسلارنىڭ ئىسمىنى قۇيۇش چەكلىنىدۇ.ئۇنىڭدىن باشقا مەنىسى كىشىنىڭ
كۆڭلىنى غەش قىلىدىغان،يېقىمسىز،يەنە كىشىنى ئالاھىدە جەلپ قىلىپ،تۇيغۇسىغا بىۋاستە
تەسىر قىلىدىغان(بۇلۇپمۇ قىزلار ئۈچۈن)،ئەخلاقسىز ئىنسانلارنىڭ ئىسمىنى دوراپ
قۇيۇش،يامان ئىش،زالىملارنىڭ ئىسمىنى دوراپ قۇيۇش،يىرگىنىچلىك ھايۋانلارنىڭ ئىسمىنى
قۇيۇش قاتارلىقلاردىن ساقلىنىش تەشەببۇس قىلىنىدۇ.بۇلار ئەلۋەتتە بالىنىڭ كېيىنكى
ئۆسۈپ يېتىلىشى ئۈچۈن پايدىلىق بۇلۇپ،چىرايلىق ئىسىم كىشىنىڭ زوقىنى
قوزغايدۇ،بالىغىمۇ سۈلكەت،غۇرۇر بېغىشلايدۇ.   

   ھازىر كۆپ ئىشلىتىلىۋاتقان
كىشى ئىسىملىرىنىڭ ئىپتىدائىي دەۋردە ئەجدادلىرىمىز ئۇلۇغلاپ قويغان
ئىسىملاردا،ھايۋانلاردىن شىر،ئارسلان،يولۋاس،توغرۇل،مېتال ۋە جاۋاھىراتلاردىن
ئالماس،ئالتۇن،كۈمۈش،مەرۋايىت،گۆھەر،پولات،تۆمۈر،تەبىئەت ھادىسىسى،شەيئىلەرنىڭ
ئىسمىدىن يۇلتۇز،ئاي بىلەن تۈزۈلگەن ئىسىملار،ئىسلام دىنغا كىرگەندىن كېيىن
ئەرەب-پارس تىلىنىڭ تەسىرىدە قۇبۇل قىلغان ئىسىملاردىن
مۇھەممەد،ئەلى،فاتىمە،ئائىشە،ھاجەر ياكى گۈل،دىل،قەمەر بىلەن تۈزۈلگەن
ئىسىملار،بالىنىڭ ھايات قېلىشىنى ئۈمىد قىلىپ قويغان
توختى،تۇرغۇن،تۇرسۇن،تۇرغۇننىسا قاتارلىق ئىسىملار،يەر ئىسلاھاتى مەزگىلىرىدىن
قالغان تاڭنۇر،ئازاد،نىجاد،ئىنقىلاب مەزگىلىدىن قالغان
غالىب،ئىلغار،قەھرىمان،ئۇچقۇن،جۇشقۇن،چەتئەللىكلەردىن قۇبۇل قىلىنغان
تېلمان،زويا،كىلارە،يېڭى زاماندا مودا بۇلۇۋاتقان
گۈلفىرە،فىلۇرە،دىلفىرە،دىلرۇس،ئېرپان دېگەندەك ھەرخىل ئىسىم ئۇچرايدۇ.ئىسىم قۇيۇش
ئاتا-ئانىنىڭ دۇنيا قارىشى،ساپاسى،قىزىقىشى،خاھىشىغا مۇناسىۋەتلىك مۇھىم
ئىش.ئۆزىڭىز ياخشى كۆرگەن،مەنىسى چوڭقۇر،يېقىشلىق ئىسىملارنى قويسىڭىز،بۇ ئىسىم
بالىڭىزغا بىر ئۆمۈر ھەمراھ بولىدۇ.ئىسىم تاللىغاندا فامىلە بىلەن
ماسلىشامدۇ،قېرىغاندا چاقىرىشقا لايىقمۇ؟ئەقىدە-ئادەتكە زىتمۇ-يوق؟مەنىسى
ياخشىمۇ؟دېگەندەك نۇرغۇن جەھەتتىن باش قاتۇرىشىمىزغا توغرا كېلىدۇ.مېنىڭ
يەكۈنۈم،ياشىغانىكەنمىز،ھاياتنىڭ ماھىيىتىنى بىلىپ ياشىشىمىز،ئىسىم
قويىدىكەنمىز،مەنىسىنى بىلىپ قۇيۇشىمىز كېرەك.
