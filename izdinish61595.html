مەن تور بىكەتلەردىكى2015 – يىلى كىتاپ ئۇقۇش يىلى بۇلشى كىرەك دىگەن تەشەببەسنى كورگەندىن كىيىن ھاياجان ئىلكىدە ئۇخلىيالمىدىم ....بىزدىمۇشۇنداق ئىنسانلار بار ئىكەن . مەرھۇم ئالىمىمىز دىگەندەك «ئېتىز بىلەن ھۇسۇل ، نامازبىلەن غۇسۇل . دۇتار بىلەن ئۇسسۇل » دىگەن ئۇچ پىرقىراتما تەپەككۇر ئىچىدە يۇريدىغان ئىنسانلاردىنمۇ باشقا بىر ئىنسانلار بار ئىكەن . ئۇزى ۋە ئوزگىنى ئويغىتىشنى مەجبۇرىيەت ھەم شەرەپ دەپ بىلىدىغان ، ئوز خەلقى ئۇچۇن مۇشۇنداق ياخشى ئىجابىي ئۇسۇللارنى تېپىپ چىقىدىغان قېرىنداشلار بار ئىكەن دىگەنئويدا چەكسىز سۇيۇندۇم .....ئۇزۇم توغۇرلۇق ئويلاندىم مەن قانچلىك كىتاپ ئۇقۇدۇم . مەن قانچلىك مىۋە بەردىم . ھاياتمنى كىتاپ ئارقىلىق قانچىلىك گۇلەندۇردۇم . مىنىڭ ھاياتمنى كىتاپ قانچىلىك ئۆزگەرتتى . مەن كىتاپلاردىن نىملەرنى ئىرىشتىم . كىتاپ توغۇرلۇق ئويلۇنۇشقا تىگىشلىك قېرىنداشلىرىمىغا نىملەرنى دەپ بىرىشىم كىرەك . بىراق بۇ سۇئاللارنىڭ تىگى ۋە جاۋابى تۇگمەيتى . چۇنكى ھەر كىمنىڭ تۇرمۇشتىكى شارائىتتى، كورگەنلىرى ئوخشمايتتى. مەن ئۇزەمگە جاۋاپ بىرەلگىنىم بىلەن قېرىنداشلىرىمغا ئوبىكتىپ جاۋاپبىرەلمەيتتىم .قانداق قىلىشىم كىرەك ؟ قىرىنداشلىرىمغا ئەگەر جاۋاب بىرىش كىرەك بولسا نىمە دەپ جاۋاب بىرىشىم كىرەك . چۇنكى مەن كىتاپ ئۇقۇيتتىم بىراق بىر نەرسە دەپ بىرەلمەيتتتىم ھەم بىرنەرسە دەپ بىرىپ باقمىغانتتىم .
ئىلىمگە ئەمەل قىلمغان ئالىم كىتاپ ئارتىلغان ئىشەككە ئوخشايتتى . ....
مەن بولسام ئاددى بىر كىتاپ ئۇقۇغۇچى ھەم كىتابى بىلىملەر بىلەن يات بىر مۇھىتىتاتىرىشپ تۇرمۇش كەچۇرۋاتقان ، كىتاپ بىلەن ئۇزەمنىڭ ھاياتنى بىزەشكەتىرىشۋاتقانلارنىڭ بىرسى . بىراق رۇزى مۇھەممەت تۇردى مۇئەللىمنىڭ كىتاپ ئۇقۇش روھىدىن ۋە كىتاپ ئۇقۇش ئۇچۇن چاقىرۋاتقان كۇچىدىن قاتتىق تەسىرلەنمەي ھەم ئۆزئۇزەمنى سوراققا تارتماي تۇرالمىدىم .
ئەدەبىيات ئوقۇتقۇچىسى ئابدۇقېيۇمىنىڭ يىتەكچىلىكىدە ئەدەبىيات ئوقۇتۇش گوروپىسىدا 30يۈۋەندىن كىتاب چىيى تەسىس قلىندى. ھەر ئايدا 15نەپەر ئوقۇتقۇچىدىن يىغىلغان پۇلغاشۇ كىشىدە يوق كىتاب سىتىۋىلىنىدىغان باشقىلارمۇ تەڭ پايدىلىنىدىغان قائىدەبىكىتىلدى. 23-ئۆمىد كۈنى بىز ئالاھىدە ۋاقىت چىقىرىپ بارلىق كىتابلارنى بىر يەرگەتوپلاپ ئوقۇغۇچىلارغا ئالدى بىلەن ئۆزىمىز ھەققىدىكى نادىر كىتابلارنى بىر قۇر تونۇشتۇرۇشتۇرغاندىكىيىنكىتاب ئوقۇش پائالىيىتىنى ئىلىپ باردۇق.بۇنىڭدا ھەر بىر ئوقۇغۇچىغا بىردىن كىتابتارقىتىپ ئۇنى ئوقۇپ بولغاندىكىيىن شۇ كىتابتىن سۇئال چىقىرىپ زىھىن سىناش شەكلىدەپائالىيەت ئىلىپ باردۇق،بۇنىڭ ئۈنىمى نەچچە سائەت چىڭقىلىپ ئۆتكەن دەرىستىن ئۈنۈملۈكرەكبولدى.ئۇيغۇر تارىخىدا بەلگىلىك قىممەتكە ئىگە ئەنە ئەشۇنداق كىتابلار جەمىيەتتىكى ساخاۋەتچىلەرنىڭ فوندىمىزغا پۇل ئىئانە قىلىشى بىلەن كۆپىيىپ ھازىرغىچە 1000پارچىدىن ئاشتى.
«مىنىڭ كىتاب يىلىم» رۇزىمۇھەممەدتۇردى ئارتۇچنىڭ شۇ ناملىق ماقالىسىدىن
شۇنىڭ بىلەن ئارتۇچ مۇئەللىمنىڭ خىيالى بىلەن ئۇزەمدىكى كىتاب ۋە ژۇرناللارنى ئاختۇرىشقاباشلىدىم .مەندىمۇ 3000 يىقىن كىتاپ ۋەژورناللار ساقلانغان بولسىمۇ ئۇقۇغانلىرىمىدىن قىلغان ھاسلاتىم كوپ ئەمەس . بىراق شۇنداق دىيەلمەنكىمئائىلەمدىكى مۇھىت ئىچىدە بارلىققا كەلگەن كىتاپ ئۇقۇش ئادىتىنىڭ ماڭا ئېلىپ كەلگەن بايلىقى ھاياتىمىدىكى ھەر قانداق ئۇتۇقتىن ئارتۇق بولمىسا كىرەك . ئائىلەمدىكى مۇرەككەپ زىدىيەت ،كەسىپ تۇرمۇش تۇسالغۇسى ۋە يۇقىرلاپ ئۇقۇيالمىغىنىم ، ماڭا ئىلىپ كەلگەن ھەر بىر زىيانلىق ئامىلىلارنىڭ تەسىرىنى مەن پەقەت كىتاپ پەقەت كىتاپ ئارقىلىق يىڭىپ چىقىتىم .شۇ سەۋەپتىن ھازىرىقى «مەن » نى كىتاپ ياراتقان دىسەم ئارتۇق كەتمەيدۇ .شۇ سەۋەبىتىن مۇشۇ چاقىرىق ئارقىلىق جەمىيتىمىزدىكى كىشلەرنى ئويغاتماقچى بولغان شۇ ئېسل كىشلىرىمىزگە بېشىم بىلەن ئىگىلىپ ئاپىرىن ئۇقۇيمەن .
تۇۋەندە كىتاپ ئۇقۇش ھەققىدە ئۇنۋىرسال پىلاننى ئوتتۇرغا قۇيۇش بىلەن بىر نەچچە تۇرلۇك كىتاپ ھەققىدىكى پىكىرلىرىمنى ، شۇنداقلا كىتاپ ئۇقۇش يىلىنى قانداق ئوتكۇزۇش تۇغۇرسىدائويلىنۋاتقان ھەر بىر قېرىندىشىم ئۇچۇن تەكلىپ قاتارىدىكىئويلۇرۇمنى ئوتتۇرغا قۇيۇپ ئۇتىمەن . دوسلارنىڭ ئاقىلانە تەپەككۇر ۋە ئاچچىق تەنقىد ئىچىدە ئويلىنشىنى ھەمدە شۇ توغۇرلۇق ئالغابېسىشنى ئۇمۇت قىلىمەن ھەمدە ئۇلۇغ ئىگەمدىن تىلەيمەن .

بۇ ئۇچ ئىستىرلكىغا بىرنەرسىلام مەنە ۋە ئەھمىيەت ئاتا قىللايدۇ . ئۇ بولسىمۇ كىتاپ .

دانانىنى دانا قىلغانكىتاپ ناداننى بار قىلغانمۇ ھەم كىتاپ .
مىللەتنىڭ خارابىلىقى ئاياللارنىڭ چۈشكۈنلىكىدىن باشلىنىدۇ ،ئاياللارنىڭ چۈشكۈنلىكىگەئەرلەرنىڭ مېھرىبان بولالماسلىقى سەۋەب بولىدۇ ،ئەرلەرنىڭ مېھرىبان بولالماسلىقىغائۇلارنىڭ ئېتىقادسىزلىقى ،كەمبەغەللىكى ،جاپا مۇشەققەت ،كۈچسىزلىكى سەۋەب بولىدۇ .
ئىنساننىڭ ئىتقادسىزلىقى كەمبەغەللىكى ...ۋاكاھازلار بىلىمسىزلىكىتىن باشلىندۇ . بىلىمىسزلىك كىتاپ ئۇقۇماسلىقنىڭ كىلىپ چىققىدىغان مۇقەرەر ئاقىۋەت .
ئەيساجان تۇردىنىڭ«كىتاپنىڭ كۆز يېشى » دىگەن ماقالسدە تىلغا ئېلىنغاندەك
ئۇيغۇرلار تارىختىن بۇيان كىتاپنى قەدرلەپ كەلگەن مىللەتلەردىن بىرى . شۇڭا قارخانىيلار سۇلالسىدىكى«ساچىيە» مەدىرىسى خانلىق كىتاپخانسى،مەۋلانى جالالىدىن رۇمى قۇرغان «كۈنىيا كۇتۇپخانىسى»، ئەلشىر نەۋائى دەۋرىدىكى«ھىرات كۇتۇپخانىسى»، يەكەن سەئىدىيە خانلىقى دەۋرىدىكى « خانلىق كۇتۇپخانىسى»،قۇمۇل ۋاڭلىقى دەۋرىدىكى «ئوردا كۇتۇپخانىسى» قاتارلىقلار قەدىمقى ئەجداتلىرىمىزنىڭ بىلىم ۋە كىتاپ خۇمارلىقىغا تەمسىلھەمدە ئەجداد روھنىڭ ئۇلۇغ بىلىدىغان مىللىتمىز ئۇچۇن ئۆرنەك بۇلالايدۇ .ئەجدالىرىمىزنىڭ قىممەتلىك ئەدەبىي مىراسلىرىنى ئەينى ۋاقىتتىكى مەتبە تەرەققىقىلمغان زامانلاردا كۇچۇرۇپ تارقىتىش ۋە ياكى ئىپتادئى باسما مەتبە ئۇسۇللىرى بىلەن قانچلىك تەسلىكتە تارقاتقانلىقى ھەمدەئۇنى ئاسىراپ بۇگۇنگە يەتكۇزۇپ بەرگەنلىكى ھەممە قېرىنداشلىرىمىزغا مەلۇم بۇلشىكىرەك ئەلۋەتتە . يۇسۈپ خاس ھاجىپنىڭ «قۇتادغۇبىلىك» نى ھەسەن بۇغراخانغا، مەھمۇدقەشقىرىنىڭ«تۈركى تىللار دىۋانى»نى ئۇبۇلقاسىم ئابدۇللا بىننىي مۇھەممىدىن مۇقتەردى بىئەمۇرۇللاغا تەقدىم قىلغانلىغىنىڭ ئۇزىلا شۇ زاماندىكى ئەڭ باي ئەڭ ھۇقۇقدار كىشلەرنىڭمۇ كىتاپنى قەدىرلەيدىغان روھىنىڭ تارىخى بايانلاردا ئەتكەن كۇرۇنىشىدۇر . ئېيتىلىشلارغا قارىغاندا سوتسيالىستىك ئىدىيىنىڭ باش پىشۋالىرىدىنبىرى بولغان لېنىن شۇ زاماندىكى ئەڭ كوپ كىتاپ ئۇقۇغان كىشى ئىكەن . دولتىمىزنىڭ داھىسى ماۋزىدۇڭنى باش پىرسۇناژ قىلىپ ئىشلەنگەن كىنو – فىلىملاردا ماۋزىدۇڭنىڭ قۇلىدىن كىتاپ چۇشمەيدىغان قىلىپ تەسۋىرلىندۇ . ئەمىليەتتە ماۋزىدۇڭ ھەققىقەتەن كىتاپ خۇمار كىشى بۇلۇپ ئامېرىكىلىق يازغۇچى تېللىر «ماۋزېدۇڭنىڭ تەرجىمىھالى»دېگەن كىتابىدا «دۇنيادىكى داھىيلار ئارىسىدا ھەممىدىن بەك كىتابخۇمار- ماۋزۇدۇڭ » دېگەن بىر سۆزنى ئوتتۇرغا قويغان . جۇڭنەنخەيدىكى ماۋزېدۇڭ كىتابخانىسىداساقلاقلىق كىتاب 90مىڭغا يېتىدىكەن. ماۋزېدۇڭ «ئەلنى ئىدارە قىلىشتىكى پايدىلىق ئۆرنەكلەر»دېگەنكىتابنى 17قېتىم ئوقۇغان، ھەتتا ئۇ «قۇرئان كەرىم»نى مەخسۇس خەنزۇچىغا تەرجىمە قىلدۇرۇپ بىر نەچچە قېتىمئوقۇپ، تەتقىق قىلىپ ئۆزىنىڭسىياسىي ھاياتى ئۈچۈن ئۈنۈملۈك پايدىلانغان.
بىراق بىزدىچۇ ؟يىللاردىن بۇيان بېسلىۋاتقان كىتاپلاردىن بىر قىسم قايتا قايتا نەشىرقىلىنۋاتقان كىتاپلارنىڭ باشقىلىرىدىن ھەممىسنىڭ تىراژى تۇۋەنلەپ بارماقتا . بۇنىڭ ئۇزى بىزدىكى كىتاپ ئۇقۇشقا قىزىقماسلىقتەك ناچار خاھىشنىڭ بازار ئىگلىكىدەكۇرۇلۋاتقان ئەينى تەسۋىرى ۋە تەسىرى . بىراق بۇخىل خاھىش بىزنى نەگە ئاپىرىدۇ ،ئەجداد كىتاپ ئۇقۇمايدۇ، ئەۋلات كىملىكنى تۇنمايدۇ، يەنە بىر ئەۋلات تىرىلمەس ئۇلۇككەئايلىندۇ .شۇنىڭ بىلەن يەر شارىدا بىر توپ ماڭقۇرت ئاۋۇيدۇ . ئاخىرىدا ھىسابات بۇگۇنكى بىر قاپ تاماكا ياكىبىر بوتولكاھاراق پۇلغا نەچچە كىتاپ سېتۋاللايدىغان ھازىرىقى زامان كىشللىرىگە مەڭگۇلۇك لەنەت تامغىسى بېسىلىدۇ خالاس.
مەن نۇرغۇن كىشلەرنى ئۇچۇراتتىم . بەلكىم تور ئالدىدىكى ھەر بىر قىرىندىشم بۇنداق ئەھۋاللارغا دۇچكەلگەن. ئۇلار ئيتىدۇكى
-ئالىي مەكتەپتە كىتاپنى تازا ئۇقۇپتىكەنمىز ، ھازىر ئۇقاي دىسەم بېشىم ئاغرىيدۇ .

بۇ خىل جاۋاپقا ئەقىللىق ئۇقۇرمەن ئەلۋەتتە قايىل بولمايدۇ .ئىنساننىڭ تۇرمۇش سۇپتىنى ھوقۇق ۋە پۇل بىلەنئولچەشنىڭ ئۇزىلا ئادىمىيلىك قىممەتتىن يىراقلاشقانلىقنىڭ ئىپادىسى .ئىنساننىڭ پايدا ئارقىلىق مەلۇم ئىش ھەركەتنىڭ ئۇنۇمنى ھىساپلاش فورملاسنى خاتا دەپ دەپكەتكىلى بولمىغاندىمۇ ئۇنىڭ يىراق ۋە يېقىن پايدىسى توغۇرلۇق ئويلانماي بولمايدۇ .ئىقتىسادۋە ھۇقۇق ھەر بىر قۇۋۇمغا مۇھىمبولغىنىدەك ئۇيغۇرمۇ ناھايتتى كوپ ئىقتىسادى كاپالەتكە ۋەئىجتمائى كۇچلەرنىڭ ھوقۇقىغا تەشنا ۋەئىھتىياجلىق . بىراق بۇ بىر ئەۋلات غۇلسىچۇ ئىقتساد بولمسا ھوقۇق ئەملىيلەشمىسچۇ .بىر تۇركۇم بىچارە پەرزەنتلەر تىلەمچلىكتىن ئىبارەت يېڭى چىقىش يۇلى تاپماي ئىلاج يوق . بىراق يىڭى دەۋىر دە كىشلەر يۇقىرى سۇرئەتىلىك تاش يولدا مىڭىپ قالسا بۇتىجارەتنىڭمۇ ئومىرى كۇتە بۇلۇپ قالامدىكىن ئىھتمالىم .
يەھۇديلارنىڭ ئويگە ئوت كەتسە كىتاپلارنى ئالدى بىلەن قۇتۇلدۇرۋالدىغان،باللىرغا تەۋراتقا ھەسەل سۇركەپ يالتىدىغانلىقىدەك ھىكايەتلەر ھىچكىمگە سىربولمسا كىرەك . يەھۇدىيلارنىڭ نۇپۇسى ئۇيغۇرلارنىڭ نۇپۇسغا يىقىن كىلىدۇ (13مىليۇن). دۇنيا نۇپۇسنىڭ 0.2 پىرسەنتىنى ئىگەللەيدۇ . بىراق دۇنياۋى تەسىرىگە ئىگەمۇكاپات بولغان نوبىل مۇكاپاتنىڭ ھەر قايسى تۇرلىرىدىن ئالغان مۇكاپاتى بۇمۇكاپاتنىڭ ئومۇمى سانى بىلەن نىسپەتلەشتۇرگەندە 45 پىرسەنتىدىن يۇقىرى . (ئۇمۇمىسان 360 نەپەر ،يەھۇدىيلار نىڭ 165 نەپەر «يەھۇديىلارنىڭ دولەت قۇرۇش جەريانى »دىگەن ماقالىدىن ئېلىندى) بۇ يەھۇدىيلارنىڭ كىتاپ ئۇقۇش ئارقىلىق ئۇزنىيۇكسەلدۇرگەنلىكىنىڭ جانلىق پاكىتى . 1884 –يىلى ئولگىلى ئىىككى مىڭ يىل بولغانئىبراي تىلدا تۇنجى چىقىرلغان «ھارۋىسد » ناملىق ھەپتلىك ژورنالنىڭ تارقىتىلشىبىلەن قىسقىغىنە 30يىل ئىچىدە يەھۇدىيلارنىڭ (پەلەستنىدىكى) تىل بىلش نىسپىتى %70گە يەتكەن . 1948 –يىلى بولسا مۇستەقىل بىر دولەتنىڭ تىلغا ئايلانغان . 1969-يىلىغا كەلگەندە ئىسرائىليىدە چىققىدىغان گېزىت ژورناللار 481 گە يەتكەن . 2012يىللىق سىتاتىسكىللارغا ئاساسلانغاندا يەھۇديلارنىڭ ھەر يىلى ئۇقۇيدىغان ئوتتۇرچەكىتاپ سانى 64 پارچە بولغان بۇلۇپ دۇنيادىكى ئەڭ كوپ كىتاپ ئۇقۇيدىغان دولەت بۇلۇپقالغان ( بىراق بىر مىليارت 300 مىليۇن نۇپۇسلۇق جۇڭگودا 5 پارچىگىمۇ يەتمىگەن ) .يىقىنىقى زامان تارىخدىن ئېلىپ ئېيىتقاندا يەھۇدىيلار دۇنيا بۇينچە ساۋاتسىزلىقنى ئەڭ بالدۇر تۇگەتكەن مىللەت ھىساپلىدىكەن . ھازىر 5 مىليۇن يەھۇدىي 250 مىليۇن ئەرەبنىڭ بېشدا ياڭاق چېقىۋاتىدۇ .دۇنيا ئىقتىسادىنىڭ 60 پىرسەنتىنى يەنە مۇشۇ خەق ئىگەللەۋاتىدۇ . مانۇ بۇ نىمنىڭ قۇدىرىتى ئەلۋەتتە كىسپ ئېيتالايمىز كىتاپ يەنە شۇ كىتاپ . باشقا مىللەتلەردىن ياپونلارنىڭ كىتاپ ئۇقۇش قىزغىنلىقى توغۇرلۇق تورداشلار خېلى خەۋەردار دەپ ئويلايمەن. مانا ھازىر ياپونلارنىڭ ئىقتسادى دۇنيادىكى ئالدىنقى قاتارغا ئوتتى .
مانا بۇلارنىڭ ھەممىسى كىتاپ ئارقىلىق ۋۇجۇدقا چىققان نەتىجىلەر ھەم كىتاپ ئۇقۇش بىلەن ئۇزنى گۇلەندۇرگەن مىللەتلەرنىڭ پارلاق نەتىجىلىرى .
ئۇنداقتا بۇ يەردە بىرمۇھىم سۇئال پەيدا بولدى . بىز رىقابەت ھەم تەرەققىيات قەدىمى يۇرتمىزدىكىدىن نەچچە ھەسسە يۇقىرى بولغان ئاشۇ يەرلەردە ياشاۋاتقان يەھۇدىيدىن شۇ ياپوندىنمۇ بەك ئالدىراشما   . توغرا بىز ئالدىراش بىراق بىز رىستۇرانغا بارىدىغان يولدا ئالدىراش . ئۇ يەردە بىزگە بىرىلىدىغان بارلىق نىسۋىلەر بار . ھاراق بار گۇزەللەر بار ، ئۇ يەردە بىززۇۋانمىزنىڭ ئىھيتياجى بىلەن ئىشتاننىڭ ئېغىنىڭ غېمنى بىر تەرەپ قىلالايمىز . ....
يىقىنىقى زامان تارخىدا كىتاپ ئۇقۇغان ئادەملىرىمىزنى جەدىت دەپ ئوزقۇلمىز بىلەن بۇغۇپ ئولتۇردۇق . ھازىر بولسا شۇنداق ياخشى شارائىت ياخشى كىتاپلىرىمىزتۇرسىمۇ يەنلا كىتاپ ئۇقۇغان ئادەملىرىمىزدىن ئۇقۇمايدىغانلىرىمىز تۇلا . بايلارخۇتۇننىڭ ئۇستىگە خۇتۇن ئېلىپھەتتا ئاشنائويناپ ئاۋارە . زىيالىلار قىمارخانا بىلەن قاۋاقخاندىن چىققالمايۋاتىدۇ .ئەمەلدارلار كۇندۇزى دوكلات بىلەن يىغىننىڭ ،كەچلىكى بولسا خۇشامەتچلەرنىڭ داستىقىندا ئالدىراش . ئانا بۇلدىغان ئاياللىرىمىز كورىيە بىلەن تۇركىيەنىڭ ئۇزۇندىن ئۇزۇن سۇزۇلدىغان كوپ قىسملىق تىلۋىزىيە تىياتىرىلىرىنى كۇرۇش بىلەن ۋاقتىنى زايە قىلۋاتىدۇ .....
يىقىندا گۇزەل ئېلى دىيارىغا جەمەتىمنىڭ نەسەپنامسى ئۇچۇن ئىزدىنش سەۋەبى بىلەن بېرىپ قىلىپ ، قۇتلۇق پۇرسەتلەرنىڭ بىرىدە مەشھۇر شائىرىمىز مۇھەممەتجان راشىدىن ئەپەندىنىڭ كىسەل بۇلۇپ قالغانلىقىنى ئاڭلاپ بۇكىشى بىلەن يۇز كۇرشۇش ھەم كىسەل يوقلاش باھانسى بىلەن ئۇنىڭ ئۇيىگە ئۇتۇپ قالدىم. سوھبەتلىرىمىزدەياشلىرىمىزدىكى كىتاپ ئۇقۇشقا دائىر بىر نەچچە مەسلە تۇغۇرلۇق پاراڭلاشقىنىم ئىسىمدە . يۇرۇتقا قايتىپ كىلىپ منى شۇ ئەزىز كىشنىڭ ئۇيىگە باشلاپ بارغان يازغۇچى مۇھەممەتجان ئەيسانىڭ ھېيتلىق تىلفۇنىنى تاپشۇرۋالغىنمدا شائىرىمىزنىڭ كىتاپ ئۇقۇشقا ھىرىسمەن ياشلارنىڭ ھېلمۇ كوپ ئىكەنلىگنى مىنىڭ ئاغزىمدىن ئاڭلاپ ناھايتتى سۇيۇنگەنلىكىنى ، ئۇزىنىڭ ياشلارغا بولغان ئىشەنچىنىڭ تېخمۇ ئاشقانلىقنى سۇزلەپ بەرگەنلىكىنى يەتكۇزدى . تىلفۇننى قۇيۇپ شۇنداق ئېسىل كىشلىرىمىزنىڭبۇگۇنكى زامان ياشلىرىمىزنىڭ كىتاپ ئۇقۇشقاكۇڭۇل بۇلشنىڭ ئارقىسدىكى نەقەدەر ئۇلۇغ ئىزگۇ تىلەكلىرىنى ھېس قىلغاندەك بولدۇم .
ئادەم ئىدىيسى ئائىلە جەمىيەت بىلەن مائارىپىنىڭ ئۇچ خىل مۇھىتنىڭ شارائىتىدا يىتلىدۇ . ھازىر مائارىپىمىزدىن قاقشاش مەسلىسى ، ئۇمۇدسىزلىنىش مەسلىسى ئېغىر . مەكتەپ دارۋازىسدىن كىرىپ كىتىپ بەش – ئالتە يىلدىن كىينمۇ ساۋاتسىزبۇلۇپ چىققان ئەھۋاللارنى كۇندە ئاڭلاپ تۇرۋاتمىز . ئائىلىدە ئاتا- ئانىلارنىڭ پەرزەنتلىرىدىن قاقشاشلىرىنى قەدەمدە بىر ئاڭلاۋاتىمىز . جەميەتمۇ كۇندىن كۇنگەمۇرەككەپلەشمەكتە .ئۇنداقتا ئائىلە ،جەمىيەتۋە مائارىپ بىرەلمىگەن تەربىيىنى قانداق تۇلۇقلاش كىرەك ؟ئۇنى پەقەت كىتاپ ئۇقۇش ئارقىلىق تۇلۇقلاش تامامەن مۇمكىن . شۇ سەۋەبتىن بالىلىرىمىزنىڭ كىتاپقابولغان قىزىقىشنى يىتلدۇرۇش ھەمدە ئۇنى جەميەت تۇسىدە ئەۋىج ئالدۇرۇش بۇگۇنكى زامان ئۇيغۇرلىرىنىڭ ئەڭ ئاقلانە تالىشى ۋە مۇقەرەر بولغان چىقىش يۇلىدۇر .
بىزدە ھازىر 10 ئادەمنىڭ توققۇزىنىڭ كىتاپ بىلەن خۇشى يوق . ئۇنداىداردىكى ئەخلەت تىمىلارنى ، دوسلار چەمبىرىكىدىكى چۇشكۇنلۇك ، يالغانچىلىققا تولغان قىسقاپارچىلارنى ، چاكىنا رەسىملەرنى كۇرۇشكە ۋاقتى يىتدۇكى ھەققى پۇتۇن مەزمۇنلۇق ، ئادەم مىڭىسگە مەلۇم بىر ئىدىيەنى ، يىڭى بىلىمنى سىڭدۇردىغان ، چۇڭقۇر تۇنۇش ۋە مول تەسەۋۇرغا ئىگە قىلدىغان بىرەر كىتاپنى ئۇقۇشقا ۋاقتىيەتمەيدۇ . قىسقىسى بىرقىسىم ئادەملىرىمىز يا ئۇقۇمايدۇ يا ئۇگەنمەيدۇ .شۇنىڭ بىلەن ئىش ھەركىتى مەنسىز ،تۇرمۇشى نىشانسىز.... سوزلەڭ دىسە مىللەت خەلىق ئۇچۇن تازا ۋالاقشپ سوزلەيغان ئادەملىرىمىز بەك تۇلا. بىراق زەن سېلىپ قارىسىڭىز مەيدانى ئېنىق ئەمەس . ساختا غورۇرى ، نائېنىق ئىتقادى مانا مەن دەپلاچېنىپ تۇرىدۇ . ئوز مەۋقەسنى ئەملىي ھەركىتى بىلەن ھەققى ئىپادلەشنى ، قوغداشنى بىلمەي بىر ئومۇر بىھۇدە ئۇرۇنۇشلار ئارىسىدا تولغۇنىدۇ . شۇنىڭ بىلەن نۇرغۇنلىرىمىزنىڭ ئېتىئۇيغۇر قىلغىنى «ئۇرغۇي» بۇلۇپ چىقىدۇ . ...
ئەلۋەتتە يۇقارقى قۇرلار كىتاپ ئۇقۇمايدىغان ، كىتاپنىڭ تۇرمۇشمىزغائەكلىدىغان پايدىلىق تەرەپلىرىنى چۇشەنمەيدىغان قېرىنداشلىرىمغا قارىتىلغان . بەزى ئاچچىق گەپلىرىم ئۇتۇپ كەتكەن يەرلىرى بولسا قېرىنداشلىرىمنىڭ ئەپۇ قىلشنى ئۇنى ئۇزى ئاچچىق لىكىن شىپالىق دورا قاتارىدا بىلشىنى ئۇمۇت قىلىمەن .
تۇۋەندە ئۇزۇمدە باربولغان ھەمدە ماڭا تەسىر كورسەتكەن كىتاپلاردىن بىر ئۇنۋىرسال كىتاپ ئۇقۇش پىلانى تۇزۇپ چىقتىم . دوسلارنىڭ پايدىلىنىشنى ئۇمۇت قىلىمەن .
بۇ پىلان تۇزۇشتىكى بىرقانچە پىرىنسىپلار
1.    تارىخى رومانلار يىقىنقى زاماندىن يىراق تارىخقا بولغان تەرتىپ بۇينىچە تىزىلدى .
2.       پىلان تارىخي رومان ، ئەدەبىي ئەسەرلەر ، تارىخى ئەسەرلەر ،پەلسەپە مەدىنيەتجەميەت ئىقتساد تۇغۇرسىدىكى كىتاپلار ،دىن تەربىيەت ھەققىدىكى كىتاپلار شئېرنەسىرلەر(بۇ تۇردىكى كىتاپلارنى تۇرگە ئايراي دىسەم تۇرلەر بەك كوپ بۇلۇپ كىتىدكەن. ھەمدە ۋاقىت تەرتىۋىگە چىقىشمايدىكەن)،چەتئەلنىڭ داڭلى رومانلىرى قاتارلىق 6 خىل تۇرگە بۇلۇنۇپ ھەپتىنى بىرلىك قىلىپ ئوتتۇرغا قۇيۇلدى .
3.       تارىخى ئەسەرلەردە ۋاقىت تەرتۋى يوق .
بۇ پىلاندىن پايدىلىنىش ئۇچۇن بىر نەچچە پىكىر
1.    بۇ پىلان ئىچىدىكى كىتاپلار بىر قەدەر زىچ ئورۇنلاشتۇرۇلغان . شۇ ھەپتىگە توغرا كەلگەن كىتاپ سانى ئوتتۇرا ھىساپ بىلەن 6پارچە ، قېرىنداشلىرىمنىڭ ئوز تۇرى بۇينىچە خالىغان بىر تۇردىكى كىتاپنى تاللاپ ئۇقۇشىغا قۇلايلىق بولسۇن ھەمدە كىتاپ ئۇقۇش يىلدا قايسى كىتاپلارنى سېتۋېلىش ئۇچۇن گاڭگىراپ قالغان دوسلارنىڭ پايدىلىنىشى ئۇچۇن شۇنداق تۇزۇلدى . ئوخشاشلا بىرئاينى ياكى بىر پەسلنى پىلان قىلساقمۇ مۇشۇ پىرىنىسىپ بۇينىچە تالىسا بۇلىدۇ .
