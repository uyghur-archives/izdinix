ئەسسلاممۇ ئەلەيكۇم ئىزدىنىش مۇنبىردىكى تورداشلار ئەممىڭلارنىڭ ئەھۋالىڭلار ياخشىمۇ، بۇلۇشۇمچە ئىزدىنىش مۇنبىردە چەتئەلگە ساياھەت ئوقۇش كاماندۇرۇپكا قاتارلىق ئىشلار بىلەن چىقىپ كىرىپ يۇرگەنلەر كۆپرەك شۇڭا تورداشلاردىن مۇشۇ ۋىزا توغۇرلۇق مەسلەھەت ئالماقچى ئىدىم ، ئەسكەرتىش مېنىڭ پاسپۇرتۇم بار :;P بۇنداق دىيىشىمدىكى گەپ بەزى تورداشلار چەتئەلدىن كىلدىغان تەكلىپ قەغەزنى ۋىزا دەپ ئالماشتۇرۇپ قۇيدىكەن، مەن ئوزەم غۇلجا شەھەردىن بىزگە بىر قەدەر يىقىنراق بۇلغان قازاقىستانغا چىقىپ كىلىش ئارزۇيۇم بار ئىدى ،يىقىندا ئالماتاغا ساياھەت دىگەن تېما يۇللانغاندىن كىيىن بىر چىقىپ ساياھەت قىلىپ كىرىش ئارزۇسىدا ئۇشبۇ تېمامنى يىزىپ ئولتۇرۇپتىمەن، ئەمدى تورداشلار قازاقىستان ۋىزسىنى بىجرىش ئۇچۇن قانداق ماتېرىياللار لازىم ، قانچە پول ئەتراپىدا پۇل كىتىدۇ،مۇشۇ ئىشلار توغۇرلۇق ئازراق يۇل كۆرسەتسەڭلار . بۇلۇشۇمچە بانكىغا ئازراق پۇل قۇيغانلىق ئىسپات كېرەكمىش .مۇشۇ ئىشتا ياردەم قىلدىغانلار بۇلسا رەھمىتىمنى بۇلدۇرگەن بۇلار ئىدىم ..ئۇندىدار نۇمۇرۇم 623502092
