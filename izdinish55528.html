پارول يۇمۇرلىرى

ئەسسالامۇئەلەيكۇم……


تۇتتى،تۇتتى، يەنە تۇتتى ما كىسەل.


ئەمدى ھەرگىز يازمايمەن دەيمەن، يەنە يېزىپ كىتىمەن…… قانداق قىلسام بولار؟   داۋاسى بارمىدۇ؟


ماۋۇ مىنىڭ جىيەنىم مانداق بىر ئوينى ئوتتۇرىغا قويدى. مۇشۇ ئىنسانلار بىلەن تەبىئەتتىكى ھايۋاناتلار بىۋاستە پاراڭلىشالىغان بولسا، ئىنسانلار ئۇلارنىڭ تىلنى تولۇق چۈشەنگەن بولسا، جىق ئىشلار ئاسان بولۇپ كىتەتتى……


مەسىلەن: 1. گۈزەل يۇرت غۇلجىنىڭ ئىشچان، جاپاكەش يېزا ئاياللىرى سەھەردە كالا ساغقىلى چىقتى،   سىيىر:


-ھۇي ئايشىمىدە، يەنە كەلدىڭىزمۇ، خويمارا، مۇشۇنداق ئەتتىگەندە يوقلاپ كەلسىڭىز خويمۇ ئوبدان ھە، ۋۇييەي ئاستا، ئاغرىتىۋەتتىڭىز. مەن دەيمەن ئايشىمدە، مۇشۇ سۈتنى سېغىپ ئېلىپ، ئاندىن يەنە قاينىتىپ ئاندىن سىنچاي دەملەپ، ئۆيىڭىزدىكىلەر ئەتكەن چاي قىلىپ ئاۋارە بولىدىكەنسىز. ئۇنىڭدىن كۆرە ھە، ماڭا ماۋۇ كىچىدە قارا چاي بىلەن تۇز يىگۈزۋىتىپ، ئەتتىگەندە ئەتكەن چاي سېغىۋالسىڭىز بولمامدۇ ھۇ ھۇ ھۇ ھۇ!!! ۋۇي ئاستا، توشتىمۇ چىلەك؟


2.ئەزىزانە قەشقەرنىڭ يېزىسىدا سەھەر بولغاندا   خوراز چىللىدى:


- قوپاڭلا！！！ قوپاڭلا خانلار، قوپاڭلا جانلا، تاڭ ئاتتىغۇ! غوجام بەك تۇراممايمىزغۇ؟ يۆلەپ قويايما؟! ماڭ قايلاڭلا～～ئاخشام ياتمايسىلە لىڭشىپ، سەدە قوپمايسىلە غىڭشىپ …… شۇنداما～～بولاڭلا، چىيىڭلىنى ئىچىڭلا……مەن بېرىپ پۈۋلەپ بېرەيما؟؟؟


3.قەدىمىي يۇرت كۇچادا بىر دېھقان ئىشەك ھارۋىسىنى ھەيدەپ، تۇققىنىنىڭ ئۆيىدىن چىقتى. يولدا ئىشەك سۆز باشلاپ قالدى:


-ماڭ قارىسىلا، بۇ تۇققانلىرىدا باغۇ زادى مېھىر يوق جۇمۇسىلا. پۇل بېرىپتۇ دەپ شۇنچە يەلۋەسىلە قاراپمۇ قويمىدى جۇمۇسىلا... دېھقان ئىشەكنى قامچىلىدى.


-ئۇرسىلا، كۈچلىرى ماڭىلا يەتكەندىكىن ھە، مەن يامان گەپ قىممىدىم سىلىگە، ياخشى بوسۇن دىدىم. بوممىسا سىلى بىي مىنىڭ تۇققىنىممۇ ئەمەس……   
4.ئۈزۈم ماكانى تۇرپاننىڭ ھويلىلىرىدا بىر كۈنلۈك تۇرۇش باشلاندى. ھويلىدا بېقىلىۋاتقان ئۆچكە پاراڭ باشلىدى:


-ئۈكەەەەەەم، ئايناقمۇ چۈپەيلىشىپ كەتمەڭلارا، يەم خەشەكنى قورقماي سەپ باقمامسەن ھە، ئۈزەڭ يەلەمىسەڭ يە، ھەي پوخۋاشلار، گە قىۋاتىمەن سەنلەرگە. ئايلا، سەن بولساڭمۇ قاراقويسىنا ماڭا……


5.قوغۇن ماكانى قۇمۇلنىڭ يېزىلىرىدا بىر ئادەم قويلىرىنى قوتانغا ھەيدەپ كىردى، قوتاننىڭ كىرىش تەرىپىدە قويلار سۆز باشلىدى:


-قېنى، ئۆزلەدىن كەلسۇن


- قېنى ئۆزلەدىن كەلسۇن


-ئاۋۇ ئىككىڭلار، مېڭىپ باقسىلاكى……
-ئارقامدىكىلەر، سۈزمىسەڭلا بولۇرمۇ مىنى……
-ماۋۇ قويچى ئەجەپ ئۇرەلەۋەردىغۇ چىۋىقنى، ئىلدام ئىلدام ماڭغايسىڭىزلەر……


6. خوتەننىڭ گۈزەل باغلىرىدا، بىر يىگىت بىر قىز بىلەن دىدارلاشتى، گۈزەل باغنىڭ ئىچىدىكى بۇلبۇل يىگىتكە قاراپ سۆز باشلىدى:


- تاتە ۋوي قولاڭنى ئەدەپسىز، ما قىز بالىنى خىجىل قىپ قويداڭيا، مىينغا ئۆتكىنە. ئونوم ئونوم ئانىڭ تۇمشۇقىنىڭ سوزۇلغىننى، ۋاي خۇدايىم ئۇياتسىز بولدى قىلىشە ئەمدى، ئادەم كەلدى ئەينا……


ئۇف،بۇ مەزمۇننى كىچىك بالا ئويلاپ چىققان، زادى بىر ياز دەپ بىشىمنى قوچۇۋىدى، شۇڭا مانداق يېزىپ باقتىم قاملاشسا قاملاشمىسا. پىكرىڭلار بولسا، ئىشىك تۈۋىدە پىكىر ساندۇقى بار، شۇنىڭغا قالدۇرۇڭلار! بەرىبىر كۆرمەيمەن……
پارول مەنۇ پارول مەن……

پارول
ئۇنىمۇ دىدۇق，بۇنىمۇ دىدۇق،مەن دەيمەن
ماۋۇ بىزنىڭ107.4تىكى سۆيۈملۈك ئاكىمىز مەۋلۇد ئىبراھىمنى بىردەم بوزەك قىلامدۇق……خوپ بولدى،مەن گەپ قىلسام ،جاۋاپ قايتۇرمىغاندىكىن……ئەرز قىلسا،قانداق قىلارمەن……
ھەممىمىزگە تونۇشلۇق بولغان داڭلىق دېكتور،رىياسەتچى،ماۋۇ شادلىق سىزگە ھەمراھ پىروگراممىسى بىلەن تۈن تىۋىشلىرىنىڭ رىياسەتچىسى مەۋلۇد ئىبراھىم ھازىر ھەممىگە تونۇشلۇق داڭلىق شەخىسلەرنىڭ بىرى بولۇپ قالدى،شۇنىڭ سەۋەبىدىن ماۋۇ بىزنىڭ مۇخبىرلار ئەل رايىنى سىناش ئۈچۈن ،جەمئىيەتكە باشچىلاپ چۆكۈپ،خەلقنىڭ مەۋلۇد ئىبراھىمغا بولغان كۆز قارىشىنى ئىگەللىدى……
1.چوڭلار
مۇخبىرنىڭ سوئالى：
سىلەر رىياسەتچى مەۋلۇد ئىبراھىمنى تونۇمسىلەر؟ سىلەرچە ئۇ قانداقراق ئادەم؟
بوۋاي A：كىمۇ مەۋلۇد ئىبراھىم دىگەن؟ تۇرسۇنئاي ئىبراھىمنىڭ ئىنىسىما؟
بوۋاي B：ھە،مەن بىلىمەن،مۇشۇ دەنشىلەردە چىقىدىغان بالا،مىنىڭ قۇلىقم ئېغىر دەڭە،شۇڭا ئۇ بالىنى دەنشىگە سېغىز چايناپلا چىقامدىكىن دەپ قالىدىكەنمەن……
مومايA：ۋاي ،مەن تازا بىلىمەن،بىزنىڭ ئالدىمىزدىلا چوڭ بولغان بالا ئۇ،شۇنداق ياخشى گەپلەرنى قىلىدۇ،شۇ گەپ قىلغاندا بىشىنى تولا مىدىراتقاچقا،بىچارە بالامنىڭ چوققىسىدا چاچ قالمىدى دە……
بوۋايC：ھەي موماي،سىنىڭ دەۋاتقىنىڭ ئاۋۇ شىنجاڭ خەۋەرلىرىدىكى غالىپ نىيازدۇ؟
مومايئB：ۋاي ياق،شۇ مەۋلۇدنى دەۋاتىدۇ ،رادىئودا گەپ قىلىدىغان،بىردەم يۇمۇر دەپ،بىردەم تەربىيە قىلىدىغان……ياخشى بالا ئۆزى،
بوۋايD：ھېسامنى بەك دورايدۇ كاساپەت،خە خە خە خە
موماي C：ئاستا كۈلسىلە بوۋاي،چىشلىرى تۆكۈلۈپ كەتمىسۇن،ھېلىمۇ چاشلىرى چۈشۈپ،مەۋلۇدقا ئوخشاپ قالدىلا……
بوۋا مومىلىرىمىز نىمداق ئوماق！！
2.ئوتتۇرا ياشلىقلار
مۇخبىرنىڭ سوئالى：
سىلەر مەۋلۇد ئىبراھىمنى تونامسىلەر؟ ئۇنىڭغا قانداق قارايسىلەر؟
ئەرA：يولداش مەۋلۇد ئىبراھ خىزمەتكە ئاكتىپ……كەسىپكە پۇختا……ھە……تۇرمۇشتا مەسئۇلىيەتچان،بالىلارغا مېھرىبان……ئۆزى تىرىشچان ياخشى كادىر،بىر ياخشى ئوقۇتقۇچى
ئەرB：ئاللا،مەۋلۇد ئىبراھىم ئوقۇتقۇچى ئەمەس،باسما زاۋۇتىنىڭ ئىشچىسدۇ؟مىنىڭ بىلىشىمچە
ئايالA：يوق گەپ قىلماڭلاا،مەۋلۇدجان رادىئو تىلىۋىزىيە ئىستانسىنىڭ داڭلىق رىياسەتچىسى،پىروگراممىلىرى بەك ياخشى،مەن بەك ياقتۇرىمەن،بولۇپمۇ سەھەردە بېرىدىغان ساغلاملىق مەسلىھەتچىسى پىروگراممىدىكى مەمەتجان دوختۇرنىڭ ئاۋازى يېقىملىق،مەۋلۇد ئىبراھىم ئۆسمۈرلەرگە مسئۇل
ئايالB：ئۇندا ئەمەس،سىز خاتا تۇتىۋاپسىز،مەۋلۇد يۇمۇر سۆزلەيدىغان،مەسلىھەت بېرىدىغان پىروگراممىدا،بولۇپمۇ ئاۋۇ مەسلىھەت بېرىدىغان پروگراممىسى بەك ياخشى،مەۋلۇدجان ئۆزىمۇ بىلىملىك،باشقىلارنى ئۇنى بۇنى دەپلا قايىل قىلىۋىتىدۇ……
ئەرC：كاساپەت ھېسامكامنى بەك دورايدۇ دە،خەخەخەخەخە
3.ياشلار……
مۇخبىرنىڭ سوئالى
سىلەر مەۋلۇد ئىبراھىمنى بىلەمسىلەر؟سىلەرچە مەۋلۇد ئىبراھىم قانداق ئادەم
70يىلدىن كىيىنكىلەر：مەۋلۇد ئاغىنەم نوچى جۇمۇ،يامان قىززىق گەپ قىلىدۇ،ئاڭلاپ قۇلاقلىرىم كۆيەپ كېتىدۇ،مەن ئاۋۇ قاتناش بىلىملىرىنى سۆزلىگىنىنى بەك ياقتۇرىمەن……
80يىلدىن كىيىنكىلەر：مەۋلۇدكامنىڭ پىروگراممىسى بەك نوچى،يۇمۇرلىرى قىززىق،كۈلدۈرۈپ ئۈچەينى ئۈزىدۇ،نەچچە ئاغىنەم ئۈچەي ئۇلاش ئوپىراتسسيىگە چۈشتى،تولا كۈلۈپ،تىككەن يەرلىرى تولا يېرىلىپ……مۇشۇ تېلىفونى ئۇلانمايدۇ……بولسا،مەۋلۇدكام بىزگە ئۇرۇپ تېلىفوننى،“يۇمۇر بارمىتى”دەپ سورىغان بولسا،نوچى بولىتى
90يىلدىن كېيىنكىلەر：مىنىڭچە مەۋلۇدنىڭ ئاۋازى ئابلاجاننىڭكىدەك نوچى ئەمەسمىكەن،ئۇ قايسى ناخشىنى ئېيتقان؟
4.قوشنىلار……
سىلەرچە مەۋلۇد ئىبراھىم قانداق قوشنا
ئۇدۇل قوشنا：بۇ رىن شى تا؟
ئاستىنقى قوشنا：مەۋلۇدكام ئۆزى نوچى ئادەم،بىراق مۇشۇ ئۆيىدىن تاقىراق تاقىراق ئاۋاز تۈگىمەيدۇ…بىر بولسا مەۋلۇدكام تىتا ئۇسسۇلى مەشىق قىلىدۇ،بىر بولسا ئۆيىدە ئات چاپتۇرىدۇ،
ئۈستۈنكى قوشنا：مەۋلۇدجان بەك ئېسىل ئادەم،قوشنىلار بىلەن ئىناق ئۆتىدۇ،سالام سائىتى جايىدا،مۇشۇ مەۋلۇدجان بويى ئىگىز بولغاچقا،بىشى تۇرۇسقا تىگىپ تۇرامدىكىن،دۇڭ دۇڭ قىلغان ئاۋاز بەك چىقىدۇ……
يان بىنا قوشنا：مەۋلۇدجان بەك مېھرىبان ئادەم،ئايالىغا بەك كۆيىنىدۇ،قاچانلا قارىسام دېرىزىدىن،ئالدىدا پەرتۇق،قولىدا ساپلىق……گاژ گاژ لا قىپ……
كەينى بىنانىڭ يىنىدىكى ئىمارەتنىڭ ئارقىدىكى كورپۇس تۆتىنچى قەۋەتتىكى قوشنا：تانىمايمىن،ئوسۇ مەلۈد دەگەن ئادەم كىم بىلمىيمىن،
5.ساۋاقداشلار……
مۇخبىرنىڭ سوئالى：
سىلەرچە ساۋاقدىشىڭلار قانداقراق ئوقۇغۇچى ئىدى
باشلانغۇچ مەكتەپتىكى ساۋاقداش：ماۋۇ كالپۇك بىلەن پۇچۇق چىشنى كۆردىڭىزما؟ شۇ مەۋلۇد كىچىك ۋاقتىدا ئوچۇق پىشانىسى بىلەن قوياپلا ئاشۇ……
باشلانغۇچ مەكتەپ مالىمى：ۋاي بەك ئەلاچى،بەك گەپ ئاڭلايدىغان بالىتى،بىراق دەرس ۋاقتىدا بەك جىق گەپ قىلىتى،مانا رىياسەتچى بوپتۇ……
ئوتتۇرا مەكتەپ ساۋاقدىشى：كۆچۈرىتى،شۇنداق،بەك كۆچۈرىتى،تاپشۇرۇق،ئىمتىھان دىگەنلەردە ئاشۇ پىشانىسىنى پارقىرىتىپ تۇرۇپ كۆچىرىتى……ۋاي ھەتتا تەنتەربىتە ئىمتىھاندىمۇ باشقىلارغا ئوخشاش يۈگۈرۈپ كۆچىرىتى……
ئوتتۇرا مەكتەپ مالىم：ھە،ساۋاقداش مەۋلۇد دەرستە ئەلا،ئۆزىگە قاتتىق تەلەپچان،ئىمتىزامچان،تەرتىپلىك،سەۋاقداشلىرىغا ياردەم قىلىدىغان،ياخشى ئىش،ياخش پائالىيەتلىرى كۆپ،ئالتۇن تىپىۋالمىسىمۇ كۆز قىرىنى سالمايدىغان،چىچى ئاز بالىتى……
ئالىي مەكتەپ ساۋاقداش：ئوقۇيتتى،تىرىشاتتى،مەكتەپتىكى پائالىيەتلەرگە ئاكتىپ قاتنىشاتتى،ئەتتىگىنى ئۇخلاپ،كەچلىرى قىز ساۋاقداشلار بىلەن شىندا خۇڭخۇ دا بەدەن چىنىقتۇراتتى ۋايت
ئالىي مەكتەپ مۇئەللىم：كىمۇ ئۇ؟ ئوقۇغۇچىلىق نۇمۇرى قايسى؟كۆچۈرۈپ تۇتۇلغان يېرى بارمىتى؟ تاڭ ئۇنداق بولسا،كىمكىن مەۋلۇد دىگەن بالا……
6.جەمئىيەت ھەر ساھە……
مۇخبىرنىڭ سوئالى
مەۋلۇد ئىبراھىمغا قانداق قارايسىلەر؟
ساتىراش：ھە،ئۇ بالا چاچ ياساتقاندا ،بىردەمدىلا ياساپ بولغاچقا،جىق پاراڭلاشمايمىز،ئادەتتە ما شەھەردە پارول دەپ يەنە بىر تاز گۇي بار،مەن ئىككىسىنىڭ بىشىنى يېىقىن چاپلاپ تۇرۇپ،ۋىژلا قىلىمەن،پۈتتى دىگەن گەپ……
شوپۇر：ئامما لېكىن مەۋلۇدكام بولغاچقا،يولدا زېرىكمەيمىز ھە،ئالامەت نوچى ئادەم……پىروگراممىسى بەك نوچى

-ۋھ ھاھاھاھاھا سىنىڭدەك ئەخمەقتىن يەنە ئونى ياكى مڭى بولغان بولسا ،بىزدەك بەختلىك ئادەم يوق جۇمۇ...

سەھرايى تارىم ھەرخىل ھەرقىسىم ئادەملەرنىڭ ھەممىسىدىن مۇشۇ سوئالنى سوراپ چىقىتى..ھاراقكەش \"يەنە بىر رومكا بولغان بولسا..ئاشۇ بەخت \"دەپ جاۋاپ بەردى..يانچۇقچى\"ھەممە ئادەمنىڭ يانچۇقى ئاستى تار،ئاغزى تەرىپى كەڭ بولسا\"دەپ جاۋاپ بەردى..جىنايەتچىلەر\"تۈرمە دىگەن بەشيۇلتۇزلۇق بولغان بولسا..\"،ئالدامچىلار\"ھەممە ئادەم ئەخمەق بولغان بولسا\"....قويمىچىلار\"ھەممە ئادەم ئۇنتۇغاق بولغان بولسا\"....ئىشقىلىپ ھەرخىل ھەرياڭزا سوئاللار ئۇنى بەكمۇ ھاياجانلاندۇردى..



ئۇ ئاخىرىدا بىر بىلىملىك،ئوقۇمۇشلۇق،چوڭ سۈپەت بىر زىيالىدىن سوردى

-سىز بەختلىكمۇ

-شۇنداق .مەن بەك بەختلىك..

-بەختكە قانداق قارايسىز..

-خاتىرجەم تۇرمۇش،ئىناق ئائىلە،خوشال خورام كەيپىيات،ئىناق جەمئىيەت، ئىجىل قېرىنداشلار،ۋاپادار بالىلار، كۆيۈمچان دوست بۇرادەر لەر ..مانا مۇشۇلار مېنى بەختلىك ھېس قىلدۇرغان ئامىللار

-سىز گە ئاپىرىن!!!!



شۇنىڭ بىلەن سەھرايى تارىم ئۆزىنىڭ يەكۈن سۆزىنى خاتىرىگە يازدى



مەيلى باي مەيلى گاداي،مەيلى قىز مەيلى ئوغۇل،،مەيلى قېرى مەيلى ياش،مەيلى ئىگىز مەيلى پاكار..ئىشقىلىپ ھەرقانداق ئادەمنىڭ بەخت تۇيغۇسى ئوخشاش بولمايدىكەن،ئىنتىلىدىغان بەخت نىشانى ئوخشاشمايدىكەن.

توغرا،بەخت دىگەن نەرسىنى كۆرگىلى بولمايدۇ،تۇتقىلى بولمايدۇ،تۇتۇپ قالغىلى تېخىمۇ بولمايدۇ،مىنىڭچە بەخت شۇ ئادەمنىڭ شۇۋاقىت،شۇزامان،شۇ مەزگىلدىكى قانائەت تېىپش،ئۆز ھاجىتى،ئۆز موھتىجىغا ھەل قىلىشى يولى ياكى ھەل قىلىش ئۇسۇلى تاپقاندىكى روھى ھالىتى ھەم خوشاللىق تۇيغۇسى ھېسابلىنىدىكەن.

مەسىلەن ،ئاچ قالغاندا،بىر پارچە نان ،ئۇسسىغاندا بىريۇتۇم سۇ،چارچىغاندا كىچىك بىر ئورۇن،ئۇيقۇسى كەلگەندە ياتقىلى بولىدىغان جاي...مانا مۇشۇلار شۇ مۇھىتتا ئادەمگە بەخت تۇيغۇسى ئېلىپ كېلىدىكەن...

ئادەم ئۈچۈن بەخت شۇنچىلىك قىممەتلىك، ھەم قەدىرلىك،بەختنى زورلاپ تاڭغىلى بولمايدۇ،شۇنداقلا باشىقىلار يارىتىپ بەرمەيدۇ،پەقەت ئىنسان ئۆز ئىقتىدارى ھەم ئۆزى قابىلىيىتىگە تايىنىپ،ئۆزىگە لايىق،ماس كېلىدىغان بەختكە ،ئىنىق ئېيتقاندا بەخت تۇيغۇسىغا ئېرىشەلەيدىكەن،كىشلەرنىڭ بەختكە بولغان كۆز قارىشى ئوخشاش بولمايدىكەن، قانائەتچان ئادەمنىڭ بەختكە بولغان تونۇشى ئاددىي بىراق ناھايىتى سۆيۈملۈك،گۈزەل،راھەت،پاكىز بولىدىكەن،ئاچكۆز ئادەمنىڭ بەختكە بولغان تونۇشى بولسا،ئەگىر توقاي،يامانلىققا تولغان،چىكى يوق،تەلىپى يۇقىرى،رەزىل بولىدىكەن،بۈگۈنكى مۇشۇ پائالىيتىمدىن بىلىۋالدىم :

خەلقنىڭ بەخت تۇيغۇسى ئاددىي ،بىراق ئۇلۇغۋار ئىكەن،بەختكە بولغان تەلىپى يۇقىرى ئەمەس،بىراق ئېرىشكەن بەخت تۇيغۇسى بەكمۇ ئالىيجاناپ ئىكەن!!!!
