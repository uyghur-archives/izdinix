سۆگەل غوجامغا بارسا دۇئا قىلىپ قويىدۇ ( شۇ جايدا ياتقان كىشىنىڭ ھۆرمىتىدىن دەپ ) تامغا لاي چاپلاپ . كىچىك چېغىمدا بەدىنىمگە سۆگەل چىقىپ قېلىپ ئۇقمىغاندىكىن مېنىمۇ ئېلىپ بارغان شۇ جايغا . بالام ( 11 ياش ) نىڭ قولىغا چىقىپ قاپتىكەن ، كېيىنچە كۆپىيىپتۇ . مەن ئۇنىڭغا ، بالام كىچىك چېغىمدا بىلمەسلىكتىن سۆگەل غوجامغا بېرىپ شۇنداق ئىشلارنى قىلىپ ساپتىمەن . ساڭا تەۋسىيەم ئاللاھ نىڭ مۇشۇ سۆگەلنى بەدىنىڭدىن يوقۇتىۋېتىشىگە چىن دىلىڭدىن ئىشەنسەڭ ئۆزۈڭ دۇئا قىل ، ھەقىقىي داۋاغا ئېرىشىسەن ، مەن سېنى يېڭىلىپ غوجامغا ئېلىپ بارمايمەن دېدىم . ئۈچ - تۆت كۈنگىچە سۆگەل يەنە تۇردى ، مەن ئۇنىڭدىن دۇئا قىلغان - قىلمىغانلىقى توغرىسىدا سوراپ تۇردۇم . ئۇ دۇئا قىلىۋاتىمەن دەيتتى . بىر ھەپتە ئۆتۈپ سۆگەل يوقىدى . ئۇنىڭ بېگىز بارمىقىغا چىققىنى گۈل ( ئېچىلىدىغان ) سۆگەل ئىكەندۇق . بۇ ھەقىقىي بولغان ۋەقە . ئاللاھ بىلەن ئوتتۇرىڭىزدا ھېچكىم يوق ، دۇئاغا قول كۆتۈرۈڭ ، ھەقىقىي داۋاغا ئېرىشىسىز .
