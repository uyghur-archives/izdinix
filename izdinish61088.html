ئۇيغۇرلارنىڭ تور يۈزىدە ئۇيغۇرچە كېپىللىك پۇل تۆلەش ئارزۇسى ....

ئۇيغۇرلارنىڭ تور يۈزىدە ئۇيغۇرچە كېپىللىك پۇل تۆلەش ئارزۇسى ئەمەلگە ئاشارمۇ؟ قاچان ئەمەلگە ئاشار؟

دەۋرىمىز سودا ئوبروتىنىڭ زور كۆپ قىسمىنى تور سودىسى ئىگىلەۋاتقان بۈگۈنكى كۈندە، بىزدىمۇ نۇرغۇنلىغان تور سودا بېكەتلىرى روياپقا چىققان بولدى ھەم توختاۋسىز بارلىققا كېلىۋاتىدۇ. بۇ ئەلۋەتتە تارىم بوستانلىقىدا ئەنئەنىۋى تىجارەت ئۇسۇلى بىلەن ياشاپ كېلىۋاتقان بىز مىللەتكە بىر يېڭىلىق ھەم قىزىقىش ئېلىپ كەلدى. خەلقىمىزدىمۇ كومپيۇتېر ۋە تورنىڭ خېلى زور كۆلەمدە ئومۇملىشىشى ئەلۋەتتە بىزدىكى بۇ تور يۈزى يېڭىچە سودا شەكلىگە بارغانسېرى گۈللىنىش ئېلىپ كەلسە توغرا بولاتتى. بەلكىم بۇ نۇقتىنى بىزدە تورسودا بېكىتى ئېچىپ باققانلار ناھايىتى چوڭقۇر ھىس قىلغان بولۇشى مۇمكىن. چۈنكى ھەر يىلقى 1111 كۈنىدە ئېلان قىلىنىۋاتقان غايەت زور ئىلگىرلەشلەر، ھەممە ياقتا تور سودىسىنىڭ راۋاج تېپىپ كەتكەنلىك، توردا مال سېتىپ بېيىپ كەتكەنلىك، توردا مال ئېلىپ بەك خۇش بوپ كەتكەنلىكتەك ... ئۇچۇرلارنى پات -پاتلا ئاڭلاپ تۇرۇۋاتىمىز. شۇنداق بولغانىكەن بىزدىكى تورسودا بېكەتلىرىمۇ ئەشۇنداق ھاياجانلىق خەۋەرلەرنى تارقىتىپ باقسا بولاتتى...  مەقسەت، بۈگۈنىمىزدىكى مۇھىم سودا شەكلى بولغان تور سودىسىدا بىز خەلقىمىزنىڭ ئۆز مەھسۇلاتلىرىنى مۇشۇ شەكىلدە سېتىپ چىقىرىش، ياقتۇرغانلىرىنى ئۆز ئانا تىلىدا ھوزۇر ئولتۇرۇپ سېتىۋېلىش ئارزۇسى رېئاللىققا ئايلانسا، ئەلۋەتتە ھەربىرىمىزنى ئەنە شۇنداق ھاياجانغا سېلىپ، يېڭىچە دۇنيانىڭ ھەقىقى ھوزۇرىنى ئالەم بىلەن بىرگە سۈرگەن بولاتتۇق.
بىراق، خۇددى سۇ يوق يەرگە تېرىلىپ قالغان زىرائەتتەك، ئاتا- ئانىسىز قالغان يىتىم بالىدەك، يىلتىزى تۇپراقسىز قېلىپ ئاپتاپتا قالغان دەرەختەك... بىر يېرىمىز بىر نىمىلا... خېلى داغدۇغا بىلەن بېكەتلەر ئېچىلىدۇيۇ، ئاۋۇ يۇقارقى ئوخشىتىشلاردىكىدەك بىر كۆكىرىپ باقىدۇيۇ... يەنە سارغىيىپ قالىدۇ....... بىر تەرەققىي قىلىۋەتكۈسى كېلىدۇيۇ، ئاۋۇ يىتىمدەك بويۇن قىسىپ يەنە بىر ياقتا قالىدۇ...
مېنىڭچە، تور سودىسىنىڭ ئۇيغۇرچە ھەم خەنزۇچە ئېلىپ بېرىلالايدىغان رېئاللىقنىڭ بىز خەلققىمۇ تەسىرى ناھايىتى زور بولارىدى دەپ ئويلايمەن.
ھازىرقى ئەڭ نىگىزلىك ئەھۋال: بىرىنچىدىن، ساتقۇچىلار ئۆز بازىرىغا ئومۇميۈزلۈك يۈزلىنەلمەيدۇ. بەلكى ئاز بىر قىسىملىرى باشقا تىلدىكى بېكەتلەر ئارقىلىق باشقىلارغا بازار ئېچىشقا يۈزلىنىدۇ. ئىككىنچىدىن، سېتىۋالغۇچىلار ئۆز بازىرىدىن سېتىۋېلىشقا ئومۇميۈزلۈك ئادەتلىنەلمەيدۇ، بەلكى ئاخىرىدا يەنىلا ئاز بىر قىسىملىرى باشقا تىلدىكى بېكەتلەردىن مال سېتىۋالىدۇ. ئۈچىنچىدىن سودا بېكىتى ئاچقانلار بولسا ساتقۇچىلار بىلەن سېتىۋالغۇچىلارنىڭ يۇقارقى ئىككى خىل ھالدىكى چېچىلما ھالىغا قاراپ زادى نىمە قىلىشنى بىلەلمەي ... ئاخىرىدا سارغىيىپ قالىدۇ...
زادى قانداق قىلىش كېرەك؟ بىز ئۇيغۇرلاردا ئۆز ئانا تىلىمىزدا مال ساتىدىغان، ئالىدىغان، ئانا تىلىدا مەھسۇلاتىنىڭ بازىرىنى توسالغۇسىز راۋانلاشتۇرىدىغان بىر سۇپا بارلىققا كەلمەسمۇ؟ ئامالسىزلىقتىن تور سودىمىز باشقا تىللىق بوپ كېتەرمۇ؟ مېنىڭچە ئاخىرقى بۇ نۇقتىنى مىللەت نۇقتىسىدا تۇرۇپ قۇبۇل قىلغىلى بولمايدۇ ئەمما ئېھتىياج نۇقتىسىدا ھازىرچە ئامالسىزمىز...

مانا يۇقارقى بۇ مەسىلىلەرنىڭ نىگىزلىك قىيىنچىلىقى باشقا يەردە ئەمەس، پەقەت دەل ،  بىزدە ئانا تىلىمىزنى قوللايدىغان بىر ئۈچىنچى تەرەپ كېپىللىك پۇل تۆلەش سۇپىسىنىڭ بولمىغانلىقىدىنلا ئىبارەت. مېنىڭچە ئۇيغۇر تور سودىسىدىكى باشقا مەسىلىلەر (مەسىلەن تەمىنلەش، توشۇش دىگەندەك)خېلى ھەل بولدى ھەم ھەل قىلغىلى بولىدۇ دەپ قارايمەن.

بۇ تېما ئارقىلىق كەڭ جامائەتچىلىك، مۇناسىۋەتلىك خادىم، كارخانىچى، بايلىرىمىزدىن شۇنى ئۇقۇپ باققۇم كېلىدۇ:
بىزدە مۇشۇ ھەقتە ئىزدىنىشلەر، قەدەملەر قانچىلىك؟ مۇشۇ ھەقتە غەم يەۋاتقانلارمۇ بارمىدۇ؟ بولسا قەدىمى قەيەرلەرگە يەتكەندۇ؟ ھەل قىلىشنىڭ ئۇسۇلى قانداق؟
قارىسام  خەلق بانكىسىنىڭ بۇ خىل پۇل تۆلەش سۇپىسىنىڭ ئىجازىتىگە قويۇلىدىغان تەلىپى ئىنتايىن يۇقىرى ئىكەن(ئىلتىماس قىلغۇچىنىڭ مەبلىغىنىڭ يۇقىرىلىقىنى دىمەيلى، ئەڭ- ئەڭ ھالقىلىق مەسىلە قانداق قىلىپ بۇ ‹گۆھەر›نى بىزنىڭ تىلىمىزغا ماسلاشتۇرۇپ بۇنى خەلق بانكىسىغا ئېتىراپ قىلغۇزۇپ..... ھەي بەك چوڭ سۆزلىۋەتمىگەندىمەن ھە؟ بۇ تەرەپلىرىنى ئويلىغاندا بۇ ئىشنىڭ بىزدىكى شەخسلەر ئارقىلىق ۋۇجۇدقا چىقىدىغان چىرايى يوق. ھۆكۈمەت يولى ئارقىلىق قىلىنسا ئەلۋەتتە بۇ بەك ئاسانغا توختايدۇ. تەس يېرىنىمۇ ۋاقىت ۋە تېخنىكا ھەل قىلىۋىتىدۇ ئەلۋەتتە. بىراق، ھۆكۈمەت، نوپۇسى ئون مىليون تور ئابونتى مىليوندىن ئاشىدىغان(ئۇيغۇر تور ئابونت سانى ھەققىدە ئېنىق ئىستاتىستىكا بولسا يوللىساڭلار) بۇ خەلق ئۈچۈن شۇنچىلىك ئىشنى قىلىپ بېرەرمۇ؟
ھەي... كاللام قېيىپ كەتتى. يەنىلا مۇشۇ نۇقتىغا كۆڭۈل بۆلىدىغان، بۆلۈۋاتقان، ياكى ئىزدىنىۋاتقان، مەلۇم تونۇشقا ئىگە بولغان تورداشلار تېخىمۇ چوڭقۇرراق پىكىرلەرنى، پىلانلارنى، ياكى لاھىيەلەرنى قالدۇرۇپ باقسا، ئاغزىمىزنى تاتلىق بولسا تاتلىق، ئاچچىق بولسا ئاچچىق قىلىپ مۇشۇ ھەقتە بىر كوتۇلدىشىپ باقساق قانداق؟
تور سودىسىدىكى مۇشۇ چېگىشلىك ھەققىدە بىر ياققا چىقالمايۋاتقان دوستۇڭلار كۆسەن

ناۋادا سىز جىفۇباۋ، سەيفۇتوڭ دىگەندەك پۇل تۆلەش سۇپىلىرىنى ئىشلىتىپ باققان ۋە بۇلارنىڭ ئالاھىدىلىكى ھەققىدە ئويلىنىپ باققانلا بولسىڭىز ئۇنىڭ قانداق نەرسىلىكىنى ئاز -تولا چۈشىنىشىڭىز مۇمكىن. ئۇ مۇنداقچە ئېيتقاندا بىر بانكا، بىراق ئىشخانىسىز، مۇلازىمەت ئورنىسىز، پەقەت تور يۈزىدىلا كىرىم- چىقىم قىلىدىغان بانكا دەپ چۈشەنسەك بولار...  بىز رېئاللىقتا كۆرۈۋاتقان بانكىلار بىزنىڭ كۆرگىنىمىزدەك. بىراق بۇ يەردىكى ئوخشاشلىق ھەر ئىككى خىل بانكىنى، يەنى رېئالىي بانكا ۋە تور بانكىسىنى خەلق بانكىسى باشقۇرىدۇ ۋە تەستىقلايدۇ. باشقا ھەرقانداق ئورۇن بۇنىڭغا ئىجازەت بېرەلمەيدۇ ۋە بېجىرەلمەيدۇ. پەقەت مۇشۇ ئىشنى قىلماقچى بولغان كارخانا خەلق بانكىسىغا ئىلتىماس يوللايدۇ. خەلق بانكىسى ئىلتىماسنى كۆرۈپ ۋە تەكشۈرۈپ تەلەپكە يەتكەن بولسا بېجىرىدۇ.
بۇ يەردە ئۈچىنچى تەرەپ كېپىللىك پۇل تۆلەش سۇپىسى (تور بانكىسى دەپمۇ تۇرايلى) ھەققىدە مەركىزى خەلق بانكىسىنىڭ پەقەت ئىلتىماس قىلغۇچىنىڭ ئىقتىسادىغىلا قويۇلىدىغان تەلىپى تۆۋەندىكىدەككەن:
央行规定，申请人拟在全国范围内从事支付业务的，其注册资本最低限额为1亿元人民币；拟在省（自治区、直辖市）范围内从事支付业务的，其注册资本最低限额为3千万元人民币。对于外商投资支付机构的业务范围、境外出资人的资格条件和出资比例等，将由中国人民银行另行规定，报国务院批准。

مېنىڭ تور ماتىلاپ كۆرۈشۈمچە  ھازىرغىچە 223 كارخانىنىڭ بۇ ئىجازەتنامىگە ئېرىشكەنلىكىنى ئېيتىپتۇغۇ.. بۇنىڭ ئىچىدە ئەڭ سۈرگىنى شۇ جىفۇباۋ. ئۇنىڭدىن قالسا سەيفۇتوڭ، ئاندىن قالسا ۋاڭيىن توردا پۇل تۆلەش سۇپىسى قاتارلىقلار ئىكەن...  بۇلارنىڭ ئىشلەش چوڭ قانۇنىيەتلىرى ئوخشاش، بىراق كىچىك پەرقلىرى،  ئالىدىغان ھەقلىرى ئوخشىمايدۇ. ئەڭ سۈرگىنى شۇ جىفۇباۋ، مۇلازىمىتىمۇ بەكلا ئەتراپلىق، ئەمما ئۇنىڭ بۇنچە سۈرۈپ كېتىشىدىكى ھەرىكەتلەندۈرگۈچ ئامىل- تاۋباۋ تورى
بىز ئۇيغۇر خەلقىگە بىر قانچىسىمۇ ئەمەس بىرسىلا بولسا پايدىلىنىپ كېتەلەيتتۇقتە. چوڭ كارخانىچىلىرىمىز بۇنداق پۇڭلاپ پۇل تاپىدىغان ئىشقا قىزىقماس ھەقىچان... بولمىسا بىرەر كارخانا باش بولۇپ ماڭسا بەلكىم بىر ياققا چىقىپ قېلىشىمۇ مۇمكىنىدى.
مېنىڭ بۇ سېستىمىنى بارلىققا كەلتۈرۈشتە ئويلىغىنىم، مەسىلەن مەلۈم بىر بانكا، شىنجاڭ بانكىسى دەمدۇ بىر بانكا ئۆزىنىڭ ئىچكى سېستىمىسىغا ئۇيغۇرچە قوشسا، ئۇيغۇر خەلقى بۇ بانكىدىن كارتا بېجىرگەندە كىملىكىدىكى ئۇيغۇچە ئۇچۇرلار بىلەنلا بېجىرەلىسە، ئاندىن ھېلىقى پۇل تۆلەش سۇپىسىمۇ  بانكىنىڭ مۇشۇ سېستىمىسىغا ئاساسەن كارتا ئىگىسى ۋە ئۇچۇرلىرىنى پەرقلەندۈرەلىسە دىمەك بۇ مەسىلە تېگىدىن ھەل بولغان بولاتتى دەپ قارىدىم. يا بانكا ھەققىدە ئارتۇق ئويلاپ قالدىممۇ بىلەلمىگىنىم، ناۋادا بانكا ئۇنداق قىلمىغان تەقدىردىمۇ، پۇل تۆلەش سۇپىسىنى يۈز پىرسەنتلىك ئۇيغۇرچە قىلىپ ئىشلىتىش مەقسىتىگە يەتكىلى بولارمۇ؟
