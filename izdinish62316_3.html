بىئارامچىلىق ئىچىگە شۇڭغۇپ ياشاشقا مەجبۇر بولغاندىن بىرى جاھاننىڭ ئىشلىرىغا قاراپ ھەيران قالىمەن . شان شۆھرەت تاماسىدا قاتىراپ باشقىلارغا سۈيقەسىت قىلىۋاتقان . تەھدىت سىلىۋاتقان . كۆزىگە سىغمىغانلارنى كۆزدىن يوقۇتۇشقا كۆزى يەتمىسە ئۇنىڭ ئەل ئىچىدىكى ئابرۇيىنى تۆكۈش ئۈچۈن كىچە كۆندۈزلەپ سۇيقەسىت ئويلاپ چىقىرىۋاتقان ئادەملەرگە قاراپ ھەيران ھەس قالىمەن . ئەجىبا بىز كۈتكەن جەمىيەت مۇشۇمىدى . بىز كىچىك چىغىمىزدا كۈتكەن كەلگۈسى مۇشۇمىدى ؟ بىز مۇشۇ ئەھمىيەتسىز تالاش تارتىشلار ئۈچۈن دۈم يىتىپ دۈم قوپوپ تىرىشىپ ئۈگىنىپ بىلىم ئالغانمىدۇق ؟ ئاساسلىق ئېىقمدا دەۋىر سۈرىۋاتقانلار پەقەت نۇمۇسنى ئۇقمىغانلىغى ئۈچۈنلا راھەتتە ياشاۋاتىدۇ . ھەقىقى ئىلىم ئەھلىنىڭ ئاۋازى تىخى بىزگە يىتىپ كەلگىنى يوق . سۆيۈش ۋە نەپرەتلىنىڭ ئارلىغىدا قېسىلىپ . قەدىرلەش بىلەن لەنەت ئوقۇشنىڭ ئارلىغىدا  تىتىما تالاڭ بۇلۇش بىزنىڭ بېشىمىزغا كىلىۋاتقان قىسمەتمىدۇ ؟ يەۋاتقان تاماقلىرمىزدىكى زەھەر ئارلاشمىلىرى ھەممىلا ئادەمنى ساقايماس كېسەللىككە ئىتتىرىۋاتىدۇ . تاماققىمۇ ساختا قىلىۋاتقان جەمىيەتتىن ۋىجداننى مىزان . كامىل ئىماننى قالقان قىلىپ توسۇش مۈمكىن ئىدى . ئەمما ئۇنى قانۇنى يول بىلەن چەكلەشنىڭ مۈمكىن ئەمەسلىگىنى ھىس قىلىۋاتىمىز . نەپەس ئىلىۋاتقان ھاۋادىن تارتىپ ئادەم ياشاشقا ماس كەلمەيدىغان موھىتتا ياشاپ . ئادەم يىيىشكە بولمايدىغان زەھەرلىك نەرسىلەرنى يەپ تۇرۇپ ھىلىمۇ مۇشۇنچىلىك ئۆمۈر كۆرىۋاتقانلىغىمىز مۆجىزىمىدۇ ؟

ۋەئەلەيكۇم ئەسسالام پەرھاتكا ، مەن مۇشۇ بىرەر مەشھۇر ئادەم ئۆلـمىگىچە شېئىرى ھىسياتى كەلـمەيدىغان شائىرلاردىن بەك تويۇپ كەتكەنلىگىم ئۈچۈن شۇنداق ئويلىسام كىرەك ، ئابدۇراھمان بىلەن كونسۇل كوچىسدا ئايلىنىپ يۈرگەن چاغلىرىم ۋە توردا پاراڭلاشقانلىرىم ئېسىمدە ، ئەمما ھەممىلا ئادەم ئۇنىڭ ئۈلۈمى ھەققىدە گەپ قىلىپ ، شېئىرلارنى يىزىپ تالاش تارتىش قىلىشقا باشلىغاندىن كىين مەندە بىر خىل خورلۇق ھىسياتى پەيدا بولدى ، گوياكى ئابدۇراھماننىڭ ئۈلۈمىدىن بىر تۈركۈم ئادەم شاتلىنىۋاتقاندەك ، پايدىلىنىۋاتقاندەك ھىس قىلىپ شۈركىنىپ كەتتىم ، ئىنسانىيەت تارىخىدا 37 ياش ئەمەس 7 ياش 17 ۋە 27 ياشتا ئۈلۈپ كەتكەنلەر بار ، مەن تۇرغان كەنىتتە تۆت ئايلىق بىر بۇۋاق تۈگەپ كېتىپتۇ ، بوۋىسى تاشتەك قانسىز، سۇلغۇن روھ بىلەن بۇ گەپنى دىگەندە ئۇنىڭ گىپى ئەمەس روھى ھالىتى ماڭا تەسىر قىلدى ، مەن ھاياتىمدىكى ئەڭ ئىغىر ئىقتىزادى قىينچىلىقتا قالغان دەۋىرلەردىمۇ چاپچاپ بازارغا ئاتاپ بىر نەرسە يىزىپ باقماپتىمەن ، شۇڭا ئابدۇراھمانغا بولغان دوسلۇق ھىسياتىم پەقەت قەلبىمدە ساقلاغلىق قالسىمۇ ئۇ قەلبىمدە ناخشا كەبى ياشايدۇ ، ئۇنىڭ ئۈچۈن كاكىراپ ناخشا ئوقۇپ يۈرۈشنىڭ ھاجىتى يوق دەپ قارىدىم ، ئۆزى ئۈچۈن ناخشا يىزىپ بولغان ئادەملەرنىڭ روھىنى تىنىچ قويماي زورلاپ ھاپاچ بۇلىۋىلىش مىنىڭچە پەسكەشلىك ، مىنىڭ قارىشىممۇ ئادەتتىكى ئادەم بىلەن مەشھۇر ئادەملەرنى باراۋەر ھالەتتە قويۇپ تەسەۋۋۇر قىلىشنى ياقىلايمەن ، مۇشۇنداق ئۈلۈم ۋە مۇسىبەتلىك تۇيغۇدا تۇرۇپ يەنە مىللەتچىلىگى چىكىگە يىتىپ مەيدىسى ئېچىشسا ۋىجدانىم قايناۋاتقان ئوخشايدۇ دەپ ئويلايدىغان ئادەملەردىن بىزارمەن ھەم ئۇلاردىن يىراق تۇرسام، كۆزۈم كۆرمىسە دەپ ئويلاپ كىتىمەن ، ئۇلار مىنىڭ نەزىرىمدە باشقىلارنىڭ ئۈلۈمىنى تاماششا قىلغۇچىلار ، تىراگىدىيەدىن  ھۇزۇر ئالغۇچىلار، داڭلىقلارنىڭ ئۈلۈمىدىن پايدىلىنىپ ئۆزىنىڭ ئىدىيسىنى يايما قىلىپ يايغۇچىلار ، شۇڭا بۇ قېتىم مەن رەھمەتلىك دوستۇمغا بولغان ھىسياتىمنى ئۈزۈمگىلا ناخشا قىلىپ ئىيتىپ بەردىم ، مەنمۇ چوڭ بولغان بولسام كىرەك ، باشقىلار مىنى ئىتىراپ قىلسىكەن دەپ ئويلاشتىن بەكرەك تەنھا ۋە يىگانە خاتىرجەملىكنى ئەھمىيتى يوق ۋاراڭ چۇرۇڭدىن ئەلا بىلگەن بولسام كىرەك ، مەن بۇنىڭدىن ئون بەش يىل بۇرۇن بىر سەنئەتچىنىڭ
