كىچەلمەيمەن سەندىن ياخشىلىق

                                                       كىچەلمەيمەن سەندىن ياخشىلىق
                  
                                                      بولسىمۇ گەر خەۋىپ خەتىرى ،
                                                     كىچەلمەيمەن سەندىن ياخشىلىق .
                                                      كەتمەك بولسام كىتەي سەن بىلەن ،
                                                      كىچەلمەيمەن سەندىن ياخشىلىق .

                                                      ياخشىلىققا يار ئىكەن ئۇيغۇر ،
                                                       ياخشىلىققا زار ئىكەن ئۇيغۇر .
                                                       بولساڭمۇ گەر يولۇڭ بەكمۇ تار ،
                                                       كىچەلمەيمەن سەندىن ياخشىلىق .

                                                       پەرىشتىلەر قەلەمگە ئالار ،
                                                       ئەل جامائەت بىرەر دۇئالار .
                                                        يۈرەكلىرىم سېنى دەپ يانار،
                                                         كىچەلمەيمەن سەندىن ياخشىلىق.

                                                         بوينۇمدىكى تۇمارسەن ئۆزۈڭ ،
                                                         ئۇيغۇرۇمغا خۇمارسەن ئۆزۈڭ .
                                                         يورۇپ تۇرسۇن سېنىڭ بۇ كۆزۈڭ،
                                                         كىچەلمەيمەن سەندىن ياخشىلىق.


                                                         تامالاردىن يىراق ياخشىلىق ،
                                                         مال دۇنياغا بۆكمەس ياخشىلىق .
                                                          قورۇقساممۇ تاللايمەن سېنى ،
                                                           كىچەلمەيمەن سەندىن ياخشىلىق.


                                                         كېرەك ئەمەس ماڭا سەپسەتە ،
                                                          كېرەك ئەمەس ماڭا پەلسەپە ،
                                                          سەنلا بولساڭ دىلىمدا نەقىش ،
                                                          كىچەلمەيمەن سەندىن ياخشىلىق.

                                                          ئاللاھ رىزا بولۇر سەن بىلەن ،
                                                          قەلىب نۇرغا چۆمەر سەن بىلەن .
                                                             ھەمراھىم بول بىر ئۆمۈر ماڭا ،
                                                           كىچەلمەيمەن سەندىن ياخشىلىق.
