تەتقصقاتتصن بايقصلصشچە ،ئەرلەر جۈرصسص بصلەن بصرياستۇققا باش قۇيۇپ ئۇخلصغاندا كاشصلصغا ئۇچرايدصكەن ،مەيلص يصقصنچصلصق قصلسۇن ياكص قصلمصسۇن ،ئەتصسص رۇيص ھالصتص بصنۇرمال بۇلۇپ ،چوڭ مصڭصسصنصڭ ئصنكاسص تۈۋەنلەپ كصتصدصكەن ،«يصڭص ئصلصم-پەن »ژۇرنصلصنصڭ ئەڭ يصڭص بصر سانصدا بصسصلغان تەتقصقاتتا دصيلصشصچە ،ئاياللار بۇخصل كاشصلصدصن مۇستەسنا ئصكەن ،ئاياللار ئەرلەرنصڭ ئەكصسچە بصرياستۇققا باش قۇيۇپ ئۇخلصسا تصخصمۇ بەھۇزۇر ئارام ئالالايدصكەن ،چۈنكص بۇخصل ھالەت ئاياللارنص غەرصق ئۇيقۇ ھالصتصگە كەلتۈردصكەن.
ۋصنا ئۇنۋصرسصتصتصنصڭ پصرۇفصسۇرص خارصت.كارۇچ يصتەكچصلصكصدصكص بصر تەتقصقات گۇرپصسص پەرزەنصت كۈرمصگەن 20نەچچە ياشلصق قصزيصگصتلەر جۈرصلەر ئۇستصدە تەتقصق ئصلصپ بارغان ،تەتقصقاتچصلارئۇلارنص 10كصچە بصر ياستۇققا باش قۇيۇپ ئۇخلصتصپ ،كصيصن يەنە 10كصچە ئايرصم ئۇخلاتقۇزغان .تەتقصقاتچصلار ئۇلارنصڭ ئۇيقۇ ھالصتصنص جەدصۋەللەشتۈرۈش ۋە قۇل بصلەن باشقۇرۇش كۈزتصش ئەسۋابص ئارقصلصق باھالصغان ،ئەتصسص يەنە ئۇلاردصن ئاددص ئەقصلص سصناق ئالغان ھەمدە ئۇلارنصڭ جصددصيلەشتۈرگۈچص ھۇرمۇن سەۋصيسصنص تەكشۈرگەن ،ئەرلەر جۈرصسص بصلەن بصر ياستۇققا باش قۇيغاندا تصخصمۇ بەھۇزۇر ئۇخلايدصغانلصقصنص ئصيصتقان بۇلسصمۇ ئەمصلصيەتتە ئۇنداق بۇلمصغان ، دصمەك ئۇلارنصڭ چوڭ مصڭصسص ئۇيقۇ جەريانصدا كاشصلصغا ئۇچرصغان .
ۋاھالەنكص ئاياللار جۇپتص بصلەن بصر ياستۇققا باش قۇيغاندا ،ئۇيقۇ سۈپصتص تصخصمۇ ياخشص بۇلغان ،ئەرلەرنصڭكصگە ئۇخشاش ئصنكاس كۈرۈلمصگەن ،ئۇلار ئادەتتە ياستۇققا باش قۇيشص بصلەنلا ئۇيقۇغا كەتكەن ،ئەرلەرگە قارغاندا ئۇيقۇغا تصخمۇ تصز غەرصق بۇلۇپ ،يالغۇز ئۇخلصغانغا قارغاندا تصخمۇ راھەتلەنگەن .
ئۇيقۇ مۇتەخەسسسص دوكتۇرص سصدصلص :«بصز جۈرصلەرنصڭ بصرياستۇققا باش قۇيۇپ ئۇخلصشصنص تەكصتلصمەيمصز ،بۇنداق قصلصش تۇلصمۇ غەلصتە بصر ئصش ،چۈنكص ئصككص ئادەم ئۈزئارا تصۋصش چصقصردۇ ،يۇتقان تالصشصپ چصقصدۇ ،مۇشۇنداق بۇلسا قانداقمۇ تاتلصق ئۇخلصغصلص بۇلسۇن ؟!»دەيدۇ .

تەتقىقاتتىن بايقىلىشچە ،ئەرلەر جۈرىسى بىلەن بىرياستۇققا باش قۇيۇپ ئۇخلىغاندا كاشىلىغا ئۇچرايدىكەن ،مەيلى يىقىنچىلىق قىلسۇن ياكى قىلمىسۇن ،ئەتىسى رۇيى ھالىتى بىنۇرمال بۇلۇپ ،چوڭ مىڭىسىنىڭ ئىنكاسى تۈۋەنلەپ كىتىدىكەن ،«يىڭى ئىلىم-پەن »ژۇرنىلىنىڭ ئەڭ يىڭى بىر سانىدا بىسىلغان تەتقىقاتتا دىيلىشىچە ،ئاياللار بۇخىل كاشىلىدىن مۇستەسنا ئىكەن ،ئاياللار ئەرلەرنىڭ ئەكىسچە بىرياستۇققا باش قۇيۇپ ئۇخلىسا تىخىمۇ بەھۇزۇر ئارام ئالالايدىكەن ،چۈنكى بۇخىل ھالەت ئاياللارنى غەرىق ئۇيقۇ ھالىتىگە كەلتۈردىكەن.
ۋىنا ئۇنۋىرسىتىتىنىڭ پىرۇفىسۇرى خارىت.كارۇچ يىتەكچىلىكىدىكى بىر تەتقىقات گۇرپىسى پەرزەنىت كۈرمىگەن 20نەچچە ياشلىق قىزيىگىتلەر جۈرىلەر ئۇستىدە تەتقىق ئىلىپ بارغان ،تەتقىقاتچىلارئۇلارنى 10كىچە بىر ياستۇققا باش قۇيۇپ ئۇخلىتىپ ،كىيىن يەنە 10كىچە ئايرىم ئۇخلاتقۇزغان .تەتقىقاتچىلار ئۇلارنىڭ ئۇيقۇ ھالىتىنى جەدىۋەللەشتۈرۈش ۋە قۇل بىلەن باشقۇرۇش كۈزتىش ئەسۋابى ئارقىلىق باھالىغان ،ئەتىسى يەنە ئۇلاردىن ئاددى ئەقىلى سىناق ئالغان ھەمدە ئۇلارنىڭ جىددىيلەشتۈرگۈچى ھۇرمۇن سەۋىيسىنى تەكشۈرگەن ،ئەرلەر جۈرىسى بىلەن بىر ياستۇققا باش قۇيغاندا تىخىمۇ بەھۇزۇر ئۇخلايدىغانلىقىنى ئىيىتقان بۇلسىمۇ ئەمىلىيەتتە ئۇنداق بۇلمىغان ، دىمەك ئۇلارنىڭ چوڭ مىڭىسى ئۇيقۇ جەريانىدا كاشىلىغا ئۇچرىغان .
ۋاھالەنكى ئاياللار جۇپتى بىلەن بىر ياستۇققا باش قۇيغاندا ،ئۇيقۇ سۈپىتى تىخىمۇ ياخشى بۇلغان ،ئەرلەرنىڭكىگە ئۇخشاش ئىنكاس كۈرۈلمىگەن ،ئۇلار ئادەتتە ياستۇققا باش قۇيشى بىلەنلا ئۇيقۇغا كەتكەن ،ئەرلەرگە قارغاندا ئۇيقۇغا تىخمۇ تىز غەرىق بۇلۇپ ،يالغۇز ئۇخلىغانغا قارغاندا تىخمۇ راھەتلەنگەن .
ئۇيقۇ مۇتەخەسسسى دوكتۇرى سىدىلى :«بىز جۈرىلەرنىڭ بىرياستۇققا باش قۇيۇپ ئۇخلىشىنى تەكىتلىمەيمىز ،بۇنداق قىلىش تۇلىمۇ غەلىتە بىر ئىش ،چۈنكى ئىككى ئادەم ئۈزئارا تىۋىش چىقىردۇ ،يۇتقان تالىشىپ چىقىدۇ ،مۇشۇنداق بۇلسا قانداقمۇ تاتلىق ئۇخلىغىلى بۇلسۇن ؟!»دەيدۇ .
