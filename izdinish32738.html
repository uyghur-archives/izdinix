ﭘﺮﯦﻤﯩﯧﺮﯨﻨﻰ ﻗﺎﻧﺪﺍﻕ ﺋﯘﮔﻪﻧﺪﯨﻢ.

ھەممىڭلار بىلىپ بولدۇڭلار ھەقىچان،مەن كومپۇتېر كەسپىدە ئوقۇمىغانتىم،خەن چىنىمۇ بىلمەيمەن، ئېنگىلىزچىنىمۇ شۇنداق،بىراق مۇشۇ كومپۇتېر ھەققىدەجىق يازمىلارنى كۇرۇپ ئۇنۇڭ ئىختىدارىغا غايىۋى ئاشىق بولۇپ يۇردۇم، ئۇ چاغلاردا مەندە  بىردانە  پاناسۇنىك6000 ماركىلىق تەقلىدى كامېرا بار ئىدى، ئۇنىڭدا تارتقان تويلارنى سىنئالغۇنىڭ كۇچۇرۇش ئىقتىدارىدىن پايدىلىنىپ كىرىشتۇرەتتىم،خەتلەرنى قەغەزگە يېزىۋىلىپ تارتىۋالاتتىم،بىراق بۇنۇڭلۇق بىلەن فىلىملىرىمىز كۇڭۇلدىكىدەك بولمايۋاتاتتى. كىسەك لېنتا مەخسەت ئەمەس ئىدى.زامان نىڭ ئالدىدا ماڭمىساق شاللىنىپ قالاتتۇق .خۇلاسە شۇكى چوقۇم كومپۇتېربىلەن قۇراللىنىش كىرەك .
شۇنداق قىلىپ يىمەي-كىيمەي يىققان پۇلۇمغا كومپۇتېرئېلىش ئۇچۇن ئۇرۇمچى نەدىسەن دەپ تەۋەككۇل قىلدىم،قىزىل بايراقنى ئارىلاپ يۇرسەم بىر كومپۇتېرخانىدا ئۇيغۇربالىلار تۇرمامدۇ،شۇنۇڭ بىلەن ئۇ يەرگە كىرىپ قاراپ باقتىم،مانا بۇ كىينكى ۋاقىتلاردا داڭ چىقارغان ئۇيغۇر سوفىت شۇ ئىدى،
شۇنداق قىلىپ 12950 سومغا ئېكرانى 14 سۇڭلۇق بىر كومپۇتېر سېتىۋالدىم ھەم دەرھال ئۇيگە قايتتىم،كىلىپلا قۇراشتۇردۇم( ئۇزۇم كەنىت توكچىسى بولغاچ تەس بىلىنمىدى)،ھەرىكەتنىمۇ باشلىۋەتتىم،ئەمما...
    ئەمما زە چاشقىنەك كۇرۇنمەي قويدى،تېخى ھېلىلا بار ئىدى! بۇزۇلۇپتۇ ئەمىسە بۇ كومپۇتېرىم.
   شۇ  كەچتە ئۇخلاۋاتقان يېرىدىن بالامنىڭ ساۋاقدىشى-قازاق خوشنىمىزنىڭ بالىسىنى چاقىرىپ چىقتىم،ئۇ ئۇرۇمچىدە تېلىۋىزىيە ئونۋېرسىتىتىدە ئوقۇيتتى،3-يىللىقتا ئىدى،كىرىپلا ( ۋاي بۇ بىزنىڭ ئوقىغان كومپۇتېرلارغا ئوخشىمايدىكەن) دەپلا قاچتى، سەۋەپ شۇكى 1998-يىللاردا ئۇلار ئاساسى ماشىنىسى ياتما قاپلىق كومپۇتېرلارنى ئىشلىتىپتىكەن،مىنىكى بولسا تىك قاپلىق ھازىرقى مودا كومپۇتېر ئىدى  بۇنۇڭ تۇرقىلا ئۇنى قاچۇرىۋەتكەن گەپ ئىكەن،ئەتىسى ئۇرۇمچى ئاشلىق مەكتەپتە ئوقۇيدىغان تۇققىنىمىزنىڭ قىزىنى تەكلىپ قىلدىم،ئۇمۇ بوسۇغىدا تۇرۇپلا ( بىزنىڭ كومپۇتېرغا ئوخشىمايدىكەن) دەپلا قاچتى،شۇ كۇنى كەچتە ئىنىمنىڭ ئۇرۇمچى ئاشلىق مەكتىۋىدە ئوقۇپ قايتقان ئوغلىنى چاقىرىپ كۇرۇپ بىقىشقا تەكلىپ قىلدىم، بۇبايقۇشمۇ كىلىپ ئۇنى قىلىپ- بۇنى قىلىپ ھىچنەرسە چىقىرالماي يانچۇقىدىن تۇت بۇرجەك قارا نەرسىنى چىقىرىپ كومپۇتېرىمنىڭ قورسىقىغا سالدى، بىردەمدىلا كىچىك بىر قونچاق دىك-دىك سەكرەپ چىقتى،باسقانسىرى يۇقۇرى ئۇرلەپ ئويۇن قىلىپ بەردى،خېلى كۇلۇشتۇق...بىراق بەرىبىر كومپۇتېرىم (سۇزلىگىلى ) ئۇنۇمىدى،
شۇنۇڭ بىلەن سورىدىم:-سىلەر مەكتەپتە 3 يىل ، ئوقۇدىڭلار،ئۇگەنگىنىڭ مۇشۇمۇ، دىسەم:- بىزنى سىنىپقا ئەكىرىپ قويۇپ قۇرۇق نەزەرىيە سۇزلەپ چىقىپ كىتىدۇ، ئامال يوق ئۇنۇڭ- بۇنۇڭدىن مۇشۇئويۇننى ئۇگۇنۇپ تېخى مەيدىمىزنى كېرىپ يۇرۇيمىز،- دىمەسما، شۇ ئاچچىقىمدا:- داداڭ شۇنداق بايما،3 يىل ئوقۇپ كومپۇتېر نى ئاچالمايسەن،- دەپ قوغلاپ چىقاردىم.
ئەندى قانداق قىلىش كىرەك؟ 13 مىڭ سوم پۇلۇمنى مۇنۇ تۇمۇر-تەرسەك يەپكىتەمدۇ،قاراپ تۇرامدىمەن؟،
ياق، ئۇز كۇچۇمگە،ئەقلىمگە تايانماقتىن باشقا ئىلاج يوق،13 مىڭ سوم پۇلنى مۇنۇ تۇمۇردىن ئۇندۇرۇۋېلىشىم كىرەك!  ناھىيەمىزدە ھىچكىمدىن ئۇگۇنۇش ئىمكانىيىتى يوق ئىدى،بىلمەيىتتى،ئۇرۇمچىدىن ئۇگۇنەي دىسەم بارالمايىتتىم، چۇنكى مەن ئەترەت نىڭ توك يەتكۇزۇپ بىرىش ئىشىنى كۇتۇرە ئالغان،تاشلاپ كىتەلمەيىتتىم.
   شۇندىن كىين جاپالىق كۇنلەر باشلاندى،خەن يېزىقلىرىنى بىرمۇ-بىر لوغەت دەپتىرىمگە خاتىرلەپ چىقتىم،ئۇ كاتەك خەتلەرنىڭ مەناسىنى رەسىم ھالىتىگە تەقلىت قىلىپ چۇشۇنۇشكە تىرىشتىم،لىكىن بەك تەس كەلدى،چۇنكى بۇ يېزىقنى بىلىش ئۇچۇن ئاۋال يېزىشنى بىلىش كىرەك،ئاندىن ئوقۇلىشىنى بىلىش كىرەك،ئاخىرىدا مەناسىنى چۇشۇنۇش كىرەك،بىلمىگەن خەتلەرنىڭ مەناسىنى ئىزدەيدىغان لوغەت مەندە يوق.بۇمۇبولماي ئېنگىلىزچە يېزىلمىلارغا دىققىتىمنى ئاغدۇردۇم،بۇ ئاسان چۇشتى چۇنكى ھەرىپ ئاران 28، ئىلگىرىمۇ بىلەتتىم،بەزى سۇزلەر تىلىمىزدىمۇ بار ئىدى،يوقلىرىنى بولسا چىكىپ بىقىپ چىققان نەتىجىگە قاراپ تەخمىن قىلدىم،شۇ بىر يىلدا ئۇگەنگەنلىرىم قىسقا فىلىملەرنى ئىشلەش،شۇ ئارقىلىق بىر قىسىملىق توي  فىلىمىنى 5-6 پارچىغا بۇلۇپ ئىشلەپ نۇر تەخسىگە كۇچۇرۇش بولدى.چۇنكى مۇشۇنداق قىلمىساق كومپۇتېرىمنىڭ ئۇ ۋاختىدىكى سەپلىمىسى فىلىم ئىشلەشكە يار بەرمەيىتتى،cpu سى ئاران330 م ب ئىدى
شۇنداق قىلىپ رەسمىيەت بىجىرىپ  بازاردىن دۇكان ئالدىم،تىجارەت باشلىدىم. بۇ 1999- يىللىرى ئىدى.

urdek يوللانغان ۋاقتى 2008-5-19 22:25:52

باشتا ئەيتىپ ئۇتكۇنۇمدەك شۇ چاغلاردا ئىشلىگەن ۈ س د لىرىمىز سۇزۇك ئەمەس ئىدى،يەنەكىلىپ بىر قىتىمدا بىرلا بېسىملاپ چىقىرىشقا سەپلىمىلەرنىڭ ئىختىدارى يەتمەيىتتى، قانداق قىلىش كىرەك؟ ئۇرۇمچى -قىزىل بايراقتىكى شېركەتلەر،ئۇيغۇرسۇفتتىكى ئالىم ئەھەت قاتارلىقلار  مەسلەھەتى بىلەن سېناززى بېسىملاش كارتىسىنى 4200 سومغا ئالدىم،بۇ كارتىدا بېسىملىغان فلىم تېز ۋەپۇتۇن چىقتى،بىراق سۇزۇكلىگى يەنىلا تۇۋەن,مەن سېناززى نى سۇزۇك چىقىرىپ بىرەمىكى دەپ ئالغان ئەمما نەتىجە بۇنداق بولۇپ چىقتى،بۇ نىمە ئىش ئەندى؟بولمىسا ئىشلىتىۋاتقان سېستىمام پىرىمىيېرى 4.2 ئىدى،كومپۇتېرنىڭ سەپلىمىلىرىنى ئۇرۇمچىدە سېتىۋاتقان ئەڭ يېڭى ئۇسكۇنىلەر بىلەن تولۇقلىدىم،ئەندى كام قىلغان قايسى ئىش قالدى؟
كىيىن بايقىشىمچە مەن كومپۇتېرغا  كىرگۇزگەن فىلىملەر تەقلىدى كامىرادا تارتىلغان فىلىملەر ئىكەن،بۇنىڭ ئىكران ئىنىقلىغى بەك تۇۋەن،ئۇرۇمچىدىكى ئىتتىپاق كوچىسىدىكى دۇكانلارغا قاراپ باقسام تېلىۋىزىيە ئىستانسىلىرىدا ئىشلىتىدىغان ،ئېغىرلىغى 25 كىلوگرام كىلىدىغان چوڭ كامىرالار بار ئىكەن،باھاسىنى 000 164 سوم دەپ يېزىپ قويۇپتۇ! ئاللا كارامەت، بۇنچىلىك پۇلۇم بولسا تېلىۋىزىيە ئىستانسىسىدىن بىرنى قۇرۇۋالاتتىم،
قانداق قىلىمىز ،يەنىلا بار ئاماللارنى قىلىپ چىقتىم،بۇنۇڭ ئىچىدىكى بىرسى شۇكى-كىسەك لېنتىدىكى فىلىمنى كومپۇتېرغا avi ھالەتتە ئەكىرىپ يەنە شۇ avi ھالەتتەئىشلەش،ئاندىن ۈ س د غا بېسىملاپ چىقىرىش،بۇنىڭدا سۇزۇكلىگى ياخشى بولغىنى بىلەن بوشلۇقنى شۇنچىلىك كۇپ ئىگەللىۋالىدىكىن قاتتىق دېسكا ئىچىگە كىرىپ ئاياق قويغىدەك يەر تاپالماي قالىسىز. ۋاقىت دىگەن ئايىماي كىتىدىكەن.
مۇشۇنداق  پالاخلاپ يۇرگەن كۇنلەرنىڭ بىرىدە قىزىلتاغ تەرەپتىكى پۇچتا بىناسىنىڭ ئۇستۇنكى قەۋەتلىرىدە  ئېلىكترون سايمانلىرى ساتىدىغان يەرلەرنى ئارىلاپ يۇرسەم ئۇزى ئالقانغا پاتقۇدەك كىچىك رەقەملىك كامىرالارغا كۇزۇم چۇشتى،ئېكران سىزىقلىغى520x400 ئەتراپىدا ئىكەن،پۇلى 12500سوم ئىكەن،شۇنۇڭ بىلەن دەرھال ئالىم ئەھەتكە تېلىفۇن قىلسام يىتىپ كەلدى،ۋە بىر-بىردىن سىناپ ئىنگىلىزچە خەتلىك چۇشەندۇرۇشلىرىنى كۇرۇپ مەشخۇلات قىلدى،يان ئىكرانىغا قارىساق شۇنداق سۇزۇك ھەم رەڭلىك ئىكەن،ئۇچاغلاردا بۇخىل رەقەملىك كامىرا ئۇرۇمچى بازىرى ئۇچۇن يېڭىلىق ئىدى،مىنى جەلىپ قىلغىنى چاققان،ئەپچىللىگى بولدى،چۇنكى يېزىلاردا ئەتىدىن كەچكىچە ئېغىر بولغان ئەۋۇ( گومىنداڭنىڭ پوتىيى) نى كۇتۇرۇپ يۇرۇش ئازاپ ئىدى.بۇخىل كامىرادىن دۇكان خۇجايىنىنىڭ سۇزىگە قارىغاندا پەقەت تۇتىلا كىرگۇزۇلۇپتىكەن،مەن بىرنى ئالدىم،كىيىن ئالىم ئەھەتمۇ (بىز دىخاندىن قالامتۇق) دەپ يەنە بىرىنى ئالدى،ھەتتائەنگىلىيەگە چىققاندا شۇ كامىرانى ئالغاچ بارغانتى.
شۇنداق قىلىپ بۇ كامىرانى ئىشلەتتىم،دەسلىۋىدە كىشىلەر رەسىم ئاپپاراتى ئوخشايدۇ دەپ قېلىشتى،يان ئىكراننى ئېچىپ ئۇزلىرىنىڭ گۇزەل چىرايلىرىنى كۇرسەتكەندىن كىين بەك ئامراق بولۇپ قېلىشتى،بىر قىشتىلا خەجلىگەن پۇلۇمنى چىقىرىۋالدىم.
ئۇ چاغلاردا رەقەملىك كامىرا ئومۇملاشمىغان،بىخ ھالىتىدە ئىدى.

2- قەۋەتتىكى mayhanaنىڭ يازمىسىدىن نەقىل

ﺷﯘ ﯞﺍﻗﯩﺘﺘﺎ ﺷﯘﻧﺪﺍﻕ ﻗﯩﻤﻤﻪﺗﻤﯩﺘﻰ ﻛﻮﻣﭙﯩﻴﻮﺗﯧﺮ؟


1989- يىلى 2- ئايدا ئۇيغۇر ئاپتونۇم رايۇنلۇق مائارىپ نازارىتى ئۇيغۇر رايۇنىدىكى  ئالىي مەكتەپلەرگە 2 دىن،  ئوتتۇرا تېخنىكوملارغا بىردىن تۇنجى قېتىم كومپيۇتېر تارقاتقان ئىدى، بىزنىڭ مەكتەپكە تەقسىم قىلغان كومپيۇتېرنىڭ ئىكرانى 14 دىيۇملۇق، رەڭسىز، مەركىزى بىرتەرەپ قىلغۇچىسى 80، قاتتىق دىسكىسى 800 مىگا بەيتىس، ئىككى دانە 5 دىيۇملۇق يۇمشاق دىسكا چۈشىدىغان ئېغىزى بار، 2 مىگابەيتىسلىق ئىچكى ساقلىغۇچىسى بار كومپيۇتېر ئىدى، شۇ چاغدا كومپيۇتېرنىڭ تالۇنىدىكى باھا 20مىڭ 600 سوم ئىدى. مەن مۇشۇ كومپيۇتېرنىڭ باھانىسىدا ئايرۇپىلانغا چىقىۋالغان ئىدى، كومپيۇتېر تەۋرەپ كەتسە بولمايدىكەن، ئۇنى ئايرۇپىلاندا ئەكىتىشىمىز كېرەككەن دەپ مەكتىپىمگە تېلېفۇن قىلسام، ئايرۇپىلاندا ئەكىلىڭ دېگەن ئىدى،  بىزگە كومپيۇتېر تارقاتقان يولداشمۇ شۇنداق ئۇقتۇرغان ئىدى، پەقەت ئىسىمدىن چىقمايدىغىنى، بىر پىشقەدەم كەسپ دىشىم، مەن خىزمەتكە چىققىلى 30 يىل بولۇپتۇ، بىرەر قېتىممۇ ئايرۇپىلاندا ئولتۇرۇپ باقمىغان ئىدىم، سىز 4 يىل بولمايلا ئايرۇپىلاندا ئولتاردىڭىز، دېگەن ئىدى، رەھمەتلىك شۇ ئۇستاز ئۆمرىدە ئايرۇپىلاندا ئولتۇرۇپ باقماي دۇنيادىن ئارماندا كەتتى.
