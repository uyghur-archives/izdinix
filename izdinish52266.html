1- سوئال: بۇ قېتىملىق يېغىندا نېمە ئۈچۈن ئاز سانلىق مىللەت ۋەكىللىرى كۆپ ؟



2- سوئال ئىلگىرى ئۇنداق ئەمەستىغۇ ؟





شىنجاڭ ئۇيغۇر ئاپتونوم رايونلۇق 12–نۆۋەتلىك خەلق قۇرۇلتىيى ۋەكىللىرىنىڭ ئىسىملىكى





     (550كىشى، ئۇيغۇر ئېلىپبە تەرتىپى بويىچە تىزىلدى)


   


ئۈرۈمچى شەھىرىدىن (77 كىشى)


   


   ئابدۇرېھىم ئابلىز، ئابدۇشۈكۈر رەھمۇتۇللا، ئابدۇغوپۇر ئابدۇگۈل، ئادىلجان خەمىت، ئاقسۇم ئاقموللا، ئالىم رەقىپ (تاتار)، ئامىنە قادىر، ئايگۈل ياقۇپ، ئەكبەر تۇرغۇن، ئەن جېڭيۈ، ئەن خۇڭۋېي، باتۇرجان يارمەمەت، پەن جىنخەي، پولات ھوشۇر، تيەن پىڭ (ئايال)، جاڭ خۇڭ، جاڭ شياۋپىڭ، جاڭ ۋېنشېڭ، جارۇللا ھېسامىدىن، جۇ خەيلۇن، جياڭ شىڭۋېي، جياۋ يىمىن، چېن دىڭۋۇ، چېڭ لىمىن، چيۇ شۇخۇا (ئايال)، خەن جيەنگاڭ، خۇ جۈنخەي، خۇاڭ ۋېي، خې پىڭ (ئايال)، خې يۇڭ، خې يىمىڭ، دولقۇن ئابدۇرۇسۇن (قازاق)، دىلبەر ئېزىز، دى مىڭ، دىڭ فۇجى (خۇيزۇ)، رەشىدە خەمىت، زۆھرە ئابلىز، سۇڭ گوجۈن(خۇيزۇ)، سۈي جىپىڭ، سۈي چۇنيۈ (مانجۇ)، شۆليان قالىيبەك (قازاق)، شېڭ يەنجياڭ، شى شىنلى، غالىپ ئىمام، فېردەۋىس سايران (ئايال، تاتار)، قادىر مەمەت، قاھار ئوسمان، گاۋ شىمېي (ئايال)، گو شياۋمېي (ئايال، رۇس)، گۈلنار ئابدۇكېرىم، لو بىن، لۇ چىن، لۇ شياڭباۋ، لى خۇيجۈەن (ئايال)، لى يى، ليۇ جيې، ليۇ شىنلەن (ئايال)، ليۇ يۈچېڭ، ما جۈن (خۇيزۇ)، مارىيە خابدۇللا (ئايال، قازاق)، ماھىنۇر، ما يۈجۇڭ (خۇيزۇ)، ما يىتۇڭ، مەھبۇبە مەنسۇر، مۆھتىرەم شېرىپ، مىنەۋەر ئىمىن، مى ئېنخۇا (خۇيزۇ)، نەسىردىن مايىت، نۇرمەمەت ھاپىز، نىجات ئابدۇرېھىم، ھەسەنجان مەمەتجان، ئۆتكۈر ئابدۇراخمان، ۋاڭ جۈن (خۇيزۇ)، ۋاڭ چىڭمىڭ، ۋاڭ يۈشىن، ۋۇرالبېك (ئايال، قازاق)، ياڭ جيې (خۇيزۇ)


    ئىلى قازاق ئاپتونوم ئوبلاستىدىن (98 كىشى)


    ئىلى ئوبلاستىغا بىۋاسىتە قاراشلىق ئورۇنلاردىن (57 كىشى)


    ئابدۇرېھىم مەمەت، ئابدۇرۇسۇل رەخىم ، ئابدۇسالام سادىق، ئابدۇشۈكۈر قارىھاجى، ئابدۇغەنى ئابدۇرېھىم (ئۆزبېك)، ئابدۇقاھار ئابدۇقادىر، ئابىلقايىر كۈندەۋلەت (قازاق)، ئامانجول ئاسانغالى (قازاق)، ئامان كۆدەباي (قازاق)، ئەركىن غازى، ئەھمەت نىياز، ئەيساجان ئۆمەر، باۋۇرجان مۇقاش (قازاق)، پەن زۇڭلىن، پۇ شىيۈ، تاجىگۈل ئابلىكىم، تاڭساغار باغىن (قازاق)، تۇڭ رۇيچىن (شىبە)، تولقىن ھەزىرباي (قازاق)، تۇرسۇنئاي سۇلتان، تۇرسۇنتاي شەمشىدىن، تۇرغان پىدا، جاۋ يۇڭلۇڭ، خالىدەم ئۆمەرجان، دىلنۇر مۇتەللىپ (ئايال، قازاق)، رىشات مۇساجان، زاكىر زوردۇن، زۇنۇن ئەكبەر، ساجىدەم ئايۇپ، سۇ شيەنگو، سېرجان ئوغۇباي (قازاق)، شۈ جۇڭ، شۈ خەيشىن، شى شىنخۇا (ئايال)، كەلىمە خەمىت، گۈلبارشىن سېرىكباي (ئايال، قازاق)، گۈلمىرا كامۇنىس (ئايال، قازاق)، لى جىيىڭ (ئايال، خۇيزۇ)، لى شۆجۈن، لى مىڭخۇي، لى يۈمېي (ئايال)، ما بەي، ما پىڭخۇڭ (خۇيزۇ)، ما جۇڭلياڭ (دۇڭشياڭ)، ما جۈن (خۇيزۇ)، ما جېڭخۇا (ئايال، خۇيزۇ)، ما سۈيلەن (ئايال، دۇڭشياڭ)، ما شىجۈن (خۇيزۇ)، مەنىن زەينىل (قازاق)، مېھراي تۇرغانجان، نۇرلان ئابىلمەجىن (قازاق)، نۈسۈپ ئايىتجان (قازاق)، ئۈزمە (موڭغۇل)، ۋاڭ چۇنگۇاڭ (شىبە)، ۋاڭ گۇاڭچياڭ، ئې گېمېي (داغۇر)، ياۋجۈن دەۋلەتباي (قازاق)


    تارباغاتاي ۋىلايىتىدىن (24 كىشى)


    ئارغىن شاپىر (قازاق)، بەردىقان قاباتاي (قازاق)، تۇرسۇنگۈل روزى، تۆلۈۋقان قابىلھاشىم (ئايال، قازاق)، تېلىۋالدى ئابدۇرېشىت (قازاق)، جاڭ بو، جاڭ جىپىڭ (ئايال)، جاۋدەت مەنسۇر (قازاق)، جۇ شىنجۈن، چى شېڭفۇ (داغۇر)، خاتىمە نىياز، خې يۈدې (ئايال)، سارقىت ئاقان (قازاق)، سالتانات قىنايات (ئايال، قازاق)، سېرىك ئارىن (قازاق)، قايرانغالى ئەھمەتجان (قازاق)، لاگۋا سەيبۇگجاپ (موڭغۇل)، لى پېڭفېي (خۇيزۇ)، ما لۇڭ (خۇيزۇ)، ماينۇر جاقان (ئايال، قازاق)، ۋاڭ يىنشەن، ئىناۋەت سايراش (قازاق)، ئىۋانوۋا ناجا (ئايال، رۇس)، يۈسۈپجان ياسىن


    ئالتاي ۋىلايىتىدىن (17 كىشى)


    ئايگۈل بايبازار (ئايال، قازاق)، ئايگۈل ياسىن، ئەركىن ئۆمەر، باغجامال غىلىمقان (ئايال، قازاق)، تالغات يۈسەن (قازاق)، تيەن ۋېن (ئايال)، چوغايۇف ئاقسۇپى (قازاق)، خۇ جيەنشىن، سېرىك بايقاسىم (قازاق)، سىمايىل سىيىلغاجى (قازاق)، قايراتبېك ئاقاي (قازاق)، قىيلان پىئونېر (قازاق)، لى جىدۇڭ، ليۇ جيەن، ماغاۋ سېيىتقامزە (قازاق)، ما يەنمېي (خۇيزۇ)، مەۋلىتبېك ئارىسباي (قازاق)


    بېيتۈن شەھىرىدىن (3 كىشى)


    قانىيبات سابىكەي (قازاق)، شۈ چىڭجېن (ئايال)، ياڭ شياۋتۇن


    بورتالا موڭغۇل ئاپتونوم ئوبلاستىدىن (14 كىشى)


    ئالتا (ئايال، موڭغۇل)، ئاۋچۇر (موڭغۇل)، بايقا (موڭغۇل)، تۇرغۇن مەمەت، جاڭ چۇنلىن، جاۋ چىڭ، چاخان (موڭغۇل)، شاڭ يۇڭچياڭ، فۇرداش (موڭغۇل)، كۈرەش مەخسۇت، لى لىپىڭ، نۇرگۈل تۇرسۇن، نيې جۇاڭ، ئۆمەرجان نوقمان (قازاق)


    قاراماي شەھىرىدىن (14 كىشى)


    بورھان قاھار، جاۋ ۋۇشېڭ، چېن جۈنخاۋ، چېن شىنفا، خو ماۋمىن (ئايال)، خې ۋېنخۇا، رەناگۈل تۇرسۇن، رەيھان ماخمۇد، شۈ ۋېيشى، گۇەن يۈنرۇي (مانجۇ)، گۈلگىنە نىياز (ئايال، ئۆزبېك)، لياڭ چاۋ، مەمەتئېلى قارىھاجىم، ۋۇ شيەن


    شىخەنزە شەھىرىدىن (15 كىشى)


    ئالىمجان، جاڭ شىنرۇڭ (ئايال)، جاۋ جىڭشياڭ، چېن يى، چې جۈن، سۇڭ شياۋلىن (ئايال)، شۈ جىشىن، كاڭ ۋېنخۇا، گو بىن، گۈلباھار، نى جىبىڭ، ۋاڭ خۇييىڭ (ئايال)، ۋاڭ يىڭخۇڭ (ئايال)، ئىلھام سابىر


    سانجى خۇيزۇ ئاپتونوم ئوبلاستىدىن (31 كىشى)


    ئەخمەت سېيىتنىياز، بەخت شاكەي (ئايال، قازاق)، بۇخەلچەم قاسىم، پېڭ گوچۇن، تۇرسۇن (قازاق)، جاڭ چۈنشيەن، جۇ جىمىڭ (خۇيزۇ)، چيەن جى، خەيرىگۈل، دۇڭ شىنگۇاڭ، دىڭ لى (ئايال، خۇيزۇ)، رېھىم، زېڭ چۈنلېي، سەي جىنپىڭ (ئايال، خۇيزۇ)، سۇڭ رۇي، شەمەي (قازاق)، لى جيەنگو، لى چېڭخۇي، ما تاۋ (خۇيزۇ)، ما جيەنشىن (خۇيزۇ)، ما دېڭفېڭ (خۇيزۇ)، ما شيۇڭچېڭ (خۇيزۇ)، ما مىڭچېڭ (خۇيزۇ)، ماھىنۇر تۇرسۇن، ئاجار كامالىم (ئايال، قازاق)، ئوسمان تاڭخەي، ئوغۇلخان، ۋاڭ جيۇجۇڭ، ۋاڭ خۇڭشىن، ياڭ جيې (خۇيزۇ)، يىن خۇڭشيا (ئايال، خۇيزۇ)


    ۋۇجياچۈ شەھىرىدىن (6 كىشى)


    بەي خۇڭبېن، چياڭ شىشۆ، شىن مىڭۋېي، لى گۇاڭشېن، ليۇ يى، ما ۋېنشيا (ئايال، خۇيزۇ)


    تۇرپان ۋىلايىتىدىن (17 كىشى)


    تۇرپان شەھىرىدىن (7 كىشى): ئەخمەت تۇرسۇن، باھادىر ئابدۇھېلىم، رابىگۈل ئابدۇۋەلى، سۇڭ ئەيرۇڭ (ئايال)، ئوسمان جېلىل، ۋۇ زىيىڭ (ئايال)، ئىلھام ئوسمان


    پىچان ناھىيەسىدىن (6 كىشى):


    تۇڭ ۋېي، زاقان قايسا (قازاق)، ليۇ شىنشېڭ، ما رۇيخۇڭ (ئايال، خۇيزۇ)، مۇختەر ھەسەن، ياسىن ئابدۇرېھىم


    توقسۇن ناھىيەسىدىن (4 كىشى):


    ئابلىز ھوشۇر، ئالىم ھەبىبۇل، ساۋ جېڭكۇەن، ليۇ يانىڭ


    قۇمۇل ۋىلايىتىدىن (23 كىشى)


    قۇمۇل شەھىرىدىن (15 كىشى): بەي جىجيې، بەي جىگې (خۇيزۇ)، تيەن جيەنرۇڭ، خۇاڭ جىگاڭ، كېرەم ھىمىت، گو ليەنشەن، گۈلشات ئابدۇقادىر، ليۇ شياڭ، مەمەت راخمان، مۇھەممەت نامان، مۇھەممەت نىياز، نۇرجاي (ئايال، قازاق)، ئىسكەندەر ياقۇپ، ياسىن سىدىق، يۈسۈپ ئېلى


    باركۆل قازاق ئاپتونوم ناھىيەسىدىن (4 كىشى): مامۇت تويمىل (قازاق)، ۋاڭ شىجياڭ، ۋاڭ يەنچىڭ (ئايال)، ئېرجان قۇسمان (قازاق)


    ئارا تۈرۈك ناھىيەسىدىن (4 كىشى): خالمۇرات غوپۇر، گۈلباھار قىياز، ۋاڭ چۇنلى، ۋاڭ يۆبىن


    بايىنغولىن موڭغۇل ئاپتونوم ئوبلاستىدىن (32 كىشى)


    ئەخمەت توختى، ئەركىن ئىسلام، باسىت ھامۇت، بورە (موڭغۇل)، بايىنكېشىك (موڭغۇل)، پېڭ جيارۇي، تۇرسۇن ھەسەن، تۇرسۇن ئىبراھىم، جۈرئەت تۇردى، جىن خۇا (ئايال، موڭغۇل)، خۇاڭ جىنجۇڭ، خۇ يۈەنجېڭ، دا سېرىنجاپ (موڭغۇل)، دولقۇن توختى، دېڭ رېنجيې، دىلنۇر ئېزىز، رابىگۈل تۇرسۇنجان، شى جيەنيۇڭ (موڭغۇل)، شى ۋېنخەي (خۇيزۇ)، غەيرەت ھەسەن، غېنى مەمەت، قۇربان ئوسمان، گۈلزار مەمەت، گۈلشەن ئابلا، لۇ پىڭ، لى مىڭيىن، لىن ۋېي (ئايال)، ليۇ فۇلىن، مىجىت ھىمىت، ھاۋاگۈل ئىبراھىم، ۋاڭ شېڭيۈن (خۇيزۇ)، ۋاڭ ۋېن (ئايال)


    ئاقسۇ ۋىلايىتىدىن (48 كىشى)


    ئاقسۇ شەھىرىدىن (10 كىشى): ئابدۇغېنى مەمەت، ئەھەت قاۋۇل، پەرھات ئەنۋەر، جۇ چاڭجيې، چىمەنگۈل نۇر، خۇاڭ سەنپىڭ، گاۋ گوفېي، گۈزەلقىز ئىسمايىل، مەمەت سادىر، ئېزىز قېيۇم


    كۇچا ناھىيەسىدىن (8 كشى): ئارزۇگۈل تۇردى، خاسىيەت ئېلى، رىسالەت روزى، قاھار ئىسمايىل، گاۋ كېپىڭ، مەردان مۇقىيت، مۇتەللىپ قاسىم، نۇرمەمەت روزى


    كەلپىن ناھىيەسىدىن (3 كىشى): ئەسقەر مەمتىمىن، زىننەتخان ئىسا، شيۇڭ شۈەنگو


    ئۇچتۇرپان ناھىيەسىدىن (5 كىشى): تۇرغۇن ئابلا، خەن يۇڭ، گو يۇڭخۇي، گۈلبۈۋى قادىر (ئايال، قىرغىز)، ئېلى يۇنۇس


    باي ناھىيەسىدىن (6 كىشى): ئاينۇر مەقسەت، ئەيسا ئىمىش، تۇرغۇن ئابدۇرېھىم، زۇلپىيە قاسىم، ۋاڭ خۇيمىن، ياڭ فاسېن


    ئونسۇ ناھىيەسىدىن (5 كىشى): ئەنۋەر سەلەي، ئەھەت توخسۇن، باھارگۈل سەمەت، گۈلئايشەم ياسىن، ئىرىسباي ئىسلام (قىرغىز)


    شايار ناھىيەسىدىن (4 كىشى): تۇرسۇنئاي باسىت، جىڭ خەييەن (ئايال)، مەمەت راخمان، ئېزىز ئەمەت


    توقسۇ ناھىيەسىدىن (3 كىشى): ئەمەرجان ئىمىن، خەيرىنىسا ياسىن، مۇ جېنجياڭ


    ئاۋات ناھىيەسىدىن (4 كىشى): ئەمەرجان ئاۋۇت، ئەنۋەر سېيىت، گۈلەيسەم مەتنىياز، نىشان ئىبراھىم
