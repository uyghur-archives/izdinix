دەرىس قىزغىن داۋاملىشىۋاتاتتى . مەن كۆزۇتكۇچى بولۇش سۇپۇتۇم بىلەن ئۇيغۇر تىلىنىڭ جەزبىسىدىن ھوزۇرلانغاچ ، خاتىرەمگە كونا ئادىتىم بويىچە نىنجى قويۇپ بىر نىمىلەرنى ئىلىپ قويۋاتاتتىم . سىرتتىن قارىماققا مېنىڭ ئالدىراش يازغانلىرىمنى كۆرگەن ئادەملەر مېنى سادىق بىر چوقۇنغۇچى دەپ ئويلاپ قېلىشى مۇمكىن ئىدى ، ئەمما مەن بەزى نىنجىلاردا ئانچە مۇنچە بۇ مۇۋاپىقيەت ئۇستازىنىڭ سۆزلىرىنى ئەينەن يازغاندىن باشقا ، خالىسام ئۇنىڭ سۆزىنى ئاڭلىغاندىن كىيىن كاللامدا شەكىللەنگەن پىكىرلەرنى ، قارىشىمنى ، ئۇنىڭغا بولغان تەستىقىمنى تەنقىتىمنى ، ھەتتا ئۇنىڭ سۆزىنى ئاڭلاپ ئىسىمگە كەلگەن باشقا ھىس تۇيغۇلۇرۇمنى قالدۇرۇپ قويىۋاتاتتىم ، خۇددى ئۇ توختىماي  چاقماق چاقاتتى ، مېنىڭ ئىچىمدە بىرنىمە گۇرۇلدەپ قېپىشىپ تۇتۇشۇپ كۆيەتتى .ئۇنىڭ ھاياجانلىق پىلىموتنىڭ ئوقىدەك چىقىۋاتقان داندان ئۇيغۇرچە ئاۋاز مېنىڭ قىممەت قارىشىم ، تارىخى مەدىنيەت بىلىمىم ئوقۇغان كۆرگەن كىتاپلىرىم ، ھايات تەجرىبىلىرىم بىلەن سوقۇشۇپ  توختىماي ئوت چاچىرتاتتى . مەن بولسام ھەر بىر ئوت چاقنىغاندا ئۇنى ئۆز پېتى خاتىرەمگە قوندۇرۇۋالاتتىم .
ھە راست ، ئەسلىدە ئاغىنەم مەسئوتجاننىڭ دۇكىنىدىن بىر خاتىرە ، قەلەم ئېلىشقا ئۇلگۇرەلمەي قۇرۇق قوللا بېرىپ قالغان ئىدىم ، ئۇلار ئىككى تۆت بەت ئېلان بار بىر يازغىلى بولىدىغان دەپتەرنى سوغا قىلدى ، شۇنىڭ بىلەن بۇ ئىشمۇ پۇتتى

مەن تۇيۇقسىز دەرىس ئاڭلىغاچ بۇرۇن قومۇلدا ئەر شەرلەر بولۇپ مودا بولغان ئەنلى دەرسى بىلەن بۇ لېكسىيەنىڭ پەرقى ۋە ئوخشاشلىقىنى سېلىشتۇرۇپ باقتىم . 99 -يىلى ياكى 2000 -يىللار بولسا كېرەك تۇنجى ئەۋلات ئۇيغۇر ئەنلى شاگىرتلىرىنىڭ نۇتۇقلىرىنى ئاڭلايمىز دەپ قايسىبىر بىنا ئۆيلەرگە بىېرىپ كىچىك كچىك سولياۋ بەندىڭلەردە مانتا تىزغاندەك تىزىلىپ ئولتۇرۇپ ئاڭلاپ كەتكەن ئىدۇق ، مەنغۇ ئەنلى دىگەنگە ئانچە قىزىقمايتتىم شۇ بويتاق بولغاچقا چىرايلىق قىزلار ئۇچراپ قالامدىكىن دەپ ئانچە مۇنچە بېرىپ قوياتتىم  شۇ يىللاردا   ،  بۇلارنىڭ ئوخشايدىغىنى  ھەر ئىككىلىسى شاگىرتلىرىنىڭ ماشىنا ئالغانلىقى ساياھەت قىلغانلىقى بىلەن پەخىرلىنىدىكەن ،ئوخشىمايدىغان يېرى ئەنلى ئاق ياقىلىق  تۇرمۇشنى ئاساس قىلىپ سۆزلىسە مۇۋاپىقيەتچى ئابدۇراخمان ئەپەندىم ئادەمنىڭ ئۆز ئۆزىنى قانداق كونتىرۇل قىلىپ مۇۋاپىقيەت سىزىقىغا قانداق دەسسىتىشنى سۆزلەيتتى .
ئەلۋەتتە قىزىقىپ قالغانلارغا ئەنلىنىڭ بەدىلى سەل ئېغىر توختاپ كەتتى شۇ يىللاردا ، ئەمما ماۋۇ ھېكمەتلىك پاراڭنى دەڭسەپ پايدىلانغۇدەك يېرىدىن پايدىلانساق بولىدىغاندەك .

مۇۋاپىقيەت ئۇستازى تىرنىرى ئابدۇراخمان ئەپەندىم ئىككى كۇن سۆزلەپ بەردى ، مېنىڭ بۇ يەردە ئۇنىڭ ھەممە سۆزلىرىنى سۆزلەپ بىرىشىمنىڭ ئورنى يوق پەقەت ماڭا ئەڭ ۋە ئەڭ تەسىرقىلغان بىر ھىكايىسىىنى سۆزلەيمەن .

يەنى ئابدۇراخمان ئەپەندىم ئىچكىردە چىن ئەنجىنىڭ پىشىنى تۇتقان شاگىرت ۋاقىتلىرىدا بىر قانچە شاگىرت بىر ئۆيدە ئۇستازىنىڭ كۆرسەتمىسى بويىچە بىر كۇنلۇك پائاليىتىنى باشلايدىكەندە ، ئەتتىگەندە بۇلار ئۆرە تۇرۇپ نۇتۇق سۆزلەش ئۇچۇن ساغلام بەدەم كېرەك دەپ چېنقىدىكەن ، ئاندىن سېتىش پىلانى نىشانىنى بىكىتىدىكەن ، بىر كۇن ساتىدىكەن ، كەچتە قايتىپ كىلىپ نىشانغا يەتكەن يەتمىگەنلىكىنى تەكشۇرەيدىكەن  . نىشانغا يەتمىسە جازاسى باركەن .جازاسى ئۆزى ئەڭ ئۆچ كۆرگەن يىمەكنى خام يىيىش ئىكەن . بىر ساۋاقدىشى ۋەزىپىنى ئورۇنلىيالماي قېشىنى چۇشۇرۋېتىپتۇ ، يەنە بىر ساۋاغدىشى ۋەزىپىنى ئورۇنلىيالماي ئۆزى ئەڭ ئۆچ ئاچچىق كاۋىدىن بەشنى خام يەپتۇ ، چىن ئەنجى ئەپەندىمنىڭ ئۇيغۇر شاگىرتى پىيازغا ئۆچ ئىكەن ، نەتىجىدە ئۇمۇ نىشاننى پىلاننى ئورۇنلىيالماي ئالمىدەك يوغان ئىككى پىيازنى خام يىيىشكە مەجبور بوپتۇ . ئۇنىڭ پىيازنى خام ياۋاتقان سۈرىتىنى كۆرۇپ بۇ ئىككى كۇن ئىچىدە ئەڭ ئىسىمدە قالغان بىر ھىكايە بولۇپ قالدى .
يەنى ۋەزىپىنى ئورۇنلىيالمىسا ئۆز ئۆزىنى جازالاش ، بەلكىم مەن بۇ ئىشنى قىلىشنى ئانچە بەك خالاپ كەتمەسمەن ، بەلكىم مەن پىلانىمنى ئورۇندىيالمىسام ئۆز ئۆزەمنى ئاسانلا كەچۇرۋېتەرمەن . شۇنداق مەن ھازىرغىچە ئۆز ئۆزەمنى بەزلەپ سەۋەپ تېپىپ ئۆز ئۆزەمنى كەچۇرۇپلا كەپتىمەن جۇما ،
بەلكىم موشۇنداق ئەمەلى ھالدا جازالاشنىڭ تېخىمۇ چوڭقۇر تەسىرى بولار ، ئۆزىنى كونتۇرۇل قىلىشقا ۋە ۋاقىتىنى كونتۇرۇل قىلىشقا .......................

يەنە بىر ئىش باشقىلارنى تەكىلىپ قىلىپ تەسرات سۆزلىتىش ، ئىككى كۇنلۇك دەرىس جەريانىدا 20 ئادەم قېتىمغا يېققىن ئادەم سەھنىگە چىقىپ ئۆز كۆز قاراش تەسراتلىرىنى سۆزلىدى ، ئەلۋەتتە توردا تەسىرات يازىدىغان توخوتۇر تېخى بۇلارنىڭ ئىچىگە كىرمەيدۇ . قاراپ باقسام شەھرى قومۇلىمىزدىمۇ نۇتۇق سۆزلەشكە قىزىقىدىغان ھەۋەس قىلدىغانلار ئاز ئەمەسكەن جۇما ، مانا نۇتۇق سۆزلەشكە قىزىقىدىغانلار ئۇچۇن  ماۋۇ سورۇن نەق بولىدىكەن جۇما يۇرەكنى پىشىرۇدىغان كاتتا سورۇنكەن ، ئالدىڭىزدا تۇرغىنى ئىلھاملاندۇرۇش ئۇستازى ، تۇرتىلىپ قالسىڭىز ئۆزى بىر قوللۇق ئوڭشايدۇ دەيمەن . ئەمىليەتتە گەپكە ئۇستا بولۇش ئامرىكىلىقلارنىڭ ئەڭ ئەقەللى ئىقتىدارلىرىدىن بىرى ،ئۆز ئويلىغانلىرىنى راۋان چۇشۇنۇشلىك تەرتىپلىك باشقىلارنى جەلىپ قىلغۇدەك ھالدا بايان قىلىپ بىرىشنىمۇ ئۆگۇنۇپ قويغاننىڭ پاراكتىكا قىلغاننىڭ زىيىنى يوق ئەلۋەتتە .
بۇ يەرلىرى يىغىنلاردىكى باشلىقلار سۆزلەپ پۇقرالار ئەسنەيدىغان ھالەتلەردىن كەسكىن پەرىقلىنىدىكەن ، ئەمما لېكىن ئالا جۇقا سەكرەشلەر قولىنى بۇلغاپ گاھىدا مۇشتىنى چىڭ تۇگۇپ ئۇستازنىڭ سۆزىنى سۆزمۇ سۆز تەكرالاشلار ئادەمگە پارتىيەگە قەسەم بىرۋاتقاندەك گاھىدا گىتلېرى دەۋرىدىكى ساراڭلارچە قىززىق قانلىقنى ئەسلىتىپمۇ قويىدىكەن .
