ئەسسالامۇ ئەلەيكۇم! مەن سىزنىڭ ئەتتىر ھەققىدىكى تېمىڭىزنى ئوقۇغان. شۇڭلاشقا ماڭا ياردەم بېرەلەيدۇ دەپ ئويلىدىم.

‏مەن ئۆزەم ئائىلە ئايالى، ماركا، مودا دېگەنلەردىن تازا خەۋىرىم يوق، يىشىم 30 دىن ئاشقىلى تۇردى، تېرە ئاسىراشقا خېلى ئەھمىيەت بېرىمەن، سەككىز يىلدىن بېرى مىيلىنكەي مەھسۇلاتلىرىنى  ئىشلىتىپ كەلدىم، ئەمدى بىر ئالماشتۇرۇپ باققۇم بار، ئۇنىڭ يېڭىلىقىنى ھىس قىلالمايدىغان بولۇپ قالدىم، ئېرىمىز، بالىمىز گۈزەللىكنى سۆيگەن ئىكەن، تاشقى گۈزەللىككىمۇ ئەھمىيەت بېرىشىمىز زۆرۈر ئىكەن، شۇ سەۋەب ياردىمىڭىزگە ئىھتىياجىم بار .رەھمەت.

‏تېرەمنىڭ تېگى پاكىز، قۇرغاققا مايىل مۆتىدىل بۇغداي ئۆڭ  تېرە، ماڭا كېرىكى ئۈنۈمى ياخشى، يېشىمغا، تېرەمگە ماس كېلىدىغان، ئىشەنچىلىك، تەننەرخى  ئۆزىگە ئەرزىسە مەيلى، تېرە چېكىنىشنىڭ ئالدىنى ئالىدىغان تەبىي پاكىز بىر  يۈرۈش  تېرە ئاسىراش ۋە رەڭلىك گىرىم مەھسۇلاتلىرى لازىم ئىدى، قايسى ماركىنى ئىشلەتسەم بولار،كىرپىك لاكنىڭمۇ ياخشىراق، يۇيسا يېيىلمايدىغان بولسا،كۆز ئاسىراشنىڭمۇ ياخشىسى كېرەك ئىدى، توردىن قارىسام تۈرى جىق ھەم  ئىشەنچىسىز دەپ ئويلاپ ئۇزۇندىن بىرى ئالالمايۋاتىمەن، ئۆزۈم گۇاڭجۇدا تۇرىمەن. سودا سارايدىن سېتىۋىلىشىم قولايلىق. ياخشى ماركىدىن تونۇشتۇرغان بولسڭىز بەك خوش بولغان بولاتتىم. بىر يىل بولدى ئىككىلىنىپ، مەتومىكەنمەن. يەنە ئائىلىدە ئىشلىتىدىغان رومانتىك، يىنىك، تەبىي، ئىسل خاسلىققا ئىگە ئەتتىردىن بىر- ئىككىنى تونۇشتۇرۇپ بەرگەن بولسىڭىز. ئۆتكەنكىسى بەك جىق ئىكەن، كۆزلىرىم ئالا- چەكمەن بولۇپ كەتتى. رەھمەت قېرىندىشىم. باھاسى 400- 700 بولسا بولاتتى، ئاۋارە قىلدىم، ئوغۇللار كۆرسە سەت بولىدىغان بولدى، خەۋىرىڭىزنى كۈتىمەن.

تېما ئىگىسى: ئەگەر بۇرۇن نۇقتىلىق بىر ماركىنى تۇتۇپ ئىشلىتىپ باقمىغان بولسىڭىز ۋە ياكى يېڭىدىن بىر خىل ماركىنى سىناپ ئىشلىتىپ باقماقچى بولسىڭىز، دەسلەپتە چوقۇم مۇقىم مال سېتىش ئورنىغا بېرىپ، تېرىڭىزنى تەكشۈرتكۈزۈپ، مەسلىھەت سوراڭ، بۇنىڭدا گەرچە بىرەر- يېرىم سائەت ۋاقتىڭىز ئىسراپ بولسىمۇ، جىق نەرسىلەرنى ئۆگىنىۋالىسىز، ئۇلار ئادەتتە تېرىڭىزنى تەپسىلى سىناپ، ئاندىن قايسى مەھسۇلاتلارنىڭ ماس كېلىدىغانلىقى ۋە ئۇ نەرسىلەرنى قانداق ئىشلىتىش ھەققىدە پىكىر بېرىدۇ. ھەر قانداق ماركىنىڭ قۇرغاق، مايلىق، زىيادە سېزىمچان.... دېگەندەك ھەر خىل تېرە تىپلىرىغا ماس كېلىدىغان خىلمۇ- خىل مەھۇلاتلىرى بار، مال ساتقۇچى سېخىراق بولسا كىچىك قۇتىلىق سىناق بويۇملارنىمۇ بېرىشى مۇمكىن. ئىشقىلىپ چوڭ سودا سارايلاردىكى فىرانسىيەنىڭ داڭلىق بويۇملىرىنىڭ ھېچقايسىسى يامان ئەمەس، لېكىن قىممەت، دەسلەپتە باھاسى سەل تۆۋەنرەك تىزىلىدىغاندىن Avene، shiseido قاتارلىقلارنىمۇ سىناپ باقسىڭىز بولىدۇ. ئۈنۈمىنى كۆرمىسىڭىز قىممىتىنى سىناپ بېقىڭ.

لاكلار ئادەتتە سۇدىن ساقلىنىدىغان ۋە ساقلانمايدىغان دەپ ئايرىلىدۇ، يۈزنى يۇيغاندا بۇزۇلۇپ كەتمەيدىغان لاكتىن 欧莱雅 نىڭ لاكىمۇ ياخشى، باھاسىمۇ مۇۋاپىق.

تېرە ئاسراشتا قايسى ماركىنى ئىشلىتىشتىنمۇ كۆرە ئىزچىللىق بەك مۇھىم، بەزى قىز- ئاياللار بىرلا قىزغىنلىق بىلەن جىقلا نەرسىلەرنى سېتىۋالىدۇ، نەچچە كۈندىن كېيىنلا ئېسىگە كەلسە ئىشلىتىپ، بولمىسا قويۇپ قويىدۇ ياكى تېرە ئاسراشتىكى بەزى باسقۇچلارغا ھورۇنلۇق قىلىدۇ. دېمەكچى، ئىزچىل بولمىسىڭىز ھەر قانداق قىممەت بويۇم ئىشلىتىشىڭىزدىن قەتئىينەزەر، ئەڭ ياخشى ئۈنۈمگە ئېرىشكىلى بولمايدۇ.

گىرىم بۇيۇملىرىغىلا تايىنىۋالماي، تەبىئىي ئۇسۇلدا ئۆزىمىزنى گۈزەللەشتۈرۈشكە كۆپرەك ئەھمىيەت بەرگىنىمىز تۈزۈك. روھىي ھالەت، تۇرمۇش ۋە خىزمەت بېسىملىرى چىراي گۈزەللىكىگە ئەڭ چوڭ تەسىر كۆرسىتىدىغان ئامىللارنىڭ بىرى، سىرتىمىزدىنلا كۈتۈنۈپ ئىچىمىزدىن كۆڭۈل بۆلمىسەك يەنىلا ئوخشاش... مەن ئاساسەن ئۆزۈمنى ئەركىن، خوشال تۇتۇش بىلەن بىرگە تەبىئىي تەركىبلىك نەرسىلەرنى كۆپرەك ئىستىمال قىلىشقا ئەھمىيەت بېرىمەن.  {:92:}
