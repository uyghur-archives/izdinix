مۇكەممەللىشىۋېتىدۇ .شەربەتنى ئىچىپ ،ئاللاھ شىپاسىنى بەرگەنلەردىن
خەۋەر كېلىۋاتىدۇ .لېكىن بىز بۇ شەربەت بىلەن كىسەل چوقۇم
ساقىيىدۇ دىمەيمىز .بۇ پەقەت ساغلاملىق مەھسۇلاتى .پەقەت ئۇزۇن ۋاقىت
ئىزچىل ئىستىمال قىلسىڭىز  ئاللاھ شىپالىق بەرسە ئاندىن ئۈنىمى
بولىدۇ .بۇ شەربەتنى ئىستىمال قىلغاندا بۇرۇنقى ياخشى  بولمىغان
تۇرمۇش ئادەتلىرىنى مەسىلەن:تۇزلۇق،مايلىق،شىلىمشىلىق،ئاچچىق
يىمەكلىكلەردىن پەرھىز تۇتىشى،تاماقنى بەك تويۇپ يەۋەلماسلىقى
ھاراق،تاماكىدىن ئۆزىنى تارتىشى لازىم .دائىم كىسىلىنىڭ شىپا تېپىشى
ئۈچۈن دۇئا قىلىپ تۇرىشى لازىم .بولمىغاندا شەربەتنىڭ ئۈنۈم
بىرىش ۋاقتىنى ئۇزارتىۋېتىدۇ .بۇ نۆۋەت 81شىشە چىقتى.يامان يېرى
ئالما ئاچچىقسۇنىڭ مەنبەسى ئۇزۇلۇپ قالدى .جۇڭگو بازىرىغا ئەكىرگەن
مەھسۇلات 6-ئاينىڭ ئوتتۇرلىرىدا ئالما سىركىسى چىقىشتىن خېلى بۇرۇنلا
تۈگەپ كەتكەن ئىدى .باشقا ئۆلكىنىڭ ۋاكالەتچىلىرى بىلەن ئالاقىلىشپ باقساممۇ
ھەقىقى راست بولۇپ چىقتى .بولۇپمۇ مۇشۇ رېتسىپنىڭ خەنزۇچىسى ئۈندىدار
لاردا بەك تەشۋىق قىلىنىپ كېتىپتۇ .شۇڭا بۇ سىركىنى تالىشىپ سېتىۋالىدىغانلار
بەك كۆپكەن .ئاڭلىسام بۇلار ئۆزىنى ئالداپ ئىشلەپ چىقارغان تۈزۈك سېتىلمايۋاتقان
ئالما سىركىلىرىمۇ بازارلىق مەھسۇلات بولۇپ قاپتىمىش . ھېلىمۇ ئاخىرقى نۆۋەت
ئالما ئاچچىقسۇنىڭ32سومغا ئۆسكەنلىكىگە قارىماي 50دانە ئېلىۋاپ
تىكەنمىز .يەنە بىز ئىشلەتكەن گوللاندىيەنىڭ لىمون مىۋىسىمۇ ئۆسۈپ
كەتتى ،ئاڭلىشىمىزچە لىمۇن كۈزگىچە ئۆسىمىش .يوقارقى سەۋەپتىن بىز
ئامالسىز بىر شىشە500مىللىلېتىرنىڭ باھاسىنى 70سومغا ئۆستۈرشنى
قارار قىلدۇق .بىزنىڭ بۇ نۆۋەتتىكى مەھسۇلاتىمىزنىڭ بىر شىشىسىنىڭ
تەننەرخى تەخمىنەن 50سوم ئۆپچۆرىسىدە چىقتى .ئەگەر ئۆزىڭىز ئىشلىسىڭىز
بەلكىم 60سوم ئەتراپىدا چىقىشى مۇمكىن .چۈنكى بىز لىموننىىڭ ياخشىسىنى
توپ ئالىمىز .زەنجىۋىلنىڭ دورىغا چىلىمىغىنىنى قىممەترەك بولسىمۇ ئالىمىز .
سىقىپ چىقىرىش ماشىنىمىز خېلى ئۈنۈملۈك .ئەگەر سىز قولدا ئىپتىدائى ئوسولدا
سىقسىڭىز چالا سىقىپ قېلىشىڭىز بۇنىڭ بىلەن تەننەرخىنى ئۆرلىتىۋېتىشىڭىز مۇمكىن .

شىپاخانىمىزنى تېخىمۇ تەشۋىق قىلىش ۋە ئەمەلىي كۈچىمىزنىڭ نامايەن قىلىش
مەقسىتىدە مەھسۇلاتىمىزنىڭ باھاسىنى ئەڭ تۆۋەن نۇقتىغا چۈشىرىۋاتىمىز .
ئۆزىمىز قول سېلىپ ئىشلىگەن بولغاچقا ،ئەمگەك ھەققى باھا ئىچىگە
كىرگۈزۈلمىدى .لېكىن سۈپەتكە ئالاھىدە دىققەت قىلدۇق .قاندا شىكەر ئارتىپ
كەتكەن قېرىنداشلارنى كۆزدە تۇتۇپ ،ھەسەل ئۆتكىلىنى چىڭ تۇتتۇق .
شۇڭا قېرىنداشلارنىڭ توغرا چۈشىنىشىنى ئۈمۈد قىلىمىز .دىمەك
6-ئاينىڭ ئوتتۇرلىرىغىچە يۈزنەچچە شىشىلا ئىشلىيەلىگىدەكمىز .چۈنكى
ئالما ئاچچىسۇنىڭ جۇڭگو ۋاكالەتچىسىنىڭ ئىسكىلاتىدا تۈگەپ كېتىپتۇ .
شۇڭا مۇشۇ شەربەت لازىم قېرىنداشلارغا بۇ ئۇچۇرنى ھەمبەھىرلەپ قويغاي
سىزلەر .بولۇپمۇ جەنوبىي شىنجاڭدىكى قېرىنداشلار
نىڭ توپلىرىغا ئەۋەتىپ قويغايسىزلەر.چۈنكى ئالما ئاچچىقسۇ جەنوبىي
شىنجاڭنىڭ كۆپ قىسىم ناھىيىلىرىگە بارالماپتۇ .بۇ شەربەت
قان تومۇر توسۇلۇش،قاندا ماي ئېشىپ كېتىش،قان تومۇر
توسۇلۇش خارەكتىرلىك كۆز قۇۋۋىتى ئاجىزلاش قاتارلىق قان تومۇر خارەكتىر
