بەك ئېچىنىپ كەتكىدەك ئىش يوق ، مۇنبەرنى غەيرى ئۇيغۇر يېزىقىدىكى مۇنبەرلەرگە سېلىشتۇرغاندا تېخىمۇ شۇنداق ، تالاش - تارتىشنى بەك سەزگۈ مەسلە دەپ قاراشمۇ ئاقىلانلىق ئەمەس ، مۇنبەرگە بىرەرسىنىڭ ئۆچمەنلىك قىلىشى ناتايىن ، ئۇنداق نىيەت ئېھتىمال ئەزا بولمىغان ، ئىزدىنىش مۇنبىرىگە قىزىقمايدىغان ، بۇ يەردىن بىرەر چاتاق ئىش چىقىپ قالسا ياخشى بولاتتى ، ئۇلار ئەركىن تالاش - تارتىش قىلسا ياكى پىكىر ئالماشتۇرۇپ ئىدىيىسى ئېچىلىپ كەتسە ياخشى بولماسمىكىن دىگەندەك ئوي - خىياللاردا بولىدىغان كىشىلەر مۇنبەرگە ئۆچمەنلىك بىلەن قارىشى مۈمكىن ، ياشاۋاتقان رىئاللىق ، كىشىلەرنىڭ پىكىر قىلىش ئۇسۇلى مۇنبەرگە تەسىر كۆرسىتىدۇ ، باشقۇرۇش ئۇسۇلىمۇ شۇ ، مۇھىمكى سەمىمىلىك ،ھازىررقى رىئالىقتا بۇنىڭدىنمۇ مۇھىم يەنە بىر نەرسە تېپىلماسلىقى مۈمكىن ، باشقىلارغا قارشى پىكىرنى ئوتتۇرغا قويۇپ ، نارازىلىقىنى ئوتتۇرغا قويۇش نۇرمال ئىش ، بىراق مۇنبەرنىڭ ساغلاملىقىنى دەپ ئېغىز تىلىدا تاپقىلى بولمايدىغان ئەسەبىي تىل - ھاقارەت بىلەن ئوخشاش بولمىغان كۆز قاراشتىكى كىشىلەرگە زەربە بېرىش ، قالپاق كەيدۈرۈش سەمىملىك بولمايدۇ ، باشقىلارنى تىللىدىڭ دەپ ئۇنىڭدىن بىر قانچە ھەسسە ئاشۈرۈپ تىللاپ سۆز قىلغىنىغا تويغۇزۋېتىش ياسالمىلىق ، تېگى - تەكتىدىن ئېيىتقاندا مۇھتاجالىق ، نامىراتلىق ، ئەتىراپتىكى ساختىلىق بىلەن تولغان ، دۇنيادا ناھايتى ئاز ئۇچىرايدىغان رىئاللىق بىلەن مۇناسىۋەتلىك . قانۇن ، سىياسىيغا مۇناسىۋەتلىك تىما، ئىنكاسلارغا بولغان كەسكلىنلىك ئېتىراپ قىلىشقا تىگىشلىك ، بۇنىڭ بىلەن مۇناسىۋەتسىز يازما ئىنكاسلارغا بولغان ئاتالمىش كەسكىنلىك ،بىز ياشاۋاتقان رىئاللقىنىڭ كىشىلەرگە كۆرسەتكەن تەسىرى ، يەنىلا تەكىتلنىدىغان گەپ ،يەنە شۇ سەمىملىك ، بىز رىئاللىقتا ساختىپەزلەرنىڭ دەستىدىن ، ساختىپەزلىككە كۆنگەن شۇڭا مۇنبەردىمۇ شۇنداق بولسا بەك كۆڭلىمىزگە ئېلىپ كەتمەيمىز، دىگەن قاراشلار ئوخشاشلار ساختىپەزلىك . بىرەر مىسال قىستۇرۇپ ئۆتسەك، شىمالدىن كەلگەن چوڭ - كىچىك ، قېرى - ياش دىگەنلەر جەنۇبىي شىنجاڭنى ئايلىنىپ چىققاندىن كىيىن بۇ تەرەپلەردىن كۆڭلى يېرىم بولىدىغان ئىش ،يېزا - بازاردىكى ئادەتتىكى باشلىقلار ئۆزىدىن بولغان ،ئۆزىگە نۇرغۇن بايلىق ، ئىناۋەت تېپىپ بەرگەن كىشىلەرنى يوق ئىش ئۈچۈن ناھايتى ئەسەببىي سۆزلەر بىلەن تىللايدىكەن ، بوزەك قىلىدىكەن ، بىچچارە دېھقان ، بۇخرالار بۇنىڭغا كۈنۈك ئىكەن ، ھەتتا شۇلارغا يېقىچىلىق قىلىدىكەن ،لىكىن ئا نۇمسىسىز باشلىقلار ئازىراق نەپكە ئېرىشىش ئۈچۈن بىر ئۆمۈر باشقىلارنىڭ ئارقىسىدىن سۆكۇلداپ ، يالاقچىلىق بىلەن ياشايدىكەن، دىگەن تەسىراتقا كىلىدىكەن ،ۇ ئايرىم ئەھۋال ئەمەس ، مۇشۇ ئىشقا قىزىقىدىغان نۇرغۇن كىشىلەرنىڭ ئورتاق كۆز قارشى ئىكەن ، باشقا مىللەت كىشىلىرىنىڭ تونۇشىمۇ مۇشىنىڭغا ئوخشاپ قالىدىكەن . مۇنبەر مۇشۇ جەھەتتىن غەيرى ئۇيغۇر يېزىقىدىكى مۇنبەرلەر بىلەن پەرقلىنىپ تۇرىشى مۈمكىن ، بىراق ئەزالارنىڭ مەخسەت ، نىيەتلىرىگە گەپ كەتمەيدۇ ، بىرەر - ئىككى تورداشنىڭ كەپسىىزلىكنى ‹‹ مېنىڭ ئىقتىدارىم بار ›› دەپ رىئالىقتا بەك ئۇچىراپ كەتمەيدىغان تېتىقىسىز ، تۆۋەن سەۋىيەدىكى ‹‹تىللار ›› بىلەن سۆپەتلەپ يۈرۈشنىڭ ھاجىتى يوق !

بۇنىڭ يەنە باشقا ئىسىملىرىمۇ بار، ئىزدەشكە ئىرىنىۋاتىمەن، تولا ئۆزەمنىڭ راس دىمىسۇن، بولمىسا چۈشتىن كېيىن ئېرىنمەي ئولتۇرۇپ تېمىلىرىمغا ھەر خىل ئىسىملاردا كىرىپ يازغان ئىنكاسلىرىنى توپلاپ چىقىپ نەشىر قىلدۇرۋىتىمەن.



ئەدەبىياتتا ئۆزەمگە تەۋە تۇتقان يولۇم بولغانلىقى ئۈچۈن،  بەزىلەرنىڭ ماڭا مۇشۇنداق قەستەن يۇقىنىۋاتقانلىقىنى بىلىمەن، ئەسلى مۇدىئاسىنى پەرەز قىلمىساممۇ ماڭا بەش قولدەك ئايان.

ئىزدىنىش مۇنبىرىدىكى ئەزالار ۋىژدانى بىلەن سۆز قېلىپ باقسۇن، مەن كىمنى ئاغزى بىزاپلىق بىلەن ئىنكاس يېزىپ ھاقارەتلەپ باقتىم؟



مەزكۆرنىڭ ھاقارەتلەش، توھمەت قېلىش، سېسىتىشلىرىغا ھېلىمۇ شۇنچە ئۈزۈن سۈكۈت قىلدىم، كارىم بولمىدى، يول قويدۇم، تاكى تۈنۈگۈنگىچە شۇنداق قىلدىم.

سەۋىرنىڭمۇ چېكى بار، مەيلىگە قويغانسىرى ھەددىدىن ئاشتى، ئاخىرى مەنمۇ يارىشىقىنى قېلىپ قويدۇم.



يەنە دىسەم يەنە شۇ،

ھېچقايسىدا مېنىڭ چىش يېرىپ بىر ئېغىز گەپ قىلغان ياكى ئۆزەمنى ئاقلاشقا ئۇرۇنغان يېرىم يوق.

ھېكايەمدە ئادەم ئۆلتۈرسەم، مەن رىئاللىقتىمۇ ئادەم ئۆلتۈرگەن بولامدىم؟

يازغىنىمدا ئىنساندىكى يۈشۈرۈن تۇيغۇلارنى تەسۋىرلىسەم، مەن ئەمىلىيەتتىمۇ بۇزۇق بولۇپ قالامدىم.

مۇنبەرنى قالايمىقان قىلىش مەقسىتىم يوق. يۇقارقى سەۋەپلىك بىرنەچچە ئىسىمدا تىزىملاتقىنىم راست.


ئا بىلەن ب تەگىشىپ قالسا ... ، ئىككىسىنىڭ تەگىشىشى چىڭغا چىقىپ ھەر ئىككىسى بەدىۋىي گەپنى قىلغاندا بىرىنى تۈرمىگە تاشلاپ يەنە بىرىگە چارە كۆرمىسە ( باشقىلارنى كۆزدە تۇتۇۋاتىمەن) ، باشقىلارنى تىللاپ- تىللىۋېلىپ يەنە باشقىلارنى تۈرمىگە تاشلاپ ھوزۇرلانسا ... بۇنداق تەڭسىز مۇئامىلىنى كۆرۈپ تۇرۇپ ئىپادە بىلدسۈرمەي يۈرسەك بولمايدىكەن بۇرادەر.


ئىزدىنىش مۇنبىرىنىڭ ئۇلىنى كولىغانلار دېسە شۇنداقلارنى دېسە بولىدۇ.


ئىزدىنىشتا ئىككى- تۆت ئېغىز دادىل گەپ قىلىشالايمىز دەپ كىرىمىز.  ئۇنىڭ ئۇلىنى كولايدىغان غەرەز ئەسلا يوق.



سىزگە يەنە ئەسكەرتىپ قۇياي ، بۇ يەردىكى مەقسەت سىزنى سازايى قىلىش ئەمەس. ھەم  بۇ تېمىنى كۆرۈۋاتقانلار ئەقلى ئىقتىدارى مېنىڭچە ئۇنچە توۋەنمۇ ئەمەس.  ئىنكاستا مەن يۇقارقى سەۋەپلىك بىرنەچچە ئىسىمدا تىزىملاتقىنىم راست.  ئەمدى قۇپۇپ بىر ئادىرىسقا نۇرغۇن كومپىيۇتېر باغلانغان دىگەندەك ئاقماس بانىلارنى كۆرسىتىپ نىمە قىلماقچى بولغىنىڭىزنى بىلگىلى بولمىدى.  بۇ تېما يالغۇز سىز گە قارىتىلمىغان . جەۋلاننىڭ ئاخىرقى گېپى سىزدەك ئادەملەرگە قارىتىلغان ،مېنىڭ شۇنداق ئاغزىمنى بۇزغۇم بار ئەمما ئاممىۋى سۇرۇندا ئامال بولمىدى. مەن يەنە تەكرارلاپ قۇياي:


ھەرقانداق ئويۇننىڭ قائىدىسى بولىدۇ ، قائىدىگە بويسۇنمىسىڭىز چەكلىنىسىز ، شاللىنىسىز ، ئۇيۇن ئويناشنى خالىمىغان بولسىڭىز ئۇ باشقا گەپ ، ئۆزىڭىز ئۇيۇن ئويناشنى خالىمىسىڭىز باشقىلارغا تەسىر قىلماي چېكىنەلىسىڭىز بۇ باشقىلارغا قىلىنغان ھۆرمەت ، ھەتتا ئوچۇق ئاشكارە تىللاپ چىقىپ كەتسىڭىزمۇ سىزنى دۈشمەن قاتارىدا كۆرمەيمىز ، پەقەت ۋە پەقەت تەلەپ قىلدىغىنىمىز شۇكى بىزگە يۇقارقىدەك سامان تەكتىدىن سۇ يۈگۈرتىدىغان خۇمسىلىقلارنى قىلماي تىنىچ چىقىپ كېتىڭ !
