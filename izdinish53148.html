ئۈرۈمچى شەھرىدە مۇسۇلمانچە يېمەكلىك كوچىسىدىن يەتتىسى بەرپا قىلىنماقچى


ئاسىيا كىندىكى تورى 16 – مارت خەۋىرى: 3 – ئاينىڭ 15 – كۈنى، مۇخبىرنىڭ
ئۈرۈمچى شەھەرلىك مۇسۇلمانچە يېمەكلىك جەمئىيىتىنىڭ 2 – نۆۋەتلىك باشلىقى
ۋە مۇئاۋىن باشلىقلىرى يۇپۇق ئېچىش مۇراسىمىدىن ئىگىلىشىچە، بۇ يىل ئۈرۈمچى
شەھىرى مۇسۇلمانچە يېمەكلىكلەر كوچىسىدىن يەتتىنى بەرپا قىلماقچىكەن،
مۇسۇلمانچە يېمەكلىكلەرنى مەركەزلىك تەرەققىي قىلدۇرۇش ئارقىلىق، ماركا ۋە
كۆلەم ھاسىل قىلىپ، ئۈرۈمچىدىكى مۇسۇلمانچە يېمەكلىكلەرنىڭ تەسىر كۈچىنى
يەنىمۇ كېڭەيتمەكچى ئىكەن.


ھازىر ئۈرۈمچى شەھەرلىك مۇسۇلمانچە يېمەكلىكلەر جەمئىيىتى سايباغ
رايونىدىكى خوتەن كوچىسى، يېڭى شەھەر رايوندىكى ۋېنجۇ يولى، بۇلاق تاغ رايونىدىكى يىكەنكۆل جەنۇبىي، شىمالىي يولى، ئۈرۈمچى ئىقتىسادىي
تېخنىكا  رايونىدىكى چېگرا ئېغىزى يولى، تەڭرىتاغ رايونىدىكى
تېنچلىق جەنۇبىي يولى، داۋانچىڭ رايونىدىكى سەيۋوپو مۇسۇلمانچە تائاملار
كوچىسى ۋە ئۈرۈمچى ناھىيەسىدىكى فاڭجيا ئېكىنزارلىقى مەزىلىك تائاملار
كوچىسى قاتارلىق يەتتە كوچا مۇسۇلمانچە تائاملار كوچىسىدىكى ئۈلگىلىك
كوچىلار قىلىنىدۇ.

«تاللىنىپ چىققان بۇ يەتتە كوچىغا قارىتا بىز مۇسۇلمانچە يېمەكلىكلەرنى
باشقۇرۇش بويىچە گۇۋاھلىقتىن ئۆتكۈزۈش خىزمىتىنى كۈچەيتىمىز. بۇندىن باشقا،
ئۈرۈمچى شەھەرلىك مۇسۇلمانچە يېمەكلىكلەر جەمئىيىتىمۇ بۇ يەتتە كوچىدىكى
يېمەك ئىچمەك كارخانىلىرىنىڭ مۇلازىمەت سەۋىيەسىنى ئۆستۈرۈپ، ئۆزىگە خاس
ئۇسلۇب بەرپا قىلىشىنى پائال قوللايدۇ، شەرتكە توشۇدىغان تىجارەتچىلەرگە
يەنە يۆلەش مەبلىغى ئىلتىماس قىلىپ بېرىپ، كارخانىنى ئۈزلۈكسىز كۈچەيتىپ،
مۇسۇلمانچە يېمەكلىك كوچىسى تاۋار ماركىسى شەكىللەندۈرۈپ، تەسىر كۈچىنى
ئاشۇرىدۇ» ئۈرۈمچى شەھەرلىك مۇسۇلمانچە يېمەكلىك جەمئىيىتىنىڭ باش كاتىبى
جاڭتاۋ مۇخبىرغا شۇنداق دىدى.

مۇخبىرنىڭ ئىگىلىشىچە، بۇ يىل ئۈرۈمچى شەھەرلىك مۇسۇلمانچە يېمەكلىك
جەمئىيىتى يەنە ھۆكۈمەت تارماقلىرى بىلەن بىرگە ئۈرۈمچى شەھىرىنىڭ
مۇسۇلمانچە يېمەكلىكلىرىنى گۇۋاھلىقتىن ئۆتكۈزۈش ئۆلچىمىنى تېزرەك تۈزۈپ
چىقىپ، ئۈرۈمچىدىكى مۇسۇلمانچە يېمەكلىك كارخانىلىرىنىڭ تېزراق چەتئەللەرگە
چىقىشى ئۈچۈن شارائىت ھازىرلاپتۇ.

بۇ يىل 3 – ئاينىڭ 1 – كۈنى، گەنسۇ، نىڭشيا، چىڭخەي، شەنشى، يۈننەن
قاتارلىق بەش ئۆلكە، رايون خەلقئارادىكى مۇسۇلمانچە يېمەكلىكلەرنى
گۇۋاھلىقتىن ئۆتكۈزۈش ئۆلچىمىدىن پايدىلانغان ئاساستا، «مۇسۇلمانچە
يېمەكلىكلەرنى گۇۋاھلىقتىن ئۆتكۈزۈش قائىدىسى» نى ئورتاق تۈزۈپ چىققان.
ھازىر ئېلىمىزدە پۈتۈن مەملىكەت بويىچە بىرلىككە كەلگەن، مۇسۇلمانچە
يېمەكلىك ئۆلچىمى يوق. 2009 – يىل، نىڭشيا «نىڭشيا خۇيزۇ ئاپتونوم
رايونىنىڭ مۇسۇلمانچە يېمەكلىكلەرنى گۇۋاھلىقتىن ئۆتكۈزۈش قائىدىسى» نى
ئېلان قىلىپ يولغا قويدى، بۇ «قائىدە» غەربتىكى بەش ئۆلكە ئىتتىپاقلىشىشتىن
بۇرۇنقى بىردىن بىر مۇسۇلمانچە يېمەكلىكلەرنى گۇۋاھلىقتىن ئۆتكۈزۈش
ئۆلچىمى بولۇپ ھېسابلىنىدۇ.
