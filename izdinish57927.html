بۇ ئەسلىدە يەمەن تىلىدا قەھۋە دېگەن نام بىلەن مەشھۇر بولغان بىر خىل قۇيۇق ھاراقنىڭ ئاتىلىشى ئىدى. كېيىن ئۇ بىر خىل ئالاھىدە مېۋىنى كۆرسىتىدىغان بولغان ، ھاراق ئىچىش ھارام بولغانلىقى ئۈچۈن ئۇنى ئىچىشنى تاشلىغان.




ماھىيىتى: ئۇ يەمەن تاغلىرى ۋە ئۇنىڭ ئەتراپىدىكى تاغلىق رايونلار ۋە ھەبەشىستاندا ئۆسىدىغان بىر خىل دەرەخنىڭ مېۋىسىدىن ئىبارەت. ئۇ زىربادات ئاراللىرىنىڭ بىرى بولغان تەباۋىيە ئارىلىدىنمۇ چىقىدۇ. ئۇنىڭ ئۇرۇقى رىم ئايلىرىدىن بولغان ئازەر ئېيىدا تېرىلىدۇ. ئاب ئېيىدا يې تىشىدۇ، پەسىللىك ۋە پەسىلسىز بولىدۇ. يەنى چۆللەردە ئۆزى ئۆسىدۇ ھەم تېرىلىدۇ. يەنى ئۇنىڭ ياۋىسى ۋە تېرىلىدىغىنى بولىدۇ. ئۇ پەرۋىش قىلىش ئارقىلىق ياخشى ئۆسىدۇ، ئۇنىڭ بەك ئىنچىكە شاخلىرى كېسىۋېتىلىدۇ. يارىمايدىغان ئاجىز گۈللىرى چۈشۈرىۋېتىلىدۇ. ياخشىلىرى سۇغۇرۇش ۋە باشقا تېگىشلىك پەرۋىشلەر ئارقىلىق ساقلاپ قېلىنىدۇ. ئۇنىڭ ياخشىلىرى يەمەنننىڭ بەيتۇل ڧەقيھ دېگەن يىرىدىن چىققانلىرىدۇر.





]ئۇ رەڭگى يېشىل، چوڭ كىچىكلىرى ئوتتۇرىھال، خۇشپۇراق، خۇشتەم بولغان ئۇرۇقىدىن ئىبارەت. ئۇنىڭ دەرىخىنىڭ غولى بار بولۇپ، باش بارماق چوڭلۇقىدا، ئىككى غۇلاچتىن ئۈچ غۇلاچقىچە ئېگىزلىكتە بولىدۇ. جىغان ئۈسۈملۈكىگە ئوخشايدۇ




گۈلى ئاق، مېۋىسى قارا رەڭلىك غىلاپ ئىچىدە بولۇپ، چوڭلىرى ھەببۇلغار چوڭلۇقىدا، كىچىكلىرى پىندۇق چوڭلۇقىدا بولىدۇ. پوستى ئىككى مېۋىنىڭ پوستىدىن چۈرۈكرەك بولىدۇ. ئىچىدىن ئىككى تال مېغىز چىقىدۇ. ھەر بىر مېغىزنىڭ ئوتتۇرىسىدا بۇغداي دېنى ۋە خورما ئۇرۇقىنىڭ يېرىقىغا ئوخشاش يېرىق بولىدۇ، ئەمما ئۇنىڭ دېنى بۇغداي دېنىدىن چوڭراق، خورما ئۇرۇقىدىن كىچىكرەك يېشىل بولىدۇ. ئىچىدە پۈرلەشكەن نىپىز پەردىسى بولۇپ، ئازراق ئاچچىق بولىدۇ.


ئۇنىڭ ياخشىلىرى يەمەندىن چىققىنى،يەنى يېشىل ،ئېغىر بولۇپ، سۇغا سالسا تېگىگە چۆكىدىغانلىرىدۇر. ئۇنىڭ ھەبەشىستاندىن چىقىدىغان قارىلىرى ناھايىتى ناچار بولۇپ، ئاچچىق،بەتتەم بولىدۇ. ئۇنىڭ تەباۋىيە ئارىلىدىن چىققىنى يەمەن ۋە ھەبەشىستاندىن چىققىنىدىن ئاقراق ۋە چوڭراق بولىدۇ، ئۇنىڭ ھۆلىنىڭ مەلۇم مىقداردا چاپلاشقاق سۇيۇقلۇقى بار بولۇپ، تاتلىق، قىرتاق بولىدۇ. قۇرۇغاندا تاتلىقلىقى ئازلايدۇ ۋە بارا بارا يوقىلىدۇ.



ئېيتىشلارغا قارىغاندا ئۇنىڭ بايقىلىشى مۇنداق بولغان ئىكەن: شەيخ ئەبۇلھەسەن شازۇلى رەھىمەھۇللاھنىڭ يەمەن تاغلىق رايونىدا ئىبادەتخانىسى بار ئىكەن . ئۇنىڭ مۇرتلىرى كېچىسى ئۇخلىماي ئىبادەت قىلىپ، كۈندۈزى رىيازەت چېكىپ ھارىدىكەن ۋە چارچايدىكەن. ئويلىمىغان يەردىن ئۇلارنىڭ بەزىسى مەزكۇر دەرەخنىڭ مېۋىسىنى ئۇچرىتىپ مەلۇم مىقداردا يەپ بېقىپتۇ. ئۇزاق ئۆتمەي ئۇلارنىڭ ھارغىنلىقى كۆتۈرۈلۈپ كېتىپتۇ. ئۇلار بۇ ئەھۋالنى باشقىلارغىمۇ سۆزلەپ بېرىپتۇ. باشقىلار يېگەن ئىكەن ئۇلارغا پايدا قىلىپتۇ. ئۇلار بۇ ئەھۋالنى پىرىغا مەلۇم قىلغان ئىكەن، پىرى ئۇنى قاينىتىپ سۈيىنى چىقىرىپ ئىچىشكە بۇيرۇپتۇ. ئۇلار مەزكۇر مېۋىنى يىغىپ ئېلىپ كىلىپ، قاينىتىپ تەمىنى چىقىرىپ ئىچكەن ئىكەن، ھەممەيلەننىڭ ھارغىنلىقى چىقىپ تېتىكلىشىپتۇ، شۇنداق قىلىپ بۇ خەۋەر يېقىن يېراقلارغا ئاستا ئاستا تارقىلىپ بۇ مېۋە تۇنۇلۇپتۇ، ھەر قايسى ئەللەردىكى سودىگەرلەر ئۇنى باشقا يەرلەرگە ئېلىپ بېرىپ سېتىشىپتۇ.

كۆپ قىسىم كىشىلەر ئۇنى ئىستىمال قىلىدۇ. ئەمما ھەر جاي ۋە شەھەرلەردىكى خەلقلەرنىڭ ئۇنى ئىستىمال قىلىشى ئوخشىمايدۇ. يەمەن ۋە ئۇنىڭ ئەتراپىدىكى رايونلارنىڭ خەلقلىرى ئۇنىڭ بولۇپمۇ ھۆلىنىڭ سىرتقى پوستىنى كۆپ سۇ بىلەن قاينىتىپ تەمىنى چىقىرىپ ،پىسەپلىرىنى سۈزۈۋېتىپ ، تائامدىن ئىلگىرى، كۆپ قىسىم كىشىلەر تائامدىن كېيىن ئىلمان ھالەتتە چوڭ پىيالىلەر بىلەن  ئىچىدۇ، ئۇلار ئۇنى ناھايىتى ياقتۇرۇپ ئىچىشىدۇ، ئۇنىڭ بولۇپمۇ ئۇنىڭ ھۆلىنىڭ تەمى تاتلىق ۋە قىرتاققا  مايىل بولىدۇ. مەككە مۇكەررەمە ۋە مەدىنە مۇنەۋۋەر(ئاللا بۇ ئىككى شەھەرنى تېخىمۇ شەرەپلىك قىلسۇن) خەلقى ۋە ئىران، تۇران، ھىندىستان، پىرانسىيە ۋە باشقا ئەرەپ ،ئەجەم شەھەر خەلقى ئۇنىڭ مېغىزىنى بەزى خام، بەزى ياڭاق پوستى رەڭگىدە چالا قورۇپ، بەزىدە كۆيگىلى ئاز قالغۇدەك دەرىجىدە بەكرەك قورۇپ، بەزىدە چالا، بەزىدە يۇمشاق سوقۇپ، سۇ بىلەن قاينىتىپ، تەمىنى بەزىدە قۇيۇق، بەزىدە سۇيۇق، بەزىدە ئوتتۇراھال، چىقىرىپ، پىسەپلىرىنى سۈزۈۋېتىپ ياكى سۈزمەي ئىچىدۇ. پىرانسۇزلار ئۇنىڭغا ئاز  مىقداردا سۈت ۋە ناۋات قوشۇپ ئىچىدۇ. بۇ ئەھۋاللارنىڭ ھەممىسى قارابادىندا تەپسىلى سۆزلەندى.

تەبىئىتى: ئۇنىڭ تەبىئىتى توغرىسىدا كۆپ ئىختىلاپلار كۆرۈلگەن،

ھەكىم مىر ئىمامۇددىن مەھمۇد ئۆزىنىڭ <<ئەپيۇنىييە>>ناملىق رىسالىسىنىڭ ئاخىرىدىكى قەھۋە ناملىق ماۋزۇسىدا : ئۇ ئىككىنچى دەرىجىدە قۇرۇق سۇغۇقتۇر، چۈنكى ئۇنىڭ تەمى، پۇرىقى ۋە رەڭگىدە ئۇنىڭ ھارارىتىنىڭ بارلىقىنى دەلىللەيدىغان بىرەر غالىپ كەيپىيات يوق. تەجرىبە ئارقىلىق ئۇنىڭ سوغۇقلۇقىنىڭ بەدەنگە زىيان يەتكۈزگۈدەك دەرىجىدە ئەمەس ،بەلكى ئوتتۇراھال ئىكەنلىكى بايقالدى . ئۇنىڭ قۇرۇقلۇقى بىلەن سۇغۇقلۇقىنىڭ قايسىسىنىڭ غالىپ ئىكەنلىكنى بىلىش پەقەت تەپەككۇرغا باغلىق دەپ يازغان.
مىر قازى ئۆزىنىڭ رىسالىسىدە : پىقىر ئۇنىڭ سۇغۇقلىقىنىڭ بىرىنچى دەرىجىدە، قۇرۇقلۇقىنىڭ ئىككىنچى دەرىجىدە ئىكەنلىكىنى بايقىدى دەپ يازغان.

ھەكىم سالىكۇددىن يەزىدىي: دورىلارنىڭ خۇسۇسىيەتلىرىدىن خەۋەردار بەزى ئەربابلار ئۇنىڭدىن كۆپ ئىستىمال قىلسا ، مېڭىگە قۇرۇقلۇق يەتكۈزىدىغانلىقى، ئۇيقۇنى قاچۇرىدىغانلىقى ۋە مىزاجىغا قۇرۇقلۇق يەتكۈزىدىغانلىقىغا ئاساسەن  ئۇنىڭ سۇغۇقلۇقىنىڭ ئىككىنچى دەرىجىنىڭ بېشىدا ئىكەنلىكى، قۇرۇقلۇقىنىڭ ئۈچىنچى دەرىجىدە ئىكەنلىكىگە ھۈكۈم قىلدى دەپ يازغان.

شەيخ داۋۇت ئەنتاكى: ئۇ بىرىنچى دەرىجىدە ئىسسىق، ئىككىنچى دەرىجىدە قۇرۇقتۇر. ئۇ قۇرۇق سۇغۇق دېگەن  كەڭ قاراش توغرا ئەمەستۇر، چۈنكى  ئۇنىڭ پوستى ئاچچىقتۇر، ئاچچىق بولغان ھەر قانداق نەرسە ئىسسىق بولىدۇ. ئېھتىمال ئۇنىڭ پوستى ئىسسىق، مېغىزى مۆتىدىل ياكى بىرىنچى دەرىجىدە سۇغۇق بولۇشى مۇمكىن.  ئۇنىڭ قىرتاقلىقى ئۇنىڭ سۇغۇقلۇقىنى دەلىللەيدۇ. تەجرىبىلەرگە قارىغاندا ، ئۇ ھۆللۈكلەرنى قۇرۇتىدىكەن، بەلغەمدىن بولغان يۆتەل نەزلىلەرگە پايدا قىلىدىكەن، توسالغۇلارنى ئاچىدىكەن، سۈيدۈك ھەيدەيدىكەن، بۇلارنىڭ ھەممىسى ئۇنىڭ قۇرۇق ئىسسىق ئىكەنلىكىنى ئىپادىلەيدۇ دېگەن.

ھەكىم مىرمۇھەممەد مۇئمىن تۆھپەتۇل مۆئمىن دېگەن كىتاۋىدا : قىياسىمچە ئۇنىڭ ئىسسىقلىقى مۆتىدىل، قۇرۇقلۇقى غالىبتەك مەلۇم بولىدۇ. پوستى ئۇنىڭ ئۆزىدىن قۇرۇق ئىسسىقراقتتۇر دەپ يازغان.
مۇئەللىپ (كىتاپ ئاپتورى) ئېھتىمال ئۇ، بولۇپمۇ يېڭىسى مۇرەككەپ قۇۋۋەتلەرگە ئىگە بولۇشى مۇمكىن،  چۈنكى ئۇنىڭدا ئىككى تەركىب بار، بىرى ھاۋاغا مەنسۇب ھۆل ئىسسىق ئۇچۇچان تەركىب، يەنە بىرى تۇپراققا مەنسۇب قۇرۇق سۇغۇق تەركىبتۇر.  ئۇنىڭ بىرىنچى تەركىبىدىن ھارارەت، ئۇسسۇزلۇق، قاننىڭ قاينىشى، سەپرانىڭ ئۆتكۈرلۈكى ۋە چىقىشىنى پەسەيتىش، ئۇيقۇ قاچۇرۇش، مېڭىگە قۇرۇقلۇق يەتكۈزۈش ۋە باشقىلارغا ئوخشاش قۇرۇقلۇقنىڭ ئالامەتلىرى پەيدا بولىدۇ. ئۇ قانچە يېڭى بولسا ئۇنىڭ بىرىنچى تەركىبى شۇنچە ئارتۇق بولىدۇ. كونىرىغانسىرى تەدرىجى ئازلايدۇ. بولۇپمۇ ئۇنىڭ پوستىنىڭ بىرىنچى تەركىبىنىڭ ئەھۋالى تېخىمۇ شۇنداق. ئۇ قورۇلسا قورۇلۇشىنىڭ  ئاز كۆپلۈكىگە قاراپ بىرىنچى تەركىبىنىڭ يوقىلىشىمۇ ئاز كۆپ بولىدۇ.
ئۇمۇمەن ئۇنىڭ يېڭىسى بولۇپمۇ خام پوستىنىڭ تەبىئىتى  قۇرۇق  ئىسسىققا مايىل، كونىسى بۇلۇپمۇ  قورۇلغۇنىنىڭ تەبىئىتى قۇرۇق سۇغۇق، ئۇ كونىرىغانسىرى قۇرۇق سۇغۇقلۇقى شۇنچە ئاشىدۇ.
