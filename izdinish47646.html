نەشپۈت يۇمىلاق قاسراقلىق قۇرتىنىڭ ئۇنىۋىرسال ئالدىنى ئېلىش يۇقىتىش تەدبىرى

نەشپۈت يۇمىلاق قاسراقلىق قۇرتىنىڭ ئۇنىۋىرسال ئالدىنى ئېلىش يۇقىتىش تەدبىرى



ئىلمىي نامى Diaspidious pernicious


خەنزۇچە نامى 梨圆介


تۈرى : ئوخشاش قاناتلىقلار ئەترىتى قالقانسىمان قاسراقلىق قۇرت ئائىلىسىگە كىرىدۇ .


زىيان سېلىش ئوبىكتى :نەشپۈت ، ئالما ، شاپتۇل ، ئالۇچا ، ئۈرۈك ، چىلان دەرىخى قاتارلىقلارغا زىيان سالىدۇ .


تاشقى تۈزۈلۈش ئالاھىدىلىگى :


چىشى ھاشارىتى يىتىشكەن قاناتسىز ، تىنى ياپىلاق ،يۇمىلاقراقشەكىلدە، سېرىق رەڭلىك ، ئېغىز ئاپپاراتى يىپسىمان ، قۇرساق يۈزىدە ئۆسكەن بۇلىدۇ . قۇرت تىنىنىڭ سىرتقى يۈزىنى كۈلرەڭ يۇمىلاق قاسراق ئوراپ تۇرىدۇ ، قاسرىقى سەل قاتتىق ، دىئامىتىرى تەخمىنەن 1.3مىللىمىتىر ، قاسراق مەركىزى ئازراق كۆتۈرۈلگەن بۇلىدۇ ، قاسراق چوققىسى سېرىق ياكى قۇڭۇر رەڭدە ، سىرتقى يۈزىدە چەمبەرسىمان سىزىقى بار .


ئەركەك يىتىشكەن ھاشارىتىنىڭ بىر جۈپ قانىتى بار ، ئاق رەڭلىك سۈزۈك كىلىدۇ， قانات تۇمۇرى ئاددى . بەدەن ئۇزۇنلۇقى تەخمىنەن 0.6 مىللىمىتىر، باش ۋە كۆكرەك قىسمى ئاپلىسىندەكقىزىل رەڭدە ، قۇرسىقى قىزغۇچ سېرىق رەڭدە ،قۇرساقنىڭ ئاخىرقى ئۇچىدا خەنجەرسىمان چېتىشىش ئەزاسى بار ، دەسلەپ تۇخۇمدىن چىققان كىچىك ياشلىق لېچىنكىلارنىڭ بەدەن ئۇزۇنلۇقى تەخمىنەن 0.2 مىللىمىتىر ، ياپىلاق ئىللىپىس شەكلىگە مايىل ، سۇس سېرىق ، بۇرۇتچىسى ، ئېغىز ئاپپاراتى، پۇتى تۇللۇق تەرەققى قىلغان ، قۇرساق ئاخىرقى قۇيرۇقىدا ئىككى تال ئۇزۇن تۈكى بار .


ئىككى ياشلىق كىچىك قۇرتنىڭ كۆزى ، بۇرۇتچىسى ، پۇتى ۋە قۇيرۇق تۈكى تۇللۇق چىكىنىپ يۇقالغان بۇلۇپ ، موملۇق قاسراق ئاجرىتىپ چىقىرىپ ئۇزىنى مۇقىملاشتۇرغان ،مىدىرلىمايدۇ .


چىشى ھاشارىتىنىڭ قاسرىقى يۇمىلاق ، ئەركەك قۇرتنىڭ قاسرىقى ئۇزۇن ئىللىپىس شەكىللىك ،قاسراق چوققىسى بىر تەرەپكە يالتايغان بۇلىدۇ ، پەقەت ئەركەك قۇرتنىڭلا قوچىقى بۇلىدۇ ، بەدەن ئۇزۇنلىقى تەخمىنەن 0.6 مىللىمىتىر ، سۇس سېرىق رەڭگە ئازراق كافى رەڭگى ئارىلاشقان ھالەتتە كۆرۈنىدۇ .


پەيدا بۇلۇش قانۇنىيىتى :


نەشپۈت يۇمىلاققاسراقلىق قۇرتى رايۇنىمىزدا يىلدا ئىككى ئەۋلادپەيدا بۇلىدۇ ، ئىككى ياشلىق ياش قۇرت ۋە ئازساندىكى يىتىلگەن چىشى ھاشارەت ھالىتىدە شاخ ئۈسىتىدە قىشتىن ئۈتىدۇ ، كىيىنىكى يىلى 4- ئاينىڭ بېشىدا ، دەرەخ سۈيى ئېقىشقا باشلىغاندا ھەرىكەتلىنىپ ئۇزۇقلىنىدۇ ، بەدەن رەڭگى سارغۇچ قارامتۇلدىن يېڭى ئاقۇچ سېرىققائۆزگىرىدۇ ،يىتىشكەن ھاشارىتى ئىككى جىنسلىق كۆپىيىدۇ ، يەنە ئاتىلىقسىزمۇ كۆپىيەلەيدۇ ، تۇخۇم ئانا تىنى ئىچىدە ئاللىبۇرۇن تۆرەلمە يتىلىشنى تاماملىغان ، شۇڭا ئانا تىنىدىن بىۋاستە ياش قۇرتلار تۇغۇلىدۇ .


تەكشۈرۈشىمىزگە ئاساسلانغاندا : 4- ئاينىڭ 18-كۈنى نەشپۈت يۇمىلاق قاسراقلىق قۇرتى تۇخۇم دانچىسى ھاسىل قىلىدۇ ، ئوخشاشلا بۇ ۋاقىتتا ئەركەك ھاشارات پەيدا بولغان ، 4- ئاينىڭ 23- كۈنى تۇخۇم تۇغۇش دەسلەپكى مەزگىلىگە كىرگەن ، تۇخۇم تۇغۇش نىسبىتى 20%گە ، ھەربىر چىشى ھاشارەت ئوتتۇرا ھىساپ بىلەن كۈنىگە 1.5 دانە تۇخۇم تۇغقان ، 5-ئاينىڭ 4-كۈنى تۇخۇم تۇغۇش نىسبىتى 80% كە ، ھەر بىر چىشى ھاشارەت 14.4 دانە تۇخۇم تۇغقان . 5-ئاينىڭ 9-كۈنى ھەر بىر چىشى قۇرت ئوتتۇرا ھىساپ بىلەن 31.3 دانە تۇخۇم تۇغقان .5-ئاينىڭ 15-كۈنى تۇخۇمدىن لېچىنكا چىقىرىش نىسبىتى 20% كە يەتكەن ، 5-ئاينىڭ 22-كۈنى تۇخۇمدىن لېچىنكا چىقىرىش نىسبىتى 50% كە يەتكەن ، ھەمدە بىر قىسىم ياش قۇرتلار شاخ ياكى مىۋە ئۈستىگە تارقىلىپ زىيان سېلىشقا باشلىغان .


دىمەك 1-ئەۋلاد ياش قۇرتنىڭ زىيان سېلىشى 5-ئاينىڭ ئاخىرىدىن 6-ئاينىڭ ئاخىرىغىچە بۇلىدۇ ، 2-ئەۋلاد ياش قۇرتنىڭ زىيان سېلىشى 7-ئاينىڭ ئاخىرىدىن 9-ئاينىڭ ئاخىرىغىچە بۇلىدۇ ، ئاساسلىقى مىۋە يۈزىدە مۇم سۇيۇقلۇقى ئاجرىتىپ چىقىرىپ زىيان سالىدۇ ، 11- ئاينىڭ ئاخىرى2 ياشلىق قۇرتى شاخ ئۈستىگە يۆتكىلىپ قىشتىن ئۆتىدۇ .


ئالدىنى ئېلىش يۇقىتىش ئۇسۇلى


1. دېھقانچىلىق تەبىرى ئارقىلىق ئالدىنى ئېلىپ يۇقىتىش : باغ باشقۇرۇشنى كۈچەيتىپ ، ئورگانىك ئۇغۇتنى ئاشۇرۇپ چېچىش ، مۇۋاپىق سۇغۇرۇپ ، ئۈسۈشنى ئىلگىرى سۈرۈش ، دەرەخ ئۈسۈش ھالىتىنى ياخشىلاش ، دەرەخنىڭ كىسەلگە قارشى تۇرۇش كۈچىنى يۇقىرى كۆتىرىش ؛ مىۋىلىك دەرەخنى چاتاشقا بىرلەشتۈرۈپ ، ياش قۇرتى ياكى يىتىلگەن ھاشارىتى توپلانغان شاخلارنى كىسىپ ، شۇ يەردىلا كۆيدۈرۈپ بىر تەرەپ قىلىش ؛ قىش ، ئەتىياز 5 – 8- ئايلاردا ياش قۇرت ۋە يىتىلگەن ھاشارەتلەرنى قىرىپ يۇقىتىش ، شۇ ئارقىلىق ھاشارەت سانىنى ئازايتىش لازىم .


2.خىمىيىلىك ئۇسۇلدا ئالدىنى ئېلىپ يۇقىتىش :


(1) مىۋىلىك دەرەخ بىخ چىقىرىشتىن بۇرۇن ( 3- ئاينىڭ باشلىرى ) 5 باۋم گىرادۇسلۇق ھاك –گۈڭگۈرت بىرىكمىسى (（石硫合剂 ياكى 45% لىك كىرىستالسىمان ھاك –گۈڭگۈرت بىرىكمىسى(（晶体石硫合剂نىڭ 300ھەسسىلىك سۇيۇقلۇقىنى تەكشى چېچىپ ، بۇ ئارقىلىق قىشتىن ئۈتىدىغان ياش قۇرتلارنىڭ سانىنى ئازايتىش لازىم . قاسراقلىق قۇرتنى قاسراق ئىچىدە بىر تەرەپ قىلالايدىغان بىردىنبىر خىمىيىلىك دۇرا ھاك – گۈڭگۈرت بىرىكمىسى . ھاك –گۈڭگۈرت بىرىكمىسى 80% ، كىرسىن 90% ئۈنۈم بىرىدۇ ، لوپىسىن ( ئامىرىكىنىڭ بىرخىل مەھسۇلاتى )100% ئۈنۈم بىرىدۇ .ئەمما بۇلارنىڭ ھىچقايسىسىنى دەرەخ بىخلىنىشتىن كىيىن قوللىنىشقا بولمايدۇ .


(2) ياش قاسراقلىق قۇرتنىڭ ئالدىنى ئېلىپ يۇقىتىش :


1-ئەۋلاد ياش قۇرتنىڭ ئالدىنى ئېلىش مەزگىلى 5-ئاينىڭ ئاخىرىدىن 6-ئاينىڭ ئاخىرىغىچە ، 2-ئەۋلاد ياش قۇرتنىڭ ئالدىنى ئېلىش مەزگىلى 7-ئاينىڭ ئاخىرىدىن 9-ئاينىڭ ئاخىرىغىچە بۇلىدۇ .


بۇنىڭدا 40%لىك مىداتىئون(杀扑磷 ) 1000-1500ھەسسە سۇيۇقلۇقى ياكى 48 %لىك لورسىبان(乐斯本 )نىڭ 1000ھەسسە سۇيۇقلۇقىياكى جىيى سى جىڭ   (蚧死净)مايلىق ئىمپولسىيسىنىڭ 800 ھەسسىلىك سۇيۇقلۇقىنى پۈركۈش ئارقىلىق ئالدىنى ئېلىپ يۇقىتىشقا بۇلىدۇ .


دىققەت قىلىدىغان ئىشلار :


40%لىك مىداتىئون(杀扑磷 ) نىڭ ئۈنۈم بىرىش دۇرا قالدۇق ۋاقتى بىر قەدەر ئۇزۇن ، ئىشلەتكەن چاغدا چۇقۇم بىخەتەر ئىشلىتىش ئارىلىقىغا دىققەت قىلىش كېرەك .
