ئىزدىنىش مۇنبىرى › تەن ۋە ساغلاملىق › دىئابىت توغرىسىدا پاراڭلىشىپ باقمامدۇق؟

دىئابىت توغرىسىدا پاراڭلىشىپ باقمامدۇق؟

ئەمدى نۆۋەت دىئابىتقا كەلدى


ئىلاۋە: ئىزدىنىش مۇنبىرى ئېچىلغاندىن بېرى كۈندە بىر كىرىپ بىر-بىرىدىن ئېسىل تېما-ئىنكاسلارنى ئوقۇپ كېلىۋاتقان ئىدىم. نۇرغۇن ھەل قىلالمىغان نەرسىلەرنى مۇشۇ ئىزدىنىشلىقلارنىڭ تېما-ئىنكاسلىرى ئارقىلىق ھەل قىلدىم. مۇنازىرە ئەخلاقىنىمۇ مۇشۇ ئىزدىنىشتىن ئۈگەندىم دىسەممۇ ئانچە ئارتۇق كەتمەيدۇ. بۈگۈن مەن بەزىلەر ئۈچۈن ئەرزىمەس، لېكىن بەزىلەر ئۈچۈن قىممىتى بار بولغان «دىئابىت» توغرىسىدا تېما يوللاپ ئولتۇرۇپتىمەن. ھۆرمەتلىك تورداشلارنىڭ ئەيىپكە بۇيرىماي تېمىغا ئاۋاز قاتناشتۇرۇپ بېقىشىنى ئۈمۈت قىلىمەن. بۇ تېما ئارقىلىق دىئابىت كېلسىلگە گىرىپتار بولغان ئاتا-ئانا، ئۇرۇق-تۇققان، قېرنداشلرىمىز ئاز تولا بولسىمۇ نەپ ئېلىپ قالسا ئەجەپ ئەمەس...


****************************


يېقىندا ئايالىم سەمرىپ كەتتىم، ئۇرۇقلىمىسام بولمىدى، ئەمدى چېنىقىپ ئۇرۇقلايمەن دەپ يۈرىۋاتقان. بىر شەنبە كۈنى ئايالىم ئۆي تازىلاۋاتقاندا ئۇنىڭ ئازراق ئۇرۇقلاپ قالغانلىقنى سېزىپ قالدىم. ئۇنىڭدىن چېنىقىۋاتامسەن دەپ سورىسام، ياق چېنىقمىدىم دەيدۇ. ئەمىسە كۆزۈمگە ئۇرۇقلاۋاتقاندەك كۆرىنىۋاتىسەنغۇ دېسەم، ۋايياق، سىزگە شۇنداق بىلىنىۋاتقان ئوخشايدۇ دىدى. مەنمۇ شۇنداق ئوخشايدۇ دەپ يۈرۋېرىتىمەن. بىر كۈنى ئايالىم تۇيۇقسىزلا ئۆزىنىڭ راست ئۇرۇقلاۋاتقانلىقىنى، لېكىن ھېچقانداق چېنىقىش تەدبىرى قوللانمىغانلىقىنى ئېيتتى. لېكىن شۇ مەزگىللەردە يىمەك-ئىچمەكنى كونتۇرۇل قىلىۋاتقانلىقىنى بىلەتتىم، شۇڭا شۇنىڭ پايدىسى بولغان بولسا كېرەك دەپ ئويلاپتىمىز.


كۈنلەر ئۆتىۋەردى، ئايالىم راستىنلا ئۇرۇقلىدى، ئۇرۇقلىسا قاندىكى ماي چۈشىدۇ دەپ بىزمۇ خۇش بولۇپ يۈرىۋېرىپتۇق. بىر كۈنى ئايالىم خىزمەت ئورنىدىلا بېشى قېيىپ يىقىلىپ چۈشۈپتۇ. بۇ خەۋەرنى ئاڭلاپ جان-پېنىم چىقىپ ئۇ ئىشلەيدىغان يەرگە بېرىپ ئۇنى دوختۇرخانىغا ئەكەلدىم. ئۇ تەكشۈرۈش، بۇ تەكشۈرۈش دەپ ئاخىرى قان ئالىدىغان بولدى. ئالسا ئالسۇن مەيلى، مۇشۇ بانادا ھەممە يېرىڭنى تەكشۈرىتىۋال، مەنمۇ تەكشۈرتەي، ھەر يىلى بىر قېتىم سالامەتلىك تەكشۈرتۈش قىلىلى دىدىم. دوختۇر قاننى ئېلىپ بولۇپ ئىككى سائەتتىن كېيىن كېلىشنى ئېيتتى. ئىككى سائەتتىن كېيىن بارساق دوختۇر: ئايالىڭىزدا دىئابىت كېلسىلى باركەن، 16 غا چىقىپتۇ دىمەسمۇ، بېشىمدىن بىر چىلەك سۇ قۇيغاندەك چۆچۈپ كەتتىم. خاتا بوپ قالمىغاندۇ؟ خەقنىڭ ئالمىشىپ قالدىمۇ يە دىگەنلىرىمنىڭ ھەممىسى بىكار بولدى. ئاخىر بولماي قايتا تەكشۈرتمەكچى بولدۇق، لېكىن نەتىجە ئوخشاش چىقتى. دەل شۇ چاغدا ئۈرۈمچىگە كاماندىروپكىغا بارىدىغان بولۇپ قالدىم، شۇڭا ئايالىمنى رۇخسەت سورىتىپ ئۈرۈمچىگە ئېلىپ بېرىپ تەكشۈرتمەكچى بولدۇق. ئۈرۈمچىدىكى ئىككىنچى دوختۇرخانىغا بېرىپ قان ئالدۇردۇق. ئىككىلىمىزدە غەم، بۇرۇنقىدەك ئۈرۈمچىگە كەلسەك نەمۇ-نە مھماندارچىلىققا قاترايدىغان ئىشلرىمىزمۇ بىر ياققا قايرىلىپ قالدى، ئۈرۈمچىدىكى دوست-بۇرادەرلەر بىلەن كۆرۈشىشكىمۇ ھەپسىلىمىز قالمىدى. دىئاگۇنۇزنىڭ خاتا بولۇشىنى ئۈمۈت قىلىپ كەتتۇق، ئەگەر دىئاگىنۇز خاتا بولۇپ قالسا زۇكام بوپ قالساقمۇ ئۈرۈمچىگە كېلىلى دىيىشىپ كەتتۇق. ئايالىمنىڭ ئۈمۈتسىز كۆزلىرىگە قاراپ ئىچىم سىيرلىپ كەتتى. تېخى قىرىققا بارماي تۇرۇپلا بۇنداق بولۇپ قالسا ئۈمۈتسىزلەنمەي تۇرالايتتىمۇ؟ ئۇنىڭ ئۈستىگە ئايالىم بەك ئاسان يىغلايدۇ. ئاخىر نەتىجىمۇ چىقتى، لېكىن دىئاگۇنۇز يۇرتتا تەكشۈرتكەن بىلەن ئوخشاش چىقتى. -دىئابىتكەن، -دېدى دوختۇر كۆز ئەينىكىنىڭ ئاستىدىن بىزگە قاراپ پەرۋاسىزلا، -بايلار كېسىلى بۇ ھازىرقى زاماندا مودا بولىۋاتقان، مودىغا ماسلىشىپسىز، لېكىن قورقماڭ، داۋاسى بار، ئۈزۈل كېسىل ساقايمايدۇ، لېكىن كونتۇرۇل قىلغىلى بولىدۇ. پالانى، پۇكىنىلار شۇنداق بولۇپ قالغىلى 20-30 يىلدەك بولۇپ قالدى، يەنە خىرامان داۋالىنىپ، ئىشلەپ پۇل تېپىپ ئويناپ، ياشاپ كېلىۋاتماتدۇ؟ ئۈمۈتسىزلەنمەڭ، بۇ دەسلەپكى باسقۇچىكەن دەپ يېتىپ داۋالىنىشنى تەۋسىيە قىلدى. لېكىن بىز ھازىر سەل ئالدىراش، بىرەر ئايدىن كېيىن  داۋالانساق دىگەچكە ئۇ نۇرغۇن مەسلىھەت بەرگەن بولدى. ئاندىن گېخۇاجى ۋە نوخىلوڭ دېگەن دورىلاردىن يېزىپ بەردى ۋە ھازىرچە نوخېلوڭنى كۈندە ئىككى ۋاق ئىچىپ بىر ھەپتىدىن كېيىن تەكشۈرتۈڭ، ئەگەر يەنە چۈشمىسە گېخۇاجىنى ئىچىڭ دىدى. شۇنداق قىلىپ ئۈرۈمچىدىمۇ پالتا-كەكىمىزنى قاقتۇرۇپ سالپايغان ھالدا قايتىپ كەلدۇق.


ئايالىم بارا-بارا خاپىغان بولۇپ قالدى. بەلكىم ياش تۇرۇپ دىئابىت بولۇپ قالغانلىقىدىن قايغۇرۇپ كەتكەن بولسا كېرەك، شۇڭا مەنمۇ ئۇنىڭ كۆڭلىنى ئاياپ ئۇنىڭ بەزى ئاچچىق گەپلىرىنىمۇ ئاڭلىماسقا سېلىپ ئۆتكۈزۈپ يۈردۈم. ئايالىم ئارىلىقتا ئۇ دورىنى كۈندە ئەتتىگەن-كەچتە ئىچكەن بولدى. كېيىنچە قارىسام سەل سەمرىدى، كېيىنچە تېخىمۇ سەمرىدى. بۇنىڭغا قاراپ قان شېكىرى تۈۋەنلىدى بولغاي دەپ ئويلىدۇق. گەپنىڭ ئوچۇقىنى قىلغاندا بىز بۇ رىئاللىققا يۈزلىنەلمەيۋاتاتتۇق. لېكىن كۆڭلۈمدە ئاللىقانداق بىر كۆلەڭگە ئەگىپ يۈرۈتتى. ئاخىرى بولماي ئايالىمغا دۇختۇرخانغىا بېرىپ مۇنتىزىم داۋالىنىشنى ئېيتتىم، باشتا ئۇ ئۇنىمىدى. مەن چىڭ تۇرۇىۋالغاندىن كېيىن تەستە بالنىستقا كىرىپ داۋالانماقچى بولدى. شۇنداق قىلىپ يەنە باشتا بىز كىرگەن دۇختۇرخانىغا كىردۇق. تەكشۈرۈشتىن مەلۇم بولدىكى نەتىجە بۇرۇنقى بىلەن ئوپمۇ-ئوخشاش 16 چىقتى. نوخېلوڭنى ئىككى ئاي مۇنتىزىم ئىچكەندىن كېيىن ئىچمىگەن ئىدى. شۇنداقتىمۇ يا تۆۋەنلىمەي يا چۈشمەي تۇرۇپتۇ. دوختۇر بۇنىڭ باشلىنىشى ئىكەنلىكىنى، ۋاقتىدا كەلگەنلىكىمىزنى، تىزگىنلىگىلى بولىدىغانلىقىنى ئېيتىپ بىزنى ئۈمۈتلەندۈرگەن بولدى. لېكىن دۇختۇرخانىغا كىرىپ دورا بىلەن داۋالايمىز دەپ ھېلىقى كۈن كۆرسە بولمايدىغان ئوكۇلدىن ئۇرغاچ گېخۇاجىنى ئىچكۈزدى. 8-9كۈن بولغاندىمۇ شېكەر مىقدارى ئاچ قوساقتا 9 دىن دىن زادىلا تۈۋەنلىمەي تۇرىۋالدى. دوختۇر ئىنسۇلىننى ئوكۇل قىلىپ ئورۇشنىكۆپ قېتىم تەۋسىيە قىلدى. ئىنسۇلىن ئۇرىلى دېسەك بەزىلەر بالدۇر ئىنسۇلىن ئۇرسا ئىنسۇلىنغا كۆنۈپ قالسا بولمايدۇ، دورا بىلەن چۈشۈرگۈلۇق دەپ مەسلىھەت بەدى. بەزىلەر ئىنسۇلىن ئۇرسا ياخشى بولىدىغانلىقىنى ئېيتتى. دوختۇرمۇ ئىنسۇلىن ئۇرسا ياخشى بولىدىغانلىقىنى تەۋسىيە قىلدى، چۈنكى گېخۇاجى نىڭ بۆرەككە تەسىرى بار دىگەندەك قىلدى، بىزمۇ باشقىلاردىن شۇنداق ئاڭلىغان ئىدۇق. بىز زادى ئىنسۇلىن ئىشلەتسە ياخشىمۇ ياكى دورا بىلەن داۋانلانسا ياخشىمۇ دەپ ئىككىلىنىپ يۈردۇق. يەنە شۇ رىئاللىققا يۈزىلىنەلمەيۋاتاتتۇق. ئاخىر دۇختۇر بىزگە جىق خىزمەت ئىشلىدى، قورقۇپ ئۈمۈتسىزلىزنىپ كەتمەسلىكىمىزنى، كېسەلگە چوقۇم ئىلمى قاراشنىڭ كېرەكلىكىنى، چوقۇم رىئاللىققا يۈزلىنىپ ئىنسۇلىن بىلەن داۋالاشنى قوبۇل قىلشىمىزنى تەۋسىيە قىلدى. بىزمۇ بارا-بارا ئۆز-ئۆز ئۆزىمىزنى ئالداپ يۈرىۋەرسەك ياخشى بولمايدىغانلىقىنى ھېس قىلىپ قالدۇق. دوختۇر بەك سەمىمىي ئايالكەن، بىز ئۇنىڭ سەمىمىي كۆيۈنىشلىرىنىڭ تۈرتكىسىدە ئنسۇلىن ئۇرۇشقا قوشۇلدۇق. رىئاللىقنى تەن ئالمىساق بولمىدى. شۇنداق قىلىپئايالىم ئىنسۇلىن ئۇرىۋاتقىلى ئۈچ كۈن بولدى، قان شېكىرىمۇ كۆرىنەرلىك تۆۋەنلىدى. ئەمدىكى مەقسەت: بەلكىم تورداشلىرىمىزنىڭ ئارىسىدىمۇ بەلكىم بۇ كېلسەل بىلەن ئاغرىغانلار، ياكى بولمىسا ئۇرۇق تۇققانلىرى، دوست بۇرادەرلىرى....ئارىسىدا بۇنداق كېسەلگە گىرىپتار بولغان بولىشى مۇمكىن. دىمىسىمۇ ھازىر بۇ بۇ كېسەلگە گىرىپتار بولغانلار ماڭدامدا بىر ئۇچرايدۇ. ھەر قېتىم دۇختۇرخانىغا قان شېكىرىنى تەكشۈرتكىلى كەلگەن ۋاقتىمىزدا نۇرۇغۇن كىشىلەرنىڭ ئۆچرەت ساقلاپ ئولتۇېىۋاتقانلىقىنى كۆرۈپ «ھەممە ئادەم دىئابىت كېسىلى بولۇپ قالدىمۇ قانداق»؟ دەپ ئويلاپ قالىمەن. شۇڭا مۇشۇ كېلسەلگە گىرىپتار بولۇپ داۋالىنىۋاتقانلار ھەم قان شېكىرى نورماللاشقانلارنىڭ بۇ توغرىدا ئۆز تەجرىبىلىرىنى بايان قىلىپ بېقىشىنى، مۇشۇ ساھەدىكى مۇتەخەسسىلەرنىڭ ئۆز مەسلىھەتلىرىنى ئايىماسلىقىنى، شۇنداقلا بۇ تېما ئارقىلىق مۇشۇنداق كېسەلگە گىرىپتار بولغان تۈمەنلىگەن قېرىنداشلىرىمىزنىڭ روھى ۋە جىسمانى جەھەتتىكى ئازاپلىرىغا ئاز بولسىمۇ مەلھەم بېرىشىنى ئۆتۈنىمەن. ئەگەر شۇنداق بولسا بۇ تېما ئۆز ۋەزىپسىنى ئادا ئەتكەن بولاتتى...

خەتلەر نورمال كۆرۈنمىدى.

arislan يوللانغان ۋاقتى 2012-12-21 21:21:57



ئەمەلىيەتتە دىئابىت كېسىلى ئۇنچىلىك قورقۇنچلۇق كېسەلمۇ ئەمەس. يېمەك- ئىچمەككە دىققەت قىلىپ بەرگەن ئاساستا قان قەنتىنى كونتىرول قىلىپ بەرسە ھەمدە ئەگەشمە كېسەللىكلەرنىڭ ئالدى ئېلىنسا ئاللاھ خالىسا ساق ۋاقتىدىكىدەك يۈرگىلى بولىدۇ.

مەن ئىلگىرى بىر مەزگىل تېببي جەۋھەر شىركىتىنىڭ ئېلان- تەشۋىقات بۆلۈمىگە مەسئۇل بولۇپ ئىشلىگەن ئىدىم. شىركەتتە تېبابەتكە مۇناسىۋەتلىك كىتابلار شۇنچىلىك كۆپ ئىدىكى، نەچە ئايلاپ چىقماي ئوقۇسىمۇ ئوقۇپ بولغىلى بولمايتتى. كىتابخۇمارلىقىم سەۋەبلىك بەزىدە ھەرخىل ماتىرىياللارنى كۆرۈپ چىقاتتىم. شۇ شىركەتتە ئىشلەش جەريانىمدا دىئابىت كېسىلىنى خېلى چوڭقۇر چۈشەندىم. شۇ شىركەتنىڭ قەنت كاپسۇلى دەيدىغان ئۇيغۇر تېبابەت مەھسۇلاتى بار. شىركەتنىڭ ئەڭ تېز يۈرۈشىۋاتقان مەھسۇلاتىمۇ شۇ. شىركەتتە تۇرۇش جەريانىدا نۇرغۇن دىئابىت بىمارلىرىنىڭ تېلېفونىنى قوبۇل قىلاتتۇق. ئۈرۈمچى شەھەر ئاھالىسىمۇ بىۋاستە شىركەتتىن سېتىۋالاتتى. ئىنسولىن ئۇرۇۋاتقىلى 10 نەچچە يىل بولغان ئادەملەرمۇ ئىنسولۇننى توختىتىپ شۇ مەھسۇلات بىلەن كونتىرول قىلىۋاتقان. مەن سىزگە شۇ مەھسۇلاتنى تەۋسىيە قىلىش بىلەن بىرگە دىئابىتكە مۇناسىۋەتلىك ماتىرىياللار ئۇلىنىشىنى قالدۇراي كۆرۈپ بېقىڭ. بۇرۇن بۇ ماتىرىياللارنى ھەرخىل پايدىلىنىش كىتابلىرى، ئۇيغۇر تېبابىتىگە ئائىت كىتابلاردىن، خەنزۇچە تور بىكەتلەردىكى ماتىرىياللارنى تەرجىمە قىلدۇرۇپ بىرمۇ- بىر ئۇرۇپ كىرگۈزگەن ئىدىم. ئايالىڭىزگە پايدىسى تىگىپ قېلىشى مۇمكىن.
