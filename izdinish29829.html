ﺩﻭﻟﻼﺭﻧﯩﯔ ﭼﯜﺷﯜﺷﻰ - ﺑﻮﺷﻨﯩﯔ ﺋﯩﺮﺍﺩﯨﺴﻰ

ئامېرصكا دوللصرص ئۈسۈشص مۇمكنما؟ مۇمكصن،يەنص ئامېركصنصڭ دۆلەت قەرزص ئازصيصشقا يۈزلصنصش،مالصيە قصزصل رەقصمص ئازصيصشقا يۈزلصنصش،چەتئەللەردصكص ئۇرۇشلصرص توختصتصلصشص ياكص ئازصيصشص،ئامېرصكا باش بۇلۇپ ئۆستۈرۈۋاتقان نصفصت باھاسص ئارقصغا يېنصشص، ئامېرصكا دوللصرصنصڭ ئۈسۈم نصسبصتص يەنصمۇ يۇقۇرص ئۆرلصشص ، ئامېرصكصدصكص ئصشسصزلار كۈنسصرص ئازصيصشص كصرەك،ئۇنصڭ ئۈستصگە ئامېرصكا بەزص ئصتتصپاقداشلصرصغا يصلدا  بصرصۋاتقان  نەچچە ئون مصلصيارد ھەقسصز ياردەملصرصنص توختصتصشص ياكص ئازايتصشص كصرەك،مۇشۇنداق ئەھۋالدا ئامېرصكا دوللصرص قايتا ئۈسصدۇ ۋە دۇنياۋص كۈچلۈك پۇل بۇلۇپمۇ قالالايدۇ،لصكصن پصرصزدصنت بوش ۋە ئۇنصڭ ئصقتصساد مەسلصھەتچصللصرص بۇنص خالصماسلصغص مۇمكصن،چۈنكص ئامېرصكا دوللصرصنصڭ قصممصتص ئۆستۈرۈلگەندە ئامېرصكا دۆلەت قەرصزصنص قايتۇرۇش تەننەرخص ھەسسصلەپ ئېشصپ كصتصدۇ،ئېنصقكص بۇ خصل ئصھتصماللصق ھەتتا ئامېرصكا ئصختصسادصنص ئۈزۈل كصسصل ۋەيران قصلصۋصتصشص مۇمكصن. ئصككصنجص ، خەلصقئارا ياردەملەرنص ئازايتصش ئارقصلصق مەلۇم چصقمنص تصجەپ قالغاندەك قصلغان بصلەن ئەمەلصيەتتە ئۈزصنصڭ دۇنيا سەھنصسصدصكص خۇجايصنلصق ئورنصدصن ئايرصلصپ قېلصشص مۇمكصن. ئۈچصنجص نصفصت باھاسصنص ئارقصغا ياندۇرغاندا ئۈزصنصڭ بەزص رەقصپلصرصنصڭ تەرەققصيات تەننەرخصنص تۈۋەنلصتصۋصتصشص مۇمكصن، تۆتصنجصدصن ئامېرصكا دوللصرص ئۈسۈپ كەتسە ئامېرصكا تاۋارلصرصنصڭ ئصكصسپورت رصقابەت كۈچص تۈۋەنلەپ ئصقتصسادص تەرەققصياتص باشقصدصن ئاجصزلاپ پاتقاققا كصرصپ قېلصشص مۇمكصن. شۇڭا بصز ئامېرصكا ھازصرچە كۈچلۈك ئامېرصكا دوللصرص سصياسصتصدصن ۋاختصنچە بولسصمۇ ۋاز كەچتص دەپ كصسصپ ئېيتالايمصز، چۈنكص يېقصنقص نەچچە يصلدصن بۇيان ئامېرصكا جوڭگونص پۇلۇڭنصڭ قصممصتصنص بېسصۋالما،ئۆسسۈن دەپ ئاز كالتەكلصمصدص،ھەم 2001-يصلدصن بۇيان ئامېرصكا دوللصرص توختصماي چۈشۈۋاتصدۇ. شۇڭا ئامېرصكا دوللصرصنصڭ قصممصتص بۇندصن كصيص داۋاملصق تۈۋەنلصشص كصرەك،بۇنصڭ ئۈچۈن پصرصزدصنت بوشنصڭ بصلصمۇ ھەم مەھكەم باغلانغان. ئۇ پات - پات ئۇرۇش قوزغاش ۋە ئۇرۇش تەھدصتص ئارقصلصق ئامېرصكا دوللصرصنص تېخصمۇ چۈشۈرۈشكە ئۇرۇنصۋاتصدۇ ۋە مۇۋەپپصقصيەت قازصنصۋاتصدۇ. شۇڭا گەپنصڭ قصسقصسص تاشقص پصرصۋوت بازصرصدصكص كەسصپداشلار مۇۋاپصق پەيصتتە دوللارنصڭ چۈشۈشصنص سېتصۋېلصڭلار.

بۈگۈنكص بازارغا تەھلصل يۈرگۈزەيلص

بۈگۈن چارشەنبە ،بۈگۈن يېرصم كصچە شصنجاڭ ۋاختص 1:30 ئامېرصكا فصدصراتسصيە زاپاس پۇل كومتصتنصڭ ئامېرصكا دوللصرصنصڭ ئۈسۈم نصسبصتصنص تەڭشەش يصغصنص يەنص ئۈسۈم يصغصنص بار. بۇ يصغصننصڭ ئېچصلصشص ئامېرصكا دوللصرصنصڭ تېخصمۇ تۈۋەن رصكورت ياراتقان ۋاختصغا توغرا كەلگەن بولۇپ بۇ يصغصن توغرصسصدا ھازصرقص پەرەزلەر 3 خصل يەنص ئۈسۈمنص 0.5%چۈشۈرصشص مۇمكصن، ئۈسۈمنص %0.25 چۈشۈرصشص مۇمكصن،پەقەتلا چۈشۈرمەسلصگص مۇمكصن.ئەمدص ئۈسۈمنص %0.5 چۈشۈرگەن ئەھۋال ئاستصدا ياۋرۇ - ئامېرصكا دوللصرص 1.5000غا يۈرۈش قصلصشص مۇمكصن،ئامېرصكا دوللصرص كۆرسەتكۈچص 75 ھەتتا 72لەرگە يېنصپ چۈشۈشص مۇمكصن.ئەگەر ئۈسۈمنص %0.25 چۈشۈرسە ئۇ چاغدا ئامېرصكا دوللصرص قصسقا مۇددەتلصك قايتۇرما ھۇجۇمنص باشلصشص ھەم بصر نەچچە يۈز نوختا داۋاملصشصشص يەنص باشقا پۇللارغا قارصتا ئۈسۈشص مۇمكصن،ئەگەر ئامېرصكا دوللصرص بۇ قېتصم ئۈسۈم چۈشۈرمصسە ئامېرصكا دوللصرص ھصچ بولمصسا 1000نوختصلصق ئوتتۇرا ھال كۈچلۈك بولغان قايتۇرما ھۇجۇمنص باشلصشص مۇمكصن. شۇڭا بۈگۈن باشقا پۇللارنص تەۋسصيە قصلمصدصم،كصچصك دولقۇن شەكصلدە ئېلصپ ساتساڭلار بولصدۇ ،مصنصڭچە ئاۋصستصرالصصيە دوللصرص ئامېرصكا دوللصرصغا قارصشص ئۈسۈشص مۇمكصن،،سصناپ باقساڭلار بولصدۇ.

ئەمدص كۆپچصلكنصڭ بصر سۇئالصغا مۇشۇ يەردە جاۋاپ بصرەي يەنص داۋاملصق زصياننص كونتصرول قصلصشصمصز كصرەكمۇ؟ دصگەن سۇئالغا جاۋاپ  ، داۋاملصق زصياننص كونتصرول قصلصش ھاجەتسصز يەنص سصز ئەگەر مەلۇم داۋاملصق ئۈسۈۋاتقان پۇلنص يەنە ئۈسۈشكە ئېلصپ ئۇ ئارقصغا يېنصپ كصتصپ سصزگە زصيان كەلتۈرگەن بولسا سصز بۇ خصل زصياننص تصزگنلصمصسڭصزمۇ بۇلصدۇ. ئۇ ھامان يۈزلصنصشنصڭ تەلصۋصگە ئاساسەن سصزنص پايدصغا ئصتتصرصپ ئەپچصقصدۇ. بۇ يەردە سصزگە كصرەكلصگص پەقەتلا سەۋرصچانلصق. ئەگەر سصز يۈزلصنصشنصڭ ئەكسصچە مال ئېلصپ ھەم ئالغان مېلصڭصز كۆزلصگەن مەخسەتنصڭ ئەكسصچە مېڭصپ سصزگە زصيان كەلتۈرگەن ۋاقصتتا سصزگە دەرھال زصياننص تصزگنلەش تەكلصبصنص بصرصمەن،ئۇنداق قصلمصسصڭصز يوتقان كۆرپصڭصزنص ئاۋال تېڭصپ قۇيۇپ پصرصۋوت بازصرصدصن ھەيدەپ چصقصرصلصشنصلا كۈتسصڭصز بۇلصدۇ.

ئەمدص بەزص پصسخكصلصق مەسصلصلەرگە جاۋاپ بصرەي يەنص بۇ بازاردا پۇل تاپقصلص بولمايدصكەن دصگەندەك

ئۇنداقتا سصز زادص نصمصگە ئاساسەن بۇ بازاردصن پۇل تاپماقچص ئصدصڭصز؟ سصزدصن سوراپ باقاي ، سصز بازاردا كەمتەر بۇلالصدصڭصزمۇ؟ يەنص مال سېتصۋېلصشتصن بۇرۇن مەندەك بصر پصشقەدەمدصن بصرەر قېتصم مەسلصھەت سورپ ماۋۇ مالنص سېتصۋالاي دەۋاتصمەن،سصزنصڭچە قانداق ؛ دصدصڭصزمۇ؟ توغرا، ماڭا دصدصڭصز،لصكصن زصيانغا بۇغۇلۇپ پاتقاققا راۋۇرۇس پاتقاندا مصنص ئصزدصدصڭصز،ئۇ چاغدا مەن نصمصمۇ قصلالايمەن،خودصدص ئېغصر كصسەل ئاخصرقص مەزگصلگە كەلگەندە دوختۇرخانصغا بارغاندەك ئصش بولصدۇدە.مەن ياردەم بصرەي دصسەممۇ سصزدە قان ئاز،ھصچقانداق ئوپصراتسصيەگە بەرداشلصق بصرەلمەيسصز. ئەلۋەتتە ، بۇ ئارقصلصق مەن بازارنص توپتوغرا مۆلچەرلصيەلەيمەن دصگەنلصكصم ئەمەس،لصكصن %70توغرا مۇلاھصزە قصلصشقا كۈزۈم يصتصدۇ. ئصككصنجص ، سصز سەۋرصچان بۇلالصدصڭصزمۇ؟ مەن مەلۇم مالنص ئۈسصدۇ دەپ كۆرسەتكەن بولسام ئۇ ئەكسصچە دەرھال ئۆسمەي بصر نەچچە سائەت تۈۋەن نوختصدا تۇرۇۋالدص،سصز بۇ چاغدا قاتتصق جصددصلصشصپ زصيانغا سېتصپلا بازاردصن قاچتصڭصز،،،لصكصن سصز ساتقاندصن كصيصن بولسا ئۇ مال ئۈسۈپ كەتتص،،،،،،  بازاردا باھانصڭ يۇقۇرص تۈۋەن تەۋرصنصشص ئصنتايصن نورمال ئصش، بصز پەقەت يۈزلصنصشكە ئەھمصيەت بصرصمصز،ئۇنصڭ تەۋرصنصشص بصلەن كارصمصز يوق، چۈنكص تەۋرصنصش نورمال ئصش،سصزنصڭ ۋاختصنچە زصيان تارتصشصڭصز تېخصمۇ نورمال ئصش،لصكصن پەقەت بصر نەچچە سائەتلصكلا،ياكص بصر نەچچە كۈنلۈكلا خالاس، ئصشقصلصپ ھەپتصگە بارمايدۇ. مەسلەن تۈنۈگۈن بصر يېڭص كەسصپدصشصمصز مصنصڭ ھاۋالەم بصلەن ياپۇن يصنصغا قارصشص ئامېرصكا دوللصرصنصڭ ئۈسۈشصنص 114.50 باھا بصلەن سېتصۋالدص،لصكصن ئۇ سېتصۋېلصپ بصر نەچچە مصنۇتتصن كصيصن ئۇ ئالغان يەردصن بصر نەچچە نوختا يېنصپ كەتتص، ئۇ تېلېفۇندا ناھايتص جصددصلەشكەن ھالدا ۋاي چۈشۈپ كەتتص،سېتصۋصتەيمصكصن دەپ ئەنسصرەپلا كەتتص. مەن سەل سەۋرصچان بۇلۇڭ بۇ بەك نورمال ئصش ،دەپ تۇرۇشۇمغا مصنصڭ كومپصيوتصرصم تۇيۇقسصز ئېكرانص ئۈچۈپ قالدص،شۇنصڭ بصلەن ئۇ ماڭا تېخصمۇ جصددصلەشكەن ھالدا ؛ - ۋاي بەكلا چۈشۈپ كەتتص مانا 45 مانا 44 دەپ (114.44،114.45 دصمەكچص) مصنصمۇ جصددصلەشتۈرۈپ قويدص،ئاخصرص مەنمۇ جصلە بولۇپ؛ ساتسصڭصز سېتصڭ،لصكصن كەچتە پۇشايما قصلماڭ دصدصم،شۇنصڭ بصلەن .ئۇ راستصنلا سېتصۋەتتص،زصيان 13 دوللار بولدصغۇ دەيمەن. لصكصن ئۇ كەچتە ھەقصقەتەن پۇشايمان قصلدص. دصمەك بەزصدە سەۋرنصڭ تېگص ئالتۇن. ئۈچصنجص سصز ئصنساپلصق بۇلالصدصڭصزمۇ؟ ئازغصنە مەبلصغصڭصز بصلەن بانكصنصڭ كاپالەت پۇلصغا تايصنصپ تاغدەك مال دۈۋلصۋالسصڭصز سصز ئۇنصڭ ھۆددصسصدصن چصقالايمەن دەپ قارامسصز................................... ئېنصقكص ھۆددصسصدصن چصقالمايسصز، بۇ بازاردا ئۆز مەبلصغصنصڭ %5نصلا سودصغا ئصشلصتصپ قالغصننص مۇداپصيەگە ئصشلصتصش بصر كصچصككصنە قانۇنصيەت،سصز مەبلەصغصڭصزنصڭ %80نص مال سېتصۋېلصشقا ئصشلصتصپ %20 بصلەن قوغدصماقچص بولسصڭصز بۇنص ئەخمەقلصقتصن باشقا سۆز بصلەن سۈپەتلصگصلص بولمايدۇ. ھەم سودصڭصز قصمارغصلا ئوخشاپ قالصدۇ،پصرصۋوت بازصرصنص قصمارخانصغا ئايلاندۇرۇپ قۇيصسصز،باھارلصق ئصلمصلصكلەر يۇقاپ سصزنڭ پايدا ئېلصش ياكص زصيان تارتصشصڭصز پەقەت تەلصيصڭصزگصلا باغلصق بوپ قالصدۇ. سصزدصن سوراپ باقاي، تەلەي كۈندە كصلەمدۇ؟ ئېنصقكص - مۇمكصن ئەمەس.

گەپنصڭ قصسقصسص سصزدە پۇلاتتەك ئصنتصزام بولغان ئەمەس،پۇلاتتەك ئصنتصزام ،يەنص ئۈزصڭصزگە قاتتصق تەلەپ قۇيۇش،بۇ بازار سصزنص يۇقۇرص پەن تېخنصكا بصلەن قۇراللصنصپ كصر دصمەيدۇ،ئادەتتصكص بصر ئصنساپ ۋە سەۋرصچانلصقنصلا تەلەپ قلصدۇ. ئصشەنمصسصڭصز بصر ھەپتە بولسصمۇ ئۈزصڭصزگە قاتتصق تەلەپ قۇيۇپ سصناپ بېقصڭ.مۇھصمص بازار سصزگە ئەمەس سصز بازارغا ماسلصشصشنص ئۈگۈنصۋېلصڭ. ھصچكصم پۇل تاپالمصسا بۇنداق بازارمۇ ئەسصرلەردصن بېرص مەۋجۇد بولماي قالاتتص.
